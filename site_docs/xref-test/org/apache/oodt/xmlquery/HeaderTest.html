85  	public void testXML() throws Exception {
86  		Document doc = XML.createDocument();
87  		Element bogus = doc.createElement("bogus");
88  		try {
89  			Header h0 = new Header(bogus);
90  			fail("Header constructor failed to throw exception when given invalid XML node");
91  		} catch (IllegalArgumentException ignored) {}
92  
93  		Header h1 = new Header("name1", "type1", "unit1");
94  		Node root = h1.toXML(doc);
95  		assertEquals("headerElement", root.getNodeName());
96  		NodeList children = root.getChildNodes();
97  		for (int i = 0; i < children.getLength(); ++i) {
98  			Node child = children.item(i);
99  			if ("elemName".equals(child.getNodeName())) {
100 				assertEquals("name1", XML.text(child));
101 			} else if ("elemType".equals(child.getNodeName())) {
102 				assertEquals("type1", XML.text(child));
103 			} else if ("elemUnit".equals(child.getNodeName())) {
104 				assertEquals("unit1", XML.text(child));
105 			} else fail("Unknown node \"" + child.getNodeName() + "\" in XML result");
106 		}
107 		Header h2 = new Header(root);
108 		assertEquals(h1, h2);
109 	}

111 	public void testMultipleHeaders() throws Exception {
112 		Document doc = XML.createDocument();
113 		Element resultHeader = doc.createElement("resultHeader");
114 		Element headerElement = doc.createElement("headerElement");
115 		resultHeader.appendChild(headerElement);
116 		XML.add(headerElement, "elemName", "name1");
117 		headerElement = doc.createElement("headerElement");
118 		resultHeader.appendChild(headerElement);
119 		XML.add(headerElement, "elemName", "name2");
120 		XML.add(headerElement, "elemType", "type2");
121 		headerElement = doc.createElement("headerElement");
122 		resultHeader.appendChild(headerElement);
123 		XML.add(headerElement, "elemName", "name3");
124 		XML.add(headerElement, "elemUnit", "unit3");
125 		headerElement = doc.createElement("headerElement");
126 		resultHeader.appendChild(headerElement);
127 		XML.add(headerElement, "elemName", "name4");
128 		XML.add(headerElement, "elemType", "type4");
129 		XML.add(headerElement, "elemUnit", "unit4");
130 
131 		List headers = Header.createHeaders(resultHeader);
132 		Header h1 = new Header("name1");
133 		Header h2 = new Header("name2", "type2", null);
134 		Header h3 = new Header("name3", null, "unit3");
135 		Header h4 = new Header("name4", "type4", "unit4");
136 		assertEquals(h1, headers.get(0));
137 		assertEquals(h2, headers.get(1));
138 		assertEquals(h3, headers.get(2));
139 		assertEquals(h4, headers.get(3));
140 	}
