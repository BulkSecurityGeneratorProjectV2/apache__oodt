41  /**
42   * @author mattmann
43   * @author bfoster
44   * @version $Revision$
45   * 
46   * <p>
47   * A simple versioning scheme that versions {@link Product}s with their
48   * production date time.
49   * </p>
50   * 
51   */
52  public class DateTimeVersioner implements Versioner {
53  
54      /* our log stream */
55      private static final Logger LOG = Logger.getLogger(DateTimeVersioner.class.getName());
56  
57      /**
58       * <p>
59       * Default Constructor
60       * </p>
61       */
62      public DateTimeVersioner() {
63          super();
64          // TODO Auto-generated constructor stub
65      }
66  
67      /*
68       * (non-Javadoc)
69       * 
70       * @see org.apache.oodt.cas.versioning.Versioner#createDataStoreReferences(org.apache.
71       *      oodt.cas.data.structs.Product)
72       */
73      public void createDataStoreReferences(Product product, Metadata metadata)
74              throws VersioningException {
75          // first, we need to know if its heirarchical, or flat
76          if (product.getProductStructure() == null) {
77              LOG
78                      .log(Level.WARNING,
79                              "DateTimeVersioner: Product Structure must be defined in order to version!");
80              return;
81          }
82  
83          SimpleDateFormat dateFormatter = new SimpleDateFormat("yyyyddMM.HHmmss");
84          String productionDateTime = metadata
85                  .getMetadata("CAS.ProductReceivedTime");
86          if (productionDateTime == null) { // generate it ourselves then
87              productionDateTime = dateFormatter.format(new Date());
88          } else {
89              Date prodDateTime;
90              try {
91                  prodDateTime = DateConvert.isoParse(productionDateTime);
92              } catch (ParseException e) {
93                  LOG.log(Level.WARNING,
94                          "Unable to parse production date time CAS.ProductReceivedTime: "
95                                  + productionDateTime
96                                  + ": generating it ourselves");
97                  prodDateTime = new Date();
98              }
99              productionDateTime = dateFormatter.format(prodDateTime);
100         }
101 
102         if (product.getProductStructure().equals(Product.STRUCTURE_FLAT)) {
103             // if its flat, just return references that include
104             // productTypeRepo/productName/fileName.productionDateTime
105             // we'll use the format: yyyy.dd.MM.HH.mm.ss
106 
107           for (Reference r : product.getProductReferences()) {
108             String dataStoreRef;
109 
110             try {
111               dataStoreRef = new File(new URI(product.getProductType()
112                                                      .getProductRepositoryPath())).toURL()
113                                                                                   .toExternalForm()
114                              + "/"
115                              + product.getProductName()
116                              + "/"
117                              + new File(new URI(r.getOrigReference())).getName()
118                              + "." + productionDateTime;
119               LOG.log(Level.FINER,
120                   "DateTimeVersioner: Generated dataStoreRef: "
121                   + dataStoreRef + " from original ref: "
122                   + r.getOrigReference());
123               r.setDataStoreReference(dataStoreRef);
124             } catch (URISyntaxException e) {
125               LOG
126                   .log(
127                       Level.WARNING,
128                       "DateTimeVersioner: Error generating URI to get name "
129                       + "of original ref while generating data store ref for orig ref: "
130                       + r.getOrigReference()
131                       + ": Message: " + e.getMessage());
132               // try and keep generating
133             } catch (MalformedURLException e) {
134               LOG.log(Level.WARNING,
135                   "DateTimeVersioner: Error getting URL for product repository path "
136                   + product.getProductType()
137                            .getProductRepositoryPath()
138                   + ": Message: " + e.getMessage());
139               // try and keep generating
140             }
141 
142           }
143         } else if (product.getProductStructure().equals(Product.STRUCTURE_STREAM)) {
144                 VersioningUtils.createBasicDataStoreRefsStream(product.getProductName(), product.getProductType().getProductRepositoryPath(),
145                         product.getProductReferences(), productionDateTime);
146         } else if (product.getProductStructure().equals(
147                 Product.STRUCTURE_HIERARCHICAL)) {
148             // if its heirarchical, then we'll version the files within the
149             // directory
150             // first, we need to get a list of all the references from the
151             // original dir
152             Reference origDirRef = (Reference) product.getProductReferences()
153                     .get(0);
154 
155             try {
156                 String dataStoreRef = new File(new URI(product.getProductType()
157                         .getProductRepositoryPath())).toURL().toExternalForm()
158                         + URLEncoder.encode(product.getProductName(), "UTF-8")
159                         + "/";
160                 LOG.log(Level.INFO,
161                         "DateTimeVersioner: generated DataStore ref: "
162                                 + dataStoreRef + " from origDirRef: "
163                                 + origDirRef.getOrigReference());
164                 origDirRef.setDataStoreReference(dataStoreRef);
165             } catch (MalformedURLException e) {
166                 LOG.log(Level.WARNING,
167                         "DateTimeVersioner: MalformedURLException while generating "
168                                 + "initial data store ref for origRef: "
169                                 + origDirRef.getOrigReference());
170                 throw new VersioningException(e);
171             } catch (URISyntaxException e) {
172                 LOG.log(Level.WARNING,
173                         "DateTimeVersioner: Error creating File reference from original dir URI: "
174                                 + origDirRef.getOrigReference() + ": Message: "
175                                 + e.getMessage());
176                 throw new VersioningException(
177                         "Unable to generate references from original dir reference for product: "
178                                 + product.getProductName() + ": Message: "
179                                 + e.getMessage());
180             } catch (UnsupportedEncodingException e) {
181                 LOG.log(Level.WARNING,
182                         "DateTimeVersioner: UnsupportedEncodingException while generating "
183                                 + "initial data store ref for origRef: "
184                                 + origDirRef.getOrigReference());
185                 throw new VersioningException(e);
186             }
187 
188             // create the basic data store refs
189             VersioningUtils.createBasicDataStoreRefsHierarchical(product
190                     .getProductReferences());
191 
192             // add the production date time
193             addProductDateTimeToReferences(product.getProductReferences(),
194                     productionDateTime);
195 
196         }
197     }
198 
199     private void addProductDateTimeToReferences(List<Reference> references,
200             String productionDateTime) {
201       for (Reference r : references) {
202         if (!r.getOrigReference().endsWith("/")) {
203           r.setDataStoreReference(r.getDataStoreReference() + "."
204                                   + productionDateTime);
205         }
206       }
207     }
