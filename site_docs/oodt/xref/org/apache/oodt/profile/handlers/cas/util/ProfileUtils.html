19  package org.apache.oodt.profile.handlers.cas.util;

21  //CAS imports
22  import java.util.List;
23  
24  import org.apache.oodt.cas.filemgr.structs.Product;

28  import org.apache.oodt.profile.EnumeratedProfileElement;
29  import org.apache.oodt.profile.Profile;
30  import org.apache.oodt.profile.ProfileAttributes;
31  import org.apache.oodt.profile.ResourceAttributes;

33  /**
34   * @author mattmann
35   * @version $Revision$
36   * 
37   * <p>
38   * A set of utility methods to aid in CAS Filemgr vocabulary conversion to OODT
39   * {@link Profile} vocabulary.
40   * </p>
41   */
42  public final class ProfileUtils {
43  
44    private static final String PROF_ID_PRE = "urn:oodt:profile:";
45  
46    private static final String PROF_TYPE = "profile";
47  
48    private ProfileUtils() throws InstantiationException {
49      throw new InstantiationException("Don't construct utility classes!");
50    }
51  
52    /**
53     * Builds a {@link Profile} from the given {@link Product}, and its
54     * {@link Metadata}.
55     * 
56     * @param p
57     *          The given {@link Product}.
58     * @param met
59     *          The given {@link Metadata}.
60     * @param dataDelivBaseUrl
61     *          A string representation of a {@link URL} to the data delivery
62     *          service for this particular {@link Product}. This is used, in
63     *          combination with the {@link Product}'s ID, to populate the
64     *          {@link ResourceAttributes} <code>resLocation</code> field.
65     * @return An OODT {@link Profile} representation pointing at the given
66     *         {@link Product}.
67     */
68    public static Profile buildProfile(Product p, Metadata met,
69        String dataDelivBaseUrl) {
70  
71      Profile prof = new Profile();
72  
73      ProfileAttributes profAttrs = new ProfileAttributes();
74      profAttrs.setID(PROF_ID_PRE + p.getProductId());
75      profAttrs.setRegAuthority("CAS");
76      profAttrs.setType(PROF_TYPE);
77  
78      prof.setProfileAttributes(profAttrs);
79  
80      ResourceAttributes resAttrs = new ResourceAttributes();
81      resAttrs.setDescription(p.getProductType().getDescription());
82      resAttrs.setIdentifier(p.getProductId());
83      resAttrs.setResClass(p.getProductType().getName());
84      resAttrs.setTitle(p.getProductName());
85      resAttrs.getPublishers().add("CAS");
86      resAttrs.getResLocations().add(
87          dataDelivBaseUrl + "?productID=" + p.getProductId());
88  
89      prof.setResourceAttributes(resAttrs);
90  
91      // build up profile elements
92      for (String key : met.getMap().keySet()) {
93        List vals = met.getAllMetadata(key);
94  
95        EnumeratedProfileElement elem = new EnumeratedProfileElement(prof);
96        System.out.println("Adding [" + key + "]=>" + vals);
97        elem.setName(key);
98        elem.getValues().addAll(vals);
99        prof.getProfileElements().put(key, elem);
100     }
101     
102     
103     return prof;
104 
105   }
