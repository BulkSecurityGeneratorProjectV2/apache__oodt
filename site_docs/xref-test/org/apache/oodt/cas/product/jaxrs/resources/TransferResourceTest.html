42  /**
43   * Implements tests for methods in the {@link TransferResource} class.
44   * @author rlaidlaw
45   * @version $Revision$
46   */
47  public class TransferResourceTest
48  {
49    /**
50     * Tests that {@link TransferResource transfer resources} are marshalled to
51     * the expected XML format.
52     * @throws IOException if the {@link Diff} constructor fails
53     * @throws JAXBException if the {@link JAXBContext} or {@link Marshaller} fail
54     * @throws MimeTypeException if {@link MimeTypes#forName(String)} fails
55     * @throws SAXException if the {@link Diff} constructor fails
56     */
57    @Test
58    public void testXmlMarshalling() throws IOException, JAXBException,
59      MimeTypeException, SAXException
60    {
61      // Create a TransferResource using ProductType, Product, Metadata, Reference
62      // and FileTransferStatus instances.
63      Hashtable metadataEntries = new Hashtable<String, Object>();
64      metadataEntries.put("CAS.ProductReceivedTime", "2013-09-12T16:25:50.662Z");
65      Metadata metadata = new Metadata();
66      metadata.addMetadata(metadataEntries);
67  
68      Reference reference = new Reference("original", "dataStore", 1000,
69        new MimeTypes().forName("text/plain"));
70  
71      ProductType productType = new ProductType("1", "GenericFile", "test type",
72        "repository", "versioner");
73  
74      Product product = new Product();
75      product.setProductId("123");
76      product.setProductName("test product");
77      product.setProductStructure(Product.STRUCTURE_FLAT);
78      product.setProductType(productType);
79  
80      FileTransferStatus status = new FileTransferStatus(reference, 1000, 100,
81        product);
82  
83      TransferResource resource = new TransferResource(product, metadata, status);
84  
85  
86      // Generate the expected output.
87      String expectedXml =
88        "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"yes\"?>"
89        + "<transfer>"
90        + "<productName>" + product.getProductName() + "</productName>"
91        + "<productId>" + product.getProductId() + "</productId>"
92        + "<productTypeName>" + productType.getName() + "</productTypeName>"
93        + "<dataStoreReference>"
94        +   reference.getDataStoreReference()
95        + "</dataStoreReference>"
96        + "<origReference>"
97        +   reference.getOrigReference()
98        + "</origReference>"
99        + "<mimeType>" + reference.getMimeType().getName() + "</mimeType>"
100       + "<fileSize>" + reference.getFileSize() + "</fileSize>"
101       + "<totalBytes>" + reference.getFileSize() + "</totalBytes>"
102       + "<bytesTransferred>"
103       +    status.getBytesTransferred()
104       + "</bytesTransferred>"
105       + "<percentComplete>"
106       +    status.computePctTransferred() * 100
107       + "</percentComplete>"
108       + "<productReceivedTime>"
109       +    metadata.getAllValues().get(0)
110       + "</productReceivedTime>"
111       + "</transfer>";
112 
113     // Set up a JAXB context and marshall the DatasetResource to XML.
114     JAXBContext context = JAXBContext.newInstance(resource.getClass());
115     Marshaller marshaller = context.createMarshaller();
116     StringWriter writer = new StringWriter();
117     marshaller.marshal(resource, writer);
118 
119     // Compare the expected and actual outputs.
120     XMLUnit.setIgnoreWhitespace(true);
121     XMLUnit.setIgnoreComments(true);
122     XMLUnit.setIgnoreAttributeOrder(true);
123     Diff diff = new Diff(expectedXml, writer.toString());
124     assertTrue("The output XML was different to the expected XML: "
125       + diff.toString(), diff.identical());
126   }
