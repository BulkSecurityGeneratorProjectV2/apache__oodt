60    private static final FileFilter ZIP_FILTER = new FileFilter() {
61      public boolean accept(File file) {
62        return file.isFile() && file.getName().endsWith(".zip");
63      }

68    public static String createDatasetZipFile(ProductType type,
69        String workingDirPath) throws IOException, CasProductException {
70      String datasetZipFileName = type.getName() + ".zip";
71      workingDirPath += workingDirPath.endsWith("/") ? "" : "/";
72      String datasetZipFilePath = workingDirPath + datasetZipFileName;
73  
74      // try and remove it first
75      if (!new File(datasetZipFilePath).delete()) {
76        LOG.log(Level.WARNING, "Attempt to remove temp dataset zip file: ["
77            + datasetZipFilePath + "] failed.");
78      }
79  
80      // get a list of all the product zip files within the temp dir
81      // assumption: the temp dir provided has a whole bunch of *.zip files
82      // that are zips of the products (and their files and metadata)
83      // belonging
84      // to this dataset
85  
86      // NOTE: it is important that this step be done BEFORE creating the zip
87      // output stream: else that will cause the generated datset zip to be
88      // included as well!
89      File[] productZipFiles = new File(workingDirPath).listFiles(ZIP_FILTER);
90      if (productZipFiles == null || productZipFiles.length == 0)
91      {
92        throw new CasProductException("No product zip files to include in dataset: ["
93            + type.getName() + "]");
94      }
95  
96      // now get a reference to the zip file that we want to write
97      ZipOutputStream out = new ZipOutputStream(new FileOutputStream(
98          datasetZipFilePath));
99  
100     for (File productZipFile : productZipFiles) {
101       String filename = productZipFile.getName();
102       FileInputStream in = new FileInputStream(productZipFile
103           .getAbsoluteFile());
104       addZipEntryFromStream(in, out, filename);
105       in.close();
106 
107       if (!productZipFile.delete()) {
108         LOG.log(Level.WARNING, "Unable to remove tempoary product zip file: ["
109                                + productZipFile.getAbsolutePath() + "]");
110       } else {
111         LOG.log(Level.INFO, "Deleting original product zip file: ["
112                             + productZipFile.getAbsolutePath() + "]");
113       }
114     }
115 
116     // add met file
117     addMetFileToProductZip(type.getTypeMetadata(), type.getName(), out);
118 
119     // Complete the ZIP file
120     out.close();
121 
122     // return the zip file path
123     return datasetZipFilePath;
124 
125   }

127   public static String createProductZipFile(Product product, Metadata metadata,
128       String workingDirPath) throws IOException {
129     String productZipFileName = product.getProductName() + ".zip";
130     workingDirPath += workingDirPath.endsWith("/") ? "" : "/";
131     String productZipFilePath = workingDirPath + productZipFileName;
132 
133     // try and remove it first
134     if (!new File(productZipFilePath).delete()) {
135       LOG.log(Level.WARNING, "Attempt to remove temp zip file: ["
136           + productZipFilePath + "] failed.");
137     }
138 
139     // now get a reference to the zip file that we want to write
140     ZipOutputStream out = new ZipOutputStream(new FileOutputStream(
141         productZipFilePath));
142 
143     for (Reference r : product.getProductReferences()) {
144       try {
145         File prodFile = new File(new URI(r.getDataStoreReference()));
146         if (prodFile.isDirectory()) {
147           LOG.log(Level.WARNING, "Data store reference is a directory. Not adding directory to the zip file: ["
148                                  + r.getDataStoreReference() + "]");
149           continue;
150         }
151         String filename = prodFile.getName();
152         FileInputStream in = new FileInputStream(prodFile.getAbsoluteFile());
153         addZipEntryFromStream(in, out, filename);
154         in.close();
155       } catch (URISyntaxException e) {
156         LOG.log(Level.WARNING, "Unable to get filename from uri: ["
157                                + r.getDataStoreReference() + "]");
158       }
159 
160     }
161 
162     // add met file
163     addMetFileToProductZip(metadata, product.getProductName(), out);
164 
165     // Complete the ZIP file
166     out.close();
167 
168     // return the zip file path
169     return productZipFilePath;
170 
171   }

173   /**
174    * Guess what the content type is from the given url.
175    * 
176    * @param name
177    *          URL to reference.
178    * @return A guessed content type.
179    */
180   public static String guessTypeFromName(String name) {
181     name = name.toLowerCase();
182     if (name.endsWith(".jpg") || name.endsWith(".jpeg")) {
183       return "image/jpeg";
184     } else if (name.endsWith(".png")) {
185       return "image/png";
186     } else if (name.endsWith(".gif")) {
187       return "image/gif";
188     } else if (name.endsWith(".doc")) {
189       return "application/msword";
190     } else if (name.endsWith(".pdf")) {
191       return "application/pdf";
192     } else if (name.endsWith(".rtf")) {
193       return "application/rtf";
194     } else if (name.endsWith(".xls")) {
195       return "application/vnd.ms-excel";
196     } else if (name.endsWith(".ppt")) {
197       return "application/vnd.ms-powerpoint";
198     } else if (name.endsWith(".html") || name.endsWith(".htm")) {
199       return "text/html";
200     } else if (name.endsWith(".xml")) {
201       return "text/xml";
202     } else if (name.endsWith(".txt")) {
203       return "text/plain";
204     }
205     return "application/octet-stream";
206   }

208   private static void addMetFileToProductZip(Metadata productMet,
209       String metFileBaseName, ZipOutputStream out) throws IOException {
210 
211     // get the product metadata, and add its met file to the stream
212     ByteArrayOutputStream metOut = new ByteArrayOutputStream();
213     SerializableMetadata serialMet = new SerializableMetadata(productMet);
214     serialMet.writeMetadataToXmlStream(metOut);
215     ByteArrayInputStream metIn = new ByteArrayInputStream(metOut.toByteArray());
216     String metFileName = metFileBaseName + ".met";
217     addZipEntryFromStream(metIn, out, metFileName);
218   }
219 
220   private static void addZipEntryFromStream(InputStream is, ZipOutputStream os,
221       String filename) throws IOException {
222     byte[] buf = new byte[INT];
223     os.putNextEntry(new ZipEntry(filename));
224 
225     int len;
226     while ((len = is.read(buf)) > 0) {
227       os.write(buf, 0, len);
228     }
229 
230     // Complete the entry
231     os.closeEntry();
232   }
