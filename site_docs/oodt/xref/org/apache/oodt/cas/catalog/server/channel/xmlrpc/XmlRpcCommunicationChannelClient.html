62  	public XmlRpcCommunicationChannelClient(URL serverUrl, int connectionTimeout, int requestTimeout, int chunkSize) {
63  		super();
64          CommonsXmlRpcTransportFactory transportFactory = new CommonsXmlRpcTransportFactory(serverUrl);
65          transportFactory.setConnectionTimeout(connectionTimeout * INT * 1000);
66          transportFactory.setTimeout(requestTimeout * INT * 1000);
67  		this.client = new XmlRpcClient(serverUrl, transportFactory);
68  		this.chunkSize = chunkSize;
69  	}

248 	public void transferUrl(URL fromUrl, URL toURL) throws CatalogException {
249 		System.out.println("Transfering '" + fromUrl + "' to '" + toURL + "'");
250         FileInputStream is = null;
251         try {
252             byte[] buf = new byte[this.chunkSize];
253 	        is = new FileInputStream(new File(fromUrl.getPath()));
254             int offset = 0;
255             int numBytes;
256 	        while ((numBytes = is.read(buf, offset, chunkSize)) != -1) {
257 			  this.transferFile(new File(toURL.getPath()).getAbsolutePath(), buf, offset, numBytes);
258 			}
259         } catch (FileNotFoundException e) {
260 		  throw new CatalogException("Transfer URL Failed: "+ e.getMessage(), e);
261 		} catch (IOException e) {
262 		  throw new CatalogException("Transfer URL Failed: "+ e.getMessage(), e);
263 		} finally {
264         	try {
265         		is.close();
266         	}catch(Exception ignored) {}
267         }
268 	}

272         Vector<Object> argList = new Vector<Object>();

275         argList.add(offset);
276         argList.add(numBytes);
277 	  try {
278 		client.execute(XmlRpcCommunicationChannelServer.class.getSimpleName() + ".xmlrpc_transferFile", argList);
279 	  } catch (XmlRpcException e) {
280 		throw new CatalogException("Transfer File Failed: "+ e.getMessage(), e);
281 	  } catch (IOException e) {
282 		throw new CatalogException("Transfer File Failed: "+ e.getMessage(), e);
283 	  }
