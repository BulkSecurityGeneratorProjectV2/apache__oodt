35    /**
36     * Tests the readNamespaces method.
37     */
38    public void testReadNamespaces()
39    {
40      String testXmlString = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
41          + "<rss:rssconf xmlns:rss=\"http://oodt.apache.org/ns/rss\">"
42          + "  <namespace prefix=\"cas\" uri=\"http://oodt.apache.org/ns/cas\"/>"
43          + "  <tag name=\"cas:source\" source=\"[ProductType]\"/>"
44          + "  <namespace prefix=\"pge\" uri=\"http://oodt.apache.org/ns/pge\"/>"
45          + "  <tag name=\"pge:tag\" source=\"[ProductType]\"/>"
46          + "  <namespace prefix=\"rss\" uri=\"http://oodt.apache.org/ns/rss\"/>"
47          + "  <tag name=\"rss:enclosure\">"
48          + "    <attribute name=\"x\" value=\"X\"/>"
49          + "    <attribute name=\"y\" value=\"Y\"/>"
50          + "    <attribute name=\"z\" value=\"Z\"/>"
51          + "  </tag>"
52          + "</rss:rssconf>";
53  
54      Document testXmlDocument =
55          XMLUtils.getDocumentRoot(new ByteArrayInputStream(testXmlString
56              .getBytes()));
57  
58      RSSConfig conf = new RSSConfig();
59      RSSConfigReader.readNamespaces(testXmlDocument.getDocumentElement(), conf);
60  
61      List<RSSNamespace> namespaceList = conf.getNamespaces();
62      assertNotNull(namespaceList);
63      assertFalse(namespaceList.isEmpty());
64      assertEquals(3, namespaceList.size());
65  
66      RSSNamespace namespaceA = namespaceList.get(0);
67      RSSNamespace namespaceB = namespaceList.get(1);
68      RSSNamespace namespaceC = namespaceList.get(2);
69  
70      assertNotNull(namespaceA);
71      assertNotNull(namespaceB);
72      assertNotNull(namespaceC);
73  
74      assertEquals("cas", namespaceA.getPrefix());
75      assertEquals("pge", namespaceB.getPrefix());
76      assertEquals("rss", namespaceC.getPrefix());
77  
78      assertEquals("http://oodt.apache.org/ns/cas",
79          namespaceA.getUri());
80  
81      assertEquals("http://oodt.apache.org/ns/pge",
82          namespaceB.getUri());
83  
84      assertEquals("http://oodt.apache.org/ns/rss",
85          namespaceC.getUri());
86    }

88    /**
89     * Tests the readTags and readAttrs methods.
90     */
91    public void testReadTagsAndAttrs()
92    {
93      String testXmlString = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
94          + "<rss:rssconf xmlns:rss=\"http://oodt.apache.org/ns/rss\">"
95          + " <namespace prefix=\"cas\" uri=\"http://oodt.apache.org/ns/cas\"/>"
96          + " <tag name=\"cas:source\" source=\"[ProductType]\"/>"
97          + " <namespace prefix=\"georss\" uri=\"http://www.georss.org/georss\"/>"
98          + " <tag name=\"georss:point\" source=\"[Latitude] [Longitude]\">"
99          + "  <attribute name=\"one\" value=\"1\"/>"
100         + " </tag>"
101         + " <tag name=\"enclosure\" source=\"\">"
102         + "  <attribute name=\"two\" value=\"2\"/>"
103         + "  <attribute name=\"three\" value=\"3\"/>"
104         + " </tag>"
105         + " <tag name=\"enclosure\">"
106         + "  <attribute name=\"four\" value=\"4\"/>"
107         + "  <attribute name=\"five\" value=\"5\"/>"
108         + "  <attribute name=\"six\" value=\"6\"/>"
109         + " </tag>"
110         + "</rss:rssconf>";
111 
112     Document testXmlDocument =
113         XMLUtils.getDocumentRoot(new ByteArrayInputStream(testXmlString
114             .getBytes()));
115 
116     RSSConfig conf = new RSSConfig();
117     RSSConfigReader.readTags(testXmlDocument.getDocumentElement(), conf);
118 
119     List<RSSTag> tagList = conf.getTags();
120     assertNotNull(tagList);
121     assertFalse(tagList.isEmpty());
122     assertEquals(4, tagList.size());
123 
124     RSSTag tagA = tagList.get(0);
125     RSSTag tagB = tagList.get(1);
126     RSSTag tagC = tagList.get(2);
127     RSSTag tagD = tagList.get(3);
128 
129     assertNotNull(tagA);
130     assertNotNull(tagB);
131     assertNotNull(tagC);
132     assertNotNull(tagD);
133 
134     assertEquals("cas:source", tagA.getName());
135     assertEquals("georss:point", tagB.getName());
136     assertEquals("enclosure", tagC.getName());
137     assertEquals("enclosure", tagD.getName());
138 
139     assertEquals("[ProductType]", tagA.getSource());
140     assertEquals("[Latitude] [Longitude]", tagB.getSource());
141     assertEquals("", tagC.getSource());
142     assertNull(tagD.getSource());
143 
144     List<RSSTagAttribute> attributesA = tagA.getAttrs();
145     List<RSSTagAttribute> attributesB = tagB.getAttrs();
146     List<RSSTagAttribute> attributesC = tagC.getAttrs();
147     List<RSSTagAttribute> attributesD = tagD.getAttrs();
148 
149     assertNotNull(attributesA);
150     assertNotNull(attributesB);
151     assertNotNull(attributesC);
152     assertNotNull(attributesD);
153 
154     assertTrue(attributesA.isEmpty());
155     assertEquals(1, attributesB.size());
156     assertEquals(2, attributesC.size());
157     assertEquals(3, attributesD.size());
158 
159     assertEquals("one", attributesB.get(0).getName());
160     assertEquals("two", attributesC.get(0).getName());
161     assertEquals("three", attributesC.get(1).getName());
162     assertEquals("four", attributesD.get(0).getName());
163     assertEquals("five", attributesD.get(1).getName());
164     assertEquals("six", attributesD.get(2).getName());
165 
166     assertEquals("1", attributesB.get(0).getValue());
167     assertEquals("2", attributesC.get(0).getValue());
168     assertEquals("3", attributesC.get(1).getValue());
169     assertEquals("4", attributesD.get(0).getValue());
170     assertEquals("5", attributesD.get(1).getValue());
171     assertEquals("6", attributesD.get(2).getValue());
172   }
