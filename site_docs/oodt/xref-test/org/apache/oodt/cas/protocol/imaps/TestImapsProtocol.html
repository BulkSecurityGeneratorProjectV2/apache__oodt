40  /**
41   * Test class for {@link ImapsProtocol}.

43   * @author bfoster

45  public class TestImapsProtocol extends TestCase {
46  
47  	private GreenMail gMail;
48  	private ImapsProtocol imapsProtocol;
49  	
50  	@Override
51  	public void setUp() {
52    	System.setProperty("mail.imaps.socketFactory.class", DummySSLSocketFactory.class.getCanonicalName());
53  		System.setProperty("mail.imaps.socketFactory.fallback", "false");
54  		gMail = new GreenMail();
55  		gMail.setUser("bfoster@google.com", "password");
56  		gMail.start();
57  		ImapsProtocol.port = gMail.getImaps().getPort();
58  		imapsProtocol = new ImapsProtocol();
59  		try {
60  			imapsProtocol.connect("localhost", new BasicAuthentication("bfoster@google.com", "password"));
61  		} catch (ProtocolException e) {
62  			fail("Failed to connect to GreenMail IMAPS server : " + e.getMessage());
63  		}
64  		assertEquals(1, ImapsProtocol.connectCalls);
65  	}
66  	
67  	@Override
68  	public void tearDown() {
69  		gMail.stop();
70  		try {
71  			imapsProtocol.close();
72  		} catch (Exception ignored) {}
73  		assertEquals(0, ImapsProtocol.connectCalls);
74  	}
75  	
76  	public void testCDAndPWD() throws ProtocolException {
77  		assertEquals("", imapsProtocol.pwd().getPath());
78  		imapsProtocol.cd(new ProtocolFile("INBOX", true));
79  		assertEquals("/INBOX", imapsProtocol.pwd().getPath());
80  	}
81  	
82  	public void testLSandGET() throws ProtocolException, IOException {
83  		GreenMailUtil.sendTextEmailSecureTest("bfoster@google.com", "tom@bumster.org", "Test Subject", "Test Body");
84  		imapsProtocol.cd(new ProtocolFile("INBOX", true));
85  		List<ProtocolFile> emails = imapsProtocol.ls();
86  		assertEquals(1, emails.size());
87  		File bogusFile = File.createTempFile("bogus", "bogus");
88  		File tmpDir = new File(bogusFile.getParentFile(), "TestImapsProtocol");
89  		bogusFile.delete();
90  		tmpDir.mkdirs();
91  		
92  		File email = new File(tmpDir, "test-email");
93  		imapsProtocol.get(emails.get(0), email);
94  		String[] splitEmail = FileUtils.readFileToString(email, "UTF-8").split("\n");
95  		assertEquals("From: tom@bumster.org", splitEmail[0]);
96  		assertEquals("To: bfoster@google.com", splitEmail[1]);
97  		assertEquals("Subject: Test Subject", splitEmail[2]);
98  		// 3 is divider text (i.e. ----- ~ Message ~ -----)
99  		assertEquals("Test Body", splitEmail[4]);
100 		tmpDir.delete();
101 	}

103 	public void testDelete() throws ProtocolException {
104 		GreenMailUtil.sendTextEmailSecureTest("bfoster@google.com", "tom@bumster.org", "Test Subject", "Test Body");

107 		assertEquals(1, emails.size());

111 	}
