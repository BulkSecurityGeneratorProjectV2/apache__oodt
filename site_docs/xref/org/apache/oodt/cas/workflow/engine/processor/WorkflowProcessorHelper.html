61    private String toString(WorkflowProcessor skeleton, String indent) {
62      StringBuilder stringModel = new StringBuilder(
63          "["
64              + (skeleton.getWorkflowInstance().getParentChildWorkflow().getId() == null ? ""
65                  : "id = '"
66                      + skeleton.getWorkflowInstance().getParentChildWorkflow()
67                          .getId()
68                      + "', name = '"
69                      + skeleton.getWorkflowInstance().getParentChildWorkflow()
70                          .getName() + "', ")
71              + "execution = '"
72              + skeleton.getWorkflowInstance().getParentChildWorkflow()
73                  .getGraph().getExecutionType() + "', state = '"
74              + skeleton.getWorkflowInstance().getState().getName() + "']");
75       if (skeleton.getPreConditions() != null) {
76        stringModel.append(indent).append("{PreCond:").append(indent).append("   ")
77                   .append(toString(skeleton.getPreConditions(), indent + "      ")).append("}");
78      }
79      if (skeleton.getPostConditions() != null) {
80        stringModel.append(indent).append("{PostCond:").append(indent).append("   ")
81                   .append(toString(skeleton.getPostConditions(), indent + "      ")).append("}");
82      }
83      if (skeleton.getSubProcessors() != null) {
84        for (WorkflowProcessor subProcessor : skeleton.getSubProcessors()) {
85          stringModel.append(indent).append(toString(subProcessor, indent + "   "));
86        }
87      }
88      return stringModel.toString();
89    }

130   public WorkflowProcessor buildProcessor(String instanceId,
131       ParentChildWorkflow workflow,
132       Map<String, Class<? extends WorkflowProcessor>> modelToProcessorMap,
133       boolean preCond) throws InstantiationException, IllegalAccessException, ClassNotFoundException {
134     List<WorkflowProcessor> subProcessors = new Vector<WorkflowProcessor>();
135     List<WorkflowCondition> conditions = preCond ? workflow.getPreConditions()
136         : workflow.getPostConditions();
137     for (WorkflowCondition cond : conditions) {
138       ParentChildWorkflow condWorkflow = new ParentChildWorkflow(new Graph());
139       condWorkflow.getGraph().setExecutionType("condition");
140       condWorkflow.getGraph().setCond(cond);
141       subProcessors.add(buildProcessor(instanceId, condWorkflow,
142           modelToProcessorMap, preCond));
143     }
144     WorkflowProcessor wp = buildProcessor(instanceId, workflow, workflow
145         .getGraph().isCondition(), modelToProcessorMap);
146     wp.setSubProcessors(subProcessors);
147     return wp;
148   }

150   public WorkflowProcessor buildProcessor(String instanceId,
151       ParentChildWorkflow model, boolean isCondition,
152       Map<String, Class<? extends WorkflowProcessor>> modelToProcessorMap)
153       throws IllegalAccessException, InstantiationException, ClassNotFoundException {
154     WorkflowProcessor wp = modelToProcessorMap.get(
155         model.getGraph().getExecutionType()).newInstance();
156     WorkflowLifecycle wLifecycle = getLifecycle(model);
157     // FIXME: I'm not sure what these excused processor Ids are. I didn't seem
158     // need them in the PackagedWorkflowRepository, so not sure what they do.
159     wp.getWorkflowInstance().setId(instanceId);
160     if (model.getPreConditions() != null) {
161       wp.setPreConditions(buildProcessor(instanceId, model,
162           modelToProcessorMap, true));
163     }
164     if (model.getPostConditions() != null) {
165       wp.setPostConditions(buildProcessor(instanceId, model,
166           modelToProcessorMap, false));
167     }
168     wp.getWorkflowInstance().setPriority(Priority.getDefault());
169     wp.setMinReqSuccessfulSubProcessors(Integer.parseInt(model.getGraph()
170         .getMinReqSuccessfulSubProcessors()));
171     wp.getWorkflowInstance().setSharedContext(new Metadata());
172     wp.getWorkflowInstance().setState(
173         wLifecycle.createState("Loaded",
174             wLifecycle.getStageForWorkflow("Loaded").getName(), ""));
175     if (wp instanceof TaskProcessor) {
176       ((TaskProcessor) wp)
177           .setInstanceClass((Class<? extends WorkflowTaskInstance>) Class
178               .forName(model.getGraph().getTask().getTaskInstanceClassName()));
179     }
180     return wp;
181   }

183   public WorkflowProcessor findSkeleton(WorkflowProcessor skeleton,
184       String modelId) {
185     if (skeleton.getWorkflowInstance().getParentChildWorkflow().getGraph()
186         .getModelId().equals(modelId)) {
187       return skeleton;
188     }
189     WorkflowProcessor found = findSkeleton(skeleton.getSubProcessors(), modelId);
190     if (found == null) {
191       if (skeleton.getPreConditions() != null) {
192         found = findSkeleton(skeleton.getPreConditions(), modelId);
193       }
194       if (found == null && skeleton.getPostConditions() != null) {
195         found = findSkeleton(skeleton.getPostConditions(), modelId);
196       }
197     }
198     return found;
199   }

201   protected WorkflowProcessor findSkeleton(List<WorkflowProcessor> skeletons,
202       String modelId) {
203     for (WorkflowProcessor skeleton : skeletons) {
204       if (skeleton.getWorkflowInstance().getParentChildWorkflow().getId()
205           .equals(modelId)) {
206         return skeleton;
207       } else {
208         skeleton = findSkeleton(skeleton, modelId);
209         if (skeleton != null) {
210           return skeleton;
211         }
212       }
213     }
214     return null;
215   }

217   public WorkflowProcessor findProcessor(WorkflowProcessor wp, String modelId) {
218     if (wp.getWorkflowInstance().getParentChildWorkflow().getId()
219         .equals(modelId)) {
220       return wp;
221     }
222     WorkflowProcessor found = findProcessor(wp.getSubProcessors(), modelId);
223     if (found == null) {
224       if (wp.getPreConditions() != null) {
225         found = findProcessor(wp.getPreConditions(), modelId);
226       }
227       if (found == null && wp.getPostConditions() != null) {
228         found = findProcessor(wp.getPostConditions(), modelId);
229       }
230     }
231     return found;
232   }

234   protected WorkflowProcessor findProcessor(List<WorkflowProcessor> processors,
235       String modelId) {
236     for (WorkflowProcessor processor : processors) {
237       if (processor.getWorkflowInstance().getId().equals(modelId)) {
238         return processor;
239       } else {
240         processor = findProcessor(processor, modelId);
241         if (processor != null) {
242           return processor;
243         }
244       }
245     }
246     return null;
247   }

259   public boolean allProcessorsSameCategory(
260       List<WorkflowProcessor> workflowProcessors, String categoryName) {
261     for (WorkflowProcessor workflowProcessor : workflowProcessors) {
262       if (!workflowProcessor.getWorkflowInstance().getState().getCategory()
263                             .getName().equals(categoryName)) {
264         return false;
265       }
266     }
267     return true;

317   public List<WorkflowProcessor> toTasks(WorkflowProcessor processor) {
318     List<WorkflowProcessor> options = new Vector<WorkflowProcessor>();
319     options.add(processor);
320     List<WorkflowProcessor> tasks = new Vector<WorkflowProcessor>();
321     while (!options.isEmpty()) {
322       WorkflowProcessor currentOption = options.remove(0);
323       if (currentOption.getSubProcessors().isEmpty()) {
324         tasks.add(currentOption);
325       } else {
326         if (currentOption.getPreConditions() != null) {
327           options.add(currentOption.getPreConditions());
328         }
329         if (currentOption.getPostConditions() != null) {
330           options.add(currentOption.getPostConditions());
331         }
332         for (WorkflowProcessor ps : currentOption.getSubProcessors()) {
333           options.add(ps);
334         }
335       }
336     }
337     return tasks;
338   }

351   public String getHostName() {
352     String host = null;
353     try {
354       host = InetAddress.getLocalHost().getHostName();
355     } catch (Exception e) {
356       try {
357         host = ExecHelper.exec(new String[] { "hostname" }).getOutput().trim();
358       } catch (Exception ignored) {
359       }
360     }
361     if (host == null) {
362       return "Unknown";
363     }
364     return host;
365   }
