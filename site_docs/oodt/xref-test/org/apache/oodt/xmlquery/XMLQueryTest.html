36  	public void testCtor() {
37  		List mimes = new ArrayList();
38  		mimes.add("text/plain");
39  		mimes.add("image/jpeg");
40  		XMLQuery q = new XMLQuery("profStatusId = UNKNOWN OR A > 3 AND RETURN = C",
41  			"id", "title", "description", "dataDictID", "resultModeID", "propType", "propLevels", 45, mimes);
42  		assertEquals("profStatusId = UNKNOWN OR A > 3 AND RETURN = C", q.getKwdQueryString());
43  		assertEquals(45, q.getMaxResults());
44  		assertEquals(1, q.getSelectElementSet().size());
45  		assertEquals(new QueryElement("elemName", "C"), q.getSelectElementSet().get(0));
46  		assertEquals(0, q.getFromElementSet().size());
47  		assertEquals(6, q.getWhereElementSet().size());
48  		assertEquals(new QueryElement("elemName", "profStatusId"), q.getWhereElementSet().get(0));
49  		assertEquals(new QueryElement("LITERAL", "UNKNOWN"), q.getWhereElementSet().get(1));
50  		assertEquals(new QueryElement("RELOP", "EQ"), q.getWhereElementSet().get(2));
51  		assertEquals(new QueryElement("elemName", "A"), q.getWhereElementSet().get(3));
52  		assertEquals(new QueryElement("LITERAL", "3"), q.getWhereElementSet().get(4));
53  		assertEquals(new QueryElement("RELOP", "GT"), q.getWhereElementSet().get(5));
54  		// Need some testing of expressions with LOGOP's, but NOT handling seems broken.
55  		assertEquals(0, q.getResults().size());
56  		assertEquals(2, q.getMimeAccept().size());
57  	}

118 	public void testXML() {
119 		NodeList children;
120 
121 		List mimes = new ArrayList();
122 		mimes.add("text/xml");
123 		mimes.add("image/gif");
124 		XMLQuery q1 = new XMLQuery("Subject < Phrenology OR A > 3 AND RETURN = C",
125 			"id", "title", "description", "dataDictID", "resultModeID", "propType", "propLevels", 45, mimes);
126 		Document doc = q1.getXMLDoc();
127 		Node root = doc.getDocumentElement();
128 		assertEquals("query", root.getNodeName());
129 
130 		Node queryAttributes = root.getFirstChild();
131 		assertEquals("queryAttributes", queryAttributes.getNodeName());
132 		children = queryAttributes.getChildNodes();
133 		for (int i = 0; i < children.getLength(); ++i) {
134 			Node child = children.item(i);
135 			String name = child.getNodeName();
136 			String text = XML.text(child);
137 			if ("queryId".equals(name)) {
138 				assertEquals("id", text);
139 			} else if ("queryTitle".equals(name)) {
140 				assertEquals("title", text);
141 			} else if ("queryDesc".equals(name)) {
142 				assertEquals("description", text);
143 			} else if ("queryType".equals(name)) {
144 				assertEquals("QUERY", text);
145 			} else if ("queryStatusId".equals(name)) {
146 				assertEquals("ACTIVE", text);
147 			} else if ("querySecurityType".equals(name)) {
148 				assertEquals("UNKNOWN", text);
149 			} else if ("queryRevisionNote".equals(name)) {
150 				assertEquals("1999-12-12 JSH V1.0 Under Development", text);
151 			} else if ("queryDataDictId".equals(name)) {
152 				assertEquals("dataDictID", text);
153 			} else fail("Unknown node <" + name + "> under <queryAttributes>");
154 		}
155 
156 		Node queryResultMode = queryAttributes.getNextSibling();
157 		assertEquals("queryResultModeId", queryResultMode.getNodeName());
158 		assertEquals("resultModeID", XML.text(queryResultMode));
159 
160 		Node propogationType = queryResultMode.getNextSibling();
161 		assertEquals("queryPropogationType", propogationType.getNodeName());
162 		assertEquals("propType", XML.text(propogationType));
163 
164 		Node propogationLevels = propogationType.getNextSibling();
165 		assertEquals("queryPropogationLevels", propogationLevels.getNodeName());
166 		assertEquals("propLevels", XML.text(propogationLevels));
167 
168 		Node mimeNode = propogationLevels.getNextSibling();
169 		assertEquals("queryMimeAccept", mimeNode.getNodeName());
170 		assertEquals("text/xml", XML.text(mimeNode));
171 
172 		mimeNode = mimeNode.getNextSibling();
173 		assertEquals("queryMimeAccept", mimeNode.getNodeName());
174 		assertEquals("image/gif", XML.text(mimeNode));		
175 
176 		Node maxResults = mimeNode.getNextSibling();
177 		assertEquals("queryMaxResults", maxResults.getNodeName());
178 		assertEquals("45", XML.text(maxResults));
179 
180 		Node results = maxResults.getNextSibling();
181 		assertEquals("queryResults", results.getNodeName());
182 		assertEquals("0", XML.text(results));
183 
184 		Node kwqString = results.getNextSibling();
185 		assertEquals("queryKWQString", kwqString.getNodeName());
186 		assertEquals("Subject < Phrenology OR A > 3 AND RETURN = C", XML.text(kwqString));
187 
188 		Node node = kwqString.getNextSibling();
189 		assertEquals("queryStatistics", node.getNodeName());
190 		node = node.getNextSibling();
191 		assertEquals("querySelectSet", node.getNodeName());
192 		node = node.getNextSibling();
193 		assertEquals("queryFromSet", node.getNodeName());
194 		node = node.getNextSibling();
195 		assertEquals("queryWhereSet", node.getNodeName());
196 		node = node.getNextSibling();
197 		assertEquals("queryResultSet", node.getNodeName());
198 		assertNull(node.getNextSibling());
199 
200 		XMLQuery q2 = new XMLQuery(root);
201 		assertEquals(q1, q2);
202 	}
