40  /**
41   * MetExtractor which uses input file's name as key for lookup into a sql database to get metadata.
42   *
43   * @author bfoster@apache.org (Brian Foster)
44   */
45  public class DataSourceMetExtractor extends AbstractMetExtractor {
46  
47    public DataSourceMetExtractor() {
48      super(new DataSourceMetExtractorConfigReader());
49    }
50  
51    @Override
52    protected Metadata extrMetadata(File file) throws MetExtractionException {
53      String key = getKey(file);
54      DataSourceMetExtractorConfig dsConfig = (DataSourceMetExtractorConfig) config;
55      DataSource dataSource = DatabaseConnectionBuilder.buildDataSource(dsConfig.getUserName(),
56          dsConfig.getPassword(), dsConfig.getDriver(), dsConfig.getDatabaseUrl());
57  
58      return getMetadata(dataSource, dsConfig.getQuery(), key);
59    }
60  
61    @VisibleForTesting
62    protected String getKey(File file) {
63      return Splitter.on(".").split(file.getName()).iterator().next();
64    }
65  
66    @VisibleForTesting
67    protected Metadata getMetadata(DataSource dataSource, String query, String key)
68        throws MetExtractionException {
69      String sqlQuery = String.format(query, key);
70  
71      Connection conn = null;
72      Statement statement = null;
73      ResultSet rs = null;
74  
75      try {
76        conn = dataSource.getConnection();
77        statement = conn.createStatement();
78        rs = statement.executeQuery(sqlQuery);
79  
80        return getMetadata(rs);
81      } catch (SQLException e) {
82        throw new MetExtractionException(
83            String.format("Failed to get metadaata for key '%s'", key), e);
84      } finally {
85        try {
86          if (conn != null) {
87            conn.close();
88          }
89        } catch (Exception e) { /* ignore */ }
90        try {
91          if (statement != null) {
92            statement.close();
93          }
94        } catch (Exception e) { /* ignore */ }
95        try {
96          if (rs != null) {
97            rs.close();
98          }
99        } catch (Exception e) { /* ignore */ }
100     }
101   }
102 
103   private Metadata getMetadata(ResultSet rs) throws SQLException {
104     Metadata metadata = new Metadata();
105     if (rs.next()) {
106       for (int i = 0; i < rs.getMetaData().getColumnCount(); i++) {
107         String metKey = rs.getMetaData().getColumnName(i+1);
108         String metVal = rs.getString(i+1);
109         metadata.addMetadata(metKey, metVal);
110       }
111     } else {
112       throw new SQLException("Failed to find metadata for result set");
113     }
114     return metadata;
115   }
