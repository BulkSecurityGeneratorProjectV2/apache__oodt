217     /**
218      * Returns the declared elements for a {@link ProductType}
219      * @param type The {@link ProductType} to get the elements for
220      * @param direct If false, return elements of the parent product types as well
221      * @return A list of all {@link Element} of the Product type
222      * @throws ValidationLayerException
223      * 				If any error occurs
224      */
225     public List<Element> getElements(ProductType type, boolean direct) {
226         List<Element> elems = new Vector<Element>();
227         String currType = type.getProductTypeId();
228         if (productTypeElementMap.containsKey(currType)) {
229             elems.addAll(productTypeElementMap.get(currType));
230         }
231 
232         if(!direct) {
233 	        while (subToSuperMap.containsKey(currType)) {
234 	            currType = subToSuperMap.get(currType);
235 	            if (productTypeElementMap.containsKey(currType)) {
236 	                elems.addAll(productTypeElementMap.get(currType));
237 	            }
238 	        }
239         }
240 
241         return elems;
242     }

277     private void saveElementsAndMappings() {
278       for (String dirUri : xmlFileDirUris) {
279         File elementDir;
280 
281         try {
282           elementDir = new File(new URI(dirUri));
283 
284           if (!elementDir.isDirectory()) {
285             LOG
286                 .log(
287                     Level.WARNING,
288                     "Element directory: "
289                     + dirUri
290                     + " is not "
291                     + "a directory: skipping element and product type map saving to it.");
292             continue;
293           }
294 
295           String elementDirStr = elementDir.getAbsolutePath();
296           if (!elementDirStr.endsWith("/")) {
297             elementDirStr += "/";
298           }
299 
300           String elementXmlFile = elementDirStr + "elements.xml";
301 
302           String productTypeMapXmlFile = elementDirStr
303                                          + "product-type-element-map.xml";
304 
305           XmlStructFactory.writeElementXmlDocument(Arrays
306                   .asList(elementMap.values().toArray(
307                       new Element[elementMap.size()])),
308               elementXmlFile);
309 
310           XmlStructFactory.writeProductTypeMapXmLDocument(
311               productTypeElementMap, subToSuperMap,
312               productTypeMapXmlFile);
313 
314         } catch (URISyntaxException e) {
315           LOG
316               .log(
317                   Level.WARNING,
318                   "URISyntaxException when saving element "
319                   + "directory URI: "
320                   + dirUri
321                   + ": Skipping Element and Product Type map saving"
322                   + "for it: Message: " + e.getMessage());
323         }
324 
325       }
326 
327     }

329     private void loadElements(List<String> dirUris) {
330       for (String dirUri1 : dirUris) {
331         File elementDir = null;
332         String dirUri = dirUri1;
333 
334         try {
335           elementDir = new File(new URI(dirUri));
336 
337           if (!elementDir.isDirectory()) {
338             LOG.log(Level.WARNING, "Element directory: " + dirUri
339                                    + " is not "
340                                    + "a directory: skipping element loading from it.");
341             continue;
342           }
343 
344           String elementDirStr = elementDir.getAbsolutePath();
345           if (!elementDirStr.endsWith("/")) {
346             elementDirStr += "/";
347           }
348 
349           String elementXmlFile = elementDirStr + "elements.xml";
350           Document elementDoc = getDocumentRoot(elementXmlFile);
351 
352           org.w3c.dom.Element elementRootElem = elementDoc
353               .getDocumentElement();
354 
355           NodeList elementNodeList = elementRootElem
356               .getElementsByTagName("element");
357 
358           if (elementNodeList != null && elementNodeList.getLength() > 0) {
359             for (int j = 0; j < elementNodeList.getLength(); j++) {
360               Node elementNode = elementNodeList.item(j);
361               Element element = XmlStructFactory
362                   .getElement(elementNode);
363               elementMap.put(element.getElementId(), element);
364             }
365           }
366 
367         } catch (URISyntaxException e) {
368           LOG.log(Level.WARNING,
369               "URISyntaxException when loading element "
370               + "directory URI: " + dirUri
371               + ": Skipping element loading"
372               + "for it: Message: " + e.getMessage());
373         }
374       }
375     }

377     private void loadProductTypeMap(List<String> dirUris) {
378       for (String dirUri1 : dirUris) {
379         File elementDir = null;
380         String dirUri = dirUri1;
381 
382         try {
383           elementDir = new File(new URI(dirUri));
384 
385           if (!elementDir.isDirectory()) {
386             LOG
387                 .log(
388                     Level.WARNING,
389                     "Element directory: "
390                     + dirUri
391                     + " is not "
392                     + "a directory: skipping product type element map loading from it.");
393             continue;
394           }
395 
396           String elementDirStr = elementDir.getAbsolutePath();
397           if (!elementDirStr.endsWith("/")) {
398             elementDirStr += "/";
399           }
400 
401           String productTypeMapXmlFile = elementDirStr
402                                          + "product-type-element-map.xml";
403           Document productTypeMapDoc = getDocumentRoot(productTypeMapXmlFile);
404 
405           org.w3c.dom.Element mapRootElem = productTypeMapDoc
406               .getDocumentElement();
407 
408           NodeList typeNodeList = mapRootElem
409               .getElementsByTagName("type");
410 
411           if (typeNodeList != null && typeNodeList.getLength() > 0) {
412             for (int j = 0; j < typeNodeList.getLength(); j++) {
413               org.w3c.dom.Element typeElement = (org.w3c.dom.Element) typeNodeList
414                   .item(j);
415               String typeId = typeElement.getAttribute("id");
416 
417               // get inheritance info
418               String typeParent = typeElement.getAttribute("parent");
419               if (typeParent != null) {
420                 subToSuperMap.put(typeId, typeParent);
421               }
422 
423               // get its element list
424               NodeList elementIdNodeList = typeElement
425                   .getElementsByTagName("element");
426 
427               // allow for 0 sized element list
428               List<Element> productTypeElementList = new Vector<Element>();
429 
430               if (elementIdNodeList != null
431                   && elementIdNodeList.getLength() > 0) {
432                 productTypeElementList = new Vector<Element>(
433                     elementIdNodeList.getLength());
434                 for (int k = 0; k < elementIdNodeList.getLength(); k++) {
435                   org.w3c.dom.Element elementIdElement = (org.w3c.dom.Element) elementIdNodeList
436                       .item(k);
437                   String elementId = elementIdElement
438                       .getAttribute("id");
439 
440                   if (elementMap.get(elementId) != null) {
441                     productTypeElementList.add(elementMap
442                         .get(elementId));
443                   }
444                 }
445               }
446 
447               productTypeElementMap.put(typeId,
448                   productTypeElementList);
449             }
450           }
451 
452         } catch (URISyntaxException e) {
453           LOG.log(Level.WARNING,
454               "URISyntaxException when loading element "
455               + "directory URI: " + dirUri
456               + ": Skipping product type map loading"
457               + "for it: Message: " + e.getMessage());
458         }
459       }
460     }

462     private Document getDocumentRoot(String xmlFile) {
463         // open up the XML file
464         DocumentBuilderFactory factory;
465         DocumentBuilder parser;
466         Document document;
467         InputSource inputSource;
468 
469         InputStream xmlInputStream;
470 
471         try {
472             xmlInputStream = new File(xmlFile).toURL().openStream();
473         } catch (IOException e) {
474             LOG.log(Level.WARNING,
475                     "IOException when getting input stream from [" + xmlFile
476                             + "]: returning null document root");
477             return null;
478         }
479 
480         inputSource = new InputSource(xmlInputStream);
481 
482         try {
483             factory = DocumentBuilderFactory.newInstance();
484             parser = factory.newDocumentBuilder();
485             document = parser.parse(inputSource);
486         } catch (Exception e) {
487             LOG.warning("Unable to parse xml file [" + xmlFile + "]."
488                     + "Reason is [" + e + "]");
489             return null;
490         }
491 
492         return document;
493     }
