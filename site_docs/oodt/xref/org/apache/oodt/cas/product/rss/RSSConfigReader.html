19  package org.apache.oodt.cas.product.rss;
20  
21  //OODT imports
22  import org.apache.oodt.commons.xml.XMLUtils;
23  
24  //JDK imports
25  import java.io.File;
26  import java.io.FileInputStream;
27  import java.io.FileNotFoundException;
28  import org.w3c.dom.Document;
29  import org.w3c.dom.Element;
30  import org.w3c.dom.NodeList;
31  
32  /**
33   * 
34   * Reader class for {@link RSSConfig}s from {@link File}s.
35   * 
36   * @author mattmann
37   * @version $Revision$
38   * 
39   */
40  public class RSSConfigReader implements RSSConfigReaderMetKeys {
41  
42    /**
43     * Reads an {@link RSSConfig} from a {@link File}.
44     * 
45     * @param file
46     *          The {@link File} representation of the {@link RSSConfig}.
47     * @return An {@link RSSConfig} initialized with the information read from the
48     *         given {@link File}.
49     * @throws FileNotFoundException
50     *           If the {@link File} is not found.
51     */
52    public static RSSConfig readConfig(File file) throws FileNotFoundException {
53      RSSConfig conf = new RSSConfig();
54  
55      Document doc = XMLUtils.getDocumentRoot(new FileInputStream(file));
56      Element rootElem = doc.getDocumentElement();
57      conf.setChannelLink(rootElem.getAttribute(CHANNEL_LINK_ATTR));
58  
59      readNamespaces(rootElem, conf);
60      readTags(rootElem, conf);
61      return conf;
62    }
63    
64    protected static void readNamespaces(Element root, RSSConfig conf) {
65      NodeList namespaceList = root.getElementsByTagName(NAMESPACE_TAG);
66      if (namespaceList != null && namespaceList.getLength() > 0) {
67        for (int i = 0; i < namespaceList.getLength(); i++) {
68          Element namespaceElem = (Element) namespaceList.item(i);
69          RSSNamespace namespace = new RSSNamespace();
70          namespace.setPrefix(namespaceElem.getAttribute(NAMESPACE_ATTR_PREFIX));
71          namespace.setUri(namespaceElem.getAttribute(NAMESPACE_ATTR_URI));
72          conf.getNamespaces().add(namespace);
73        }
74      }
75    }
76  
77    protected static void readTags(Element root, RSSConfig conf) {
78      NodeList tagList = root.getElementsByTagName(TAG_TAG);
79      if (tagList != null && tagList.getLength() > 0) {
80        for (int i = 0; i < tagList.getLength(); i++) {
81          Element tagElem = (Element) tagList.item(i);
82          RSSTag tag = new RSSTag();
83          tag.setName(tagElem.getAttribute(TAG_ATTR_NAME));
84  
85          // check to see if it has a source
86          if (tagElem.getAttributeNode(TAG_ATTR_SOURCE) != null) {
87            tag.setSource(tagElem.getAttribute(TAG_ATTR_SOURCE));
88          }
89  
90          readAttrs(tagElem, tag);
91          conf.getTags().add(tag);
92        }
93      }
94    }
95  
96    protected static void readAttrs(Element tagElem, RSSTag tag) {
97      NodeList attrTagList = tagElem.getElementsByTagName(ATTRIBUTE_TAG);
98  
99      if (attrTagList != null && attrTagList.getLength() > 0) {
100       for (int i = 0; i < attrTagList.getLength(); i++) {
101         Element attrTag = (Element) attrTagList.item(i);
102         RSSTagAttribute attr = new RSSTagAttribute();
103         attr.setName(attrTag.getAttribute(ATTRIBUTE_ATTR_NAME));
104         attr.setValue(attrTag.getAttribute(ATTRIBUTE_ATTR_VALUE));
105         tag.getAttrs().add(attr);
106       }
107     }
108 
109   }
