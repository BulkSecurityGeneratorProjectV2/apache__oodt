101   public void init(ServletConfig config) throws ServletException {
102     super.init(config);
103     try {
104       this.rdfConf = RDFUtils.initRDF(config);
105     } catch (FileNotFoundException e) {
106       LOG.log(Level.SEVERE, e.getMessage());
107       throw new ServletException(e.getMessage());
108     }
109 
110     String fileManagerUrl;
111     try {
112       fileManagerUrl = PathUtils.replaceEnvVariables(config.getServletContext().getInitParameter(
113           "filemgr.url") );
114     } catch (Exception e) {
115       throw new ServletException("Failed to get filemgr url : " + e.getMessage(), e);
116     }
117     
118     if (fileManagerUrl == null) {
119       // try the default port
120       fileManagerUrl = "http://localhost:9000";
121     }
122 
123     fClient = null;
124 
125     try {
126       fClient = new XmlRpcFileManagerClient(new URL(fileManagerUrl));
127     } catch (MalformedURLException e) {
128       LOG.log(Level.SEVERE,
129           "Unable to initialize file manager url in RDF Servlet: [url="
130               + fileManagerUrl + "], Message: " + e.getMessage());
131     } catch (ConnectionException e) {
132       LOG.log(Level.SEVERE,
133           "Unable to initialize file manager url in RDF Servlet: [url="
134               + fileManagerUrl + "], Message: " + e.getMessage());
135     }
136   }

148   public void doIt(HttpServletRequest req, HttpServletResponse resp)
149       throws ServletException {
150 
151     // need to know the type of product to get products for
152     String productTypeName = req.getParameter("type");
153     String productTypeId = req.getParameter("id");
154     ProductType type = null;
155 
156     List<Product> products;
157 
158     try {
159       if (productTypeName.equals("ALL")) {
160         products = aggregatePagedProducts();
161       } else {
162 
163         try {
164           type = fClient.getProductTypeById(productTypeId);
165         } catch (RepositoryManagerException e) {
166           LOG.log(Level.SEVERE,
167               "Unable to obtain product type from product type id: ["
168                   + productTypeId + "]: Message: " + e.getMessage());
169           return;
170         }
171 
172         products = fClient.getProductsByProductType(type);
173       }
174 
175     } catch (CatalogException e) {
176       LOG.log(Level.SEVERE, e.getMessage());
177       LOG
178           .log(Level.WARNING,
179               "Exception getting products from Catalog: Message: "
180                   + e.getMessage());
181       return;
182     }
183 
184     String requestUrl = req.getRequestURL().toString();
185     String base = requestUrl.substring(0, requestUrl.lastIndexOf('/'))
186         + "/data";
187 
188     // allow override through the "baseUrl" GET parameter
189     if (req.getParameter("baseUrl") != null) {
190       base = req.getParameter("baseUrl");
191     }
192 
193     if (products != null && products.size() > 0) {
194       outputRDF(products, type, base, resp);
195     }
196 
197   }

199   public void outputRDF(List<Product> products, ProductType type, String base,
200       HttpServletResponse resp) throws ServletException {
201 
202     try {
203       DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
204       factory.setNamespaceAware(true);
205       Document doc = factory.newDocumentBuilder().newDocument();
206 
207       Element rdf = XMLUtils.addNode(doc, doc, "rdf:RDF");
208       RDFUtils.addNamespaces(doc, rdf, this.rdfConf);
209 
210       for (Product p : products) {
211         String productTypeIdStr = p.getProductType().getProductTypeId();
212         ProductType productType;
213 
214         if (type != null) {
215           productType = type;
216         } else {
217           try {
218             productType = fClient.getProductTypeById(productTypeIdStr);
219           } catch (RepositoryManagerException e) {
220             LOG.log(Level.SEVERE, e.getMessage());
221             LOG.log(Level.SEVERE,
222                 "Unable to obtain product type from product type id: ["
223                 + ((Product) products.get(0)).getProductType()
224                                              .getProductTypeId() + "]: Message: " + e.getMessage());
225             return;
226           }
227         }
228 
229         p.setProductType(productType);
230 
231         Element productRdfDesc = XMLUtils.addNode(doc, rdf, this.rdfConf
232                                                                 .getTypeNs(productType.getName())
233                                                             + ":" + productType.getName());
234         XMLUtils.addAttribute(doc, productRdfDesc, "rdf:about", base
235                                                                 + "?productID=" + p.getProductId());
236 
237         // now add all its metadata
238         Metadata prodMetadata = safeGetMetadata(p);
239 
240         // for all of its metadata keys and values, loop through them
241         // and add RDF nodes underneath the RdfDesc for this product
242 
243         if (prodMetadata != null) {
244           for (String key : prodMetadata.getMap().keySet()) {
245             List<String> vals = prodMetadata.getAllMetadata(key);
246 
247             if (vals != null && vals.size() > 0) {
248 
249               for (String val : vals) {
250                 String outputKey = key;
251                 if (outputKey.contains(" ")) {
252                   outputKey = StringUtils.join(WordUtils.capitalizeFully(outputKey).split(
253                       " "));
254                 }
255 
256                 val = StringEscapeUtils.escapeXml(val);
257                 Element rdfElem = RDFUtils.getRDFElement(outputKey, val,
258                     this.rdfConf, doc);
259                 productRdfDesc.appendChild(rdfElem);
260               }
261 
262             }
263           }
264         }
265 
266       }
267 
268       DOMSource source = new DOMSource(doc);
269       TransformerFactory transFactory = TransformerFactory.newInstance();
270       Transformer transformer = transFactory.newTransformer();
271       transformer.setOutputProperty("indent", "yes");
272       StreamResult result = new StreamResult(resp.getOutputStream());
273       resp.setContentType("text/xml");
274       transformer.transform(source, result);
275 
276     } catch (ParserConfigurationException e) {
277       throw new ServletException(e);
278     } catch (TransformerException e) {
279       throw new ServletException(e);
280     } catch (IOException e) {
281       throw new ServletException(e);
282     }
283   }

285   private List<Product> aggregatePagedProducts() {
286     List<ProductType> types = safeGetProductTypes();
287     List<Product> products = null;
288 
289     if (types != null && types.size() > 0) {
290       products = new Vector<Product>();
291       for (ProductType type : types) {
292         ProductPage page;
293 
294         try {
295           page = fClient.getFirstPage(type);
296 
297           if (page != null) {
298 
299             while (true) {
300               products.addAll(page.getPageProducts());
301               if (!page.isLastPage()) {
302                 page = fClient.getNextPage(type, page);
303               } else {
304                 break;
305               }
306             }
307           }
308         } catch (Exception ignore) {
309         }
310 
311       }
312     }
313 
314     return products;
315   }

317   private Metadata safeGetMetadata(Product p) {
318     Metadata met = null;
319 
320     try {
321       met = fClient.getMetadata(p);
322     } catch (CatalogException e) {
323       LOG.log(Level.SEVERE, e.getMessage());
324       LOG.log(Level.WARNING, "Error retrieving metadata for product: ["
325           + p.getProductId() + "]: Message: " + e.getMessage());
326     }
327 
328     return met;
329   }

331   private List<ProductType> safeGetProductTypes() {
332     List<ProductType> types = null;
333 
334     try {
335       types = fClient.getProductTypes();
336     } catch (RepositoryManagerException e) {
337       LOG.log(Level.SEVERE, e.getMessage());
338       LOG.log(Level.WARNING, "Error retrieving product types: Message: "
339           + e.getMessage());
340     }
341 
342     return types;
343   }
