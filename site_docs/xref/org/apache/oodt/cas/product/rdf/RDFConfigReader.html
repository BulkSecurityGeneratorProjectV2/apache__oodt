19  package org.apache.oodt.cas.product.rdf;

21  //OODT imports
22  import org.apache.oodt.commons.xml.XMLUtils;
23  import static org.apache.oodt.cas.product.rdf.RDFConfigReaderMetKeys.*;
24  
25  //JDK imports
26  import java.io.File;
27  import java.io.FileInputStream;
28  import java.io.FileNotFoundException;
29  import org.w3c.dom.Document;
30  import org.w3c.dom.Element;
31  import org.w3c.dom.NodeList;
32  
33  /**
34   * 
35   * Reader class for {@link RDFConfig}s from {@link File}s.
36   * 
37   * @author mattmann
38   * @version $Revision$
39   * 
40   */
41  public class RDFConfigReader {
42  
43    /**
44     * Reads an {@link RDFConfig} from a {@link File}.
45     * 
46     * @param file
47     *          The {@link File} representation of the {@link RDFConfig}.
48     * @return An {@link RDFConfig} initialized with the information read from the
49     *         given {@link File}.
50     * @throws FileNotFoundException
51     *           If the {@link File} is not found.
52     */
53    public static RDFConfig readConfig(File file) throws FileNotFoundException {
54      RDFConfig conf = new RDFConfig();
55  
56      Document doc = XMLUtils.getDocumentRoot(new FileInputStream(file));
57      Element rootElem = doc.getDocumentElement();
58  
59      readNamespaces(rootElem, conf);
60      readRewriteRules(rootElem, conf);
61      readResourceLinks(rootElem, conf);
62      readKeyNsMap(rootElem, conf);
63      readTypesNsMap(rootElem, conf);
64  
65      return conf;
66    }
67  
68    protected static void readNamespaces(Element root, RDFConfig conf) {
69      Element namespacesRoot = XMLUtils.getFirstElement(NS_OUTER_TAG, root);
70      NodeList nsList = namespacesRoot.getElementsByTagName(NS_TAG);
71      if (nsList != null && nsList.getLength() > 0) {
72        for (int i = 0; i < nsList.getLength(); i++) {
73          Element nsElem = (Element) nsList.item(i);
74          conf.getNsMap().put(nsElem.getAttribute(NS_NAME_ATTR),
75              nsElem.getAttribute(NS_VALUE_ATTR));
76        }
77      }
78    }
79  
80    protected static void readRewriteRules(Element root, RDFConfig conf) {
81      Element rewriteRoot = XMLUtils.getFirstElement(REWRITE_OUTER_TAG, root);
82      NodeList rewriteList = rewriteRoot.getElementsByTagName(REWRITE_KEY_TAG);
83      if (rewriteList != null && rewriteList.getLength() > 0) {
84        for (int i = 0; i < rewriteList.getLength(); i++) {
85          Element rewriteKeyElem = (Element) rewriteList.item(i);
86          conf.getRewriteMap().put(
87              rewriteKeyElem.getAttribute(REWRITE_FROM_ATTR),
88              rewriteKeyElem.getAttribute(REWRITE_TO_ATTR));
89        }
90      }
91    }
92  
93    protected static void readResourceLinks(Element root, RDFConfig conf) {
94      Element resLinkRoot = XMLUtils.getFirstElement(RESOURCE_LINK_TAG, root);
95      NodeList linkList = resLinkRoot.getElementsByTagName(RESLINK_KEY_TAG);
96      if (linkList != null && linkList.getLength() > 0) {
97        for (int i = 0; i < linkList.getLength(); i++) {
98          Element reslinkKeyElem = (Element) linkList.item(i);
99          conf.getResLinkMap().put(
100             reslinkKeyElem.getAttribute(RESLINK_KEY_TAG_NAME_ATTR),
101             reslinkKeyElem.getAttribute(RESLINK_KEY_TAG_LINK_ATTR));
102       }
103     }
104   }
105 
106   protected static void readKeyNsMap(Element root, RDFConfig conf) {
107     Element keyNsRoot = XMLUtils.getFirstElement(KEY_NSMAP_TAG, root);
108     conf.setDefaultKeyNs(keyNsRoot.getAttribute(KEY_NSMAP_DEFAULT_ATTR));
109     NodeList keyNsList = keyNsRoot.getElementsByTagName(KEY_NSMAP_KEY_TAG);
110     if (keyNsList != null && keyNsList.getLength() > 0) {
111       for (int i = 0; i < keyNsList.getLength(); i++) {
112         Element keyNsElem = (Element) keyNsList.item(i);
113         conf.getKeyNsMap().put(
114             keyNsElem.getAttribute(KEY_NSMAP_KEY_TAG_NAME_ATTR),
115             keyNsElem.getAttribute(KEY_NSMAP_KEY_TAG_NS_ATTR));
116       }
117     }
118   }
119 
120   protected static void readTypesNsMap(Element root, RDFConfig conf) {
121     Element typeNsRoot = XMLUtils.getFirstElement(TYPE_NSMAP_TAG, root);
122     conf.setDefaultTypeNs(typeNsRoot.getAttribute(TYPE_NSMAP_DEFAULT_ATTR));
123     NodeList typeNsList = typeNsRoot.getElementsByTagName(TYPE_NSMAP_TYPE_TAG);
124     if (typeNsList != null && typeNsList.getLength() > 0) {
125       for (int i = 0; i < typeNsList.getLength(); i++) {
126         Element typeNsElem = (Element) typeNsList.item(i);
127         conf.getTypesNsMap().put(
128             typeNsElem.getAttribute(TYPE_NSMAP_TYPE_NAME_ATTR),
129             typeNsElem.getAttribute(TYPE_NSMAP_TYPE_NS_ATTR));
130       }
131     }
132   }
