18  package org.apache.oodt.commons.activity;
19  
20  import java.io.ByteArrayInputStream;
21  import java.io.IOException;
22  import java.io.InvalidClassException;
23  import java.io.ObjectInputStream;
24  import java.io.OptionalDataException;
25  import java.io.StreamCorruptedException;
26  import java.net.DatagramPacket;
27  import java.net.DatagramSocket;
28  import java.util.concurrent.ConcurrentHashMap;
29  import java.util.Map;
30  
31  /**
32   * The datagram logger accepts incidents in UDP datagrams and saves them in a {@link
33   * Storage}.  The datagrams contain serialized {@link Incident}s, and are probably sent by
34   * a {@link DatagramLoggingActivityFactory}.
35   *
36   * @author Kelly
37   * @version $Revision: 1.1 $
38   */
39  class DatagramLogger {
40  
41    public static final int VAL = 4556;
42    public static final int INT = 2048;
43  
44    public static void main(String[] argv) throws Throwable {
45  		if (argv.length > 0) {
46  			System.err.println("This program takes NO command line arguments.");
47  			System.err.println("Set the activity.port property to adjust the port number.");
48  			System.err.println("Set the activity.storage property to set the Storage class to use.");
49  			System.exit(1);
50  		}
51  		int port = Integer.getInteger("activity.port", VAL);
52  		String className = System.getProperty("activity.storage");
53  		if (className == null) {
54  			System.err.println("No Storage class defined via the `activity.storage' property; exiting...");
55  			System.exit(1);
56  		}
57  		Class storageClass = Class.forName(className);
58  		storage = (Storage) storageClass.newInstance();
59  		DatagramSocket socket = new DatagramSocket(port);
60  		byte[] buf = new byte[INT];
61  		DatagramPacket packet = new DatagramPacket(buf, buf.length);
62  		for (;;) {
63  			socket.receive(packet);
64  			byte[] received = new byte[packet.getLength()];
65  			System.arraycopy(packet.getData(), packet.getOffset(), received, 0, packet.getLength());
66  			new ReceiverThread(received).start();
67  		} 
68  	}
69  
70  	/** Long term storage for incidents. */
71  	private static Storage storage;
72  
73  	/** History awaiting long-term storage.  Keys are {@link String} activity IDs, and values are {@link History} objects. */
74  	private static Map histories = new ConcurrentHashMap();
75  
76  	/**
77  	 * Thread that saves off an incident into a history so the main thread can go back
78  	 * to receiving more datagrams.
79  	 */
80  	private static class ReceiverThread extends Thread {
81  		/**
82  		 * Creates a new {@link ReceiverThread} instance.
83  		 *
84  		 * @param data Copy of bytes received in a datagram.
85  		 */
86  		private ReceiverThread(byte[] data) {
87  			this.data = data;
88  		}
89  
90  		/**
91  		 * Reconstitute the incident in the received byte array and store it into
92  		 * a {@link History}.
93  		 */
94  		public void run() {
95  			try {
96  				ObjectInputStream in = new ObjectInputStream(new ByteArrayInputStream(data));
97  				Incident incident = (Incident) in.readObject();
98  				String id = incident.getActivityID();
99  				in.close();
100 				History history;
101 				synchronized (histories) {
102 					history = (History) histories.get(id);
103 					if (history == null) {
104 						histories.put(id, new History(incident, storage));
105 						return;
106 					}
107 				}
108 				history.addIncident(incident);
109 			} catch (ClassNotFoundException ex) {
110 				System.err.println("Dropping Incident of unknown class: " + ex.getMessage());
111 			} catch (InvalidClassException ex) {
112 				System.err.println("Dropping Incident of invalid class: " + ex.getMessage());
113 			} catch (StreamCorruptedException ex) {
114 				System.err.println("Unable to read Incident from packet: " + ex.getMessage());
115 			} catch (OptionalDataException ex) {
116 				System.err.println("Primitive data instead of Incident in packet: " + ex.getMessage());
117 			} catch (IOException ex) {
118 				throw new IllegalStateException("Unexpected IOException: " + ex.getMessage());
119 			}
120 		}
121 
122 		/** Bytes received in a datagram. */
123 		private byte[] data;
124 	}
