49      private List<Cluster> gridConfiguration;

51      /**
52       * {@inheritDoc}
53       * Read gangliaXMLdump.xml and build the grid configuration
54       */
55      protected void setUp() throws MonitorException, IOException {
56          StringBuilder stringBuffer = new StringBuilder();
57  
58          try {
59              BufferedReader reader = new BufferedReader(new FileReader("." + File.separator +
60                      "src" + File.separator + "test" + File.separator + "resources" + File.separator + "resourcemon"
61                      + File.separator + "gangliaXMLdump.xml"));
62              String line = reader.readLine();
63              while (line != null) {
64                  stringBuffer.append(line);
65                  line = reader.readLine();
66              }
67              reader.close();
68          } catch (IOException e) {
69              throw new IOException("Unable to read the sample monitoring report from the file: "
70                      + e.getMessage());
71          }
72  
73          String buffer = stringBuffer.toString().trim();
74          SAXParserFactory factory = SAXParserFactory.newInstance();
75          javax.xml.parsers.SAXParser parser;
76          GangliaXMLParser gangliaXMLParser;
77          try {
78              parser = factory.newSAXParser();
79              gangliaXMLParser = new GangliaXMLParser();
80              parser.parse(new InputSource(new StringReader(buffer)), gangliaXMLParser);
81              gridConfiguration = gangliaXMLParser.getGridConfiguration();
82          } catch (ParserConfigurationException e) {
83              throw new MonitorException("Error while parsing: " + e.getMessage());
84          } catch (SAXException e) {
85              throw new MonitorException("Error while parsing the XML: " + e.getMessage());
86          } catch (IOException e) {
87              throw new MonitorException("I/O error: " + e.getMessage());
88          }
89      }

91      /**
92       * test the "<CLUSTER>" tag data
93       * <CLUSTER NAME="" LOCALTIME="" OWNER="" LATLONG="" URL="">
94       */
95      public void testClusterTag() {
96          Cluster cluster = gridConfiguration.get(0);

98          assertEquals("1370187645", cluster.getLocaltime());
99          assertEquals("oodt", cluster.getOwner());
100         assertEquals("N32.87 W117.22", cluster.getLatLong());
101         assertEquals("http://www.mycluster.org/", cluster.getUrl());
102     }

120     /**
121      * test the "<METRIC>" tag data
122      * <METRIC NAME="" VAL="" TYPE="" UNITS="" TN="" TMAX="" DMAX="" SLOPE="" SOURCE="">
123      * <EXTRA_DATA>
124      * <EXTRA_ELEMENT NAME="" VAL=""/>
125      * <EXTRA_ELEMENT NAME="" VAL=""/>
126      * ........
127      * </EXTRA_DATA>
128      * </METRIC>
129      */
130     public void testMetricTag() {
131         Metric metric = ((((gridConfiguration.get(0)).getHosts()).get(0)).getMetrics()).get(0);
132         assertEquals("disk_free", metric.getName());
133         assertEquals("307.790", metric.getValue());
134         assertEquals("double", metric.getType());
135         assertEquals("GB", metric.getUnits());
136         assertEquals("143", metric.getTn());
137         assertEquals("180", metric.getTmax());
138         assertEquals("0", metric.getDmax());
139         assertEquals("both", metric.getSlope());
140         assertEquals("gmond", metric.getSource());
141 
142         //extra data of the metric
143         ConcurrentHashMap<String, String> extraData = metric.getExtraData();
144         assertEquals("disk", extraData.get(GangliaMetKeys.GROUP));
145         assertEquals("Total free disk space", extraData.get(GangliaMetKeys.DESC));
146         assertEquals("Disk Space Available", extraData.get(GangliaMetKeys.TITLE));
147     }
