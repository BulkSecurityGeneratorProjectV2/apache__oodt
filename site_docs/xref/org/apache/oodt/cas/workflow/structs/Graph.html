79    public Graph(Element graphElem, Metadata staticMetadata) throws WorkflowException {
80      this();
81      this.modelId = graphElem.getAttribute("id");
82      this.modelName = graphElem.getAttribute("name");
83      this.clazz = graphElem.getAttribute("class");
84      this.modelIdRef = graphElem.getAttribute("id-ref");
85      this.excused.addAll(Arrays.asList(graphElem.getAttribute("excused").split(
86          ",")));
87      this.alias = graphElem.getAttribute("alias");
88      this.minReqSuccessfulSubProcessors = graphElem.getAttribute("min");
89      this.executionType = graphElem.getAttribute("execution");
90      this.timeout = Long.valueOf(graphElem.getAttribute("timeout") != null
91          && !graphElem.getAttribute("timeout").equals("") ? graphElem
92          .getAttribute("timeout") : "-1");
93      this.optional = Boolean.valueOf(graphElem.getAttribute("optional"));
94  
95      NamedNodeMap attrMap = graphElem.getAttributes();
96      for (int i = 0; i < attrMap.getLength(); i++) {
97        Attr attr = (Attr) attrMap.item(i);
98        if (attr.getName().startsWith("p:")) {
99          staticMetadata.replaceMetadata(attr.getName().substring(2),
100             attr.getValue());
101       }
102     }
103 
104     if ((graphElem.getNodeName().equals("workflow") || graphElem.getNodeName()
105         .equals("conditions")) && this.executionType.equals("")) {
106       throw new WorkflowException("workflow model '" + graphElem.getNodeName()
107           + "' missing execution type");
108     } else {
109       this.executionType = graphElem.getNodeName();
110     }
111 
112     if (!processorIds.contains(this.executionType)) {
113       throw new WorkflowException("Unsupported execution type id '"
114                                   + this.executionType + "'");
115     }
116 
117     if (!checkValue(this.modelId) && !checkValue(this.modelIdRef)) {
118       this.modelId = UUID.randomUUID().toString();
119     }
120 
121     if (this.alias != null && !this.alias.equals("")) {
122       this.modelId = this.alias;
123     }
124   }
