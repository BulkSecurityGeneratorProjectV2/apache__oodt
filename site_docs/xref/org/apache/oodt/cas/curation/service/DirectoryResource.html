43  public class DirectoryResource extends CurationService {

52    @GET
53    @Path("staging")
54    @Produces("text/plain")
55    public String showStagingArea(
56        @DefaultValue("/") @QueryParam("path") String path,
57        @DefaultValue("true") @QueryParam("showFiles") boolean showFiles,
58        @DefaultValue(FORMAT_HTML) @QueryParam("format") String format) {
59      if (FORMAT_HTML.equals(format)) {
60        return this.getDirectoryAreaAsHTML(
61            CurationService.config.getStagingAreaPath(), path, showFiles);
62  
63      }
64      return this.getDirectoryAreaAsJSON(
65          CurationService.config.getStagingAreaPath(), path, showFiles);
66    }

72        @DefaultValue("true") @QueryParam("showFiles") boolean showFiles) {

74          path, showFiles);
75    }

89    public String getDirectoryAreaAsHTML(String base, String path,
90        boolean showFiles) {
91      StringBuilder html = new StringBuilder();
92      String relativePath;
93      try {
94        relativePath = this.cleansePath(path);
95      } catch (Exception e) {
96        LOG.log(Level.SEVERE, e.getMessage());
97        LOG.log(Level.WARNING, "Error decoding path: [" + path + "]: Message: "
98            + e.getMessage());
99        return html.toString();
100     }
101 
102     String startingPath = (base + "/" + relativePath);
103     String f[] = this.getFilesInDirectory(startingPath, showFiles);
104 
105     html.append("<ul class=\"fileTree\">\r\n");
106     // Loop through and list directories first. Nicer for UI to get these first
107 
108     if (f != null) {
109       for (String aF1 : f) {
110         if (new File(startingPath + "/" + aF1).isDirectory()) {
111           html.append(" <li class=\"directory collapsed\">");
112           html.append("<a href=\"#\" rel=\"").append(relativePath).append("/")
113               .append(aF1).append("\">").append(aF1).append("</a>");
114           html.append("</li>\r\n");
115         }
116       }
117       // If we are showing files now loop through and show files
118       if (showFiles) {
119         for (String aF : f) {
120           if (new File(startingPath + "/" + aF).isFile()) {
121             String filename = new File(startingPath + "/" + aF).getName();
122             String ext = filename.substring(filename.lastIndexOf('.') + 1);
123             html.append(" <li class=\"file draggy ext_").append(ext)
124                 .append("\">");
125             html.append("<a href=\"#\" rel=\"").append(relativePath)
126                 .append("/").append(aF).append("\">").append(aF)
127                 .append("</a>");
128             html.append("</li>\r\n");
129           }
130         }
131       }
132 
133     }
134     html.append("</ul>");
135 
136     return html.toString();
137   }
