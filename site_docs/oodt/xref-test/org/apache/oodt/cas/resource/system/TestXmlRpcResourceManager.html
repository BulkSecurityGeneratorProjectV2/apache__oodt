57    public void testDynSetNodeCapacity() {
58      XmlRpcResourceManagerClient rmc = null;
59      try {
60        rmc = new XmlRpcResourceManagerClient(new URL("http://localhost:"
61            + RM_PORT));
62      } catch (Exception e) {
63        fail(e.getMessage());
64      }
65  
66      assertNotNull(rmc);
67      try {
68        rmc.setNodeCapacity("localhost", 8);
69      } catch (MonitorException e) {
70        fail(e.getMessage());
71      }
72  
73      int setCapacity = -1;
74      try {
75        setCapacity = rmc.getNodeById("localhost").getCapacity();
76      } catch (Exception e) {
77        fail(e.getMessage());
78      }
79      assertEquals(8, setCapacity);
80  
81    }

105   private void deleteAllFiles(String startDir) {
106     File startDirFile = new File(startDir);
107     File[] delFiles = startDirFile.listFiles();
108 
109     if (delFiles != null && delFiles.length > 0) {
110       for (File delFile : delFiles) {
111         delFile.delete();
112       }
113     }
114 
115     startDirFile.delete();
116 
117   }

119     private void generateTestConfiguration() throws IOException {
120     Properties config = new Properties();
121 
122     String propertiesFile = "." + File.separator + "src" + File.separator +
123             "test" + File.separator + "resources" + File.separator + "test.resource.properties";
124     System.getProperties().load(new FileInputStream(new File(propertiesFile)));
125 
126     // stage policy
127     File tmpPolicyDir = null;
128     try {
129       tmpPolicyDir = File.createTempFile("test", "ignore").getParentFile();
130     } catch (Exception e) {
131       fail(e.getMessage());
132     }
133     for (File policyFile : new File("./src/test/resources/policy")
134         .listFiles(new FileFilter() {

136           @Override
137           public boolean accept(File pathname) {
138             return pathname.isFile() && pathname.getName().endsWith(".xml");
139           }
140         })) {
141       try {
142         FileUtils.copyFileToDirectory(policyFile, tmpPolicyDir);
143       } catch (Exception e) {
144         fail(e.getMessage());

146     }

151         tmpPolicyDir.toURI().toString());

153     System.getProperties().putAll(config);
154     this.tmpPolicyDir = tmpPolicyDir;
