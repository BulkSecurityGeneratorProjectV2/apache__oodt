16  package org.apache.oodt.commons.util;

18  import java.io.*;
19  import org.apache.oodt.commons.io.*;

21  /**
22  	The <code>ScreenLogger</code> class is intended to be used with the
23  	{@link Log} class in order to setup the System.err as a {@link LogListener}.
24  
25  	@author S. Hardman
26  	@version $Revision: 1.1 $
27  */
28  public class ScreenLogger extends WriterLogger {
29  
30  	/**
31  		Constructor given no arguments.
32  
33  		This constructor will specify System.err as the output stream
34  		and then call the other constructor with the output stream argument.
35  	*/
36  	public ScreenLogger () {
37  		this(System.err);
38  	}
39  
40  
41  	/**
42  		Constructor given an output stream.
43  
44  		This constructor calls the {@link WriterLogger} constructor with the
45  		same signature.
46  
47  		@param outputStream The output stream representing the log destination.
48  	*/
49  	public ScreenLogger (OutputStream outputStream) {
50  		super(outputStream);
51  	}
52  }
