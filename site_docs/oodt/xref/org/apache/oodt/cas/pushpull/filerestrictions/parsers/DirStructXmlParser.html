66      public VirtualFileStructure parse(FileInputStream xmlFile, Metadata metadata)
67              throws ParserException {
68          try {
69              String initialCdDir = "/";
70              VirtualFile root = null;
71              NodeList list = (DocumentBuilderFactory.newInstance()
72                      .newDocumentBuilder().parse(new InputSource(xmlFile)))
73                      .getDocumentElement().getChildNodes();
74              VirtualFile currentFile;
75              for (int i = 0; i < list.getLength(); i++) {
76                  Node node = list.item(i);
77                  if (node.getNodeName().equals("dirstruct")) {
78  
79                      // parse out starting path
80                      String startingPath = ((Element) node)
81                              .getAttribute("starting_path");
82                          root = (currentFile = new VirtualFile(
83                                  initialCdDir = startingPath, true))
84                                  .getRootDir();
85                          VirtualFile temp = currentFile.getParentFile();
86                          while (temp != null) {
87                              temp.setNoDirs(true);
88                              temp.setNoFiles(true);
89                              temp = temp.getParentFile();
90                          }
91                      // parse the directory structure
92                      parseDirstructXML(node.getChildNodes(), currentFile);
93  
94                  } else if (node.getNodeName().equals("variables")) {
95                      parseVariablesXML(node.getChildNodes());
96                  } else if (node.getNodeName().equals("methods")) {
97                      parseMethodsXML(node.getChildNodes());
98                  }
99              }
100             return new VirtualFileStructure(initialCdDir, root);
101         } catch (Exception e) {
102             LOG.log(Level.SEVERE, e.getMessage());
103             throw new ParserException("Failed to parse XML file : "
104                     + e.getMessage());
105         }
106     }

108     private void parseDirstructXML(NodeList list, VirtualFile currentLoadFile) {
109         for (int i = 0; i < list.getLength(); i++) {
110             Node dir = list.item(i);
111             if (dir.getNodeName().equals("dir")) {
112                 String dirName = replaceVariablesAndMethods(((Element) dir)
113                         .getAttribute("name"));
114                 currentLoadFile.addChild(new VirtualFile(dirName, true));
115                 NodeList children;
116                 if ((children = dir.getChildNodes()).getLength() > 0) {
117                     parseDirstructXML(children, currentLoadFile.getChild(
118                             dirName, true));
119                 }
120             } else if (dir.getNodeName().equals("nodirs")) {
121                 currentLoadFile.setNoDirs(true);
122             } else if (dir.getNodeName().equals("nofiles")) {
123                 currentLoadFile.setNoFiles(true);
124             } else if (dir.getNodeName().equals("file")) {
125                 VirtualFile vf = new VirtualFile(
126                         replaceVariablesAndMethods(((Element) dir)
127                                 .getAttribute("name")), false);
128                 vf.setNoDirs(true);
129                 currentLoadFile.addChild(vf);
130             }
131         }
132     }

134     private String replaceVariablesAndMethods(String input) {
135         for (int i = 0; i < input.length(); i++) {
136             char c = input.charAt(i);
137             switch (c) {
138             case '$':
139                 try {
140                     if (input.charAt(i + 1) == '{') {
141                         StringBuilder variable = new StringBuilder("");
142                         for (int j = i + 2; j < input.length(); j++) {
143                             char ch = input.charAt(j);
144                             if ((ch <= 'Z' && ch >= 'A')
145                                     || (ch <= 'z' && ch >= 'a')
146                                     || (ch <= '9' && ch >= '0') 
147                                     || ch == '_') {
148                               variable.append(ch);
149                             } else {
150                               break;
151                             }
152                         }
153                         Variable v = GlobalVariables.ConcurrentHashMap.get(variable
154                                 .toString());
155                         if (v == null) {
156                           throw new Exception("No variable defined with name '" + variable.toString() + "'");
157                         }
158                         input = input.replaceFirst("\\$\\{" + variable + "\\}", v.toString());
159                         i = i + v.toString().length();
160                     }
161                 } catch (Exception e) {
162                 	LOG.log(Level.WARNING, "Failed to replace variable in '" + input + " for i = '" + i + "' : " + e.getMessage(), e);
163                 }
164                 break;
165             case '%':
166                 try {
167                     StringBuilder method = new StringBuilder("");
168                     int j = i + 1;
169                     for (; j < input.length(); j++) {
170                         char ch = input.substring(j, j + 1).charAt(0);
171                         if ((ch <= 'Z' && ch >= 'A')
172                                 || (ch <= 'z' && ch >= 'a')
173                                 || (ch <= '9' && ch >= '0') || ch == '_') {
174                           method.append(ch);
175                         } else {
176                           break;
177                         }
178                     }
179 
180                     if (input.substring(j, j + 1).charAt(0) == '(') {
181                         Method m = methodRepo.get(method.toString());
182                         StringTokenizer st = new StringTokenizer(input
183                                 .substring(j, input.substring(j).indexOf(")")
184                                         + j), "#\", ()");
185                         while (st.hasMoreTokens()) {
186                             String arg = st.nextToken();
187                             m.addArg(null, arg);
188                         }
189                         String returnValue = m.execute().toString();
190                         input = input.substring(0, i)
191                                 + returnValue
192                                 + input.substring(input.substring(i).indexOf(
193                                         ")")
194                                         + 1 + i);
195                         i = i + returnValue.length();
196                     } else {
197                         LOG.log(Level.SEVERE, "Invalid method signature in "
198                                 + input + " near " + method);
199                         break;
200                     }
201                 } catch (Exception ignored) {
202                 }
203                 break;
204             }
205         }
206         return input;
207     }

209     private void parseVariablesXML(NodeList list) throws ParseException, CommonsException, CasMetadataException {
210 
211         // loop through all variable elements
212         for (int i = 0; i < list.getLength(); i++) {
213             Node node = list.item(i);
214 
215             // parse variable element
216             if (node.getNodeName().equals("variable")) {
217                 NodeList children = node.getChildNodes();
218 
219                 // create Variable Object
220                 String variableName = ((Element) node).getAttribute("name");
221                 Variable variable = new Variable(variableName);
222 
223                 // loop through to fill Variable
224                 String type = null, value = null;
225                 for (int j = 0; j < children.getLength(); j++) {
226                     Node child = children.item(j);
227 
228                     // get the Variable's name
229                     if (child.getNodeName().equals("type")) {
230                         type = XMLUtils.getSimpleElementText((Element) child,
231                                 true).toLowerCase();
232 
233                         // get the Variable's value
234                     } else if (child.getNodeName().equals("value")) {
235                         value = PathUtils.doDynamicReplacement(XMLUtils
236                                 .getSimpleElementText((Element) child, false));
237 
238                         // get the Variable's value's precision infomation
239                     } else if (child.getNodeName().equals("precision")) {
240                         NodeList grandChildren = child.getChildNodes();
241                         for (int k = 0; k < grandChildren.getLength(); k++) {
242                             Node grandChild = grandChildren.item(k);
243                             // get the precision
244                             if (grandChild.getNodeName().equals("locations")) {
245                                 variable.setPrecision(Integer.parseInt(XMLUtils
246                                         .getSimpleElementText((Element) grandChild, true)));
247                                 // get the fill character to meet the precision
248                                 // [optional]
249                             } else if (grandChild.getNodeName().equals("fill")) {
250                                 variable.setFillString(
251                                     XMLUtils.getSimpleElementText((Element) grandChild, false));
252                                 // get the side for which the fill character
253                                 // will be applied [optional]
254                             } else if (grandChild.getNodeName().equals("side")) {
255                                 variable.setFillSide(
256                                         (XMLUtils.getSimpleElementText((Element) grandChild, true)
257                                             .toLowerCase().equals("front")) 
258                                                 ? variable.FILL_FRONT
259                                                 : variable.FILL_BACK);
260                             }
261                         }
262                     }
263                 }
264                 // determine if variable is an Integer or a String
265                 if (type.equals("int")) {
266                     variable.setValue(Integer.valueOf(value));
267                 } else {
268                   variable.setValue(value);
269                 }
270 
271                 // store Variable in list of Variables
272                 GlobalVariables.ConcurrentHashMap.put(variable.getName(), variable);
273             }
274         }
275     }

277     private void parseMethodsXML(NodeList list) {
278 
279         // loop though all method elements
280         for (int i = 0; i < list.getLength(); i++) {
281             Node node = list.item(i);
282             // parse method element
283             if (node.getNodeName().equals("method")) {
284                 NodeList children = node.getChildNodes();
285 
286                 // create Method Object
287                 String methodName = ((Element) node).getAttribute("name");
288                 Method method = new Method(methodName);
289 
290                 // loop through to fill Method Object
291                 for (int j = 0; j < children.getLength(); j++) {
292                     Node child = children.item(j);
293 
294                     // get the Method's behavoir
295                     if (child.getNodeName().equals("action")) {
296                         method.setBehavoir(XMLUtils.getSimpleElementText(
297                                 (Element) child, false));
298 
299                         // get the Method's arguments
300                     } else if (child.getNodeName().equals("args")) {
301                         String name, argType = null;
302                         NodeList grandChildren = child.getChildNodes();
303 
304                         // loop for every arg element
305                         for (int k = 0; k < grandChildren.getLength(); k++) {
306                             Node grandChild = grandChildren.item(k);
307 
308                             // parse arg element
309                             if (grandChild.getNodeName().equals("arg")) {
310                                 name = ((Element) grandChild)
311                                         .getAttribute("name");
312 
313                                 // get arg element properties
314                                 NodeList greatGrandChildren = grandChild
315                                         .getChildNodes();
316                                 for (int l = 0; l < greatGrandChildren
317                                         .getLength(); l++) {
318                                     Node greatGrandChild = greatGrandChildren
319                                             .item(l);
320                                     if (greatGrandChild.getNodeName().equals(
321                                             "type")) {
322                                         argType = XMLUtils.getSimpleElementText(
323                                                 (Element) greatGrandChild,
324                                                 true);
325                                     }
326                                 }
327 
328                                 // create argument signature in Method
329                                 method
330                                         .addArgSignature(
331                                                 name,
332                                                 ((argType.toLowerCase()
333                                                         .equals("int")) ? method.INT
334                                                         : method.STRING));
335                             }
336                         }
337                     }
338                 }
339 
340                 // store Method in list of Methods
341                 methodRepo.put(method.getName(), method);
342             }
343         }
344     }
