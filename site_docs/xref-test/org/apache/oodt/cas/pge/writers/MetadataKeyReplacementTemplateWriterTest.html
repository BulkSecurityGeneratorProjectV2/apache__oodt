43  public class MetadataKeyReplacementTemplateWriterTest extends TestCase {
44  
45    private static final Logger LOG = Logger
46        .getLogger(MetadataKeyReplacementTemplateWriterTest.class.getName());
47  
48    private static final String expected = "Welcome to ApacheCon Paul and Chris! You are a member of the following projects OODT,Tika,SIS,Gora.";
49  
50    public void testGenerateTemplate() throws IOException {
51      MetadataKeyReplacerTemplateWriter writer = new MetadataKeyReplacerTemplateWriter();
52      String templateSourcePath = "./src/test/resources/metkeyreplace.template";
53      String outPath = File.createTempFile("foo", "bar").getParentFile()
54          .getAbsolutePath();
55      if (!outPath.endsWith("/"))
56        outPath += "/";
57  
58      outPath += "generated.template";
59  
60      Metadata met = new Metadata();
61      met.addMetadata("Person1", "Paul");
62      met.addMetadata("Person2", "Chris");
63      met.addMetadata("Projects", "OODT");
64      met.addMetadata("Projects", "Tika");
65      met.addMetadata("Projects", "SIS");
66      met.addMetadata("Projects", "Gora");
67  
68      try {
69        writer.generateFile(outPath, met, LOG, templateSourcePath);
70      } catch (Exception e) {
71        LOG.log(Level.SEVERE, e.getMessage());
72        fail(e.getMessage());
73      }
74  
75      assertTrue(new File(outPath).exists());
76      String fileString = FileUtils.readFileToString(new File(outPath));
77      assertNotNull(fileString);
78      assertEquals(expected, fileString);
79    }
