17  package org.apache.oodt.cas.workflow.cli.action;
18  
19  //JDK imports
20  import java.util.List;
21  
22  //OODT imports
23  import org.apache.oodt.cas.cli.exception.CmdLineActionException;
24  import org.apache.oodt.cas.workflow.structs.WorkflowInstance;
25  import org.apache.oodt.cas.workflow.structs.WorkflowInstancePage;
26  import org.apache.oodt.cas.workflow.system.XmlRpcWorkflowManagerClient;
27  
28  /**
29   * A {@link CmdLineOption} which gets the last page of workflows.
30   * 
31   * @author bfoster (Brian Foster)
32   */
33  public class GetLastPageCliAction extends WorkflowCliAction {
34  
35     private String status;
36  
37     @SuppressWarnings("unchecked")
38     @Override
39     public void execute(ActionMessagePrinter printer)
40           throws CmdLineActionException {
41        try {
42           XmlRpcWorkflowManagerClient client = getClient();
43           WorkflowInstancePage page;
44           if (status != null && !status.equals("")) {
45              WorkflowInstancePage firstPage = client.paginateWorkflowInstances(
46                    1, status);
47              page = client.paginateWorkflowInstances(firstPage.getTotalPages(),
48                    status);
49           } else {
50              page = client.getLastPage();
51           }
52  
53           printer.println("Page: [num=" + page.getPageNum() + ","
54                 + "pageSize=" + page.getPageSize() + ",totalPages="
55                 + page.getTotalPages() + "]");
56           if (page.getPageWorkflows() == null) {
57              throw new Exception(
58                    "WorkflowManager returned null page of workflows");
59           }
60           for (WorkflowInstance inst : (List<WorkflowInstance>) page
61                 .getPageWorkflows()) {
62              printer.println("Instance: [id=" + inst.getId() + ", status="
63                    + inst.getStatus() + ", currentTask="
64                    + inst.getCurrentTaskId() + ", workflow="
65                    + inst.getWorkflow().getName() + ",wallClockTime="
66                    + client.getWorkflowWallClockMinutes(inst.getId())
67                    + ",currentTaskWallClockTime="
68                    + client.getWorkflowCurrentTaskWallClockMinutes(inst.getId())
69                    + "]");
70           }
71        } catch (Exception e) {
72           throw new CmdLineActionException(
73                 "Failed to get last page of workflows with " + "status '"
74                       + status + "' : " + e.getMessage(), e);
75        }
76     }
77  
78     public void setStatus(String status) {
79        this.status = status;
80     }
81  }
