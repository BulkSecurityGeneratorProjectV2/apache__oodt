33  import junit.framework.TestCase;

35  public class TestXmlHelper extends TestCase {
36  
37     private static final String BLANK_PGE_CONFIG = "src/test/resources/blank-pge-config.xml";
38     private static final String IMPORTS_ONLY_PGE_CONFIG = "src/test/resources/imports-only-pge-config.xml";
39  
40     // Tests that only root node is required.
41     public void testLoadBlankXmlFile() throws FileNotFoundException {
42        Element elem = getRootElement(BLANK_PGE_CONFIG);
43        assertNotNull(elem);
44        assertEquals(1, elem.getChildNodes().getLength());
45        assertEquals("#text", elem.getChildNodes().item(0).getNodeName());
46        assertEquals("", elem.getChildNodes().item(0).getNodeValue().trim());
47     }
48  
49     public void testGetImports() throws Exception {
50        Element elem = getRootElement(IMPORTS_ONLY_PGE_CONFIG);
51        Metadata metadata = new Metadata();
52        List<Pair<String, String>> imports = getImports(elem, metadata);
53        assertEquals(2, imports.size());
54        assertEquals("blank", imports.get(0).getFirst());
55        assertEquals("blank-pge-config.xml", imports.get(0).getSecond());
56        assertEquals(null, imports.get(1).getFirst());
57        assertEquals("pge-config.xml", imports.get(1).getSecond());
58     }
59  
60     public void testGetNamespace() throws Exception {
61        Element elem = getRootElement(IMPORTS_ONLY_PGE_CONFIG);
62        Metadata metadata = new Metadata();
63  
64        NodeList importTags = elem.getElementsByTagName(IMPORT_TAG);
65        assertEquals("blank", getNamespace((Element) importTags.item(0), metadata));
66        assertEquals("blank-pge-config.xml", getFile((Element) importTags.item(0), metadata));
67        assertNull(getNamespace((Element) importTags.item(1), metadata));
68        assertEquals("pge-config.xml", getFile((Element) importTags.item(1), metadata));
69     }
