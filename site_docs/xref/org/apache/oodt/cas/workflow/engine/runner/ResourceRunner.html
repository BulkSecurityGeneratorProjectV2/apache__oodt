53    protected XmlRpcResourceManagerClient rClient;

55    private String currentJobId;

57    public ResourceRunner(URL resUrl, WorkflowInstanceRepository instRep) {
58      super();
59      this.rClient = new XmlRpcResourceManagerClient(resUrl);
60    }
61  
62    /* (non-Javadoc)
63     * @see org.apache.oodt.cas.workflow.engine.runner.EngineRunner#execute(org.apache.oodt.cas.workflow.engine.processor.TaskProcessor)
64     */
65    @Override
66    public void execute(TaskProcessor taskProcessor) {
67      Job workflowTaskJob = new Job();
68      WorkflowTask workflowTask = getTaskFromProcessor(taskProcessor);
69      workflowTaskJob.setName(workflowTask.getTaskId());
70      workflowTaskJob
71          .setJobInstanceClassName("org.apache.oodt.cas.workflow.structs.TaskJob");
72      workflowTaskJob
73          .setJobInputClassName("org.apache.oodt.cas.workflow.structs.TaskJobInput");
74      workflowTaskJob.setLoadValue(2);
75      workflowTaskJob.setQueueName(workflowTask.getTaskConfig().getProperty(
76          QUEUE_NAME) != null ? workflowTask.getTaskConfig().getProperty(
77          QUEUE_NAME) : DEFAULT_QUEUE_NAME);
78  
79      if (workflowTask.getTaskConfig().getProperty(TASK_LOAD) != null) {
80        workflowTaskJob.setLoadValue(Integer.valueOf(workflowTask.getTaskConfig()
81            .getProperty(TASK_LOAD)));
82      }
83  
84      TaskJobInput in = new TaskJobInput();
85      in.setDynMetadata(taskProcessor.getWorkflowInstance().getSharedContext());
86      in.setTaskConfig(workflowTask.getTaskConfig());
87      in.setWorkflowTaskInstanceClassName(workflowTask.getTaskInstanceClassName());
88  
89      try {
90        this.currentJobId = rClient.submitJob(workflowTaskJob, in);
91      } catch (JobExecutionException e) {
92        LOG.log(Level.WARNING,
93            "Job execution exception using resource manager to execute job: Message: "
94                + e.getMessage());
95      }
96  
97    }

132   protected boolean safeCheckJobComplete(String jobId) {
133     try {
134       return rClient.isJobComplete(jobId);
135     } catch (Exception e) {
136       LOG.log(Level.WARNING, "Exception checking completion status for job: ["
137           + jobId + "]: Messsage: " + e.getMessage());
138       return false;
139     }
140   }
141 
142   protected boolean stopJob(String jobId) {
143     if (this.rClient != null && this.currentJobId != null) {
144       if (!this.rClient.killJob(this.currentJobId)) {
145         LOG.log(Level.WARNING, "Attempt to kill " + "current resmgr job: ["
146             + this.currentJobId + "]: failed");
147         return false;
148       } else {
149         return true;
150       }
151     } else {
152       return false;
153     }
154   }
