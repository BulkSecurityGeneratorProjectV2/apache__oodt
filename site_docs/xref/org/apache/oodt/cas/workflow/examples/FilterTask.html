18  package org.apache.oodt.cas.workflow.examples;
19  
20  import java.util.Arrays;
21  import java.util.logging.Level;
22  import java.util.logging.Logger;
23  
24  import org.apache.oodt.cas.metadata.Metadata;
25  import org.apache.oodt.cas.workflow.structs.WorkflowTaskConfiguration;
26  import org.apache.oodt.cas.workflow.structs.WorkflowTaskInstance;
27  import org.apache.oodt.cas.workflow.structs.exceptions.WorkflowTaskInstanceException;
28  
29  /**
30   * Filters dynamic {@link Metadata} from the provided metadata in the
31   * {@link #run(Metadata, WorkflowTaskConfiguration)} method.
32   * 
33   * The config parameter in {@link #run(Metadata, WorkflowTaskConfiguration)}
34   * defines metadata parameters to remove and rename via the following
35   * directives:
36   * 
37   * <ul>
38   * <li>Remove_Key - a comma separated list of keys to remove</li>
39   * <li>Rename_[Key] - set the value of this property to the new name of the
40   * [Key] parameter. E.g., having a key in the task configuration named
41   * Rename_Filename with value set to Prior_Filename will rename the key in the
42   * dynamic metadata named Filename to the new name of Prior_Filename, preserving
43   * its prior value.</li>
44   * </ul>
45   * 
46   * Note that the Rename keys will be evaluated first, before the Remove_Key
47   * configuration parameter is evaluated.
48   * 
49   */
50  public class FilterTask implements WorkflowTaskInstance {
51  
52  	private final static String REMOVE_KEY = "Remove_Key";
53  	private static final Logger LOG = Logger.getLogger(FilterTask.class
54  			.getName());
55  
56  	@Override
57  	public void run(Metadata metadata, WorkflowTaskConfiguration config)
58  			throws WorkflowTaskInstanceException {
59  
60  		// evaluate renames
61  		for (Object configKey : config.getProperties().keySet()) {
62  			String configKeyName = (String) configKey;
63  			if (configKeyName.startsWith("Rename")) {
64  				String renameOrigKeyName = configKeyName.split("_")[1];
65  				String renameKeyName = config.getProperty(configKeyName);
66  				// check to see if key exists
67  				if (metadata.containsKey(renameOrigKeyName)) {
68  					LOG.log(Level.INFO, "Renaming key: [" + renameOrigKeyName
69  							+ "] to [" + renameKeyName + "]: values: "
70  							+ metadata.getAllMetadata(renameOrigKeyName));
71  					metadata.replaceMetadata(renameKeyName,
72  							metadata.getAllMetadata(renameOrigKeyName));
73  					metadata.removeMetadata(renameOrigKeyName);
74  				} else {
75  					LOG.log(Level.WARNING, "Request to rename key: ["
76  							+ renameOrigKeyName + "] to [" + renameKeyName
77  							+ "]: orig key does not exist in dynamic metadata!");
78  				}
79  			}
80  		}
81  
82  		// evaluate removes
83  		if (config.getProperties().containsKey(REMOVE_KEY)) {
84  			String removeMetKeyNames = config.getProperty(REMOVE_KEY);
85  			for (String keyName : Arrays.asList(removeMetKeyNames.split(","))) {
86  				// handle whitespace
87  				keyName = keyName.trim();
88  				if (metadata.containsKey(keyName)) {
89  					LOG.log(Level.INFO,
90  							"Removing key from workflow metadata: [" + keyName
91  									+ "]: values: "
92  									+ metadata.getAllMetadata(keyName));
93  					metadata.removeMetadata(keyName);
94  				} else {
95  					LOG.log(Level.WARNING, "Request to remove key: [" + keyName
96  							+ "]: key does not exist in workflow metadata!");  
97  				}
98  			}
99  		}
100 
101 	}
102 
103 }
