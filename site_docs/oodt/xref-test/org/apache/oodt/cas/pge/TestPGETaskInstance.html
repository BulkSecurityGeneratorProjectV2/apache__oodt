163             "some.prop.adder.classpath"));

165             "some.other.classpath"));

167       // Verify still works when only one property adder is specified.

206       assertEquals(Level.INFO.getLocalizedName() + ": pge1 message2", messages.get(3));
207       assertEquals(Level.INFO.getLocalizedName() + ": pge1 message3", messages.get(5));
208       logDir = new File(pgeTask2.pgeConfig.getExeDir() + "/logs");
209       assertTrue(logDir.exists());
210       messages = FileUtils.readLines(logDir.listFiles(
211          new FileFilter() {
212             @Override
213             public boolean accept(File pathname) {
214                return pathname.getName().endsWith(".log");
215             }
216          })[0], "UTF-8");

350   @Test
351    public void testDumpMetadataIfRequested() throws Exception {
352       PGETaskInstance pgeTask = createTestInstance();
353       File dumpMetFile = new File(pgeTask.getDumpMetadataPath());
354       pgeTask.dumpMetadataIfRequested();
355       assertFalse(dumpMetFile.exists());
356       pgeTask.pgeMetadata.replaceMetadata(DUMP_METADATA, "true");
357       pgeTask.dumpMetadataIfRequested();
358       assertTrue(dumpMetFile.exists());
359       @SuppressWarnings("unchecked")
360       String expectedMetString =    "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
361     		+ "<cas:metadata xmlns:cas=\"http://oodt.jpl.nasa.gov/1.0/cas\">"
362       		+ "<keyval type=\"vector\">"
363     		+ "<key>PGETask_Name</key>"
364       		+ "<val>"
365     		+ pgeTask.pgeMetadata.getMetadata(NAME)
366     		+ "</val>"
367     		+ "</keyval>"
368     		+ "<keyval type=\"vector\">"
369     		+ "<key>PGETask_DumpMetadata</key>"
370     		+ "<val>"
371     		+ pgeTask.pgeMetadata.getMetadata(DUMP_METADATA)
372     		+ "</val>"
373     		+ "</keyval>"
374     		+ "</cas:metadata>";
375        
376        Document dumpMetDoc = parseXmlFile(dumpMetFile);
377        Document expectedMetDoc = parseXmlString(expectedMetString);
378        
379        
380        Element dumpMetRoot, expectedMetRoot;
381        dumpMetRoot = dumpMetDoc.getDocumentElement();
382        expectedMetRoot = expectedMetDoc.getDocumentElement();
383        
384        
385        NodeList dumpKeyList = dumpMetRoot.getElementsByTagName("key");
386        NodeList expectedKeyList = expectedMetRoot.getElementsByTagName("key");
387        
388        NodeList dumpValList = dumpMetRoot.getElementsByTagName("val");
389        NodeList expectedValList = expectedMetRoot.getElementsByTagName("val");
390        
391        assertEquals(dumpKeyList.getLength(), expectedKeyList.getLength());
392        assertEquals(dumpValList.getLength(), expectedValList.getLength());
393        
394        
395        Map<String, String> dumpKeyValMap = new ConcurrentHashMap<String, String>();
396        Map<String, String> expectedKeyValMap = new ConcurrentHashMap<String, String>();
397        
398        for (int i = 0; i < dumpKeyList.getLength(); i++) {
399            
400            Node k1 = dumpKeyList.item(i);
401            Node k2 = expectedKeyList.item(i);
402            
403            Node v1 = dumpValList.item(i);
404            Node v2 = expectedValList.item(i);
405            
406            dumpKeyValMap.put(k1.getFirstChild().getNodeValue(), v1.getFirstChild().getNodeValue());
407            expectedKeyValMap.put(k2.getFirstChild().getNodeValue(), v2.getFirstChild().getNodeValue());
408            
409        }
410        
411        assertTrue(dumpKeyValMap.equals(expectedKeyValMap));
412        
413    }
