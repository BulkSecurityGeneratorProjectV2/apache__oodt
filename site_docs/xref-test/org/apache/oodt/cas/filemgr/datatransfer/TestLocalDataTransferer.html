24  import org.apache.oodt.cas.filemgr.structs.Reference;

27  //JDK imports
28  import java.io.File;

31  import java.util.UUID;
32

34  import junit.framework.TestCase;

36  /**
37   * Test class for {@link LocalDataTransferer}.
38   *
39   * @author mattmann (Chris Mattmann)
40   * @author bfoster (Brian Foster)
41   */
42  public class TestLocalDataTransferer extends TestCase {
43  
44     private LocalDataTransferer transfer;
45  
46     private File origFile;
47     private File testDir;
48     private File repoDir;
49     private File repoFile;
50     private File destDir;
51     private File destFile;
52  
53     public void setUp() throws Exception {
54        transfer = (LocalDataTransferer) new LocalDataTransferFactory()
55           .createDataTransfer();
56        URL url = this.getClass().getResource("/test.txt");
57        origFile = new File(url.getFile());
58        File testFile = File.createTempFile("test", ".txt");
59        testDir = new File(testFile.getParentFile(), UUID.randomUUID().toString());
60        repoDir = new File(testDir, "repo");
61        if (!repoDir.mkdirs()) {
62           throw new Exception("Failed to create repo directory!");
63        }
64        repoFile = new File(repoDir, "test.txt"); 
65        destDir = new File(testDir, "dest");
66        if (!destDir.mkdirs()) {
67           throw new Exception("Failed to create destination directory!");
68        }
69        destFile = new File(destDir, "test.txt"); 
70     }
71  
72     public void tearDown() throws Exception {
73        FileUtils.forceDelete(testDir);
74     }
75  
76     public void testTransferAndRetrieve() throws DataTransferException, IOException {
77        Product testProduct = createDummyProduct();
78  
79        // Test transfer.
80        transfer.transferProduct(testProduct);
81  
82        // Check that file was successfully transfered.
83        assertTrue("Repo file does not exist", repoFile.exists());
84        assertTrue("Repo file does not have the same contents as orig file",
85              FileUtils.contentEquals(origFile, repoFile));
86  
87        // Test retrieve
88        transfer.retrieveProduct(testProduct, destDir);
89  
90        // Check that file was successfully transfered.
91        assertTrue("Destination file does not exist", destFile.exists());
92        assertTrue("Destination file does not have the same contents as orig file",
93              FileUtils.contentEquals(origFile, destFile));
94     }
95  
96     private Product createDummyProduct() {
97        Product testProduct = Product.getDefaultFlatProduct("test",
98              "urn:oodt:GenericFile");
99        testProduct.getProductReferences().add(
100             new Reference(origFile.toURI().toString(), new File(repoDir,
101                   "test.txt").toURI().toString(), origFile.length()));
102       return testProduct;
103    }
