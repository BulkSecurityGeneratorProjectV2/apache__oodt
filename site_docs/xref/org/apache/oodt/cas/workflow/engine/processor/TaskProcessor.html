32  /**
33   * 
34   * WorkflowProcessor which handles running task workflows.
35   * 
36   * @author bfoster
37   * @author mattmann
38   * 
39   * @version $Revision$
40   */
41  public class TaskProcessor extends WorkflowProcessor {
42  
43    public static final double DOUBLE = 0.1;
44    public static final int INT = 60;
45    private Class<? extends WorkflowTaskInstance> instanceClass;
46    private String jobId;
47    
48    public TaskProcessor(WorkflowLifecycleManager lifecycleManager, WorkflowInstance instance) {
49      super(lifecycleManager, instance);
50    }
51  
52    public Class<? extends WorkflowTaskInstance> getInstanceClass() {
53      return this.instanceClass;
54    }
55  
56    public void setJobId(String jobId) {
57      this.jobId = jobId;
58    }
59  
60    public String getJobId() {
61      return this.jobId;
62    }
63  
64    public void setInstanceClass(
65        Class<? extends WorkflowTaskInstance> instanceClass) {
66      this.instanceClass = instanceClass;
67    }
68  
69    @Override
70    public void setWorkflowInstance(WorkflowInstance instance) {
71      instance.setPriority(Priority
72          .getPriority(instance.getPriority().getValue() + DOUBLE));
73      super.setWorkflowInstance(instance);
74    }
75  
76    @Override
77    public List<TaskProcessor> getRunnableWorkflowProcessors() {
78      List<TaskProcessor> tps = super.getRunnableWorkflowProcessors();
79      if (tps.size() == 0) {
80        if (this.getWorkflowInstance().getState().getName().equals("Blocked")) {
81          String requiredBlockTimeElapseString = this.getWorkflowInstance()
82              .getCurrentTask().getTaskConfig().getProperty("BlockTimeElapse");
83          int requiredBlockTimeElapse = 2;
84          if (requiredBlockTimeElapseString != null) {
85            try {
86              requiredBlockTimeElapse = Integer
87                  .parseInt(requiredBlockTimeElapseString);
88            } catch (Exception ignored) {
89            }
90          }
91          Calendar calendar = Calendar.getInstance();
92          calendar.setTime(this.getWorkflowInstance().getState().getStartTime());
93          long elapsedTime = ((System.currentTimeMillis() - calendar
94              .getTimeInMillis()) / 1000) / INT;
95          if (elapsedTime >= requiredBlockTimeElapse) {
96            tps.add(this);
97          }
98        } else if (this.isAnyState("Loaded", "Queued", "PreConditionSuccess") && 
99            !this.isAnyState("Executing") && this.passedPreConditions()){
100         tps.add(this);
101       }
102     }
103     return tps;
104   }
105
