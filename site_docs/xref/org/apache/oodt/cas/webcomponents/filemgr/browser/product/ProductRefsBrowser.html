44  /**
45   *
46   * Component responsible for showing a {@link org.apache.oodt.cas.filemgr.structs.Product}s
47   * {@link List} of {@link Reference}s.
48   *
49   * @author mattmann
50   * @version $Revision$
51   *
52   */
53  public class ProductRefsBrowser extends Panel {
54    
55    private static final long serialVersionUID = -2278188309669737798L;
56  
57    private static final Logger LOG = Logger.getLogger(ProductRefsBrowser.class.getName());
58    
59    private FileManagerConn fm;
60    
61    public ProductRefsBrowser(String componentId, String fmUrlStr, String productId){
62      super(componentId);
63      this.fm = new FileManagerConn(fmUrlStr);
64      List<Reference> refs = null;
65      try {
66        refs = fm.getProductReferences(fm.getFm().getProductById(productId));
67      } catch (CatalogException e1) {
68        e1.printStackTrace();
69      }
70      if (refs != null && refs.size() > 0) {
71        add(new Label("no_prod_ref_display") {
72          /*
73           * (non-Javadoc)
74           * 
75           * @see org.apache.wicket.Component#isVisible()
76           */
77          @Override
78          public boolean isVisible() {
79            return false;
80          }
81        });
82  
83        add(new ListView<Reference>("ref_list", new ListModel<Reference>(refs)) {
84          /*
85           * (non-Javadoc)
86           * 
87           * @see
88           * org.apache.wicket.markup.html.list.ListView#populateItem(org.apache
89           * .wicket.markup.html.list.ListItem)
90           */
91          @Override
92          protected void populateItem(ListItem<Reference> refItem) {
93            Reference r = refItem.getModelObject();
94            String filePath = null;
95            try {
96              filePath = new File(new URI(r.getDataStoreReference()))
97                  .getAbsolutePath();
98            } catch (Exception ignore) {
99            }
100 
101           refItem.add(new Label("ref_file_path", filePath));
102           refItem.add(new Label("ref_file_size", String
103               .valueOf(r.getFileSize())));
104           try {
105             refItem.add(new Label("ref_pct_transferred", NumberFormat
106                 .getPercentInstance().format(
107                     fm.getFm()
108                         .getRefPctTransferred(r))));
109           } catch (DataTransferException e) {
110             LOG.log(Level.WARNING,
111                 "Unable to determine product reference size: Reason: "
112                     + e.getMessage());
113             refItem.add(new Label("ref_pct_transferred", "N/A"));
114           }
115 
116         }
117       });
118     } else {
119       add(new Label("no_prod_ref_display", "No Product References!"));
120     }    
121   }
