35   * <p>
36   * A Configuration File Writer for PGEs.
37   * </p>

43  public final class PGEConfigFileWriter implements PGEConfigFileKeys,
44      PGEDataParseKeys {

46    /* the PGE configuration file that we're writing */
47    private PGEConfigurationFile configFile = null;

50    private static Logger LOG = Logger.getLogger(PGEConfigFileWriter.class
51        .getName());

53    /*
54     * whether or not the values in the XML file should be URLEncoded: if true,
55     * the values will be encoded using UTF-8.
56     */
57    private boolean urlEncoding = false;

60    private String schemaLocation = null;

90    /**
91     * 
92     * @return An XML DOM {@link Document} representation of the internal
93     *         PGEConfigurationFile.
94     * @throws Exception
95     *           If any error occurs.
96     */
97    public Document getConfigFileXml() throws Exception {
98      DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
99      factory.setNamespaceAware(true);
100     Document document;
101 
102     try {
103       DocumentBuilder builder = factory.newDocumentBuilder();
104       document = builder.newDocument();
105 
106       Element root = (Element) document.createElement(PGE_INPUT_TAG_NAME);
107       root.setAttribute("xmlns:xsi",
108           "http://www.w3.org/2001/XMLSchema-instance");
109       root
110           .setAttribute(
111               "xsi:noNamespaceSchemaLocation",
112               (schemaLocation == null || (schemaLocation.equals("")) ? "input.xsd"
113                   : schemaLocation));
114       document.appendChild(root);
115 
116       if (configFile != null) {
117         // write the PGE Name group
118         if (configFile.getPgeName() != null) {
119           PGEGroup pgeNameGroup = new PGEGroup(PGE_NAME_GROUP);
120           pgeNameGroup.addScalar(configFile.getPgeName());
121           root.appendChild(getGroupElement(pgeNameGroup, document));
122         }
123 
124         // write the input product files
125         root.appendChild(getGroupElement(configFile.getInputProductFiles(),
126             document));
127 
128         // write the static file identification files
129         root.appendChild(getGroupElement(configFile
130             .getStaticFileIdentificationFiles(), document));
131 
132         // write the dynamic auxilliary files
133         root.appendChild(getGroupElement(configFile
134             .getDynamicAuxiliaryInputFiles(), document));
135 
136         // write the recorded auxilliary files
137         root.appendChild(getGroupElement(configFile
138             .getRecordedAuxiliaryInputFiles(), document));
139 
140         // write the product path group
141         if (configFile.getProductPath() != null) {
142           PGEGroup productPathGroup = new PGEGroup(PRODUCT_PATH_GROUP);
143           productPathGroup.addScalar(configFile.getProductPath());
144           root.appendChild(getGroupElement(productPathGroup, document));
145         }
146 
147         // write the monitor level group
148         root.appendChild(getGroupElement(configFile.getMonitorLevelGroup(),
149             document));
150 
151         // write the monitor group
152         if (configFile.getMonitorFilenameFormat() != null
153             && configFile.getMonitorPath() != null) {
154           PGEGroup monitorGroup = new PGEGroup(MONITOR_GROUP);
155           monitorGroup.addScalar(configFile.getMonitorPath());
156           monitorGroup.addScalar(configFile.getMonitorFilenameFormat());
157           root.appendChild(getGroupElement(monitorGroup, document));
158         }
159 
160         // write the pge specific groups
161         for (String pgeSpecificGroupName : configFile.getPgeSpecificGroups().keySet()) {
162           PGEGroup pgeSpecificGroup = (PGEGroup) configFile
163               .getPgeSpecificGroups().get(pgeSpecificGroupName);
164 
165           root.appendChild(getGroupElement(pgeSpecificGroup, document));
166         }
167 
168       }
169 
170       return document;
171 
172     } catch (ParserConfigurationException pce) {
173       LOG.log(Level.WARNING, "Error generating pge configuration file!: "
174           + pce.getMessage());
175       throw new Exception("Error generating pge configuration file!: "
176           + pce.getMessage());
177     } catch (Exception e) {
178       LOG.log(Level.SEVERE, e.getMessage());
179       throw e;
180     }
181 
182   }

184   private Element getGroupElement(PGEGroup group, Document document)
185       throws Exception {
186     Element groupElem = document.createElement(GROUP_TAG_NAME);
187     groupElem.setAttribute(NAME_ATTR, group.getName());
188 
189     if (group.getNumScalars() > 0) {
190       for (String scalarName : group.getScalars().keySet()) {
191         PGEScalar scalar = group.getScalar(scalarName);
192 
193         Element scalarElem = document.createElement(SCALAR_TAG_NAME);
194         scalarElem.setAttribute(NAME_ATTR, scalar.getName());
195 
196         if (scalar.getValue() == null) {
197           throw new Exception("Attempt to write null value for scalar: ["
198                               + scalarName + "] to PGE config file!");
199         }
200 
201         if (urlEncoding) {
202           try {
203             scalarElem.appendChild(document.createTextNode(URLEncoder.encode(
204                 scalar.getValue(), "UTF-8")));
205           } catch (UnsupportedEncodingException e) {
206             LOG.log(Level.WARNING,
207                 "Error creating text node for scalar element: "
208                 + scalar.getName() + " in pge group: " + group.getName()
209                 + " Message: " + e.getMessage());
210           }
211 
212         } else {
213           scalarElem.appendChild(document.createTextNode(scalar.getValue()));
214         }
215 
216         groupElem.appendChild(scalarElem);
217       }
218 
219     }
220 
221     if (group.getNumVectors() > 0) {
222       for (String vectorName : group.getVectors().keySet()) {
223         PGEVector vector = group.getVector(vectorName);
224 
225         Element vectorElem = document.createElement(VECTOR_TAG_NAME);
226         vectorElem.setAttribute(NAME_ATTR, vector.getName());
227 
228         for (Object o : vector.getElements()) {
229           String element = (String) o;
230 
231           if (element == null) {
232             throw new Exception("Attempt to write null value for vector: ["
233                                 + vectorName + "] to PGE config file!");
234           }
235 
236           Element elementElem = document.createElement(VECTOR_ELEMENT_TAG);
237           if (urlEncoding) {
238             try {
239               elementElem.appendChild(document.createTextNode(URLEncoder
240                   .encode(element, "UTF-8")));
241             } catch (UnsupportedEncodingException e) {
242               LOG.log(Level.WARNING,
243                   "Error creating text node for vector element: "
244                   + vector.getName() + " in pge group: " + group.getName()
245                   + " Message: " + e.getMessage());
246             }
247           } else {
248             elementElem.appendChild(document.createTextNode(element));
249           }
250 
251           vectorElem.appendChild(elementElem);
252         }
253 
254         groupElem.appendChild(vectorElem);
255       }
256     }
257 
258     if (group.getNumMatrixs() > 0) {
259       for (String matrixName : group.getMatrixs().keySet()) {
260         PGEMatrix matrix = group.getMatrix(matrixName);
261 
262         Element matrixElem = document.createElement(MATRIX_TAG_NAME);
263         matrixElem.setAttribute(NAME_ATTR, matrix.getName());
264 
265         int rowNum = 0;
266         for (List<Object> objects : matrix.getRows()) {
267           List rowValues = (List) objects;
268 
269           Element rowElem = document.createElement(MATRIX_ROW_TAG);
270 
271           int colNum = 0;
272           for (Object rowValue : rowValues) {
273             String colValue = (String) rowValue;
274             Element colElem = document.createElement(MATRIX_COL_TAG);
275 
276             if (colValue == null) {
277               throw new Exception("Attempt to write null value for matrix: ["
278                                   + matrixName + "]: " + "(" + rowNum + "," + colNum + ")");
279             }
280 
281             if (urlEncoding) {
282               try {
283                 colElem.appendChild(document.createTextNode(URLEncoder.encode(
284                     colValue, "UTF-8")));
285               } catch (UnsupportedEncodingException e) {
286                 LOG.log(Level.WARNING,
287                     "Error creating node for matrix element: "
288                     + matrix.getName() + " (" + rowNum + "," + colNum
289                     + ") in pge group: " + group.getName() + " Message: "
290                     + e.getMessage());
291               }
292 
293             } else {
294               colElem.appendChild(document.createTextNode(colValue));
295             }
296 
297             colNum++;
298           }
299 
300           rowNum++;
301         }
302 
303         groupElem.appendChild(matrixElem);
304       }
305     }
306 
307     if (group.getNumGroups() > 0) {
308       for (String groupName : group.getGroups().keySet()) {
309         PGEGroup subgroup = group.getGroup(groupName);
310         Element subgroupElem = getGroupElement(subgroup, document);
311         groupElem.appendChild(subgroupElem);
312       }
313 
314     }
315 
316     return groupElem;
317 
318   }
