65      public QueryTool(URL fmUrl) throws InstantiationException {
66          try {
67              client = new XmlRpcFileManagerClient(fmUrl);
68          } catch (ConnectionException e) {
69              throw new InstantiationException(e.getMessage());
70          }
71      }

74      public List query(org.apache.oodt.cas.filemgr.structs.Query query) {
75          List prodIds = new Vector();
76          List products;
77  
78          List productTypes = safeGetProductTypes();
79  
80          if (productTypes != null && productTypes.size() > 0) {
81              for (Object productType : productTypes) {
82                  ProductType type = (ProductType) productType;
83                  try {
84                      products = client.query(query, type);
85                      if (products != null && products.size() > 0) {
86                          for (Object product1 : products) {
87                              Product product = (Product) product1;
88                              prodIds.add(product.getProductId());
89                          }
90                      }
91                  } catch (CatalogException e) {
92                      LOG.log(Level.WARNING, "Exception querying for: ["
93                                             + type.getName() + "] products: Message: "
94                                             + e.getMessage());
95                  }
96  
97              }
98  
99          }
100 
101         return prodIds;
102 
103     }

106     public static Query parseQuery(String query) {
107       QueryParser parser;
108       // note that "__FREE__" is a control work for free text searching
109       parser = new QueryParser(freeTextBlock, new CASAnalyzer());
110       Query luceneQ = null;
111       try {
112           luceneQ = (Query) parser.parse(query);
113       } catch (ParseException e) {
114           System.out.println("Error parsing query text.");
115           System.exit(-1);
116       }
117       return luceneQ;
118     }

120     public void generateCASQuery(
121         org.apache.oodt.cas.filemgr.structs.Query casQuery,
122         Query luceneQuery) {
123     if (luceneQuery instanceof TermQuery) {
124         Term t = ((TermQuery) luceneQuery).getTerm();
125         if (!t.field().equals(freeTextBlock)) {
126             casQuery.addCriterion(new TermQueryCriteria(t.field(), 
127                     t.text()));
128         }
129     } else if (luceneQuery instanceof PhraseQuery) {
130         Term[] t = ((PhraseQuery) luceneQuery).getTerms();
131         if (!t[0].field().equals(freeTextBlock)) {
132             for (Term aT : t) {
133                 casQuery.addCriterion(new TermQueryCriteria(
134                     aT.field(), aT.text()));
135             }
136         }
137     } else if (luceneQuery instanceof TermRangeQuery) {
138         BytesRef startT = ((TermRangeQuery) luceneQuery).getLowerTerm();
139         BytesRef endT = ((TermRangeQuery) luceneQuery).getUpperTerm();
140         casQuery.addCriterion(new RangeQueryCriteria(((TermRangeQuery) luceneQuery).getField(), startT.utf8ToString(), endT.utf8ToString()));
141     } else if (luceneQuery instanceof BooleanQuery) {
142         List<BooleanClause> clauses = ((BooleanQuery) luceneQuery).clauses();
143         for (BooleanClause clause : clauses) {
144             generateCASQuery(casQuery, (clause).getQuery());
145         }
146     } else {
147         throw new RuntimeException(
148                 "Error parsing query! Cannot determine clause type: ["
149                         + luceneQuery.getClass().getName() + "] !");
150     }
151 }

153     public static void main(String[] args)
154         throws MalformedURLException, InstantiationException, CatalogException, QueryFormulationException,
155         ConnectionException {
156         String usage = "Usage: QueryTool [options] \n"
157             + "options: \n"
158             + "--url <fm url> \n"
159             + "  Lucene like query options: \n"
160             + "    --lucene \n"
161             + "         -query <query> \n"
162             + "  SQL like query options: \n"
163             + "    --sql \n"
164             + "         -query <query> \n"
165             + "         -sortBy <metadata-key> \n"
166             + "         -outputFormat <output-format-string> \n";
167                 
168         String fmUrlStr = null, queryStr = null, sortBy = null, outputFormat = null, delimiter = null;
169         QueryType queryType = null;
170         for (int i = 0; i < args.length; i++) {
171             if (args[i].equals("--lucene")) {
172                 if (queryType != null) {
173                     exit("ERROR: Can only perform one query at a time! \n" + usage);
174                 }
175                 if (args[++i].equals("-query")) {
176                     queryStr = args[++i];
177                 } else {
178                     exit("ERROR: Must specify a query! \n" + usage);
179                 }
180                 queryType = QueryType.LUCENE;
181             }else if (args[i].equals("--sql")) {
182                 if (queryType != null) {
183                     exit("ERROR: Can only perform one query at a time! \n" + usage);
184                 }
185                 if (args[++i].equals("-query")) {
186                     queryStr = args[++i];
187                 } else {
188                     exit("ERROR: Must specify a query! \n" + usage);
189                 }
190                 for (; i < args.length; i++) {
191                     if (args[i].equals("-sortBy")) {
192                         sortBy = args[++i];
193                     } else if (args[i].equals("-outputFormat")) {
194                         outputFormat = args[++i];
195                     } else if (args[i].equals("-delimiter")) {
196                         delimiter = args[++i];
197                     }
198                 }
199                 queryType = QueryType.SQL;
200             }else if (args[i].equals("--url")) {
201                 fmUrlStr = args[++i];
202             }
203         }
204 
205         if (queryStr == null || fmUrlStr == null) {
206             exit("Must specify a query and filemgr url! \n" + usage);
207         }
208         
209         if (queryType == QueryType.LUCENE) {
210             URL fmUrl = new URL(fmUrlStr);
211             QueryTool queryTool = new QueryTool(fmUrl);
212             org.apache.oodt.cas.filemgr.structs.Query casQuery = new org.apache.oodt.cas.filemgr.structs.Query();
213             queryTool.generateCASQuery(casQuery, parseQuery(queryStr));
214     
215             List prodIds = queryTool.query(casQuery);
216             if (prodIds != null && prodIds.size() > 0) {
217                 for (Object prodId1 : prodIds) {
218                     String prodId = (String) prodId1;
219                     System.out.println(prodId);
220                 }
221             }
222         }else {
223             System.out.println(performSqlQuery(queryStr, sortBy, outputFormat, delimiter != null ? delimiter : "\n", fmUrlStr));
224         }
225 
226     }

229     private List safeGetProductTypes() {
230         List prodTypes = null;
231 
232         try {
233             prodTypes = client.getProductTypes();
234         } catch (RepositoryManagerException e) {
235             LOG.log(Level.WARNING,
236                     "Error obtaining product types from file manager: ["
237                             + client.getFileManagerUrl() + "]: Message: "
238                             + e.getMessage());
239         }
240 
241         return prodTypes;
242     }

244     private static String performSqlQuery(String query, String sortBy, String outputFormat, String delimiter, String filemgrUrl) 
245             throws MalformedURLException, CatalogException, ConnectionException, QueryFormulationException {
246         ComplexQuery complexQuery = SqlParser.parseSqlQuery(query);
247         complexQuery.setSortByMetKey(sortBy);
248         complexQuery.setToStringResultFormat(outputFormat);
249         List<QueryResult> results = new XmlRpcFileManagerClient(new URL(filemgrUrl)).complexQuery(complexQuery);
250         StringBuilder returnString = new StringBuilder("");
251         for (QueryResult qr : results) {
252             returnString.append(qr.toString()).append(delimiter);
253         }
254         return returnString.substring(0, returnString.length() - delimiter.length());
255     }
