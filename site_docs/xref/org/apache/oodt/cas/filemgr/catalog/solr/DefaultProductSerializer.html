46  /**
47   * Default implementation of {@link ProductSerializer} 
48   * that transforms a CAS product into a single Solr document based on the following rules: 
49   * o) the core product attributes are used to generate Solr fields starting with "CAS...."
50   * o) the product references are converted to Solr fields starting with "CAS.Reference..." or "CAS.RootReference..."
51   * o) all other metadata fields are converted into Solr fields with the same name and number of values. 
52   *    Note that the field multiplicity must be consistent with its definition in the Solr schema.xml.
53   *    
54   * This class generates all Solr documents in XML format.
55   * 
56   * @author Luca Cinquini
57   *
58   */
59  public class DefaultProductSerializer implements ProductSerializer {
60

70  	/**
71  	 * {@inheritDoc}
72  	 */
73  	@Override
74  	public List<String> serialize(Product product, boolean create) {
75  		
76  		Map<String, List<String>> fields = new ConcurrentHashMap<String, List<String>>();
77  		List<String> docs = new ArrayList<String>();
78  		
79  		// add core product attributes to map
80  		this.addKeyValueToMap(fields, Parameters.PRODUCT_ID, product.getProductId());
81  		this.addKeyValueToMap(fields, Parameters.PRODUCT_NAME, product.getProductName());
82  		this.addKeyValueToMap(fields, Parameters.PRODUCT_STRUCTURE, product.getProductStructure());
83  		this.addKeyValueToMap(fields, Parameters.PRODUCT_TRANSFER_STATUS, product.getTransferStatus());
84  		ProductType productType = product.getProductType();
85  		if (productType!=null) {
86  			this.addKeyValueToMap(fields, Parameters.PRODUCT_TYPE_NAME, productType.getName());
87  			this.addKeyValueToMap(fields, Parameters.PRODUCT_TYPE_ID, productType.getProductTypeId());				
88  		}
89  		if (create) {
90  			// only insert date/time when product is first created
91  			Date productDateTime = new Date(); // current datetime
92  			this.addKeyValueToMap(fields, Parameters.PRODUCT_RECEIVED_TIME, Parameters.SOLR_DATE_TIME_FORMATTER.format(productDateTime));
93  		}
94  		
95  		// create new product: use Solr id == CAS id
96  		if (create) {			
97  			docs.add( this.generateInsertDocuments(product.getProductId(), fields) );
98  			
99  		// update existing product
100 		} else {			
101 			docs.addAll( this.generateUpdateDocuments(product.getProductId(), fields, true) ); // replace=true
102 		
103 		}
104 		
105 		return docs;
106 		
107 	}

112 	public List<String> serialize(String productId, Reference rootReference, List<Reference> references, boolean replace) {
113 		
114 		Map<String, List<String>> fields = new ConcurrentHashMap<String, List<String>>();
115 		
116 		// product root reference
117 		if (rootReference!=null) {
118 			
119 			addKeyValueToMap(fields, Parameters.ROOT_REFERENCE_ORIGINAL, rootReference.getOrigReference());
120 			addKeyValueToMap(fields, Parameters.ROOT_REFERENCE_DATASTORE, rootReference.getDataStoreReference());
121 			addKeyValueToMap(fields, Parameters.ROOT_REFERENCE_FILESIZE, ""+rootReference.getFileSize());
122 			addKeyValueToMap(fields, Parameters.ROOT_REFERENCE_MIMETYPE, rootReference.getMimeType().toString());
123 			
124 		}
125 		
126 		// all other product references
127 	  // note that Solr will preserve the indexing order.
128 		for (Reference reference : references) {
129 			
130 			addKeyValueToMap(fields, Parameters.REFERENCE_ORIGINAL, reference.getOrigReference());
131 			addKeyValueToMap(fields, Parameters.REFERENCE_DATASTORE, reference.getDataStoreReference());
132 			addKeyValueToMap(fields, Parameters.REFERENCE_FILESIZE, ""+reference.getFileSize());
133 			addKeyValueToMap(fields, Parameters.REFERENCE_MIMETYPE, reference.getMimeType().toString());
134 			
135 		}
136 		
137 		return generateUpdateDocuments(productId, fields, replace);
138 		
139 	}

144 	public QueryResponse deserialize(String xml) throws CatalogException {
145 		
146 		try {
147 			
148 			QueryResponse queryResponse = new QueryResponse();
149 			
150 			// parse XML into DOM
151 			Document document = parseXml(xml);
152 			
153 			// extract information from DOM to Product
154 			Element response = document.getDocumentElement();
155 			Node result = response.getElementsByTagName("result").item(0);
156 			queryResponse.setNumFound( Integer.parseInt( ((Element)result).getAttribute("numFound") ) );
157 			queryResponse.setStart( Integer.parseInt( ((Element)result).getAttribute("start") ) );			
158 			NodeList docs = result.getChildNodes();
159 			for (int i=0; i< docs.getLength(); i++) {
160 				Node node = docs.item(i);
161 				if (node.getNodeName().equals("doc")) {
162 					Element doc = (Element)node;					
163 					CompleteProduct cp = this.deserialize(doc);
164 					queryResponse.getCompleteProducts().add(cp);
165 				}
166 			}		
167 			return queryResponse;
168 		
169 		} catch(Exception e) {
170 			LOG.log(Level.SEVERE, e.getMessage());
171 			throw new CatalogException(e.getMessage(), e);
172 		}
173 		
174 	}

178 	 */
179 	public List<String> serialize(String productId, Metadata metadata, boolean replace) {
180 		
181 		Map<String, List<String>> fields = new ConcurrentHashMap<String, List<String>>();
182 		
183 		for (String key : metadata.getKeys()) {
184 			if (! (key.startsWith(Parameters.NS)              // skip metadata keys starting with reserved namespace
185 				     || Parameters.PRODUCT_TYPE_NAME.contains(key)
186 				   // skip 'ProductType' as already stored as 'CAS.ProductTypeName'
187 				     || Parameters.PRODUCT_STRUCTURE.contains(key))) { // skip 'ProductType' as already stored as 'CAS.ProductStructure'
188 				for (String value : metadata.getAllMetadata(key)) {
189 					this.addKeyValueToMap(fields, key, value);
190 				}
191 			}
192 		}
193 		
194 		return this.generateUpdateDocuments(productId, fields, replace);
195 		
196 	}

199 	 * Method to add a (key, value) to a multi-valued map with appropriate checks.
200 	 * @param map
201 	 * @param key
202 	 * @param value
203 	 */
204 	protected void addKeyValueToMap(Map<String, List<String>> map, String key, String value) {
205 		
206 		if (!map.containsKey(key)) {
207 			map.put(key, new ArrayList<String>());
208 		}
209 		if (value!=null) {
210 			map.get(key).add(value);
211 		} else {
212 			// use special value to trigger key removal
213 			map.get(key).add(Parameters.NULL);
214 		}
215 	}

217 	/**
218 	 * Utility method to generate a Solr insert document.
219 	 * 
220 	 * @param productId
221 	 * @param fields
222 	 * @return
223 	 */
224 	protected String generateInsertDocuments(String productId, Map<String,List<String>> fields) {
225 		
226 		StringBuilder doc = new StringBuilder();
227 		doc.append("<doc>");
228 		
229 		// product Solr id field
230 		doc.append( encodeIndexField(Parameters.ID, productId) );
231 		
232 		// all other fields
233 		for (Map.Entry<String, List<String>> key : fields.entrySet()) {
234 			for (String value : key.getValue()) {
235 				doc.append( encodeIndexField(key.getKey(), value) );
236 			}
237 		}
238 
239 		doc.append("</doc>");
240 		return doc.toString();
241 	}

243 	/**
244 	 * Utility method to generate Solr update documents.
245 	 * Note that the requests for setting/adding/deleting fields must be sent as separate documents to Solr
246 	 * @param productId
247 	 * @param fields
248 	 * @param replace
249 	 * @return
250 	 */
251 	protected List<String> generateUpdateDocuments(String productId, Map<String,List<String>> fields, boolean replace) {
252 		
253 		// list for different instruction types
254 		List<String> setFields = new ArrayList<String>();
255 		List<String> addFields = new ArrayList<String>();
256 		List<String> delFields = new ArrayList<String>();
257 		
258 		// encode update instructions
259 		for (Map.Entry<String, List<String>> key : fields.entrySet()) {
260 			
261 			List<String> values = key.getValue();
262 			
263 			if (replace) {
264 				
265 				if (values.isEmpty()) {
266 					// use special value to flag removal
267 					delFields.add( this.encodeUpdateField(key.getKey(), Parameters.NULL, true) );
268 					
269 				} else {
270 					for (String value : values) {
271 						setFields.add( this.encodeUpdateField(key.getKey(), value, true) );
272 					}
273 				}
274 				
275 			} else {
276 				for (String value : values) {
277 					addFields.add( this.encodeUpdateField(key.getKey(), value, false) );
278 				}
279 			}
280 			
281 		}
282 		
283 		List<String> docs = new ArrayList<String>();
284 		if (!delFields.isEmpty()) {
285 		  docs.add(toDoc(productId, delFields));
286 		}
287 		if (!setFields.isEmpty()) {
288 		  docs.add(toDoc(productId, setFields));
289 		}
290 		if (!addFields.isEmpty()) {
291 		  docs.add(toDoc(productId, addFields));
292 		}
293 		return docs;
294 		
295 	}

298 	 * Utility method to merge field update instructions into a single document.
299 	 * @param productId
300 	 * @param updates
301 	 * @return
302 	 */
303 	private String toDoc(String productId, List<String> updates) {
304 		
305 		StringBuilder doc = new StringBuilder();
306 		doc.append("<doc>");
307 		
308 		// reference product record id
309 		doc.append( encodeIndexField(Parameters.ID, productId) );
310 		
311 		// loop over field update instructions
312 		for (String update : updates) {
313 			doc.append(update);
314 		}
315 		
316 		doc.append("</doc>");
317 		
318 		return doc.toString();
319 		
320 	}

323 	 * Method to encode a Solr field indexing instruction.
324 	 * If the value is null, the empty string is returned.
325 	 * @param key
326 	 * @param value
327 	 * @return
328 	 */
329 	protected String encodeIndexField(String key, String value) {
330 		if (value==null || value.equals(Parameters.NULL)) {
331 			return "";
332 		} else {
333 			return "<field name=\""+key+"\">" + value + "</field>";
334 		}
335 	}

337 	/**
338 	 * Method to encode a field update instruction for the three possible cases:
339 	 * add new values to a key (1), replace current values for a key (2), remove all values for a key (3).
340 	 * 
341 	 * @param key
342 	 * @param value
343 	 * @param replace
344 	 * @return
345 	 */
346 	protected String encodeUpdateField(String key, String value, boolean replace) {
347 		StringBuilder sb = new StringBuilder();
348 		sb.append("<field name=\"").append(key).append("\"");
349 		
350 		if (replace) {
351 			
352 			if (value==null || value.equals(Parameters.NULL)) {
353 				
354 				// (3) remove all values for given key
355 				sb.append(" update=\"set\" null=\"true\" />");
356 				
357 			} else {
358 				
359 				// (2) replace existing values with new values
360 				sb.append(" update=\"set\">").append(value).append("</field>");
361 			}
362 			
363 		} else {
364 			
365 			// (1) add new values to existing values
366 			sb.append(" update=\"add\">").append(value).append("</field>");
367 			
368 		}
369 
370 		return sb.toString();
371 	}

375 	/**
376 	 * Method that parses a single Solr document snippet
377 	 * to extract Product and Metadata attributes.
378 	 * 
379 	 * @param doc
380 	 * @return
381 	 */
382 	private CompleteProduct deserialize(Element doc) {
383 		
384 		CompleteProduct cp = new CompleteProduct();
385 		Product product = cp.getProduct();
386 		ProductType productType = product.getProductType();
387 		Metadata metadata = cp.getMetadata();
388 		List<Reference> references = product.getProductReferences();
389 		Reference rootReference = product.getRootRef();
390 		
391 		NodeList children = doc.getChildNodes();
392 		for (int j=0; j<children.getLength(); j++) {
393 			
394 			Node child = children.item(j);
395 			Element element = (Element)child;
396 			String name = element.getAttribute("name");
397 			
398 			/**
399 			 *<arr name="ScanPointingSource">
400 			 *	<str>G073.65+0.19</str>
401 			 *	<str>J2015+3410</str>
402 	     *  ..........
403 			 */
404 			if (child.getNodeName().equals("arr")) {
405 				
406 				NodeList values = element.getChildNodes();
407 				List<String> vals = new ArrayList<String>();
408 				for (int k=0; k<values.getLength(); k++) {
409 					String value = ((Element)values.item(k)).getTextContent();
410 					vals.add(value);
411 				}
412 				// CAS.reference.... fields
413 				if (name.startsWith(Parameters.NS)) {					
414 						for (int k=0; k<values.getLength(); k++) {
415 							// create this reference
416 							if (references.size()<=k) {
417 							  references.add(new Reference());
418 							}
419 							if (name.equals(Parameters.REFERENCE_ORIGINAL)) {
420 								references.get(k).setOrigReference(vals.get(k));
421 							} else if (name.equals(Parameters.REFERENCE_DATASTORE)) {
422 								references.get(k).setDataStoreReference(vals.get(k));
423 							} else if (name.equals(Parameters.REFERENCE_FILESIZE)) {
424 								references.get(k).setFileSize(Long.parseLong(vals.get(k)));
425 							} else if (name.equals(Parameters.REFERENCE_MIMETYPE)) {
426 								references.get(k).setMimeType(vals.get(k));
427 							}
428 						}
429 				// all other multi-valued fields
430 				} else {
431 					this.deserializeMultiValueField(name, vals, metadata);
432 				}
433 				
434 			/**
435 			 * 	<str name="id">6684d79d-a011-4bc0-b3b3-4f11817091c8</str>
436 			 *  <str name="CAS.ProductId">6684d79d-a011-4bc0-b3b3-4f11817091c8</str>
437 			 *  <str name="CAS.ProductName">tns_br145x4_20</str>
438 			 *  <str name="FileLocation">/usr/local/ska-dc/data/archive</str>
439        *  ...........
440 			 */
441 			} else {
442 				
443 				String value = element.getTextContent();
444 				
445 				// core CAS fields
446 				if (name.startsWith(Parameters.NS)) {
447 					if (name.equals(Parameters.PRODUCT_ID)) {
448 						product.setProductId(value);
449 					} else if (name.equals(Parameters.PRODUCT_NAME)) {
450 						product.setProductName(value);
451 					} else if (name.equals(Parameters.PRODUCT_STRUCTURE)) {
452 						product.setProductStructure(value);
453 					} else if (name.equals(Parameters.PRODUCT_TRANSFER_STATUS)) {
454 						product.setTransferStatus(value);
455 					} else if (name.equals(Parameters.PRODUCT_TYPE_NAME)) {
456 						productType.setName(value);
457 					} else if (name.equals(Parameters.PRODUCT_TYPE_ID)) {
458 						productType.setProductTypeId(value);
459 					} else if (name.equals(Parameters.PRODUCT_RECEIVED_TIME)) {
460 						product.setProductRecievedTime(value);
461 						metadata.addMetadata(name, value);
462 						// CAS root reference
463 					} else if (name.startsWith(Parameters.NS+Parameters.ROOT)) {
464 						if (rootReference==null) {
465 						  rootReference = new Reference();
466 						}
467 						if (name.equals(Parameters.ROOT_REFERENCE_ORIGINAL)) {
468 							rootReference.setOrigReference(value);
469 						} else if (name.equals(Parameters.ROOT_REFERENCE_DATASTORE)) {
470 							rootReference.setDataStoreReference(value);
471 						} else if (name.equals(Parameters.ROOT_REFERENCE_FILESIZE)) {
472 							rootReference.setFileSize(Long.parseLong(value));
473 						} else if (name.equals(Parameters.ROOT_REFERENCE_MIMETYPE)) {
474 							rootReference.setMimeType(value);
475 						}
476 
477 					}
478 					
479 				// non core single-valued fields
480 				} else {
481 					this.deserializeSingleValueField(name, value, metadata);
482 				} // "CAS".... or not
483 				
484 			} // "arr" or anything else
485 			
486 		} // loop over <doc> children
487 		
488 		return cp;
489 		
490 	}

493 		
494 		DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
495 		DocumentBuilder parser = factory.newDocumentBuilder();
496 	  return parser.parse( new InputSource(new StringReader(xml)) );

505 	 * @param value : the Solr field single value

509 	  	// ignore Solr internal identifier (as it is duplicate information of CAS.ProductId)

511 			metadata.addMetadata(name, value);
