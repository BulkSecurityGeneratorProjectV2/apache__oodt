50  /**
51   * A {@link Provider} that writes {@link DatasetResource dataset resources} to
52   * output streams for HTTP responses with "application/rss+xml" content-type.
53   * @author rlaidlaw
54   * @version $Revision$
55   */
56  @Provider
57  @Produces("application/rss+xml")
58  public class DatasetRssWriter extends RssWriter
59    implements MessageBodyWriter<DatasetResource>
60  {
61    private static final Logger LOGGER = Logger.getLogger(DatasetRssWriter.class
62      .getName());
63  
64  
65  
66    @Override
67    public long getSize(DatasetResource resource, Class<?> type,
68      Type genericType, Annotation[] annotations, MediaType mediaType)
69    {
70      return -1;
71    }
72  
73  
74  
75    @Override
76    public boolean isWriteable(Class<?> type, Type genericType,
77      Annotation[] annotations, MediaType mediaType)
78    {
79      return true;
80    }
81  
82  
83  
84    @Override
85    public void writeTo(DatasetResource resource, Class<?> type, Type genericType,
86      Annotation[] annotations, MediaType mediaType,
87      MultivaluedMap<String, Object> httpHeaders, OutputStream entityStream)
88        throws IOException, WebApplicationException
89    {
90      String productTypeName = resource.getName();
91      String productTypeId = resource.getId();
92      String base = getBaseUri() + "dataset";
93      String query = "?productTypeId=" + productTypeId;
94      String idLink = base + query;
95      String selfLink = base + ".rss" + query;
96      String currentDate = dateFormatter.format(new Date());
97  
98      DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
99      factory.setNamespaceAware(true);
100 
101     RssConfiguration configuration = getConfiguration();
102 
103     try
104     {
105       Document doc = factory.newDocumentBuilder().newDocument();
106       Element rss = XMLUtils.addNode(doc, doc, "rss");
107       XMLUtils.addAttribute(doc, rss, "version", "2.0");
108       XMLUtils.addAttribute(doc, rss, "xmlns:atom",
109         "http://www.w3.org/2005/Atom");
110 
111       if (configuration != null)
112       {
113         // Add namespaces defined in the configuration file.
114         for (RssNamespace namespace : configuration.getNamespaceList())
115         {
116           XMLUtils.addAttribute(doc, rss, "xmlns:" + namespace.getPrefix(),
117             namespace.getUriString());
118         }
119 
120         // Get channel link information, if specified in the configuration.
121         String channelLink = configuration.getChannelLink();
122         if (channelLink != null && !channelLink.equals(""))
123         {
124           Metadata channelMetadata = new Metadata();
125           channelMetadata.addMetadata("ProductType", productTypeName);
126           channelMetadata.addMetadata("ProductTypeId", productTypeId);
127           channelMetadata.addMetadata("BaseUrl", base);
128 
129           idLink = PathUtils.replaceEnvVariables(channelLink, channelMetadata);
130         }
131       }
132 
133       Element channel = XMLUtils.addNode(doc, rss, "channel");
134       XMLUtils.addNode(doc, channel, "title", productTypeName);
135 
136       Element atomLink = XMLUtils.addNode(doc, channel, "atom:link");
137       XMLUtils.addAttribute(doc, atomLink, "href", selfLink);
138       XMLUtils.addAttribute(doc, atomLink, "rel", "self");
139       XMLUtils.addAttribute(doc, atomLink, "type", "application/rss+xml");
140 
141       XMLUtils.addNode(doc, channel, "link", idLink);
142       XMLUtils.addNode(doc, channel, "description", productTypeName);
143 
144       XMLUtils.addNode(doc, channel, "language", LANGUAGE);
145       XMLUtils.addNode(doc, channel, "copyright", COPYRIGHT);
146       XMLUtils.addNode(doc, channel, "pubDate", currentDate);
147       XMLUtils.addNode(doc, channel, "category", productTypeName);
148       XMLUtils.addNode(doc, channel, "generator", GENERATOR);
149       XMLUtils.addNode(doc, channel, "lastBuildDate", currentDate);
150 
151       for (ProductResource productResource : resource.getProductResources())
152       {
153         String productLink = getBaseUri() + "product?productId="
154           + productResource.getProductId();
155         Element item = XMLUtils.addNode(doc, channel, "item");
156         XMLUtils.addNode(doc, item, "title", productResource.getProductName());
157         XMLUtils.addNode(doc, item, "link", productLink);
158         XMLUtils.addNode(doc, item, "description", productResource
159           .getProductTypeName());
160         XMLUtils.addNode(doc, item, "guid", productLink);
161 
162         Metadata metadata = productResource.getMetadataResource().getMetadata();
163         Date productReceivedTime = DateConvert.isoParse(metadata
164           .getMetadata("CAS.ProductReceivedTime"));
165         if (productReceivedTime != null)
166         {
167           XMLUtils.addNode(doc, item, "pubDate", dateFormatter
168             .format(productReceivedTime));
169         }
170 
171         // Append additional tags defined in the configuration file.
172         if (configuration != null)
173         {
174           configuration.appendTags(metadata, doc, item);
175         }
176       }
177 
178       XMLUtils.writeXmlToStream(doc, entityStream);
179     }
180     catch (ParserConfigurationException e)
181     {
182       String message = "Unable to create RSS XML document for RSS response.";
183       LOGGER.log(Level.WARNING, message, e);
184       throw new InternalServerErrorException(message);
185     }
186     catch (ParseException e)
187     {
188       String message = "Unable to create RSS XML document for RSS response.";
189       LOGGER.log(Level.WARNING, message, e);
190       throw new InternalServerErrorException(message);
191     }
192   }
