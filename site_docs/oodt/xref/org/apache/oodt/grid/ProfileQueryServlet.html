40  /**
41   * Profile query servlet handles profile queries.  It returns every matching profile from
42   * every query handler that provides matching profiles.  If no handlers are installed,
43   * then it returns an empty &lt;profiles&gt; document.
44   * 
45   */
46  public class ProfileQueryServlet extends QueryServlet {
47  	/** {@inheritDoc} */
48  	protected List getServers(Configuration config) {
49  		return config.getProfileServers();
50  	}
51  
52  	/** {@inheritDoc} */
53  	protected void handleQuery(XMLQuery query, List handlers, HttpServletRequest req, HttpServletResponse res)
54  		throws IOException {
55  		// Find if the query should be targeted to specific handlers.
56  		Set ids = new HashSet();
57  		if (query.getFromElementSet() != null) {
58  		  for (Object o : query.getFromElementSet()) {
59  			QueryElement qe = (QueryElement) o;
60  			if ("handler".equals(qe.getRole()) && qe.getValue() != null) {
61  			  ids.add(qe.getValue());
62  			}
63  		  }
64  		}
65  		
66  		res.setContentType("text/xml");					       // XML, comin' at ya
67  		res.getWriter().println("<?xml version='1.0' encoding='UTF-8'?>");     // UTF-8 no less.  Boo-ya.
68  		res.getWriter().println("<!DOCTYPE profiles PUBLIC '"		       // Get a doctype in there for the highly ...
69  			+ Profile.PROFILES_DTD_FPI + "' '"			       // ... anal ...
70  			+ Profile.PROFILES_DTD_URL + "'>");		               // ... retentive.
71  		res.getWriter().println("<profiles>");				       // Start tag for the whole /sbin/fsck mess
72  		Transformer transformer = null;					       // Start out w/no transformer and no doc
73  		Document doc = null;						       // Don't make 'em if we don't need 'em
74  		boolean sentAtLeastOne = false;					       // Track if we send any profiles at all
75  		Exception exception = null;					       // And save any exception
76  	  for (Object handler1 : handlers) {
77  		try {           // To iterate over each handler
78  		  ProfileHandler handler = (ProfileHandler) handler1;            // Get the handler
79  		  String id = handler.getID();                                   // Get the ID, and if targeting to IDs
80  		  if (!ids.isEmpty() && !ids.contains(id)) {
81  			continue;             // ... and it's not one we want, skip it.
82  		  }
83  		  List results = handler.findProfiles(query);                    // Have it find profiles
84  		  if (results == null) {
85  			results = Collections.EMPTY_LIST;         // Assume nothing
86  		  }
87  		  for (Object result1 : results) {          // For each matching profile
88  			Profile profile = (Profile) result1;                   // Get the profile
89  			if (transformer == null) {                       // No transformer/doc yet?
90  			  transformer = createTransformer();             // Then make the transformer
91  			  doc = Profile.createProfileDocument();         // And the doc
92  			}                                   // And use the doc ...
93  			Node profileNode = profile.toXML(doc);                   // To render the profile into XML
94  			DOMSource source = new DOMSource(profileNode);         // And the XML becomes is source
95  			StreamResult result = new StreamResult(res.getWriter()); // And the response is a result
96  			transformer.transform(source, result);                   // And serialize into glorious text
97  			sentAtLeastOne = true;                       // OK, we got at least one out the doo
98  		  }
99  		} catch (Exception ex) {                           // Uh oh
100 		  exception = ex;                               // OK, just hold onto it for now
101 		}
102 	  }
103 		if (!sentAtLeastOne && exception != null) {			       // Got none out the door and had an error?
104 			res.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR,    // Then we can report it.
105 				exception.getMessage());
106 			return;
107 		}
108 
109 		// However, if we get here, we got at least one profile out the door. In
110 		// that case, hide any error that might've occurred from subsequent
111 		// handlers.  (Or, if we get here, there were no errors.  Yay.)
112 		res.getWriter().println("</profiles>");				       // End tag for the whole fsck'n mess
113 	}

115 	/**
116 	 * Create a transformer, properly configured for XML text serialization.
117 	 *
118 	 * @return a <code>Transformer</code> value.
119 	 * @throws TransformerException if an error occurs.
120 	 */
121 	private static Transformer createTransformer() throws TransformerException {
122 		Transformer transformer;
123 		synchronized (TRANSFORMER_FACTORY) {
124 			transformer = TRANSFORMER_FACTORY.newTransformer();
125 		}
126 		transformer.setOutputProperty(OutputKeys.METHOD, "xml");
127 		transformer.setOutputProperty(OutputKeys.VERSION, "1.0");
128 		transformer.setOutputProperty(OutputKeys.ENCODING, "UTF-8");
129 		transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
130 		transformer.setOutputProperty(OutputKeys.STANDALONE, "no");
131 		transformer.setOutputProperty(OutputKeys.INDENT, "yes");
132 		transformer.setOutputProperty(OutputKeys.MEDIA_TYPE, "text/xml");
133 		transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "4");
134 
135 		return transformer;
136 	}
137 
138 	/** Sole transformer factory this class will ever need. */
139 	private static final TransformerFactory TRANSFORMER_FACTORY = TransformerFactory.newInstance();
140 }
