17  package org.apache.oodt.cas.catalog.page;
18  
19  //JDK imports
20  import java.util.Collections;
21  import java.util.List;
22  import java.util.Vector;
23  
24  /**
25   * @author bfoster
26   * @version $Revision$
27   *
28   * <p>
29   * A pager for paging through query results
30   * <p>
31   */
32  public class QueryPager extends IndexPager {
33  
34  	protected List<TransactionReceipt> receipts;
35  	
36  	public QueryPager(List<TransactionReceipt> receipts) {
37  		super();
38  		this.receipts = new Vector<TransactionReceipt>(receipts);
39  		this.totalPages = this.caculateTotalPages();
40  		this.numOfHits = receipts.size();
41  	}
42  
43  	protected int caculateTotalPages() {
44  		return (int) Math.ceil((double) receipts.size() / (double) this.getPageSize());
45  	}
46  	
47  	public void setPageInfo(PageInfo pageInfo) {
48  		this.pageSize = Math.max(pageInfo.getPageSize(), 0);
49  		this.totalPages = this.caculateTotalPages();
50  		if (this.totalPages == 0) {
51  		  this.pageNum = 0;
52  		} else {
53  		  this.pageNum = (pageInfo.getPageNum() == PageInfo.LAST_PAGE || pageInfo.getPageNum() >= this.totalPages)
54  						 ? this.totalPages : pageInfo.getPageNum();
55  		}
56  	}
57  		
58  	public List<TransactionReceipt> getTransactionReceipts() {
59  		return Collections.unmodifiableList(this.receipts);
60  	}
61  	
62  	public List<TransactionReceipt> getCurrentPage() {
63  		List<TransactionReceipt> currentPage = new Vector<TransactionReceipt>();
64  		if (this.pageNum > 0) {
65  		  for (int i = (this.getPageNum() - 1) * this.getPageSize();
66  			   i < receipts.size() && i < this.getPageNum() * this.getPageSize(); i++) {
67  			currentPage.add(receipts.get(i));
68  		  }
69  		}
70  		return currentPage;
71  	}
72  	
73  }
