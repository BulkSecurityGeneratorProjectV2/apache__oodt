19  package org.apache.oodt.cas.pushpull.daemon;
20  
21  //JDK imports
22  
23  import java.rmi.Naming;
24  import java.rmi.RemoteException;
25  import java.util.logging.Level;
26  import java.util.logging.Logger;
27  
28  /**
29   * @author bfoster
30   * 
31   * <p>
32   * After the CrawlDaemon has been started, this class can control the
33   * CrawlDaemon through webserver communication, thus it can be control from
34   * computers other than the one it is executing on. This class has control over
35   * pausing, resuming, and killing the CrawlDaemon process. It also has the
36   * ability of getting the status on several variables within the CrawlDaemon
37   * class. See CrawlDaemon for more info.
38   * </p>.
39   */
40  public class DaemonController {
41      private static Logger LOG = Logger.getLogger(DaemonController.class.getName());
42      private DaemonRmiInterface daemon;
43  
44      /**
45       * Constructor -- initializes the XmlRpcClient
46       * 
47       * @throws InstantiationException
48       */
49      public DaemonController(String rmiUrl) {
50          try {
51              daemon = (DaemonRmiInterface) Naming.lookup(rmiUrl);
52              System.out.println(!daemon.getHasBeenToldToQuit());
53          } catch (Exception e) {
54              LOG.log(Level.SEVERE, e.getMessage());
55          }
56      }
57  
58      /**
59       * Get the average time for each run of the Crawler controlled by the
60       * CrawlDaemons
61       * 
62       * @return Average runtime of Crawler
63       * @throws RemoteException
64       * @throws XmlRpcCommunicationException
65       *             Error communicating with server
66       */
67      public long getAverageRunTime() throws RemoteException {
68          return daemon.getAverageRunTime();
69      }
70  
71      /**
72       * Gets the total milliseconds for which the Crawler in CrawlDaemon has been
73       * crawling since the CrawlDaemon was created.
74       * 
75       * @return The total milliseconds which the Crawler has been crawling
76       * @throws RemoteException
77       * @throws XmlRpcCommunicationException
78       *             Error communicating with server
79       */
80      public long getMillisCrawling() throws RemoteException {
81          return daemon.getMillisCrawling();
82      }
83  
84      /**
85       * Gets the time between the start of each scheduled run.
86       * 
87       * @return The time between the start of each execution of the Crawler
88       *         controlled by the CrawlDaemon
89       * @throws RemoteException
90       * @throws XmlRpcCommunicationException
91       *             Error communicating with server
92       */
93      public long getWaitInterval() throws RemoteException {
94          return daemon.getTimeInterval();
95      }
96  
97      /**
98       * Gets the numbers of times the Crawler has been run by the CrawlDaemon
99       * 
100      * @return The of times Crawler has executed
101      * @throws RemoteException
102      * @throws XmlRpcCommunicationException
103      *             Error communicating with server
104      */
105     public int getNumCrawls() throws RemoteException {
106         return daemon.getNumCrawls();
107     }
108 
109     /**
110      * Used to determine if the Crawler in the CrawlDaemon is running
111      * 
112      * @return true is Crawler is running
113      * @throws RemoteException
114      * @throws XmlRpcCommunicationException
115      *             Error communicating with server
116      */
117     public boolean isRunning() throws RemoteException {
118         return daemon.isRunning();
119     }
120 
121     /**
122      * Wakes the CrawlDaemon up and tells it to start crawling again. If stop()
123      * was called, this method will have the CrawlDaemon continue from where it
124      * left off. This method will also wake up the CrawlDaemon from its sleep
125      * between scheduled runs.
126      * 
127      * @throws RemoteException
128      * @throws XmlRpcCommunicationException
129      *             Error communicating with server
130      */
131     public void resume() throws RemoteException {
132         daemon.resume();
133     }
134 
135     public void quit() throws RemoteException {
136         daemon.quit();
137     }
138 
139     /**
140      * Gets the time in milliseconds of when the CrawlDaemon was created
141      * 
142      * @return The time the CrawlDaemon was created
143      * @throws RemoteException
144      * @throws XmlRpcCommunicationException
145      *             Error communicating with server
146      */
147     public long getDaemonCreationTime() throws RemoteException {
148         return daemon.getDaemonCreationTime();
149     }
150 
151     /**
152      * Driver method -- starts the CrawlDaemonController
153      * 
154      * @param args
155      *            See documentation
156      * @throws Exception
157      *             On error! :)
158      */
159     public static void main(String[] args) throws RemoteException {
160         String avgCrawlOperation = "--getAverageRunTime\n";
161         String getMilisCrawlOperation = "--getMilisCrawling\n";
162         String getNumCrawlsOperation = "--getNumCrawls\n";
163         String getWaitIntervalOperation = "--getWaitInterval\n";
164         String isRunningOperation = "--isRunning\n";
165         String stopOperation = "--stop\n";
166         String usage = "CrawlController --url <url to xml rpc service> "
167                 + "--operation [<operation> [params]]\n" + "operations:\n"
168                 + avgCrawlOperation + getMilisCrawlOperation
169                 + getNumCrawlsOperation + getWaitIntervalOperation
170                 + isRunningOperation + stopOperation;
171         String operation = null, url = null;
172 
173         for (int i = 0; i < args.length; i++) {
174             if (args[i].equals("--operation")) {
175                 operation = args[++i];
176             } else if (args[i].equals("--url")) {
177                 url = args[++i];
178             }
179         }
180         if (operation == null) {
181             System.err.println(usage);
182             System.exit(1);
183         }
184 
185         // create the controller
186         DaemonController controller = new DaemonController(url);
187         if (operation.equals("--getAverageRunTime")) {
188             double avgCrawlTime = controller.getAverageRunTime();
189             System.out.println("Average Crawl Time: [" + avgCrawlTime + "]");
190         } else if (operation.equals("--getMilisCrawling")) {
191             long crawlTime = controller.getMillisCrawling();
192             System.out.println("Total Crawl Time: [" + crawlTime
193                     + "] miliseconds");
194         } else if (operation.equals("--getNumCrawls")) {
195             int numCrawls = controller.getNumCrawls();
196             System.out.println("Num Crawls: [" + numCrawls + "]");
197         } else if (operation.equals("--getWaitInterval")) {
198             long waitInterval = controller.getWaitInterval();
199             System.out.println("Wait Interval: [" + waitInterval + "]");
200         } else if (operation.equals("--isRunning")) {
201             boolean running = controller.isRunning();
202             System.out.println(running ? "Yes" : "No");
203         } else {
204             throw new IllegalArgumentException("Unknown Operation!");
205         }
206     }
