82  	 /** Get the singleton configuration.
83  	  *
84  	  * This method returns the singleton configuration object, or creates it if it
85  	  * doesn't yet exist.  To create it, it reads the configuration file specified by
86  	  * the system property <code>org.apache.oodt.commons.Configuration.url</code> or the file in the
87  	  * user's home directory called .edarc.xml if the system property isn't
88  	  * specified.  It parses the file and returns a <code>Configuration</code> object
89  	  * initialized with the data specified therein.
90  	  *
91  	  * @throws IOException If reading the configuration file fails.
92  	  * @throws SAXException If parsing the configuration file fails.
93  	  * @throws MalformedURLException If the URL specification is invalid.
94  	  * @return An initialized configuration object.
95  	  */
96  	 public static Configuration getConfiguration() throws IOException, SAXException {
97  		 // Got one?  Use it.
98  		 if (configuration != null) {
99  		   return configuration;
100 		 }
101 
102 		 URL url;
103 
104 		 // First preference: URL via the org.apache.oodt.commons.Configuration.url prop.
105 		 String urlString = System.getProperty("org.apache.oodt.commons.Configuration.url");
106 		 if (urlString != null) {
107 			 url = new URL(urlString);
108 		 } else {
109 			 File file = null;
110 
111 			 // Second preference: file via the org.apache.oodt.commons.Configuration.file prop.
112 			 String filename = System.getProperty("org.apache.oodt.commons.Configuration.file");
113 			 if (filename != null) {
114 				 file = new File(filename);
115 				 if (!file.exists()) {
116 				   throw new IOException("File " + file + " not found");
117 				 }
118 			 } else {
119 				 List candidates = new ArrayList();
120 
121 				 // Third preference: ~/.edarc.xml
122 				 File homedir = new File(System.getProperty("user.home", "/"));
123 				 File homedirfile = new File(homedir, DEFAULT_CONFIG_FILE);
124 				 candidates.add(homedirfile);
125 
126 				 // Fourth preference: ~/.oodtrc.xml
127 				 File alt = new File(homedir, ALT_CONFIG_FILE);
128 				 candidates.add(alt);
129 
130 				 // Fifth and sixth preferences: $EDA_HOME/conf/edarc.xml and $EDA_HOME/etc/edarc.xml
131 				 String edaHome = System.getProperty("eda.home");
132 				 if (edaHome != null) {
133 					 File edaHomeDir = new File(edaHome);
134 					 candidates.add(new File(new File(edaHomeDir, "conf"), "edarc.xml"));
135 					 candidates.add(new File(new File(edaHomeDir, "etc"), "edarc.xml"));
136 				 }
137 
138 				 // Seventh preference: JAVA_HOME/lib/edarc.xml
139 				 candidates.add(LIB_CONFIG_FILE);
140 
141 				 // Final preference: JAVA_HOME/../lib/edarc.xml (to get out of JRE)
142 				 candidates.add(ALT_LIB_CONFIG_FILE);
143 
144 				 // Now find one.
145 				 boolean found = false;
146 			   for (Object candidate : candidates) {
147 				 file = (File) candidate;
148 				 if (file.exists()) {
149 				   found = true;
150 				   break;
151 				 }
152 			   }
153 				 if (found && file == alt) {
154 				   System.err.println("WARNING: Using older config file " + alt + "; rename to "
155 									  + homedirfile + " as soon as possible.");
156 				 }
157 				 if (!found) {
158 					 return getEmptyConfiguration();
159 				 }
160 			 }
161 			 url = file.toURL();
162 		 }
163 
164 		 return getConfiguration(url);
165 
166 	 }

247 	Configuration(InputSource inputSource) throws IOException, SAXException {
248 		String systemID = inputSource.getSystemId();
249 		if (systemID == null) {
250 		  inputSource.setSystemId("file:/unknown");
251 		}
252 
253 		// Get the document
254 		DOMParser parser = XML.createDOMParser();
255 		parser.setEntityResolver(new ConfigurationEntityResolver());
256 		parser.setErrorHandler(new ErrorHandler() {
257 			public void error(SAXParseException ex) throws SAXException {
258 				throw ex;
259 			}
260 			public void warning(SAXParseException ex) {
261 				System.err.println("Warning: " + ex.getMessage());
262 			}
263 			public void fatalError(SAXParseException ex) throws SAXException {
264 				System.err.println("Fatal parse error: " + ex.getMessage());
265 				throw ex;
266 			}
267 		});
268 		parser.parse(inputSource);
269 		Document document = parser.getDocument();
270 		XML.removeComments(document);
271 		document.normalize();
272 		
273 		// See if this really is a <configuration> document.
274 		if (!document.getDocumentElement().getNodeName().equals("configuration")) {
275 		  throw new SAXException("Configuration " + inputSource.getSystemId() + " is not a <configuration> document");
276 		}
277 
278 		NodeList list = document.getDocumentElement().getChildNodes();
279 		for (int eachChild = 0; eachChild < list.getLength(); ++eachChild) {
280 			Node childNode = list.item(eachChild);
281 			if (childNode.getNodeName().equals("webServer")) {
282 				NodeList children = childNode.getChildNodes();
283 				for (int i = 0; i < children.getLength(); ++i) {
284 					Node node = children.item(i);
285 					if ("host".equals(node.getNodeName())) {
286 					  webHost = XML.unwrappedText(node);
287 					} else if ("port".equals(node.getNodeName())) {
288 					  webPort = XML.unwrappedText(node);
289 					} else if ("dir".equals(node.getNodeName())) {
290 					  webServerDocumentDirectory = new File(XML.unwrappedText(node));
291 					}
292 				}					
293 				properties.setProperty("org.apache.oodt.commons.Configuration.webServer.baseURL", getWebServerBaseURL());
294 				if (webServerDocumentDirectory == null) {
295 				  webServerDocumentDirectory = new File(System.getProperty("user.home", "/")
296 														+ "/dev/htdocs");
297 				}
298 			} else if (childNode.getNodeName().equals("nameServer")) {
299 				Element nameServerNode = (Element) childNode;
300 				String nameServerStateFrequencyString = nameServerNode.getAttribute("stateFrequency");
301 				if (nameServerStateFrequencyString == null || nameServerStateFrequencyString.length() == 0) {
302 				  nameServerStateFrequency = 0;
303 				} else {
304 				  try {
305 					nameServerStateFrequency = Integer.parseInt(nameServerStateFrequencyString);
306 				  } catch (NumberFormatException ex) {
307 					throw new SAXException("Illegal nun-numeric value \"" + nameServerStateFrequencyString
308 										   + "\" for stateFrequency attribute");
309 				  }
310 				}
311 				if (childNode.getFirstChild().getNodeName().equals("rir")) {
312 					nameServerUsingRIRProtocol = true;
313 					NodeList children = childNode.getFirstChild().getChildNodes();
314 					nameServerObjectKey = children.getLength() == 1? XML.unwrappedText(children.item(0)):null;
315 				} else {
316 					nameServerUsingRIRProtocol = false;
317 					nameServerVersion = null;
318 					nameServerPort = null;
319 					// Must be same as CORBAMgr.NS_OBJECT_KEY:
320 					nameServerObjectKey = "StandardNS/NameServer%2DPOA/_root";
321 					NodeList children = childNode.getFirstChild().getChildNodes();
322 					for (int i = 0; i < children.getLength(); ++i) {
323 						Node node = children.item(i);
324 						if (node.getNodeName().equals("version")) {
325 						  nameServerVersion = XML.unwrappedText(node);
326 						} else if (node.getNodeName().equals("host")) {
327 						  nameServerHost = XML.unwrappedText(node);
328 						} else if (node.getNodeName().equals("port")) {
329 						  nameServerPort = XML.unwrappedText(node);
330 						} else if (node.getNodeName().equals("objectKey")) {
331 						  nameServerObjectKey = XML.unwrappedText(node);
332 						}
333 					}
334 				}
335 			} else if (childNode.getNodeName().equals("xml")) {
336 				NodeList children = childNode.getChildNodes();
337 				for (int i = 0; i < children.getLength(); ++i) {
338 					Node xmlNode = children.item(i);
339 					if ("entityRef".equals(xmlNode.getNodeName())) {
340 						NodeList dirNodes = xmlNode.getChildNodes();
341 						StringBuilder refDirs = new StringBuilder(System.getProperty(ENTITY_DIRS_PROP, ""));
342 						for (int j = 0; j < dirNodes.getLength(); ++j) {
343 						  refDirs.append(',').append(XML.unwrappedText(dirNodes.item(j)));
344 						}
345 						if (refDirs.length() > 0) {
346 						  System.setProperty(ENTITY_DIRS_PROP, refDirs.charAt(0) == ',' ?
347 															   refDirs.substring(1) : refDirs.toString());
348 						}
349 					}
350 				}
351 			} else if ("serverMgr".equals(childNode.getNodeName())) {
352 				serverMgrPort = Integer.parseInt(XML.unwrappedText(childNode.getFirstChild()));
353 			} else if (childNode.getNodeName().equals("properties")) {
354 				loadProperties(childNode, properties);
355 			} else if (childNode.getNodeName().equals("programs")) {
356 				NodeList children = childNode.getChildNodes();
357 				for (int i = 0; i < children.getLength(); ++i) {
358 					// They're all of type execServer---for now.
359 					ExecServerConfig esc = new ExecServerConfig(children.item(i));
360 					esc.getProperties().setProperty("org.apache.oodt.commons.Configuration.url", inputSource.getSystemId());
361 					execServers.add(esc);
362 				}
363 			}
364 		}
365 
366 		initializeContext();
367 	}

369 	private void initializeContext() {
370 		contextEnvironment.put(Context.INITIAL_CONTEXT_FACTORY, "org.apache.oodt.commons.object.jndi.ObjectCtxFactory");
371 		String registryList = System.getProperty("org.apache.oodt.commons.rmiregistries", System.getProperty("rmiregistries"));
372 		if (registryList == null) {
373 			String host = System.getProperty("rmiregistry.host", "localhost");
374 			int port = Integer.getInteger("rmiregistry.port", Registry.REGISTRY_PORT);
375 			registryList = "rmi://" + host + ":" + port;
376 		}
377 		contextEnvironment.put("rmiregistries", registryList);
378 	}

391 	/** 
392 	 *
393 	 * @param document The document to which the XML structure will belong.
394 	 * @return The root node representing this configuration.
395 	 * @throws DOMException If an error occurs constructing the XML structure.
396 	 */
397 	public Node toXML(Document document) throws DOMException {
398 		// <configuration>
399 		Element configurationNode = document.createElement("configuration");
400 
401 		// <webServer>
402 		Element webServerNode = document.createElement("webServer");
403 		configurationNode.appendChild(webServerNode);
404 
405 		// <webServer>
406 		//   <host>...</host><port>...</port><dir>...</dir>
407 		XML.add(webServerNode, "host", webHost);
408 		XML.add(webServerNode, "port", webPort);
409 		XML.add(webServerNode, "dir", webServerDocumentDirectory.toString());
410 
411 		// <nameServer>
412 		Element nameServerNode = document.createElement("nameServer");
413 		nameServerNode.setAttribute("stateFrequency", String.valueOf(nameServerStateFrequency));
414 		configurationNode.appendChild(nameServerNode);
415 
416 		// <nameServer>
417 		//   <rir> or <iiop>
418 		if (nameServerUsingRIRProtocol) {
419 			Element rirNode = document.createElement("rir");
420 			nameServerNode.appendChild(rirNode);
421 			if (nameServerObjectKey != null) {
422 			  XML.add(rirNode, "objectKey", nameServerObjectKey);
423 			}
424 		} else {
425 			Element iiopNode = document.createElement("iiop");
426 			nameServerNode.appendChild(iiopNode);
427 			if (nameServerVersion != null) {
428 			  XML.add(iiopNode, "version", nameServerVersion);
429 			}
430 			XML.add(iiopNode, "host", nameServerHost);
431 			if (nameServerPort != null) {
432 			  XML.add(iiopNode, "port", nameServerPort);
433 			}
434 			if (nameServerObjectKey != null) {
435 			  XML.add(iiopNode, "objectKey", nameServerObjectKey);
436 			}
437 		}
438 
439 		// <xml><entityRef><dir>...
440 		if (!getEntityRefDirs().isEmpty()) {
441 			Element xmlNode = document.createElement("xml");
442 			configurationNode.appendChild(xmlNode);
443 			Element entityRefNode = document.createElement("entityRef");
444 			xmlNode.appendChild(entityRefNode);
445 			XML.add(entityRefNode, "dir", getEntityRefDirs());
446 		}
447 
448 		// <serverMgr><port>...
449 		if (getServerMgrPort() != 0) {
450 			Element serverMgrNode = document.createElement("serverMgr");
451 			configurationNode.appendChild(serverMgrNode);
452 			XML.add(serverMgrNode, "port", String.valueOf(getServerMgrPort()));
453 		}
454 
455 		// Global <properties>...</properties>
456 		if (properties.size() > 0) {
457 		  dumpProperties(properties, configurationNode);
458 		}
459 
460 		// <programs>...
461 		if (execServers.size() > 0) {
462 			Element programsNode = document.createElement("programs");
463 			configurationNode.appendChild(programsNode);
464 
465 		  for (Object execServer : execServers) {
466 			ExecServerConfig esc = (ExecServerConfig) execServer;
467 			Element execServerNode = document.createElement("execServer");
468 			programsNode.appendChild(execServerNode);
469 			XML.add(execServerNode, "class", esc.getClassName());
470 			XML.add(execServerNode, "objectKey", esc.getObjectKey());
471 			XML.add(execServerNode, "host", esc.getPreferredHost().toString());
472 			if (esc.getProperties().size() > 0) {
473 			  dumpProperties(esc.getProperties(), execServerNode);
474 			}
475 		  }
476 		}
477 
478 		return configurationNode;
479 	}

482 	 *
483 	 * Properties that already exist in the <var>targetProps</var> won't be
484 	 * overwritten.
485 	 *
486 	 * @param targetProps The target properties.
487 	 */
488 	public void mergeProperties(Properties targetProps) {
489 	  for (Map.Entry<Object, Object> objectObjectEntry : properties.entrySet()) {
490 		Map.Entry entry = (Map.Entry) objectObjectEntry;
491 		if (!targetProps.containsKey(entry.getKey())) {
492 		  targetProps.put(entry.getKey(), entry.getValue());
493 		}
494 	  }
495 	}

534                 }
535                 return execServerConfig;

586 	/** Get the object context.
587 	 *
588 	 * @return The object context based on this configuration.
589 	 * @throws NamingException If the context can't be created.
590 	 */
591 	public Context getObjectContext() throws NamingException {
592 		Context c;
593 		final String className = (String) contextEnvironment.get(javax.naming.Context.INITIAL_CONTEXT_FACTORY);
594 		if (className == null) {
595 		  c = new InitialContext(contextEnvironment);
596 		} else {
597 		  try {
598 			// Work around iPlanet bug.  JNDI uses the thread's context class
599 			// loader to load the initial context factory class.  For some
600 			// reason, that loader fails to find things in iPlanet's
601 			// classpath, such as the EDA initial context factory.  Here, we
602 			// cut a new thread and explicitly set its context class loader to
603 			// the application class loader.  When JNDI looks up the initial
604 			// context factory, the thread's context class loader is the app
605 			// class loader, which succeeds.
606 			Class clazz = Class.forName(className);
607 			final ClassLoader loader = clazz.getClassLoader();
608 			InitialContextThread thread = new InitialContextThread(loader);
609 			thread.start();
610 			try {
611 			  thread.join();
612 			} catch (InterruptedException ex) {
613 			  throw new NoInitialContextException("Initial context thread interrupted: " + ex.getMessage());
614 			}
615 			c = thread.getContext();
616 			if (c == null) {
617 			  throw thread.getException();
618 			}
619 		  } catch (ClassNotFoundException ex) {
620 			throw new NoInitialContextException("Class " + className + " not found");
621 		  }
622 		}
623 		return c;
624 	}
