58      private Metadata getMetadataFromFileAttributes(Product product) {
59          Metadata met = new Metadata();
60  
61          File file = null;
62          try {
63              file = getProductFile(product);
64          } catch (MetExtractionException e) {
65              LOG.severe(e.getMessage());
66          }
67  
68          if (file != null) {
69              Map<String, Object> attrMap = new HashMap<String, Object>();
70              try {
71                  attrMap = Files.readAttributes(file.toPath(), attributes);
72              } catch (IOException e) {
73                  LOG.severe(e.getMessage());
74              }
75  
76              for (Map.Entry<String, Object> entry : attrMap.entrySet()) {
77                  met.addMetadata(entry.getKey(), String.valueOf(entry.getValue()));
78              }
79          }
80          return met;
