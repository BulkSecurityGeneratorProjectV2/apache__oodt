165 	 *
166 	 * @param keywordQuery  The DIS style keyword query string.
167 	 * @param id A query identifier.
168 	 * @param title A terse description of the query for display.
169 	 * @param desc A description of the query.
170 	 * @param ddId The data dictionary identifier.    
171 	 * @param resultModeId Indicates the return of INSTANCE, PROFILE, or CLASS.
172 	 * @param propType Indicates query BROADCAST or PROPOGATE.
173 	 * @param propLevels Number of propogation levels.
174 	 * @param maxResults Maximum number of results to be returned.
175 	 * @param mimeAccept List of acceptable MIME types.
176 	 * @param parseQuery Indicates whether query should be parsed
177 	 */
178 	public XMLQuery(String keywordQuery, String id, String title, String desc, String ddId, String resultModeId,
179 		String propType, String propLevels, int maxResults, 
180 		List mimeAccept, boolean parseQuery) {
181 		if (mimeAccept == null) {
182 			mimeAccept = new ArrayList();
183 			mimeAccept.add("*/*");
184 		}
185 		if (keywordQuery == null) {
186 		  keywordQuery = "UNKNOWN";
187 		}
188 		// init query header (object attributes)
189 		if (id == null) {
190 		  id = "UNKNOWN";
191 		}
192 		if (title == null) {
193 		  title = "UNKNOWN";
194 		}
195 		if (desc == null) {
196 		  desc = "UNKNOWN";
197 		}
198 		if (ddId == null) {
199 		  ddId = "UNKNOWN";
200 		}
201 		queryHeader = new QueryHeader(id, title, desc, /*type*/"QUERY", /*status*/"ACTIVE", /*security*/"UNKNOWN",
202 			/*revision*/"1999-12-12 JSH V1.0 Under Development", ddId);
203 
204 		// init query attributes
205 		if (resultModeId == null) {
206 		  resultModeId = "ATTRIBUTE";
207 		}
208 		if (propType == null) {
209 			propType = "BROADCAST";
210 			propLevels = "N/A";
211 		}
212 		this.resultModeId = resultModeId;
213 		this.propogationType = propType;
214 		this.propogationLevels = propLevels;
215 		this.maxResults = maxResults;
216 		this.kwqString = keywordQuery;
217 		this.mimeAccept = mimeAccept;
218         
219 		// parse the keyword query string
220 		if (! parseQuery) {
221 			queryHeader.setStatusID("NOTPARSED");
222 		} else if (!parseKeywordString(keywordQuery)) {
223 			queryHeader.setStatusID("ERROR");
224 		}
225 	}

241 		if (xmlQueryString == null) {
242 		  xmlQueryString = BLANK;
243 		}
244 		try {
245 			DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
246 			factory.setCoalescing(false);
247 			factory.setNamespaceAware(false);
248 			factory.setValidating(false);
249 			DocumentBuilder builder = factory.newDocumentBuilder();
250 			builder.setEntityResolver(ENTITY_RESOLVER);
251 			Document doc = builder.parse(new ByteArrayInputStream(xmlQueryString.getBytes()));
252 			loadXMLQueryStruct(doc.getDocumentElement());
253 		} catch (IOException ex) {
254 			System.err.println("Unexpected no good bloody IOException while parsing doc: " + xmlQueryString);
255 			throw new IllegalStateException("Unexpected no good bloody IOException: " + ex.getMessage() + ", during"
256 				+ " parse of " + xmlQueryString);
257 		} catch (ParserConfigurationException ex) {
258 			throw new IllegalStateException("Unexpected ParserConfigurationException: " + ex.getMessage());
259 		}
260    	}

412     /**
413      * Parse keyword query string (simple top down recursion).
414      */
415     private boolean parseKeywordString (String kwdQueryString) {
416 	    previous_token = "";
417 	    lpart = "";
418 	    rpart = "";
419 
420             tokens = new java.io.StreamTokenizer(new StringReader(kwdQueryString));
421             tokens.resetSyntax();
422             tokens.ordinaryChar('/');
423             tokens.wordChars('#', '#');
424             tokens.wordChars(':', ':');
425             tokens.wordChars('0', '9');
426             tokens.wordChars('.', '.');
427             tokens.wordChars('-', '-');
428             tokens.wordChars('[', '[');
429             tokens.wordChars(']', ']');
430             tokens.wordChars('a', 'z');
431             tokens.wordChars('A', 'Z');
432             tokens.wordChars('_', '_');
433             tokens.wordChars('/', '/');
434             tokens.eolIsSignificant(true);
435             tokens.whitespaceChars(0, ' ');
436             tokens.quoteChar('"');
437             tokens.quoteChar('\'');
438 
439 	  return kqOrParse();
440 	}

442     /**
443      * Parse OR logical operator.
444      */
445 	private boolean kqOrParse ()
446 	{
447 	    boolean lflag, rflag;
448 
449 	    lflag = kqAndParse();
450 	    while (isTokenEqual("OR") | isTokenEqual("or") | isTokenEqual("|")) {
451             rflag = kqAndParse();
452             lflag &= rflag;
453             appendLogOperator("LOGOP", "OR");
454         }
455         return lflag;
456 	}
457 
458     /**
459      * Parse AND logical operator.
460      */
461    	private boolean kqAndParse ()
462 	{
463 	    boolean lflag, rflag;
464 
465 	    lflag = kqNotParse();
466 	    while (isTokenEqual("AND") | isTokenEqual("and") | isTokenEqual("&")) {
467             rflag = kqNotParse();
468             lflag &= rflag;
469             appendLogOperator("LOGOP", "AND");
470         }
471         return lflag;
472 	}
473 
474     /**
475      * Parse NOT logical operator.
476      */
477 	private boolean kqNotParse ()
478 	{
479 	    boolean lflag, rflag;
480 
481 	    if (isTokenEqual("NOT") | isTokenEqual("not") | isTokenEqual("!")) {
482             lflag = kqFactorParse();
483             appendLogOperator("LOGOP", "NOT");
484         } else {
485             lflag = kqFactorParse ();
486         }
487         return lflag;
488 	}
489 
490     /**
491      * Parse logical operator operand and handle nesting (parens)
492      * also logic for differentiating SELECT, WHERE, and FROM
493      */
494 	private boolean kqFactorParse ()
495 	{
496 	    String ropn, enm;
497 	    if (isTokenEqual("(")) {
498 	        return (kqOrParse () & isTokenEqual(")"));
499 	    } else {
500             lpart = getNextToken ();
501             if (lpart.compareTo("") != 0) {
502                  ropn = getNextToken ();
503                  if (ropn.compareTo("EQ") == 0 || ropn.compareTo("LT") == 0 || ropn.compareTo("LE") == 0 || ropn.compareTo("GT") == 0 || ropn.compareTo("GE")== 0 || ropn.compareTo("NE") == 0 || ropn.compareToIgnoreCase("LIKE") == 0 || ropn.compareToIgnoreCase("NOTLIKE") == 0 || ropn.compareTo("IS") == 0 || ropn.compareTo("ISNOT") == 0) {
504                     rpart = getNextToken ();
505                     if (rpart.compareTo("") != 0) {
506                         lit = 0;
507                         enm = lpart;
508                         if (enm.compareTo("RETURN") == 0) {
509 			    selectElementSet.add(new QueryElement("elemName", rpart));
510                         } else if (isFromToken(lpart)) {
511                             qfsc++;
512                             lit = 1;
513 			    fromElementSet.add(new QueryElement("elemName", lpart));
514 			    fromElementSet.add(new QueryElement("LITERAL", rpart));
515 			    fromElementSet.add(new QueryElement("RELOP", ropn));
516                         } else {
517                             lit = 2;
518                             qwsc++;
519 			    whereElementSet.add(new QueryElement("elemName", lpart));
520 			    whereElementSet.add(new QueryElement("LITERAL", rpart));
521 			    whereElementSet.add(new QueryElement("RELOP", ropn));
522                         }
523                         return true;
524                     }
525                 }
526             }
527         }
528         return false;
529 	}

531     /**
532      * Create Logical Operator DOM node for WHERE clause.
533      */
534     private boolean appendLogOperator (String tt, String ts)
535 	{
536 	    if (lit == 1 & qfsc > 1) {
537 		    fromElementSet.add(new QueryElement(tt, ts));
538 
539     	} else if (lit == 2 & qwsc > 1) {
540 		whereElementSet.add(new QueryElement(tt, ts));
541     	}
542 	    return true;
543 	}

545     /**
546      * Lexical analyzer - find a FROM token.
547      */
548     private static boolean isFromToken (String s1)
549 	{
550 	  for (String FROM_TOKEN : FROM_TOKENS) {
551 		if (s1.compareTo(FROM_TOKEN) == 0) {
552 		  return true;
553 		}
554 	  }
555 	    return false;
556 	}

558     /**
559      * Lexical analyzer - return next token - allows lookahead.
560      */
561     private String getNextToken ()
562 	{
563 	    String ts;
564 
565     	    ts = previous_token;
566             if (previous_token.compareTo("") == 0) {
567                     ts = getNextTokenFromStream();
568         	    return ts;
569             } else {
570         	    previous_token = "";
571         	    return ts;
572             }
573 	}

575     /**
576      * Lexical analyzer - check if next token equal to argument.
577      */
578     private boolean isTokenEqual (String s1) {
579 	    String ts;
580 	    boolean rc;
581 
582 	  if (previous_token.compareTo("") == 0) {
583        	        ts = getNextTokenFromStream ();
584 		if (ts.compareTo("") == 0) {
585                     rc = false;
586                 } else {
587            	    if (ts.compareTo(s1) == 0) {
588      	                previous_token = "";
589        	                rc = true;
590        	            } else {
591        	                previous_token = ts;
592        	                rc = false;
593                     }
594        	        }
595       	    } else {
596            	    if (previous_token.compareTo(s1) == 0) {
597        	            previous_token = "";
598        	            rc = true;
599        	        } else {
600        	            rc = false;
601        	        }
602      	    }
603 	    return rc;
604 	}

606     /**
607      * Lexical analyzer - return next token from instance of StreamTokenizer.
608      */
609    private String getNextTokenFromStream () {
610             int c, c2;
611 	    String rc;
612             try {
613                 switch (c=tokens.nextToken()) {
614                     case StreamTokenizer.TT_EOF:
615                         rc = "";
616 			break;
617                     case StreamTokenizer.TT_EOL:
618                         rc = "";
619 			break;
620                     case StreamTokenizer.TT_NUMBER: // not currently set
621 			rc = "";
622 			break;
623                     case StreamTokenizer.TT_WORD:
624                         rc = tokens.sval;
625 			break;
626                     case '(':
627                     case ')':
628 			rc = String.valueOf((char)c);
629 			break;
630                     case '"':
631                     case '\'':
632                         rc = tokens.sval;
633 			break;
634                     case '=':
635                         rc = "EQ";
636 			break;
637                     case '&':
638                         rc = "AND";
639 			break;
640                     case '|':
641                         rc = "OR";
642 			break;
643                     case '<':
644                         c2 = tokens.nextToken();
645                         if (c2 == '=') {
646 				rc = "LE";
647                         } else {
648 				tokens.pushBack();
649 				rc = "LT";
650                         }
651 			break;
652                     case '>':
653                         c2 = tokens.nextToken();
654                         if (c2 == '=') {
655 				rc = "GE";
656                         } else {
657                                 tokens.pushBack();
658 				rc = "GT";
659                         }
660 			break;
661                     case '!':
662                         c2 = tokens.nextToken();
663                         if (c2 == '=') {
664 				rc = "NE";
665                         } else {
666                                 tokens.pushBack();
667 				rc = "NOT";
668                         }
669 			break;
670                     default:
671                         rc = "";
672                     }
673                 } catch (IOException e1) {
674 			rc = "";
675 		}
676 	    return rc;
677    }

679     /**
680      * Replace character with string
681      */
682     private String replaceCharWithString (String s1, char c, String rs)
683 	{
684         int p1, p2, s1l;
685     	StringBuilder s2;
686     	
687     	p1 = 0;
688 	  s1l = s1.length();
689     	s2 = new StringBuilder();
690     	p2 = s1.indexOf(c, p1);
691     	while (p2 >= 0) {
692     	    s2.append(s1.substring(p1, p2)).append(rs);
693             p1 = p2 + 1;
694             p2 = s1.indexOf(c, p1);
695         }
696         if (p1 < s1l) {
697             s2.append(s1.substring(p1, s1l));
698         }
699         return s2.toString();
700 	}

702     /**
703      * Creates a XML DOM structure from a keyword query string.
704      * Set up the basic structure then calls the query parser.
705      */
706 	private void createXMLDoc (Document doc) {
707 	    int loc;
708 	    String kwd;
709 	    String type;
710 	    String val;
711 	    Element item;
712 	    Element elem;
713 	    
714 	    Element query = doc.getDocumentElement();
