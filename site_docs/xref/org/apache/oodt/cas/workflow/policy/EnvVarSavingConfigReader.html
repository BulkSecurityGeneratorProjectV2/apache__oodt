6   package org.apache.oodt.cas.workflow.policy;
7   
8   //JDK imports
9   import org.w3c.dom.Element;
10  import org.w3c.dom.Node;
11  import org.w3c.dom.NodeList;
12  
13  /**
14   * @author mattmann
15   * @version $Revision$
16   * 
17   * <p>
18   * Provides a utility method to read in a {@link WorkflowConfiguration} from an
19   * XML {@link Node}, preserving its envReplace information.
20   * </p>.
21   */
22  public class EnvVarSavingConfigReader {
23  
24      private EnvVarSavingConfigReader() throws InstantiationException {
25          throw new InstantiationException("Don't construct readers!");
26      }
27  
28      public static EnvSavingConfiguration getConfiguration(Node node) {
29          Element configNode = (Element) node;
30  
31          NodeList configProperties = configNode.getElementsByTagName("property");
32  
33          EnvSavingConfiguration config = null;
34  
35          if (configProperties == null) {
36              return null;
37          }
38  
39          config = new EnvSavingConfiguration();
40          for (int i = 0; i < configProperties.getLength(); i++) {
41              Element propElem = (Element) configProperties.item(i);
42              String value = propElem.getAttribute("value");
43              boolean doReplace = Boolean.valueOf(
44                      propElem.getAttribute("envReplace")).booleanValue();
45  
46              config.addConfigProperty(propElem.getAttribute("name"), value,
47                      doReplace);
48          }
49  
50          return config;
51      }
52  
53  }
