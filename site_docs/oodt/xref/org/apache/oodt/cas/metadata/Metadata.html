313   /**
314    * Get all keys whose leaf key name is equal to the given arg
315    * @param key leaf key name
316    * @return list of keys with the given leaf key name
317    */
318   public List<String> getAllKeysWithName(String key) {
319 	  List<String> keys = new Vector<String>();
320 	  Stack<Group> stack = new Stack<Group>();
321 	  stack.add(this.root);
322 	  while (!stack.empty()) {
323 		  Group curGroup = stack.pop();
324 		  if (curGroup.getName().equals(key) && curGroup.hasValues()) {
325             keys.add(curGroup.getFullPath());
326           }
327 		  stack.addAll(curGroup.getChildren());
328 	  }
329 	  return keys;
330   }

450   protected Group getGroup(String key, boolean create) {
451     if (key == null) {
452       return this.root;
453     }
454     StringTokenizer tokenizer = new StringTokenizer(key, "/");
455     Group curGroup = this.root;
456     while (tokenizer.hasMoreTokens()) {
457       String groupName = tokenizer.nextToken();
458       Group childGroup = curGroup.getChild(groupName);
459       if (childGroup == null) {
460         if (!create) {
461           return null;
462         }
463         childGroup = new Group(groupName);
464         curGroup.addChild(childGroup);
465       }
466       curGroup = childGroup;
467     }
468     return curGroup;
469   }
