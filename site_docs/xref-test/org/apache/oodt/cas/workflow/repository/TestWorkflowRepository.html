271             config = workflowRepository
272                     .getConfigurationByTaskId("urn:oodt:PropReplaceTask");
273         } catch (RepositoryException e) {

277

279         String replacedPath = config.getProperty("PathToReplace");
280         String expectedValue = System.getenv("HOME") + "/my/path";
281         assertEquals("The path: [" + replacedPath
282                 + "] does not equal the expected" + "path: [" + expectedValue
283                 + "]", replacedPath, expectedValue);
284 
285         String notReplacedPath = config.getProperty("DontReplaceMe");
286         String notReplacedPathNoSpec = config
287                 .getProperty("DontReplaceMeNoSpec");
288 
289         assertNotNull(notReplacedPath);
290         assertNotNull(notReplacedPathNoSpec);
291         assertEquals(notReplacedPath, notReplacedPathNoSpec);
292 
293         String expected = "[HOME]/my/path";
294         assertEquals("The path: [" + notReplacedPath + "] is not equal to the "
295                 + "expected value: [" + expected + "]", expected,
296                 notReplacedPath);

300     public void testMultipleConditions() {
301         WorkflowTask multiTask = null;
302 
303         try {
304             multiTask = workflowRepository
305                     .getWorkflowTaskById("urn:oodt:TestMultiConditionTask");
306         } catch (Exception e) {
307             fail(e.getMessage());
308         }
309 
310         assertNotNull(multiTask);
311         assertNotNull(multiTask.getConditions());
312         assertEquals(2, multiTask.getConditions().size());
313 
314         boolean gotTrueCond = false, gotFalseCond = false;
315 
316         for (int i = 0; i < multiTask.getConditions().size(); i++) {
317             WorkflowCondition c = (WorkflowCondition) multiTask.getConditions()
318                     .get(i);
319             assertNotNull(c);
320             if (c.getConditionName().equals("True Condition")) {
321                 gotTrueCond = true;
322             } else if (c.getConditionName().equals("False Condition")) {
323                 gotFalseCond = true;
324             }
325         }
326 
327         assertTrue(gotTrueCond && gotFalseCond);
328     }

352     private void validateBackwardsWorkflow(Workflow w) {
353         assertNotNull(w);
354         assertNotNull(w.getId());
355         assertEquals("urn:oodt:backwardsTestWorkflow", w.getId());
356         assertNotNull(w.getName());
357         assertEquals("backwardsTestWorkflow", w.getName());
358         validateBackwardsWorkflowTasks(w.getTasks());

361     private void validateBackwardsWorkflowTasks(List tasks) {
362         assertNotNull(tasks);
363         assertEquals(2, tasks.size());
364 
365         WorkflowTask t1 = (WorkflowTask) tasks.get(0);
366         assertEquals("Goodbye World", t1.getTaskName());
367         assertEquals("urn:oodt:GoodbyeWorld", t1.getTaskId());
368         assertEquals(1, t1.getOrder());
369         validateTaskConfiguration(t1.getTaskConfig());
370         validateTaskCondition(t1.getConditions());
371 
372         WorkflowTask t2 = (WorkflowTask) tasks.get(1);
373         assertEquals("Hello World", t2.getTaskName());
374         assertEquals("urn:oodt:HelloWorld", t2.getTaskId());
375         assertEquals(2, t2.getOrder());
376         validateTaskConfiguration(t2.getTaskConfig());
377         validateTaskCondition(t2.getConditions());
378 
379     }

381     private void validateTaskCondition(List conditions) {
382         assertNotNull(conditions);
383         assertEquals(1, conditions.size());
384         WorkflowCondition c = (WorkflowCondition) conditions.get(0);
385         assertEquals("urn:oodt:TrueCondition", c.getConditionId());
386         assertEquals("True Condition", c.getConditionName());
387         assertEquals(1, c.getOrder());
388         assertEquals(-1L, c.getTimeoutSeconds());
389         assertEquals(false, c.isOptional());
390     }
