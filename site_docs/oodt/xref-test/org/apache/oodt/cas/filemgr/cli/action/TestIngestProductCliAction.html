129    public void testClientTransTrueAndFlatProduct() throws CmdLineActionException, IOException {
130       ActionMessagePrinter printer = new ActionMessagePrinter();
131       MockIngestProductCliAction cliAction = new MockIngestProductCliAction();
132       cliAction.setProductName(PRODUCT_NAME);
133       cliAction.setProductStructure(Product.STRUCTURE_FLAT);
134       cliAction.setProductTypeName(PRODUCT_TYPE_NAME);
135       cliAction.setMetadataFile(metadataFile.getAbsolutePath());
136       cliAction.setDataTransferer(DATA_TRANSFERER_FACTORY);
137       cliAction.setReferences(Lists.newArrayList(flatRefFile.getAbsolutePath()));
138       cliAction.execute(printer);
139       assertEquals(2, printer.getPrintedMessages().size());
140       assertEquals("ingestProduct: Result: " + PRODUCT_ID, printer.getPrintedMessages().get(0)); 
141       assertEquals("\n", printer.getPrintedMessages().get(1));
142       assertEquals(PRODUCT_NAME, clientSetProduct.getProductName());
143       assertEquals(Product.STRUCTURE_FLAT, clientSetProduct.getProductStructure());
144       assertEquals(PRODUCT_TYPE_NAME, clientSetProduct.getProductType().getName());
145       assertEquals(1, clientSetProduct.getProductReferences().size());
146       assertEquals("file:" + flatRefFile.getAbsolutePath(), clientSetProduct.getProductReferences().get(0).getOrigReference());
147       assertEquals(DATA_TRANSFERER, clientSetDataTransferer.getClass().getCanonicalName());
148       assertEquals(2, clientSetMetadata.getAllKeys().size());
149       assertEquals(FILENAME_MET_VAL, clientSetMetadata.getMetadata(FILENAME_MET_KEY));
150       assertEquals(NOMINAL_DATE_MET_VAL, clientSetMetadata.getMetadata(NOMINAL_DATE_MET_KEY));
151    }

176    public void testClientTransTrueAndHierProduct() throws CmdLineActionException, IOException {
177       ActionMessagePrinter printer = new ActionMessagePrinter();
178       MockIngestProductCliAction cliAction = new MockIngestProductCliAction();
179       cliAction.setProductName(PRODUCT_NAME);
180       cliAction.setProductStructure(Product.STRUCTURE_HIERARCHICAL);
181       cliAction.setProductTypeName(PRODUCT_TYPE_NAME);
182       cliAction.setMetadataFile(metadataFile.getAbsolutePath());
183       cliAction.setDataTransferer(DATA_TRANSFERER_FACTORY);
184       cliAction.setReferences(Lists.newArrayList(hierRefFile.getAbsolutePath()));
185       cliAction.execute(printer);
186       assertEquals(2, printer.getPrintedMessages().size());
187       assertEquals("ingestProduct: Result: " + PRODUCT_ID, printer.getPrintedMessages().get(0)); 
188       assertEquals("\n", printer.getPrintedMessages().get(1));

191       assertEquals(PRODUCT_TYPE_NAME, clientSetProduct.getProductType().getName());
192       assertEquals(3, clientSetProduct.getProductReferences().size());
193       Collections.sort(clientSetProduct.getProductReferences(), new Comparator<Reference>() {
194          @Override

196             return ref1.getOrigReference().compareTo(ref2.getOrigReference());

198       });
