17  package org.apache.oodt.cas.filemgr.datatransfer;

19  //OODT imports
20  import org.apache.oodt.cas.filemgr.structs.Product;
21  import org.apache.oodt.cas.filemgr.structs.exceptions.ConnectionException;
22  import org.apache.oodt.cas.filemgr.structs.exceptions.DataTransferException;
23  import org.apache.oodt.cas.filemgr.system.XmlRpcFileManagerClient;
24  
25  //JDK imports
26  import java.io.File;
27  import java.io.IOException;
28  import java.util.logging.Logger;
29  import java.util.logging.Level;
30  import java.net.URL;

32  /** 
33   * An implementation of the {@link DataTransfer} interface that leaves
34   * products in the same place (i.e., "in place") and doesn't transfer
35   * them at all.
36   *
37   * @author mattmann@apache.org (Chris Mattmann)
38   */
39  public class InPlaceDataTransferer implements DataTransfer {
40  
41    private static final Logger LOG = Logger.getLogger(InPlaceDataTransferer.class.getName());
42  
43    private XmlRpcFileManagerClient client = null;
44  
45    public InPlaceDataTransferer() {}
46  
47    public void setFileManagerUrl(URL url) {
48      try {
49        client = new XmlRpcFileManagerClient(url);
50        LOG.log(Level.INFO, "In Place Data Transfer to: [" + client.getFileManagerUrl().toString()
51            + "] enabled");
52      } catch (ConnectionException e) {
53        LOG.log(Level.WARNING, "Connection exception for filemgr: [" + url + "]");
54      }
55    }
56  
57    @Override
58    public void transferProduct(Product product) throws DataTransferException, IOException {
59      // do nothing
60    }
61  
62    @Override
63    public void retrieveProduct(Product product, File directory) throws DataTransferException,
64        IOException {
65      // do nothing
66    }
67  
68    @Override
69    public void deleteProduct(Product product) throws DataTransferException, IOException {
70      // do nothing
71    }
72  }
