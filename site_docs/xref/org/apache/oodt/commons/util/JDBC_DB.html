95  	public void openConnection(
96  		String username,
97  		String password,
98  		String database) throws SQLException
99  	{
100 		String url, classname;
101 
102 		if (stmt != null) {
103 		  stmt.close();
104 		}
105 
106 		if (rs != null) {
107 		  rs.close();
108 		}
109 
110 		if (keep_connect_open) {
111 		  return;
112 		}
113 
114 		if (connect != null) {
115 		  connect.close();
116 		}
117 
118 
119 		rs_meta = null;
120 		connect = null;
121 		stmt = null;
122 		rs = null;
123 
124 		Properties props = new Properties();
125 		props.put("user", username);
126 
127 		if (password != null) {
128 		  props.put("password", password);
129 		}
130 
131 
132 		classname = serverProps.getProperty("org.apache.oodt.commons.util.JDBC_DB.driver", "oracle.jdbc.driver.OracleDriver");
133 		try {
134 			System.err.println("Attempting to load class " + classname);
135 			Class.forName(classname);
136 			System.err.println("Loaded " + classname);
137 		} catch (ClassNotFoundException e) {
138 			System.err.println("Can't load JDBC driver \"" + classname + "\": " + e.getMessage());
139 			LOG.log(Level.SEVERE, e.getMessage());
140 		}
141 		url = serverProps.getProperty("org.apache.oodt.commons.util.JDBC_DB.url", "jdbc:oracle:@");
142 		try {
143 			if (database != null) {
144 				System.err.println("Connecting to url+database combo: " + url + database);
145 				connect = DriverManager.getConnection(url+database, props);
146 			} else {
147 				System.err.println("Connecting to full url: " + url);
148 				connect = DriverManager.getConnection(url, props);
149 			}
150 		} catch (SQLException e) {
151 			System.err.println("SQL Exception during connection creation: " + e.getMessage());
152 			LOG.log(Level.SEVERE, e.getMessage());
153 			while (e != null) {
154 				System.err.println(e.getMessage());
155 				e = e.getNextException();
156 			}
157 		}
158 
159 		connect.setAutoCommit(autoCommitMode);
160 
161 	}

202 	public void executeSQLCommand(String cmd) throws SQLException
203 	{
204 
205 
206 		/*
207 		** Get the string, and create the statement
208 		*/
209 		sql_command = cmd;
210 
211 		if (stmt!=null) {
212 		  stmt.close();
213 		}
214 
215 		if (connect == null) {
216 		  openConnection();
217 		}
218 		if (connect == null) {
219 			keep_connect_open = false;
220 			openConnection();
221 		}
222 		if (connect == null) {
223 		  throw new IllegalStateException("Connection is null!!!");
224 		}
225 		
226 		if (connect.isClosed()) {
227 			connect = null;
228 			keep_connect_open = false;
229 			openConnection();
230 		}
231 		if (connect == null) {
232 		  throw new IllegalStateException("Connection is still null!!!");
233 		}
234 		if (connect.isClosed()) {
235 		  throw new IllegalStateException("Connection got closed!");
236 		}
237 
238 		stmt = connect.createStatement();
239 		affected = stmt.executeUpdate(sql_command);
240 	}
