157 	public CatalogReceipt ingest(Metadata metadata) throws CatalogException {
158 		try {
159 			if (this.isIngestable()) {
160 				List<TermBucket> termBuckets = this.getTermBuckets(metadata);
161 				if (termBuckets.size() > 0) {
162 					LOG.log(Level.INFO, "Catalog '" + this + "' attemping ingest metadata");
163 					return new CatalogReceipt(((IngestService) this.index).ingest(termBuckets), this.getId());
164 				}else {
165 					LOG.log(Level.WARNING, "Catalog '" + this + "' dictionaries did not generate any TermBuckets from Metadata");
166 					return null;
167 				}
168 			}else {
169 				LOG.log(Level.WARNING, "Catalog '" + this + "' is not ingestable");
170 				return null;
171 			}
172 		}catch (Exception e) {
173 			throw new CatalogException(e.getMessage(), e);
174 		}
175 	}

177 	public CatalogReceipt update(TransactionId<?> transactionId, Metadata metadata) throws CatalogException {
178 		try {
179 			if (this.isIngestable()) {
180 				List<TermBucket> termBuckets = this.getTermBuckets(metadata);
181 				if (termBuckets.size() > 0) {
182 					LOG.log(Level.INFO, "Catalog '" + this + "' attemping update metadata for catalog TransactionId [id = '" + transactionId + "']");
183 					IngestReceipt ingestReceipt = ((IngestService) this.index).update(transactionId, termBuckets);
184 					if (ingestReceipt != null) {
185 					  return new CatalogReceipt(ingestReceipt, this.getId());
186 					} else {
187 					  return null;
188 					}
189 				}else {
190 					LOG.log(Level.WARNING, "Catalog '" + this + "' did not generate any TermBuckets from Metadata for catalog TransactionId [id = '" + transactionId + "']");
191 					return null;
192 				}
193 			}else {
194 				LOG.log(Level.WARNING, "Catalog '" + this + "' is not ingestable");
195 				return null;
196 			}
197 		}catch (Exception e) {
198 			throw new CatalogException(e.getMessage(), e);

301 	public Map<TransactionId<?>, Metadata> getMetadata(List<TransactionId<?>> transactionIds) throws CatalogException {
302 		try {
303 			Map<TransactionId<?>, Metadata> metadataMap = new ConcurrentHashMap<TransactionId<?>, Metadata>();
304 			if (this.isQueriable()) {
305 				QueryService queryService = (QueryService) this.index;
306 				Map<TransactionId<?>, List<TermBucket>> termBucketMap = queryService.getBuckets(transactionIds);
307 
308 			  for(Map.Entry<TransactionId<?>, List<TermBucket>> entry :  termBucketMap.entrySet()){
309 				metadataMap.put(entry.getKey(), this.getMetadataFromBuckets(entry.getValue()));
310 			  }
311 			}else {
312 				LOG.log(Level.WARNING, "Catalog '" + this + "' is not queriable");
313 			}
314 			return metadataMap;

316 			throw new CatalogException(e.getMessage(), e);

337 	protected Metadata getMetadataFromBuckets(List<TermBucket> termBuckets) {
338 		Metadata metadata = new Metadata();
339 		for (TermBucket termBucket : termBuckets) {
340 			if (this.dictionaries != null) {
341 				for (Dictionary dictionary : this.dictionaries) {
342 				  metadata.addMetadata(dictionary.reverseLookup(termBucket));
343 				}
344 			}else {
345 				metadata.addMetadata(this.asMetadata(termBuckets));
346 			}
347 		}
348 		return metadata;
349 	}

351 	protected Metadata asMetadata(List<TermBucket> termBuckets) {
352 		Metadata m = new Metadata();
353 		for (TermBucket bucket : termBuckets) {
354 		  for (Term term : bucket.getTerms()) {
355 			m.addMetadata(term.getName(), term.getValues());
356 		  }
357 		}
358 		return m;
359 	}

361 	protected List<TermBucket> getTermBuckets(Metadata metadata) {
362 		List<TermBucket> termBuckets = new Vector<TermBucket>();
363 		if (this.dictionaries != null) {
364 			for (Dictionary dictionary : this.dictionaries) {
365 				TermBucket termBucket = dictionary.lookup(metadata);
366 				if (termBucket != null) {
367 				  termBuckets.add(termBucket);
368 				}
369 			}
370 		}else {
371 			LOG.log(Level.WARNING, "Catalog '" + this + "' has no dictionaries defined, attempting to send all Metadata in a default TermBucket");
372 			TermBucket bucket = new TermBucket();
373 			for (String key : metadata.getAllKeys()) {
374 			  bucket.addTerm(new Term(key, metadata.getAllMetadata(key)));
375 			}
376 			termBuckets.add(bucket);
377 		}
378 		return termBuckets;
379 	}
