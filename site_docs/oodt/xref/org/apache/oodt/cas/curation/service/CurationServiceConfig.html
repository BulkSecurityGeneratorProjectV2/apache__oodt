38  /**
39   * 
40   * 
41   * Configures the {@link CurationService} by reading the parameters out of the
42   * <code>context.xml</code> file, doing a
43   * {@link PathUtils#replaceEnvVariables(String)} on each property in the
44   * context.xml. This allows you to specify environment variables using the
45   * traditional CAS syntax of:
46   * 
47   * <pre>
48   *   [VAR_NAME], e.g., [CAS_CURATOR_HOME]/some/other/path
49   * </pre>
50   * 
51   * The configuration parameters are read once upon loading the instance of this
52   * object.
53   * 
54   * @author pramirez
55   * @author mattmann
56   * @version $Revision$
57   * 
58   */
59  public class CurationServiceConfig implements CuratorConfMetKeys {
60    private static CurationServiceConfig instance;
61    private final Map<String, String> parameters = new ConcurrentHashMap<String, String>();
62  
63    private XmlRpcFileManagerClient fmClient = null;
64  
65    private static final Logger LOG = Logger
66        .getLogger(CurationServiceConfig.class.getName());
67  
68    /**
69     * Gets a singleton static instance of the global
70     * {@link CurationServiceConfig} for the CAS Curator Webapp.
71     * 
72     * @param conf
73     *          The {@link ServletConfig} read on startup of the webapp. This is
74     *          typically specified in a <code>context.xml</code> file, but can
75     *          also be specified in <code>web.xml</code>.
76     * @return A singleton instance of the global {@link CurationServiceConfig}.
77     * @throws InstantiationException
78     *           If there is any error constructing the config.
79     */
80    public static CurationServiceConfig getInstance(ServletConfig conf) {
81      if (instance == null) {
82        instance = new CurationServiceConfig(conf);
83      }
84      return instance;
85    }

185   private String evaluateParameter(String name) {
186     return PathUtils.replaceEnvVariables(this.parameters.get(name));
187   }
188 
189   // Note that the constructor is private
190   private CurationServiceConfig(ServletConfig conf) {
191     readContextParams(conf.getServletContext());
192     try {
193       fmClient = new XmlRpcFileManagerClient(new URL(this.getFileMgrURL()));
194     } catch (Exception e) {
195       LOG.log(Level.SEVERE, e.getMessage());
196       LOG.log(Level.WARNING, "Unable to build CurationServiceConfig: Message: " + e.getMessage());
197     }
198   }
199 
200   @SuppressWarnings("unchecked")
201   private void readContextParams(ServletContext context) {
202     for (Enumeration<String> names = context.getInitParameterNames(); names
203         .hasMoreElements();) {
204       String name = names.nextElement();
205       parameters.put(name, context.getInitParameter(name));
206     }
207     LOG.log(Level.INFO, "Init Parameters: " + parameters);
208   }
