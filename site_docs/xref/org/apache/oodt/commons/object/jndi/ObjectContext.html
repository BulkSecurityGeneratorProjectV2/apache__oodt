49  	 * Construct the object context.
50  	 *
51  	 * @param environment Its environment, currently unused.
52  	 */
53  	ObjectContext(Hashtable environment) {
54  		this.environment = environment != null? (Hashtable) environment.clone() : new Hashtable();
55  
56  		// Add the CORBA context, but by name so we don't get a compile-time
57  		// coupling with the edm-corba component, and only if CORBA's available.
58  		try {
59  			Class clazz = Class.forName("org.apache.oodt.commons.object.jndi.CORBAContext");
60  			Constructor ctor = clazz.getConstructor(new Class[]{Hashtable.class});
61  			Object corbaContext = ctor.newInstance(new Object[]{this.environment});
62  			contexts.add(corbaContext);
63  		} catch (Throwable ignored) {}
64  
65  		String registryList = (String) (environment != null ? environment.get("rmiregistries") : null);
66  		if (registryList != null) {
67  		  for (Iterator i = Utility.parseCommaList(registryList); i.hasNext(); ) {
68  			Hashtable rmiEnv = (Hashtable) this.environment.clone();
69  			URI uri = URI.create((String) i.next());
70  			rmiEnv.put("host", uri.getHost());
71  			rmiEnv.put("port", uri.getPort());
72  			contexts.add(new RMIContext(rmiEnv));
73  		  }
74  		}
75  
76  		Hashtable httpEnv = (Hashtable) this.environment.clone();
77  		contexts.add(new HTTPContext(httpEnv));
78  
79  		String className = null;
80  		for (Iterator i = org.apache.oodt.commons.util.Utility.parseCommaList(System.getProperty("org.apache.oodt.commons.object.contexts", ""));
81  		        i.hasNext();) {
82  		  try {
83  			className = (String) i.next();
84  			Class clazz = Class.forName(className);
85  			contexts.add(clazz.newInstance());
86  		  } catch (ClassNotFoundException ex) {
87  			System.err.println("Ignoring not-found context class `" + className + "': " + ex.getMessage());
88  		  } catch (InstantiationException ex) {
89  			System.err.println("Ignoring non-instantiable context class `" + className + "': " + ex.getMessage());
90  		  } catch (IllegalAccessException ex) {
91  			System.err.println("Ignoring context class `" + className + "' with non-accessible no-args c'tor: "
92  							   + ex.getMessage());
93  		  }
94  		}
95  
96  		installAliases();
97  		System.err.println("Object context ready; delegating to: " + contexts);
98  	}

113 	/**
114 	 * Returns the object to which the given name is bound.  Because this context
115 	 * delegates to multiple other contexts, the lookup returns the first successful
116 	 * match.
117 	 *
118 	 * @param name a <code>String</code> value.
119 	 * @return an <code>Object</code> value.
120 	 * @throws NamingException if an error occurs.
121 	 */
122 	public Object lookup(String name) throws NamingException {
123 		if (name == null) {
124 		  throw new IllegalArgumentException("Name required");
125 		}
126 		if (name.length() == 0) {
127 		  return this;
128 		}
129 
130 		// Let alias redirection do its magic
131 		String alias = aliases.getProperty(name);
132 		if (alias != null) {
133 		  name = alias;
134 		}
135 
136 	  for (Object context : contexts) {
137 		Context c = (Context) context;
138 		try {
139 		  return c.lookup(name);
140 		} catch (InvalidNameException ignore) {
141 		} catch (NameNotFoundException ignore) {
142 		} catch (NamingException ignore) {
143 		}
144 	  }
145 		throw new NameNotFoundException(name + " not found in any managed subcontext");
146 	}

152 	public synchronized void bind(String name, Object obj) throws NamingException {
153 		if (name == null) {
154 		  throw new IllegalArgumentException("Name required");
155 		}
156 		if (name.length() == 0) {
157 		  throw new InvalidNameException("Cannot bind object named after context");
158 		}
159 
160 		// If it's an alias name, stop here.
161 		if (aliases.containsKey(name)) {
162 		  throw new NameAlreadyBoundException("Name \"" + name + "\" already bound as an aliased name");
163 		}
164 
165 		// Make sure it isn't bound anywhere
166 		for (NamingEnumeration e = list(""); e.hasMore();) {
167 			NameClassPair nameClassPair = (NameClassPair) e.next();
168 			if (name.equals(nameClassPair.getName())) {
169 			  throw new NameAlreadyBoundException("Name \"" + name + "\" already bound by a managed subcontext");
170 			}
171 		}
172 		doRebind(name, obj);
173 	}

181 		if (name == null) {
182 		  throw new IllegalArgumentException("Name required");
183 		}
184 		if (name.length() == 0) {
185 		  throw new InvalidNameException("Cannot rebind object named after context");
186 		}
187 
188 		// If it's an alias name, remove the alias
189 		if (aliases.containsKey(name)) {
190 		  aliases.remove(name);
191 		}
192 
193 		doRebind(name, obj);
194 	}

196 	/**
197 	 * Rebind the given name to the given object.
198 	 *
199 	 * @param name Name to rebind
200 	 * @param obj Object to which it's bound
201 	 * @throws NamingException if an error occurs.
202 	 */
203 	private void doRebind(String name, Object obj) throws NamingException {
204 		boolean bound = false;
205 	  for (Object context : contexts) {
206 		Context c = (Context) context;
207 		try {
208 		  c.rebind(name, obj);
209 		  bound = true;
210 		} catch (NamingException ignored) {
211 		}
212 	  }
213 		if (!bound) {
214 		  throw new InvalidNameException("Name \"" + name + "\" not compatible with any managed subcontext");
215 		}
216 	}

254 	public void rename(String oldName, String newName) throws NamingException {
255 		if (oldName == null || newName == null) {
256 		  throw new IllegalArgumentException("Name required");
257 		}
258 		if (oldName.length() == 0 || newName.length() == 0) {
259 		  throw new InvalidNameException("Cannot rename object named after context");
260 		}
261 
262 		// See if it's an aliased name
263 		String oldValue = (String) aliases.remove(oldName);
264 		if (oldValue != null) {
265 			aliases.setProperty(newName, oldName);
266 			return;
267 		}
268 
269 		boolean renamed = false;
270 	  for (Object context : contexts) {
271 		Context c = (Context) context;
272 		try {
273 		  c.rename(oldName, newName);
274 		  renamed = true;
275 		} catch (NamingException ignore) {
276 		}
277 	  }
278 		if (!renamed) {
279 		  throw new InvalidNameException("Names not compatible with any managed subcontext");
280 		}
281 	}

287 	public NamingEnumeration list(final String name) throws NamingException {
288 		final Iterator eachContext = contexts.iterator();
289 		return new NamingEnumeration() {
290 			private NamingEnumeration enumeration
291 				= eachContext.hasNext()? ((Context) eachContext.next()).list(name) : null;
292 			private boolean open = true;
293 			public Object next() throws NamingException {
294 				if (!open) {
295 				  throw new NamingException("closed");
296 				}
297 				if (enumeration != null && enumeration.hasMore()) {
298 				  return enumeration.next();
299 				} else if (eachContext.hasNext()) {
300 					enumeration = ((Context) eachContext.next()).list(name);
301 					if (enumeration.hasMore()) {
302 					  return enumeration.next();
303 					}
304 				}
305 				throw new NoSuchElementException("No more objects in context");
306 			}
307 			public Object nextElement() {
308 				Object rc = null;
309 				try {
310 					rc = next();
311 				} catch (NamingException ignore) {}
312 				return rc;
313 			}
314 			public boolean hasMore() throws NamingException {
315 				if (!open) {
316 				  return false;
317 				}
318 				if (enumeration == null) {
319 				  return false;
320 				} else if (enumeration.hasMore()) {
321 				  return true;
322 				} else if (eachContext.hasNext()) {
323 					enumeration = ((Context) eachContext.next()).list(name);
324 					return hasMore();
325 				}
326 				return false;
327 			}
328 			public boolean hasMoreElements() {
329 				boolean h = false;
330 				try {
331 					h = hasMore();
332 				} catch (NamingException ignore) {}
333 				return h;
334 			}
335 			public void close() throws NamingException {
336 				open = false;
337 				if (enumeration != null) {
338 				  enumeration.close();
339 				}
340 			}
341 		};
342 	}

481 	/**
482 	 * Install aliases specified in the properties file.  The properties file simply
483 	 * maps a string object name to a new object name.  Use the system property
484 	 * <code>org.apache.oodt.commons.object.jndi.aliases</code> (preferred) or simply
485 	 * <code>aliases</code> to tell the location of the properties file.
486 	 */
487 	private void installAliases() {
488 		String aliasFileName = System.getProperty("org.apache.oodt.commons.object.jndi.aliases", System.getProperty("aliases"));
489 		if (aliasFileName != null && aliasFileName.length() > 0) {
490 			FileInputStream in = null;
491 			try {
492 				in = new FileInputStream(aliasFileName);
493 				aliases.load(in);
494 			} catch (IOException ex) {
495 				throw new IllegalStateException("Cannot handle I/O exception reading alias file " + aliasFileName
496 					+ ": " + ex.getMessage());
497 			} finally {
498 				if (in != null) {
499 				  try {
500 					in.close();
501 				  } catch (IOException ignore) {
502 				  }
503 				}
504 			}
505 		}
506 	}
