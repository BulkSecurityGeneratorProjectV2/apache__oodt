44  public class WorkflowTaskViewer extends Panel {
45  
46    private static final long serialVersionUID = -1895109076537364697L;
47  
48    private WorkflowMgrConn wm;
49  
50    /**
51     * @param id
52     */
53    public WorkflowTaskViewer(String id, String wmUrlStr, String taskId, final Class<? extends WebPage> conditionPage) {
54      super(id);
55      this.wm = new WorkflowMgrConn(wmUrlStr);
56      final WorkflowTask task = this.wm.safeGetTaskById(taskId);
57  
58      add(new Label("workflow_task_id", task.getTaskId()));
59      add(new Label("workflow_task_name", task.getTaskName()));
60      add(new Label("workflow_task_class", task.getTaskInstanceClassName()));
61  
62      List<String> taskConfigMetKeyNames = Arrays.asList(task.getTaskConfig()
63          .getProperties().keySet().toArray(
64              new String[task.getTaskConfig().getProperties().size()]));
65      Collections.sort(taskConfigMetKeyNames);
66  
67      add(new ListView<String>("workflow_config", new ListModel<String>(taskConfigMetKeyNames)){
68      
69        @Override
70        protected void populateItem(ListItem<String> item) {
71          String configMetKeyName = item.getModelObject();
72          String configMetKeyValue = task.getTaskConfig().getProperty(configMetKeyName);
73          
74          item.add(new Label("workflow_config_pname", configMetKeyName));
75          item.add(new Label("workflow_config_pvalue", configMetKeyValue));
76        }
77      });
78      
79      add(new ListView<WorkflowCondition>("workflow_conditions", new ListModel<WorkflowCondition>(task.getConditions())){
80        /* (non-Javadoc)
81         * @see org.apache.wicket.markup.html.list.ListView#populateItem(org.apache.wicket.markup.html.list.ListItem)
82         */
83        @Override
84        protected void populateItem(ListItem<WorkflowCondition> item) {
85           WorkflowCondition cond = item.getModelObject();
86           final PageParameters params = new PageParameters();
87           params.add("id", cond.getConditionId());
88           Link condLink = new Link("condition_page_link"){
89                 /* (non-Javadoc)
90                 * @see org.apache.wicket.markup.html.link.Link#onClick()
91                 */
92                @Override
93                public void onClick() {
94                  setResponsePage(conditionPage, params);
95                }
96            };
97            condLink.add(new Label("condition_plink_name", cond.getConditionName()));
98            item.add(condLink);
99        }
100     });
101   }
