43  /**
44   * {@link DataTransfer} which put/gets files in/from Amazon S3 storage.
45   * 
46   * @author bfoster@apache.org (Brian Foster)
47   */
48  public class S3DataTransferer implements DataTransfer {
49  
50  	private final AmazonS3 s3Client;
51  	private final String bucketName;
52  	private final boolean encrypt;
53  
54  	public S3DataTransferer(AmazonS3 s3Client, String bucketName, boolean encrypt) {
55  		this.s3Client = checkNotNull(s3Client);
56  		this.bucketName = checkNotNull(bucketName);
57  		this.encrypt = encrypt;
58  	}
59  
60  	@Override
61  	public void setFileManagerUrl(URL url) {}
62  
63  	@Override
64  	public void transferProduct(Product product) throws DataTransferException, IOException {
65  		for (Reference ref : product.getProductReferences()) {
66        String origRef = stripProtocol(ref.getOrigReference(), false);
67  		  String dataStoreRef = stripProtocol(ref.getDataStoreReference(), true);
68  			try {
69  			  PutObjectRequest request = new PutObjectRequest(
70  			      bucketName, dataStoreRef, new File(origRef));
71  			  if (encrypt) {
72    				ObjectMetadata requestMetadata = new ObjectMetadata();
73    				requestMetadata.setServerSideEncryption(AES_256_SERVER_SIDE_ENCRYPTION);     
74    				request.setMetadata(requestMetadata);
75  			  }
76          s3Client.putObject(request);
77  			} catch (AmazonClientException e) {
78  				throw new DataTransferException(String.format(
79  				    "Failed to upload product reference %s to S3 at %s", origRef,
80  				    dataStoreRef), e);
81  			}
82  		}
83  	}
84  
85  	@Override
86  	public void retrieveProduct(Product product, File directory) throws DataTransferException,
87  	    IOException {
88  		for (Reference ref : product.getProductReferences()) {
89        GetObjectRequest request = new GetObjectRequest(bucketName, stripProtocol(
90            ref.getDataStoreReference(), true));
91  			S3Object file = s3Client.getObject(request);
92  			stageFile(file, ref, directory);
93  		}
94  	}
95  
96    @Override
97    public void deleteProduct(Product product) throws DataTransferException, IOException {
98      for (Reference ref : product.getProductReferences()) {
99        DeleteObjectRequest request = new DeleteObjectRequest(bucketName, stripProtocol(
100           ref.getDataStoreReference(), true));
101       try {
102         s3Client.deleteObject(request);
103       } catch (AmazonClientException e) {
104         throw new DataTransferException(String.format(
105             "Failed to delete product reference %s from S3", ref.getDataStoreReference()), e);
106       }
107     }
108   }
109 
110 	private void stageFile(S3Object file, Reference ref, File directory) throws IOException {
111 		S3ObjectInputStream inStream = null;
112 		FileOutputStream outStream = null;
113 		try {
114 			inStream = file.getObjectContent();
115 			outStream = new FileOutputStream(new File(directory, new File(
116 			    stripProtocol(ref.getDataStoreReference(), false)).getName()));
117 			IOUtils.copy(inStream, outStream);
118 		} finally {
119 			try { inStream.close(); } catch (Exception ignored) {}
120 			try { outStream.close(); } catch (Exception ignored) {}
121 		}
122 	}
123 
124 	private String stripProtocol(String ref, boolean stripLeadingSeparator) throws IOException {
125 	  try {
126       String path = new URI(ref).getPath();
127       if (stripLeadingSeparator) {
128         path = path.replaceAll("^/", "");
129       }
130       return path;
131     } catch (URISyntaxException e) {
132       throw new IOException(e);
133     }
134 	}
