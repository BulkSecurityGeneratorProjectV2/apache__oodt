51  /**
52   * An instance of the {@link WorkflowProcessorThread} that processes through an
53   * iterative {@link WorkflowInstance}. This class keeps an <code>Iterator</code>
54   * that allows it to move from one end of a sequential {@link Workflow}
55   * processing pipeline to another. This class should only be used to process
56   * science pipeline style {@link Workflow}s, i.e., those which resemble an
57   * iterative processing pipelines, with no forks, or concurrent task executions.
58   * 
59   * @author mattmann
60   * @version $Revision$
61   * 
62   */
63

115   public IterativeWorkflowProcessorThread(WorkflowInstance wInst,
116       WorkflowInstanceRepository instRep, URL wParentUrl) {
117     workflowInst = wInst;
118     taskIterator = workflowInst.getWorkflow().getTasks().iterator();
119     this.instanceRepository = instRep;
120 
121     /* start out the gates running */
122     running = true;
123 
124     /*
125      * get the amount of wait time inbetween checking for task pre-condition
126      * satisfaction
127      */
128     waitForConditionSatisfy = Long.getLong(
129         "org.apache.oodt.cas.workflow.engine.preConditionWaitTime", 10);
130 
131     pollingWaitTime = Long.getLong(
132         "org.apache.oodt.cas.workflow.engine.resourcemgr.pollingWaitTime", 10);
133 
134     wmgrParentUrl = wParentUrl;
135   }

137   /*
138    * (non-Javadoc)
139    * 
140    * @see java.lang.Runnable#run()
141    */
142   public void run() {
143     /*
144      * okay, we got into the run method, mark the start date time for the
145      * workflow instance here
146      */
147     String startDateTimeIsoStr = DateConvert.isoFormat(new Date());
148     workflowInst.setStartDateTimeIsoStr(startDateTimeIsoStr);
149     // persist it
150     persistWorkflowInstance();
151 
152     while (running && taskIterator.hasNext()) {
153       if (pause) {
154         LOG.log(Level.FINE,
155             "IterativeWorkflowProcessorThread: Skipping execution: Paused: CurrentTask: "
156                 + getTaskNameById(workflowInst.getCurrentTaskId()));
157         continue;
158       }
159 
160       WorkflowTask task = (WorkflowTask) taskIterator.next();
161       workflowInst.setCurrentTaskId(task.getTaskId());
162 
163       // now persist it
164       persistWorkflowInstance();
165 
166       // check to see if req met fields are present
167       // if they aren't, set the status to METERROR, and then fail
168       if (!checkTaskRequiredMetadata(task, this.workflowInst.getSharedContext())) {
169         this.workflowInst.setStatus(METADATA_MISSING);
170         persistWorkflowInstance();
171         // now break out of this run loop
172         return;
173       }
174 
175       // this is where the pre-conditions come in
176       // only execute the below code when it's passed all of its
177       // pre-conditions
178       if (task.getConditions() != null) {
179         while (!satisfied(task.getConditions(), task.getTaskId())
180             && !isStopped()) {
181 
182           // if we're not paused, go ahead and pause us now
183           if (!isPaused()) {
184             pause();
185           }
186 
187           LOG.log(Level.FINEST,
188               "Pre-conditions for task: " + task.getTaskName()
189                   + " unsatisfied: waiting: " + waitForConditionSatisfy
190                   + " seconds before checking again.");
191           try {
192             Thread.currentThread().sleep(waitForConditionSatisfy * 1000);
193           } catch (InterruptedException ignore) {
194           }
195 
196           // check to see if we've been resumed, if so, break
197           // the loop and start
198           if (!isPaused()) {
199             break;
200           }
201         }
202 
203         // check to see if we've been killed
204         if (isStopped()) {
205           break;
206         }
207 
208         // un pause us (if needed)
209         if (isPaused()) {
210           resume();
211         }
212       }
213 
214       // task execution
215       LOG.log(
216           Level.FINEST,
217           "IterativeWorkflowProcessorThread: Executing task: "
218               + task.getTaskName());
219 
220       WorkflowTaskInstance taskInstance = GenericWorkflowObjectFactory
221           .getTaskObjectFromClassName(task.getTaskInstanceClassName());
222       // add the TaskId and the JobId and ProcessingNode
223       // TODO: unfake the JobId
224       workflowInst.getSharedContext()
225           .replaceMetadata(TASK_ID, task.getTaskId());
226       workflowInst.getSharedContext().replaceMetadata(WORKFLOW_INST_ID,
227           workflowInst.getId());
228       workflowInst.getSharedContext().replaceMetadata(JOB_ID,
229           workflowInst.getId());
230       workflowInst.getSharedContext().replaceMetadata(PROCESSING_NODE,
231           getHostname());
232       workflowInst.getSharedContext().replaceMetadata(WORKFLOW_MANAGER_URL,
233           this.wmgrParentUrl.toString());
234       workflowInst.getSharedContext().replaceMetadata(WORKFLOW_ID,
235               workflowInst.getParentChildWorkflow().getId());
236       workflowInst.getSharedContext().replaceMetadata(WORKFLOW_NAME,
237               workflowInst.getParentChildWorkflow().getName());
238 
239 
240       if (rClient != null) {
241         // build up the Job
242         // and the Job Input
243         Job taskJob = new Job();
244         taskJob.setName(task.getTaskId());
245         taskJob
246             .setJobInstanceClassName("org.apache.oodt.cas.workflow.structs.TaskJob");
247         taskJob
248             .setJobInputClassName("org.apache.oodt.cas.workflow.structs.TaskJobInput");
249         taskJob.setLoadValue(task.getTaskConfig().getProperty(TASK_LOAD) != null ? 
250             Integer.parseInt(task.getTaskConfig().getProperty(TASK_LOAD)): 2);
251         taskJob
252             .setQueueName(task.getTaskConfig().getProperty(QUEUE_NAME) != null ? task
253                 .getTaskConfig().getProperty(QUEUE_NAME) : DEFAULT_QUEUE_NAME);
254 
255         TaskJobInput in = new TaskJobInput();
256         in.setDynMetadata(workflowInst.getSharedContext());
257         in.setTaskConfig(task.getTaskConfig());
258         in.setWorkflowTaskInstanceClassName(task.getTaskInstanceClassName());
259 
260         workflowInst.setStatus(RESMGR_SUBMIT);
261         persistWorkflowInstance();
262 
263         try {
264           // this is * NOT * a blocking operation so when it returns
265           // the job may not actually have finished executing
266           // so we go into a waiting/sleep behavior using the passed
267           // back job id to wait until the job has actually finished
268           // executing
269 
270           this.currentJobId = rClient.submitJob(taskJob, in);
271 
272           while (!safeCheckJobComplete(this.currentJobId) && !isStopped()) {
273             // sleep for 5 seconds then come back
274             // and check again
275             try {
276               Thread.currentThread().sleep(pollingWaitTime * 1000);
277             } catch (InterruptedException ignore) {
278             }
279           }
280 
281           // okay job is done: TODO: fix this hack
282           // the task update time was set remotely
283           // by remote task, so let's read it now
284           // from the instRepo (which will have the updated
285           // time)
286 
287           if (isStopped()) {
288             // this means that this workflow was killed, so
289             // gracefully exit
290             break;
291           }
292 
293           WorkflowInstance updatedInst;
294           try {
295             updatedInst = instanceRepository
296                 .getWorkflowInstanceById(workflowInst.getId());
297             workflowInst = updatedInst;
298           } catch (InstanceRepositoryException e) {
299             LOG.log(Level.SEVERE, e.getMessage());
300             LOG.log(Level.WARNING, "Unable to get " + "updated workflow "
301                 + "instance record " + "when executing remote job: Message: "
302                 + e.getMessage());
303           }
304 
305         } catch (JobExecutionException e) {
306           LOG.log(Level.WARNING,
307               "Job execution exception using resource manager to execute job: Message: "
308                   + e.getMessage());
309         }
310       } else {
311         // we started, so mark it
312         workflowInst.setStatus(STARTED);
313         // go ahead and persist the workflow instance, after we
314         // save the current task start date time
315         String currentTaskIsoStartDateTimeStr = DateConvert
316             .isoFormat(new Date());
317         workflowInst
318             .setCurrentTaskStartDateTimeIsoStr(currentTaskIsoStartDateTimeStr);
319         workflowInst.setCurrentTaskEndDateTimeIsoStr(null); /*
320                                                              * clear this out
321                                                              * until it's ready
322                                                              */
323         persistWorkflowInstance();
324         executeTaskLocally(taskInstance, workflowInst.getSharedContext(),
325             task.getTaskConfig(), task.getTaskName());
326         String currentTaskIsoEndDateTimeStr = DateConvert.isoFormat(new Date());
327         workflowInst
328             .setCurrentTaskEndDateTimeIsoStr(currentTaskIsoEndDateTimeStr);
329         persistWorkflowInstance();
330       }
331 
332       LOG.log(
333           Level.FINEST,
334           "IterativeWorkflowProcessorThread: Completed task: "
335               + task.getTaskName());
336 
337     }
338 
339     LOG.log(Level.FINEST,
340         "IterativeWorkflowProcessorThread: Completed workflow: "
341             + workflowInst.getWorkflow().getName());
342     if (!isStopped()) {
343       stop();
344     }
345 
346   }

352   public synchronized void stop() {
353     running = false;
354     // if the resource manager is active
355     // then kill the current job there
356     if (this.rClient != null && this.currentJobId != null) {
357       if (!this.rClient.killJob(this.currentJobId)) {
358         LOG.log(Level.WARNING, "Attempt to kill " + "current resmgr job: ["
359             + this.currentJobId + "]: failed");
360       }
361     }
362 
363     workflowInst.setStatus(FINISHED);
364     String isoEndDateTimeStr = DateConvert.isoFormat(new Date());
365     workflowInst.setEndDateTimeIsoStr(isoEndDateTimeStr);
366     persistWorkflowInstance();
367   }

471   private boolean checkTaskRequiredMetadata(WorkflowTask task,
472       Metadata dynMetadata) {
473     if (task.getRequiredMetFields() == null || (task.getRequiredMetFields()
474                                                     .size() == 0)) {
475       LOG.log(Level.INFO, "Task: [" + task.getTaskName()
476           + "] has no required metadata fields");
477       return true; /* no required metadata, so we're fine */
478     }
479 
480     for (Object o : task.getRequiredMetFields()) {
481       String reqField = (String) o;
482       if (!dynMetadata.containsKey(reqField)) {
483         LOG.log(Level.SEVERE, "Checking metadata key: [" + reqField
484                               + "] for task: [" + task.getTaskName()
485                               + "]: failed: aborting workflow");
486         return false;
487       }
488     }
489 
490     LOG.log(Level.INFO, "All required metadata fields present for task: ["
491         + task.getTaskName() + "]");
492 
493     return true;
494   }

506   private boolean satisfied(List conditionList, String taskId) {
507     for (Object aConditionList : conditionList) {
508       WorkflowCondition c = (WorkflowCondition) aConditionList;
509       WorkflowConditionInstance cInst;
510 
511       // see if we've already cached this condition instance
512       if (CONDITION_CACHE.get(taskId) != null) {
513         ConcurrentHashMap conditionMap = (ConcurrentHashMap) CONDITION_CACHE.get(taskId);
514 
515         /*
516          * okay we have some conditions cached for this task, see if we have the
517          * one we need
518          */
519         if (conditionMap.get(c.getConditionId()) != null) {
520           cInst = (WorkflowConditionInstance) conditionMap.get(c
521               .getConditionId());
522         }
523         /* if not, then go ahead and create it and cache it */
524         else {
525           cInst = GenericWorkflowObjectFactory
526               .getConditionObjectFromClassName(c
527                   .getConditionInstanceClassName());
528           conditionMap.put(c.getConditionId(), cInst);
529         }
530       }
531       /* no conditions cached yet, so set everything up */
532       else {
533         ConcurrentHashMap conditionMap = new ConcurrentHashMap();
534         cInst = GenericWorkflowObjectFactory.getConditionObjectFromClassName(c
535             .getConditionInstanceClassName());
536         conditionMap.put(c.getConditionId(), cInst);
537         CONDITION_CACHE.put(taskId, conditionMap);
538       }
539 
540       // actually perform the evaluation
541       if (!cInst.evaluate(workflowInst.getSharedContext(), c.getTaskConfig())) {
542         return false;
543       }
544     }
545 
546     return true;
547   }

569   private void executeTaskLocally(WorkflowTaskInstance instance, Metadata met,
570       WorkflowTaskConfiguration cfg, String taskName) {

581   private boolean safeCheckJobComplete(String jobId) {
582     try {
583       return rClient.isJobComplete(jobId);
