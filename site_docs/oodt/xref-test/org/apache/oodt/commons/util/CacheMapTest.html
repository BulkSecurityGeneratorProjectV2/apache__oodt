16  package org.apache.oodt.commons.util;
17  
18  import junit.framework.*;
19  
20  /** Unit test the {@link CacheMap} class.
21   *
22   * @author Kelly
23   */ 
24  public class CacheMapTest extends TestCase {
25  	/** Construct the test case for the {@link CacheMap} class. */
26  	public CacheMapTest(String name) {
27  		super(name);
28  	}
29  
30  	/** Test the caching operation. */
31  	public void testCache() {
32  		CacheMap c = new CacheMap(3);
33  		c.put("alpha", "1");
34  		c.put("beta", "2");
35  		c.put("gamma", "3");
36  		// gamma beta alpha
37  		assertEquals("2", c.get("beta"));
38  		// beta gamma alpha
39  		c.put("delta", "4");
40  		// delta beta gamma
41  		assertNull(c.get("alpha"));
42  	}
43  
44  	/** Test the {@link CacheMap#size} and {@link CacheMap#isEmpty} methods. */
45  	public void testSize() {
46  		CacheMap c = new CacheMap(3);
47  		assertEquals(0, c.size());
48  		assertTrue(c.isEmpty());
49  		c.put("alpha", "1");
50  		// alpha
51  		assertEquals(1, c.size());
52  		c.put("beta", "2");
53  		// beta alpha
54  		assertEquals(2, c.size());
55  		c.put("gamma", "3");
56  		// gamma beta alpha
57  		assertEquals(3, c.size());
58  		c.put("delta", "4");
59  		// delta gamma beta
60  		assertEquals(3, c.size());
61  		c.clear();
62  		assertTrue(c.isEmpty());
63  	}
64  
65  	/** Test the {@link CacheMap#containsKey} and {@link CacheMap#containsValue} methods. */
66  	public void testContains() {
67  		CacheMap c = new CacheMap(3);
68  		c.put("alpha", "1");
69  		c.put("beta", "2");
70  		c.put("gamma", "3");
71  
72  		assertTrue(c.containsKey("alpha"));
73  		assertTrue(!c.containsKey("hungus"));
74  		assertTrue(c.containsValue("2"));
75  		assertTrue(!c.containsValue("x"));
76  	}
77  
78  	/** Test value replacement for the same key. */
79  	public void testRePut() {
80  		CacheMap c = new CacheMap(3);
81  		c.put("alpha", "1");
82  		c.put("beta", "2");
83  		c.put("gamma", "3");
84  		// (gamma, 3) (beta, 2) (alpha, 1)
85  		c.put("alpha", "x");
86  		// (alpha, x), (gamma, 3) (beta, 2)
87  		assertEquals("x", c.get("alpha"));
88  		// (alpha, x), (gamma, 3) (beta, 2)
89  		c.put("delta", "y");
90  		// (delta, y) (alpha, x), (gamma, 3)
91  		assertEquals(3, c.size());
92  		assertNull(c.get("beta"));
93  	}
94  
95  	/** Test the {@link CacheMap#remove} method. */
96  	public void testRemove() {
97  		CacheMap c = new CacheMap(3);
98  		c.put("alpha", "1");
99  		c.put("beta", "2");
100 		c.put("gamma", "3");
101 		// gamma beta alpha
102 		c.put("delta", "4");
103 		// delta gamma beta
104 		assertEquals("3", c.remove("gamma"));
105 		// delta beta
106 		assertNull(c.remove("gamma"));
107 		assertNull(c.remove("alpha"));
108 	}
109 }
