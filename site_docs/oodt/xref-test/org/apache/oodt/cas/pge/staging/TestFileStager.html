112    public void testAsURI() throws URISyntaxException {
113       String absoluteHttpUri = "http://somewhere.com/path/to/data.dat";
114       String absoluteFileUri = "file:///path/to/data.dat";
115       String relativePath = "path/to/data.dat";
116       String absolutePath = "/path/to/data.dat";
117 
118       assertEquals("http://somewhere.com/path/to/data.dat",
119             FileStager.asURI(absoluteHttpUri).toString());
120       assertEquals("file:///path/to/data.dat", FileStager
121             .asURI(absoluteFileUri).toString());
122       assertEquals("file://" + new File("").getAbsolutePath()
123             + "/path/to/data.dat", FileStager.asURI(relativePath).toString());
124       assertEquals("file:///path/to/data.dat", FileStager.asURI(absolutePath)
125             .toString());
126    }

128    public static class ProductIdMatcher implements IArgumentMatcher {
129 
130       private String productId;
131 
132       public ProductIdMatcher(String productId) {
133          this.productId = productId;
134       }
135 
136       @Override
137       public void appendTo(StringBuffer buffer) {
138          buffer.append("eqProduct(");
139          buffer.append(Product.class.getName());
140          buffer.append(" with product id [");
141          buffer.append(productId).append("])");
142       }
143 
144       @Override
145       public boolean matches(Object obj) {
146          if (obj instanceof Product) {
147             return productId.equals(((Product) obj).getProductId());
148          }
149          return false;
150       }
151 
152       public static Product eqProductId(String productId) {
153          reportMatcher(new ProductIdMatcher(productId));
154          return null;
155       }
156    }
