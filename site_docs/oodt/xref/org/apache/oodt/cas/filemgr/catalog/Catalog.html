34  /**
35   * @author mattmann
36   * @author bfoster
37   * @version $Revision$
38   * 
39   * <p>
40   * A Catalog is the front end object interface for a back end peristance layer
41   * for storing product references and metadata. Classes implementing this
42   * interface implement methods to retrieve and store product references and
43   * metadata from a particular backend repository.
44   * </p>
45   * 
46   */
47  public interface Catalog extends Pagination {
48  
49      String X_POINT_ID = Catalog.class.getName();

52       * <p>

54       * the given <code>productId</code>.
55       * </p>
56       * 
57       * @param m
58       *            The {@link Metadata} to ingest.
59       * @param product
60       *            The product to add the metadata for.
61       * @throws CatalogException
62       *             If any general error occurs.
63       */
64      void addMetadata(Metadata m, Product product)
65              throws CatalogException;
66  
67      /**
68       * <p>
69       * Removes {@link Metadata} from the metadata store, and disassociates it
70       * from the given <code>productId</code>.
71       * </p>
72       * 
73       * @param m
74       *            The {@link Metadata} to remove.
75       * @param product
76       *            The product for which the metadata is to be removed.

78       *             If any general error occurs.
79       */
80      void removeMetadata(Metadata m, Product product)

88       * @param product

91       *             If any error occurs during the add.

93      void addProduct(Product product) throws CatalogException;

95      /**

106     void modifyProduct(Product product) throws CatalogException;

108     /**

130     void setProductTransferStatus(Product product)

140      * @param product

146     void addProductReferences(Product product) throws CatalogException;

261     List<String> query(Query query, ProductType type) throws CatalogException;
262 
263     /**
264      * <p>
265      * Performs a query against the underlying {@link Catalog}, and then
266      * properly formulates a page of results to send back to the user. This
267      * method is useful when you would like to conserve memory and not send back
268      * the entire list of results, nor load them into memory. Of course, this
269      * method assumes that queries are deterministic, i.e., the same query

272      * </p>
273      * 
274      * @param query
275      *            The query to perform against the underlying Catalog.
276      * @param type
277      *            The {@link ProductType} that you are querying for.
278      * @param pageNum
279      *            The number of the {@link ProductPage} to return back to the
280      *            user.
281      * @return The requested {@link ProductPage} of results.
282      * @throws CatalogException
283      *             If any error occurs.
284      */
