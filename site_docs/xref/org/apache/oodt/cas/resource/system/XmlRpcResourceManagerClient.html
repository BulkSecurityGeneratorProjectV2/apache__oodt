73      /**
74       * <p>
75       * Constructs a new XmlRpcResourceManagerClient with the given
76       * <code>url</code>.
77       * </p>
78       * 
79       * @param url
80       *            The url pointer to the xml rpc resource manager service.
81       */
82      public XmlRpcResourceManagerClient(URL url) {
83          // set up the configuration, if there is any
84          if (System.getProperty("org.apache.oodt.cas.resource.properties") != null) {
85              String configFile = System
86                      .getProperty("org.apache.oodt.cas.resource.properties");
87              LOG.log(Level.INFO,
88                      "Loading Resource Manager Configuration Properties from: ["
89                              + configFile + "]");
90              try {
91                  System.getProperties().load(
92                          new FileInputStream(new File(configFile)));
93              } catch (Exception e) {
94                  LOG.log(Level.INFO,
95                          "Error loading configuration properties from: ["
96                                  + configFile + "]");
97              }
98  
99          }
100 
101         CommonsXmlRpcTransportFactory transportFactory = new CommonsXmlRpcTransportFactory(
102                 url);
103         int connectionTimeoutMins = Integer
104             .getInteger(
105                 "org.apache.oodt.cas.resource.system.xmlrpc.connectionTimeout.minutes",
106                 VAL);
107         int connectionTimeout = connectionTimeoutMins * INT * 1000;
108         int requestTimeoutMins = Integer
109             .getInteger(
110                 "org.apache.oodt.cas.resource.system.xmlrpc.requestTimeout.minutes",
111                 VAL1);
112         int requestTimeout = requestTimeoutMins * INT1 * 1000;
113         transportFactory.setConnectionTimeout(connectionTimeout);
114         transportFactory.setTimeout(requestTimeout);
115         client = new XmlRpcClient(url, transportFactory);
116         resMgrUrl = url;
117     }

526   public static String getReadableJobStatus(String status) {
527     if (status.equals(JobStatus.SUCCESS)) {
528       return "SUCCESS";
529     } else if (status.equals(JobStatus.FAILURE)) {
530       return "FAILURE";
531     } else if (status.equals(JobStatus.EXECUTED)) {
532       return "EXECUTED";
533     } else if (status.equals(JobStatus.QUEUED)) {
534       return "QUEUED";
535     } else if (status.equals(JobStatus.SCHEDULED)) {
536       return "SCHEDULED";
537     } else if (status.equals(JobStatus.KILLED)) {
538       return "KILLED";
539     }
540     else {
541         return null;
542     }
543   }
