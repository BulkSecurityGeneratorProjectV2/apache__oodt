84  	/**
85  	 * Constructor reads in the configuration and initiates the connection to the
86  	 * Solr instance.
87  	 * 
88  	 * @param solrUrl
89  	 *          URL for the Solr instance.
90  	 * @param fmUrl
91  	 *          URL for the File Manager instance.
92  	 */
93  	public SolrIndexer(String solrUrl, String fmUrl)
94  	    throws InstantiationException {
95  		InputStream input = null;
96  		String filename;
97  
98  		try {
99  			LOG.info("System property " + SOLR_INDEXER_CONFIG + " set to "
100 			    + System.getProperty(SOLR_INDEXER_CONFIG));
101 			filename = System.getProperty(SOLR_INDEXER_CONFIG);
102 			if (filename != null) {
103 				LOG.info("Reading config from " + filename);
104 				input = new FileInputStream(filename);
105 			} else {
106 				LOG.info("Config file not found, reading config from classpath");
107 				input = SolrIndexer.class.getResourceAsStream("/indexer.properties");
108 			}
109 			config = new IndexerConfig(input);
110 		} catch (IOException e) {
111 			LOG
112 			    .severe("Could not read in configuration for indexer from classpath or file");
113 			throw new InstantiationException(e.getMessage());
114 		} finally {
115 			if (input != null) {
116 				try {
117 					input.close();
118 				} catch (IOException e) {
119 					// no op
120 				}
121 			}
122 		}
123 
124 		this.solrUrl = solrUrl;
125 		if (this.solrUrl == null) {
126 			this.solrUrl = config.getProperty(SOLR_URL);
127 		}
128 
129 		this.fmUrl = fmUrl;
130 		if (this.fmUrl == null) {
131 			this.fmUrl = config.getProperty(FILEMGR_URL);
132 		}
133 
134 		LOG.info("Using Solr: " + this.solrUrl + " FileManager: " + this.fmUrl);
135 
136 
137 		try {
138 			server = new CommonsHttpSolrServer(this.solrUrl);
139 		} catch (MalformedURLException e) {
140 			LOG.severe("Could not connect to Solr server " + this.solrUrl);
141 			throw new InstantiationException(e.getMessage());
142 		}
143 
144 	}

167 	/**
168 	 * This method transforms the product metadata into a Solr document.
169 	 * 
170 	 * @param metadata
171 	 *          The metadata object for the product to index.
172 	 * @return Returns the SolrInputDocument containing product metadata.
173 	 */
174 	private SolrInputDocument getSolrDocument(Metadata metadata) {
175 		SolrInputDocument doc = new SolrInputDocument();
176 		// Only grab metadata which have a mapping in the indexer.properties
177 		for (Object objKey : config.getMapProperties().keySet()) {
178 			// The key in the metadata object
179 			String key = (String) objKey;
180 			// The solr field name this metadata key will be mapped to
181 			String fieldName = config.getMapProperties().getProperty(key);
182 			List<String> values = metadata.getAllMetadata(key);
183 			if (values != null) {
184 				for (String value : values) {
185 					// Add each metadata value into the
186 					if (value != null && !config.getIgnoreValues().contains(value.trim())) {
187 						LOG.fine("Adding field: " + fieldName + " value: " + value);
188 						doc.addField(fieldName, value);
189 					}
190 				}
191 			}
192 		}
193 		return doc;
194 	}

196 	/**
197 	 * This method adds a single product extracted from a metadata file to the
198 	 * Solr index.
199 	 * 
200 	 * @param file
201 	 *          The file containing product metadata.
202 	 * @param delete
203 	 *          Flag indicating whether the entry should be deleted from the
204 	 *          index.
205 	 * @throws SolrServerException
206 	 *           When an error occurs communicating with the Solr server instance.
207 	 */
208 	public void indexMetFile(File file, boolean delete)
209 	    throws
210 		SolrServerException {
211 		LOG.info("Attempting to index product from metadata file.");
212 		try {
213 			SerializableMetadata metadata = new SerializableMetadata("UTF-8", false);
214 			metadata.loadMetadataFromXmlStream(new FileInputStream(file));
215 			metadata.addMetadata("id", metadata.getMetadata("CAS."
216 			    + CoreMetKeys.PRODUCT_ID));
217 			metadata.addMetadata(config.getProperty(ACCESS_KEY), config
218 			    .getProperty(ACCESS_URL)
219 			    + metadata.getMetadata("CAS." + CoreMetKeys.PRODUCT_ID));
220 			if (delete) {
221 				server
222 				    .deleteById(metadata.getMetadata("CAS." + CoreMetKeys.PRODUCT_ID));
223 			}
224 			server.add(this.getSolrDocument(metadata));
225 			LOG.info("Indexed product: "
226 			    + metadata.getMetadata("CAS." + CoreMetKeys.PRODUCT_ID));
227 		} catch (InstantiationException e) {
228 			LOG.severe("Could not instantiate metadata object: " + e.getMessage());
229 		} catch (FileNotFoundException e) {
230 			LOG.severe("Could not find metadata file: " + e.getMessage());
231 		} catch (IOException e) {
232 			LOG.severe("Could not delete product from index: " + e.getMessage());
233 		}
234 	}

236 	/**
237 	 * This method indexes all product types retrieved from the File Manager to
238 	 * the Solr index.
239 	 * 
240 	 * @param delete
241 	 *          Flag indicating whether each product type retrieved from the File
242 	 *          Manager should be deleted from the index.
243 	 * @throws SolrServerException
244 	 *           When an error occurs communicating with the Solr server instance.
245 	 */
246 	public void indexProductTypes(boolean delete) {
247 		LOG.info("Indexing product types...");
248 		try {
249 			XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(new URL(
250 			    this.fmUrl));
251 			LOG.info("Retrieving list of product types.");
252 			List<ProductType> types = fmClient.getProductTypes();
253 			for (ProductType type : types) {
254 				if (!config.getIgnoreTypes().contains(type.getName().trim())) {
255 					Metadata metadata = new Metadata();
256 					metadata.addMetadata("id", type.getProductTypeId());
257 					metadata.addMetadata("CAS.ProductTypeId", type.getProductTypeId());
258 					metadata.addMetadata("CAS.ProductTypeDescription", type
259 					    .getDescription());
260 					metadata.addMetadata("CAS.ProductTypeRepositoryPath", type
261 					    .getProductRepositoryPath());
262 					metadata.addMetadata("CAS.ProductTypeVersioner", type.getVersioner());
263 					metadata.addMetadata("CAS.ProductTypeName", type.getName());
264 					metadata.addMetadata("ProductType", "ProductType");
265 					metadata.replaceMetadata(type.getTypeMetadata());
266 					if (delete) {
267 						try {
268 							server.deleteById(type.getProductTypeId());
269 						} catch (Exception e) {
270 							LOG.severe("Could not delete product type " + type.getName()
271 							    + " from index: " + e.getMessage());
272 						}
273 					}
274 					try {
275 						performSubstitution(metadata);
276 						server.add(this.getSolrDocument(metadata));
277 						LOG.info("Indexed product type: " + type.getName());
278 					} catch (Exception e) {
279 						LOG.severe("Could not index " + type.getName() + ": "
280 						    + e.getMessage());
281 					}
282 				} else {
283 					LOG.info("Ignoring product type: " + type.getName());
284 				}
285 			}
286 		} catch (MalformedURLException e) {
287 			LOG.severe("File Manager URL is malformed: " + e.getMessage());
288 		} catch (ConnectionException e) {
289 			LOG.severe("Could not connect to File Manager: " + e.getMessage());
290 		} catch (RepositoryManagerException e) {
291 			LOG.severe("Could not retrieve product types from File Manager: "
292 			    + e.getMessage());
293 		}
294 		LOG.info("Finished indexing product types.");
295 	}

310 	/**
311 	 * This method indexes all products retrieved from the File Manager to the
312 	 * Solr index. Metadata from the product's associated ProductType is also
313 	 * included.
314 	 * 
315 	 * @param delete
316 	 *          Flag indicating whether each product retrieved from the File
317 	 *          Manager should be deleted from the index.
318 	 * @throws SolrServerException
319 	 *           When an error occurs communicating with the Solr server instance.
320 	 */
321 	public void indexAll(boolean delete) {
322 		LOG.info("Indexing products...");
323 		try {
324 			XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(new URL(
325 			    this.fmUrl));
326 			LOG.info("Retrieving list of product types.");
327 			List<ProductType> types = fmClient.getProductTypes();
328 			for (ProductType type : types) {
329 				if (!config.getIgnoreTypes().contains(type.getName().trim())) {
330 					LOG.info("Paging through products for product type: "
331 					    + type.getName());
332 					ProductPage page = safeFirstPage(fmClient, type); 
333 					while (page != null) {
334 					    for (Product product : page.getPageProducts()) {
335 							try {
336 								this.indexProduct(product.getProductId(), fmClient
337 								    .getMetadata(product), type.getTypeMetadata());
338 							} catch (Exception e) {
339 								LOG.severe("Could not index " + product.getProductId() + ": "
340 								    + e.getMessage());
341 							}
342 						}
343 					    if (page.isLastPage()) {
344 					        break;
345 					    }
346 					    page = fmClient.getNextPage(type, page);
347 					}
348 				}
349 			}
350 			LOG.info("Finished indexing products.");
351 		} catch (MalformedURLException e) {
352 			LOG.severe("File Manager URL is malformed: " + e.getMessage());
353 		} catch (ConnectionException e) {
354 			LOG.severe("Could not connect to File Manager: " + e.getMessage());
355 		} catch (CatalogException e) {
356 			LOG.severe("Could not retrieve products from File Manager: "
357 			    + e.getMessage());
358 		} catch (RepositoryManagerException e) {
359 			LOG.severe("Could not retrieve product types from File Manager: "
360 			    + e.getMessage());
361 		}
362 	}

381 			Metadata productMetadata = fmClient.getMetadata(product);
382 			indexProduct(product.getProductId(), productMetadata, product
383 			    .getProductType().getTypeMetadata());
384 		} catch (MalformedURLException e) {
385 			LOG.severe("File Manager URL is malformed: " + e.getMessage());
386 		} catch (ConnectionException e) {
387 			LOG.severe("Could not connect to File Manager: " + e.getMessage());
388 		} catch (CatalogException e) {
389 			LOG.severe("Could not retrieve product from File Manager: "
390 			    + e.getMessage());
391 		} catch (java.text.ParseException e) {
392 			LOG.severe("Could not format date: " + e.getMessage());
393 		}

397 	 * This method adds a single product retrieved from the File Manager by its
398 	 * product name to the Solr index. Metadata from the ProductType is also
399 	 * included.
400 	 * 
401 	 * @param productName
402 	 *          The identifier of the product (CAS.ProductId).
403 	 * @param delete
404 	 *          Flag indicating whether the entry should be deleted from the
405 	 *          index.
406 	 * @throws SolrServerException
407 	 *           When an error occurs communicating with the Solr server instance.
408 	 */
409 	public void indexProductByName(String productName, boolean delete) throws SolrServerException {
410 		
411 		LOG.info("Attempting to index product: " + productName);
412 		try {
413 
414 			// Try to delete product by name
415 			// Note: the standard field "CAS.ProductName" must be mapped to some Solr field in file indexer.properties
416 			if (delete) {
417 				try {
418 					String productNameField = config.mapProperties.getProperty(PRODUCT_NAME);
419 					if (StringUtils.hasText(productNameField)) {
420 						server.deleteByQuery(productNameField+":"+productName);
421 					} else {
422 						LOG.warning("Metadata field "+PRODUCT_NAME+" is not mapped to any Solr field, cannot delete product by name");
423 					}
424 				} catch(Exception e) {
425 					LOG.warning("Could not delete product: "+productName+" from Solr index");
426 				}
427 			}
428 
429 			XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(new URL(
430 			    this.fmUrl));
431 			Product product = fmClient.getProductByName(productName);
432 			Metadata productMetadata = fmClient.getMetadata(product);
433 			// NOTE: delete (by id) is now false
434 			indexProduct(product.getProductId(), productMetadata, product.getProductType().getTypeMetadata());
435 			
436 		} catch (MalformedURLException e) {
437 			LOG.severe("File Manager URL is malformed: " + e.getMessage());
438 		} catch (ConnectionException e) {
439 			LOG.severe("Could not connect to File Manager: " + e.getMessage());
440 		} catch (CatalogException e) {
441 			LOG.severe("Could not retrieve product from File Manager: "
442 			    + e.getMessage());
443 		} catch (java.text.ParseException e) {
444 			LOG.severe("Could not format date: " + e.getMessage());
445 		}
446 	}

448 	private void indexProduct(String productId, Metadata productMetadata,
449 	    Metadata typeMetadata) throws SolrServerException,
450 	    java.text.ParseException {
451 		Metadata metadata = new Metadata();
452 		metadata.addMetadata("id", productId);
453 		// Add in product type metadata
454 		if (typeMetadata != null) {
455 			metadata.addMetadata(typeMetadata);
456 		}
457 		if (productMetadata != null) {
458 			String accessKey = config.getProperty(ACCESS_KEY);
459 			// Product metadata takes precedence
460 			metadata.replaceMetadata(productMetadata);
461 			// If there is an access url configured add it to the metadata
462 			if (config.getProperty(ACCESS_URL) != null) {
463 				metadata.addMetadata(accessKey, config.getProperty(ACCESS_URL));
464 			}
465 			// Replace values for metadata keys specified in config. This allows
466 			// for metadata substitution. For instance, if a key named "product_url"
467 			// has a value of
468 			// http://localhost:8080/cas-product/data?productID=[CAS.ProductId]
469 			// the value in brakets will be updated with the value from the
470 			// CAS.ProductId.
471 			performSubstitution(metadata);
472 			try {
473 				server.add(this.getSolrDocument(metadata));
474 				LOG.info("Indexed product: " + productId);
475 			} catch (IOException e) {
476 				LOG.severe("Could not index product: " + productId);
477 			}
478 		} else {
479 			LOG.info("Could not find metadata for product: " + productId);
480 		}
481 	}

522 	/**
523 	 * Quick helper method to do substitution on the keys specified in the config
524 	 * 
525 	 * @param metadata
526 	 *          to substitute on
527 	 * @throws java.text.ParseException
528 	 */
529 	private void performSubstitution(Metadata metadata)
530 	    throws java.text.ParseException {
531 		// Do metadata replacement
532 		for (String key : config.getReplacementKeys()) {
533 			List<String> values = metadata.getAllMetadata(key);
534 			if (values != null) {
535 				List<String> newValues = new ArrayList<String>();
536 				for (String value : values) {
537 					newValues.add(PathUtils.replaceEnvVariables(value, metadata));
538 				}
539 				metadata.removeMetadata(key);
540 				metadata.addMetadata(key, newValues);
541 			}
542 		}
543 		// Format dates
544 		for (Object key : config.getFormatProperties().keySet()) {
545 			String keyString = (String) key;
546 			if (metadata.containsKey(keyString)) {
547 				List<String> values = metadata.getAllMetadata(keyString);
548 				if (values != null) {
549 					List<String> newValues = new ArrayList<String>();
550 					SimpleDateFormat format = new SimpleDateFormat(config
551 					    .getFormatProperties().getProperty(keyString).trim());
552 					for (String value : values) {
553 						newValues.add(formatDate(format, value));
554 					}
555 					metadata.removeMetadata(keyString);
556 					metadata.addMetadata(keyString, newValues);
557 				}
558 			}
559 		}
560 	}
561 
562 	private String formatDate(SimpleDateFormat format, String value)
563 	    throws java.text.ParseException {
564 		// Ignore formating if its an ignore value
565 		if (config.getIgnoreValues().contains(value.trim())) {
566 		  return value;
567 		}
568 		return solrFormat.format(format.parse(value));
569 	}

572 	 * This method builds the command-line options.
573 	 * 
574 	 * @return Returns the supported Options.
575 	 */
576 	@SuppressWarnings("static-access")
577 	public static Options buildCommandLine() {
578 		Options options = new Options();
579 
580 		options.addOption(new Option("h", "help", false, "Print this message"));
581 		options.addOption(new Option("o", "optimize", false,
582 		    "Optimize the Solr index"));
583 		options.addOption(new Option("d", "delete", false,
584 		    "Delete item before indexing"));
585 		options.addOption(OptionBuilder.withArgName("Solr URL").hasArg()
586 		    .withDescription("URL to the Solr instance").withLongOpt("solrUrl")
587 		    .create("su"));
588 		options.addOption(OptionBuilder.withArgName("Filemgr URL").hasArg()
589 		    .withDescription("URL to the File Manager").withLongOpt("fmUrl")
590 		    .create("fmu"));
591 
592 		OptionGroup group = new OptionGroup();
593 		Option all = new Option("a", "all", false,
594 		    "Index all products from the File Manager");
595 		Option product = OptionBuilder.withArgName("productId").hasArg()
596 		    .withDescription("Index the product from the File Manager")
597 		    .withLongOpt("product").create("p");
598 		Option met = OptionBuilder.withArgName("file").hasArg().withDescription(
599 		    "Index the product from a metadata file").withLongOpt("metFile")

721 		// Used to hold general properties for indexer configuration

723 		// Used to hold mappings for filemanager -> solr for keys
724 		private Properties mapProperties = new Properties();
725 		// Used to define the date format for a field

728 		private List<String> ignoreValues = new ArrayList<String>();
729 		private List<String> replacementKeys = new ArrayList<String>();
730 
731 		public IndexerConfig(InputStream inputStream) throws IOException {
732 			Properties props = new Properties();
733 			props.load(inputStream);
734 			for (Object objKey : props.keySet()) {
735 				String key = (String) objKey;
736 				if (key.startsWith(PREFIX_CONFIG)) {
737 					properties.put(key.substring(PREFIX_CONFIG.length()), props
738 					    .getProperty(key));
739 				} else if (key.startsWith(PREFIX_MET)) {
740 					mapProperties.put(key.substring(PREFIX_MET.length()), props
741 					    .getProperty(key));
742 				} else if (key.startsWith(PREFIX_FORMAT)) {
743 					formatProperties.put(key.substring(PREFIX_FORMAT.length()), props
744 					    .getProperty(key));
745 				}
746 			}
747 
748 			if (properties.getProperty(IGNORE_TYPES) != null) {
749 				String[] values = properties.getProperty(IGNORE_TYPES).trim()
750 				    .split(",");
751 			  Collections.addAll(ignoreTypes, values);
752 			}
753 
754 			if (properties.getProperty(IGNORE_VALUES) != null) {
755 				String[] values = properties.getProperty(IGNORE_VALUES).trim().split(
756 				    ",");
757 			  Collections.addAll(ignoreValues, values);
758 			}
759 
760 			if (properties.getProperty(REPLACEMENT_KEYS) != null) {
761 				String[] values = properties.getProperty(REPLACEMENT_KEYS).trim()
762 				    .split(",");
763 			  Collections.addAll(replacementKeys, values);
764 			}
765 		}
