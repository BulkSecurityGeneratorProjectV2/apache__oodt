21  import java.net.MalformedURLException;
22  import java.net.URL;
23  import java.util.Arrays;
24  import java.util.Date;
25  import java.util.Iterator;
26  
27  //OODT imports
28  import org.apache.oodt.cas.metadata.Metadata;
29  import org.apache.oodt.cas.resource.structs.Job;
30  import org.apache.oodt.cas.resource.structs.exceptions.JobQueueException;
31  import org.apache.oodt.cas.resource.system.XmlRpcResourceManagerClient;
32  import org.apache.oodt.cas.workflow.structs.WorkflowConditionConfiguration;
33  import org.apache.oodt.cas.workflow.structs.WorkflowConditionInstance;
34  
35  /**
36   * @author singhk
37   * @version $Revision$
38   *
39   * <p>A Simple condition that evaluates to false as many times as the prerequisite job in Resource Manager
40   * isn't completed. After that, the condition returns
41   * true.</p>
42   *
43   */
44  public class PrerequisiteCondition implements WorkflowConditionInstance {
45  
46  	public PrerequisiteCondition() {
47  		super();
48  	}
49  	
50  	/**
51  	 * Check if there is any job queued in Resource Manager with name equivalent to <code>jobName</code>
52  	 * @param client
53  	 * @param jobName
54  	 * @return true if job is present, otherwise false
55  	 */
56  	@SuppressWarnings("unchecked")
57  	private boolean isJobPresentInQueue(XmlRpcResourceManagerClient client, String jobName){
58  		Iterator<Job> iter;
59  		try {
60  			iter = (Iterator<Job>)client.getQueuedJobs().iterator();
61  			while(iter.hasNext()){
62  				Job job = iter.next();
63  				if(job.getName().equals(jobName))
64  					return true;
65  			}
66  		} catch (JobQueueException e) {
67  			e.printStackTrace();
68  		}
69  		return false;
70  	}
71  
72  	/* (non-Javadoc)
73  	 * @see org.apache.oodt.cas.workflow.structs.WorkflowConditionInstance#evaluate(org.apache.oodt.cas.metadata.Metadata)
74  	 */
75      public boolean evaluate(Metadata metadata, WorkflowConditionConfiguration config) {
76  		
77  		String[] jobs = (config.getProperty("JOBS")).split(",");
78  		String resourceManagerUrl = config.getProperty("RESMGR_URL");
79  		boolean flag = true;
80  
81  		System.out.println(new Date() + " PrerequisiteCondition: Jobs: "+ Arrays.toString(jobs));
82  		System.out.println(new Date() + " PrerequisiteCondition: Resource Manager: "+ resourceManagerUrl);
83  		
84  		XmlRpcResourceManagerClient client = null;
85  		try {
86  			client = new XmlRpcResourceManagerClient(new URL(resourceManagerUrl));
87  			flag = true;
88  			for (String job : jobs)
89  				flag = flag && !isJobPresentInQueue(client, job);
90  			if(flag)
91  				return true;
92  		} catch (MalformedURLException e) {
93  			e.printStackTrace();
94  		}
95  		return false;
96  	}
97  }
