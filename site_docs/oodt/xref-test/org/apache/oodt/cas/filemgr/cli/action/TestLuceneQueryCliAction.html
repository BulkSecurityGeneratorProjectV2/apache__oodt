72     public void testClientTransTrueAndFlatProduct()
73           throws CmdLineActionException, IOException {
74        ActionMessagePrinter printer = new ActionMessagePrinter();
75        MockLuceneQueryCliAction cliAction = new MockLuceneQueryCliAction();
76  
77        cliAction.setQuery("ProductId:TestProductId");
78        cliAction.execute(printer);
79        assertEquals(1, clientSetQuery.getCriteria().size());
80        assertEquals("ProductId", ((TermQueryCriteria) clientSetQuery
81              .getCriteria().get(0)).getElementName());
82        assertEquals("TestProductId", ((TermQueryCriteria) clientSetQuery
83              .getCriteria().get(0)).getValue());
84  
85        cliAction.setQuery("ProductId:TestProductId ProductName:TestProductName");
86        cliAction.execute(printer);
87        assertEquals(1, clientSetQuery.getCriteria().size());
88        BooleanQueryCriteria bqc = (BooleanQueryCriteria) clientSetQuery
89              .getCriteria().get(0);
90        assertEquals(2, bqc.getTerms().size());
91        assertEquals(BooleanQueryCriteria.OR, bqc.getOperator());
92        assertEquals("ProductId",
93              ((TermQueryCriteria) bqc.getTerms().get(0)).getElementName());
94        assertEquals("TestProductId",
95              ((TermQueryCriteria) bqc.getTerms().get(0)).getValue());
96        assertEquals("ProductName",
97              ((TermQueryCriteria) bqc.getTerms().get(1)).getElementName());
98        assertEquals("TestProductName", ((TermQueryCriteria) bqc.getTerms()
99              .get(1)).getValue());
100 
101       cliAction
102             .setQuery("ProductId:TestProductId NominalDate:[20020101 TO 20030101]");
103       cliAction.execute(printer);
104       assertEquals(1, clientSetQuery.getCriteria().size());
105       bqc = (BooleanQueryCriteria) clientSetQuery.getCriteria().get(0);
106       assertEquals(2, bqc.getTerms().size());
107       assertEquals(BooleanQueryCriteria.OR, bqc.getOperator());
108       assertEquals("ProductId",
109             ((TermQueryCriteria) bqc.getTerms().get(0)).getElementName());
110       assertEquals("TestProductId",
111             ((TermQueryCriteria) bqc.getTerms().get(0)).getValue());
112       assertEquals("NominalDate",
113             ((RangeQueryCriteria) bqc.getTerms().get(1)).getElementName());
114       assertEquals("20020101",
115             ((RangeQueryCriteria) bqc.getTerms().get(1)).getStartValue());
116       assertEquals("20030101",
117             ((RangeQueryCriteria) bqc.getTerms().get(1)).getEndValue());
118       assertTrue(((RangeQueryCriteria) bqc.getTerms().get(1)).getInclusive());
119 
120       cliAction
121             .setQuery("ProductId:TestProductId NominalDate:{20020101 TO 20030101}");
122       cliAction.execute(printer);
123       cliAction.execute(printer);
124       assertEquals(1, clientSetQuery.getCriteria().size());
125       bqc = (BooleanQueryCriteria) clientSetQuery.getCriteria().get(0);
126       assertEquals(2, bqc.getTerms().size());
127       assertEquals(BooleanQueryCriteria.OR, bqc.getOperator());
128       assertEquals("ProductId",
129             ((TermQueryCriteria) bqc.getTerms().get(0)).getElementName());
130       assertEquals("TestProductId",
131             ((TermQueryCriteria) bqc.getTerms().get(0)).getValue());
132       assertEquals("NominalDate",
133             ((RangeQueryCriteria) bqc.getTerms().get(1)).getElementName());
134       assertEquals("20020101",
135             ((RangeQueryCriteria) bqc.getTerms().get(1)).getStartValue());
136       assertEquals("20030101",
137             ((RangeQueryCriteria) bqc.getTerms().get(1)).getEndValue());
138       assertFalse(((RangeQueryCriteria) bqc.getTerms().get(1)).getInclusive());
139 
140       cliAction
141             .setQuery("ProductId:TestProductId AND ProductName:TestProductName");
142       cliAction.execute(printer);
143       assertEquals(1, clientSetQuery.getCriteria().size());
144       bqc = (BooleanQueryCriteria) clientSetQuery.getCriteria().get(0);
145       assertEquals(2, bqc.getTerms().size());
146       assertEquals(BooleanQueryCriteria.AND, bqc.getOperator());
147       assertEquals("ProductId",
148             ((TermQueryCriteria) bqc.getTerms().get(0)).getElementName());
149       assertEquals("TestProductId",
150             ((TermQueryCriteria) bqc.getTerms().get(0)).getValue());
151       assertEquals("ProductName",
152             ((TermQueryCriteria) bqc.getTerms().get(1)).getElementName());
153       assertEquals("TestProductName", ((TermQueryCriteria) bqc.getTerms()
154             .get(1)).getValue());
155 
156       cliAction
157             .setQuery("ProductId:TestProductId OR ProductName:TestProductName");
158       cliAction.execute(printer);
159       assertEquals(1, clientSetQuery.getCriteria().size());
160       bqc = (BooleanQueryCriteria) clientSetQuery.getCriteria().get(0);
161       assertEquals(2, bqc.getTerms().size());
162       assertEquals(BooleanQueryCriteria.OR, bqc.getOperator());
163       assertEquals("ProductId",
164             ((TermQueryCriteria) bqc.getTerms().get(0)).getElementName());
165       assertEquals("TestProductId",
166             ((TermQueryCriteria) bqc.getTerms().get(0)).getValue());
167       assertEquals("ProductName",
168             ((TermQueryCriteria) bqc.getTerms().get(1)).getElementName());
169       assertEquals("TestProductName", ((TermQueryCriteria) bqc.getTerms()
170             .get(1)).getValue());
171 
172       cliAction
173             .setQuery("(ProductId:TestProductId OR ProductName:TestProductName) AND NominalDate:20110120");
174       cliAction.execute(printer);
175       assertEquals(1, clientSetQuery.getCriteria().size());
176       bqc = (BooleanQueryCriteria) clientSetQuery.getCriteria().get(0);
177       assertEquals(2, bqc.getTerms().size());
178       assertEquals(BooleanQueryCriteria.AND, bqc.getOperator());
179       BooleanQueryCriteria subBqc = (BooleanQueryCriteria) bqc.getTerms()
180             .get(0);
181       assertEquals("ProductId",
182             ((TermQueryCriteria) subBqc.getTerms().get(0)).getElementName());
183       assertEquals("TestProductId",
184             ((TermQueryCriteria) subBqc.getTerms().get(0)).getValue());
185       assertEquals("ProductName",
186             ((TermQueryCriteria) subBqc.getTerms().get(1)).getElementName());
187       assertEquals("TestProductName", ((TermQueryCriteria) subBqc.getTerms()
188             .get(1)).getValue());
189       assertEquals("NominalDate",
190             ((TermQueryCriteria) bqc.getTerms().get(1)).getElementName());
191       assertEquals("20110120",
192             ((TermQueryCriteria) bqc.getTerms().get(1)).getValue());
193    }

195    public class MockLuceneQueryCliAction extends LuceneQueryCliAction {
196       @Override
197       public XmlRpcFileManagerClient getClient() throws MalformedURLException,
198             ConnectionException {
199          return new XmlRpcFileManagerClient(new URL("http://localhost:9000"),
200                false) {
201             @Override
202             public List<QueryResult> complexQuery(ComplexQuery complexQuery) {
203                clientSetQuery = complexQuery;
204                Product p = new Product();
205                p.setProductId("TestProductId");
206                Metadata m = new Metadata();
207                m.addMetadata("Filename", TEST_FILENAME);
208                QueryResult qr = new QueryResult(p, m);
209                qr.setToStringFormat(complexQuery.getToStringResultFormat());
210                return Lists.newArrayList(qr);
211             }
212          };
213       }
