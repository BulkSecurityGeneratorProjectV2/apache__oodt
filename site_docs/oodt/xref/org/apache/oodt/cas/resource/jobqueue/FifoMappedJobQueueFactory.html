20  import java.io.File;

24  import org.apache.oodt.cas.metadata.util.PathUtils;

31  /**
32   * This factory class reads in properties set in the resource.properties file
33   * and read in via the command line and uses those properties to create a
34   * {@link FifoMappedJobQueue}.
35   * 
36   * @author resneck
37   *
38   */
39  public class FifoMappedJobQueueFactory implements JobQueueFactory {
40  
41  	private int stackSize = -1;
42  	private JobRepository repo;
43  	
44  	private static final Logger LOG =
45  			Logger.getLogger(FifoMappedJobQueueFactory.class.getName());
46  	
47  	public FifoMappedJobQueueFactory() {
48  		try{
49  			String stackSizeStr = System.getProperty(
50  					"gov.nasa.smap.spdm.resource.jobqueue.fifomappedjobqueue.maxstacksize");
51  	
52  			if (stackSizeStr != null) {
53  				stackSize = Integer.parseInt(stackSizeStr);
54  			}
55  		    
56  			String jobRepoFactoryClassStr = System.getProperty(
57  					"resource.jobrepo.factory",
58  					"gov.nasa.smap.spdm.resource.jobrepo.SmapMemoryJobRepositoryFactory");
59  			this.repo = GenericResourceManagerObjectFactory.
60  					getJobRepositoryFromServiceFactory(jobRepoFactoryClassStr);
61  		}catch(Exception e){
62  			LOG.log(Level.SEVERE, "An error occurred while creating a " +
63  					"FifoMappedJobQueue: " + e.getMessage());
64  		}
65  
66  	}
67  	
68  	/**
69  	 * @see org.apache.oodt.cas.resource.jobqueue.JobQueueFactory#createQueue()
70  	 */
71  	public JobQueue createQueue() {
72  		return new FifoMappedJobQueue(stackSize, repo);
73  	}
74  	
75  }
