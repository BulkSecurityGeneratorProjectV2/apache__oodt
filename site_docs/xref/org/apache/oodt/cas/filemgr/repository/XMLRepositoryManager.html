69      /**
70       * 
71       */
72      public XMLRepositoryManager(List<String> uris) throws InstantiationException {
73          if (uris == null) {
74              throw new InstantiationException(
75                      "Attempt to construct XMLRepositoryManager with a NULL list of  product type home URIs!");
76          }
77  
78          this.productTypeHomeUris = uris;
79          loadProductTypes(productTypeHomeUris);
80      }

134       for (Map.Entry<String, ProductType> typeId : productTypeMap.entrySet()) {

136         if (type.getName().equals(productTypeName)) {

142                 "XMLRepositoryManager: Unable to find product type: ["

157     private void saveProductTypes() {
158       for (String dirUri : productTypeHomeUris) {
159         File productTypeDir;
160 
161         try {
162           productTypeDir = new File(new URI(dirUri));
163 
164           if (!productTypeDir.isDirectory()) {
165             LOG
166                 .log(
167                     Level.WARNING,
168                     "Product type directory: "
169                     + dirUri
170                     + " is not "
171                     + "a directory: skipping product type saving to it.");
172             continue;
173           }
174 
175           String productTypeDirStr = productTypeDir.getAbsolutePath();
176           if (!productTypeDirStr.endsWith("/")) {
177             productTypeDirStr += "/";
178           }
179 
180           String productTypeXmlFile = productTypeDirStr
181                                       + "product-types.xml";
182           XmlStructFactory.writeProductTypeXmlDocument(Arrays
183                   .asList(productTypeMap.values().toArray(new ProductType[productTypeMap.values().size()])),
184               productTypeXmlFile);
185         } catch (URISyntaxException e) {
186           LOG.log(Level.WARNING,
187               "URISyntaxException when saving product "
188               + "type directory URI: " + dirUri
189               + ": Skipping Product Type saving"
190               + "for it: Message: " + e.getMessage());
191         }
192 
193       }
194 
195     }

197     private void loadProductTypes(List<String> dirUris) {
198       for (String dirUri1 : dirUris) {
199         File productTypeDir = null;
200         String dirUri = dirUri1;
201 
202         try {
203           productTypeDir = new File(new URI(dirUri));
204 
205           if (!productTypeDir.isDirectory()) {
206             LOG
207                 .log(
208                     Level.WARNING,
209                     "Product type directory: "
210                     + dirUri
211                     + " is not "
212                     + "a directory: skipping product type loading from it.");
213             continue;
214           }
215 
216           String productTypeDirStr = productTypeDir.getAbsolutePath();
217           if (!productTypeDirStr.endsWith("/")) {
218             productTypeDirStr += "/";
219           }
220 
221           String productTypeXmlFile = productTypeDirStr
222                                       + "product-types.xml";
223           Document productTypeDoc = getDocumentRoot(productTypeXmlFile);
224 
225           // now load the product types from it
226           if (productTypeDoc != null) {
227             Element productTypeRoot = productTypeDoc
228                 .getDocumentElement();
229 
230             NodeList productTypeNodeList = productTypeRoot
231                 .getElementsByTagName("type");
232 
233             if (productTypeNodeList != null
234                 && productTypeNodeList.getLength() > 0) {
235               for (int j = 0; j < productTypeNodeList.getLength(); j++) {
236                 Node productTypeNode = productTypeNodeList.item(j);
237                 ProductType type = XmlStructFactory
238                     .getProductType(productTypeNode);
239                 LOG.log(Level.FINE,
240                     "XMLRepositoryManager: found product type: ["
241                     + type.getName() + "]");
242                 productTypeMap.put(type.getProductTypeId(), type);
243               }
244             }
245           }
246         } catch (URISyntaxException e) {
247           LOG.log(Level.WARNING,
248               "URISyntaxException when loading product "
249               + "type directory URI: " + dirUri
250               + ": Skipping Product Type loading"
251               + "for it: Message: " + e.getMessage());
252         }
253       }
254     }

256     private Document getDocumentRoot(String xmlFile) {
257         // open up the XML file
258         DocumentBuilderFactory factory;
259         DocumentBuilder parser;
260         Document document;
261         InputSource inputSource;
262 
263         InputStream xmlInputStream;
264 
265         try {
266             xmlInputStream = new File(xmlFile).toURI().toURL().openStream();
267         } catch (IOException e) {
268             LOG.log(Level.WARNING,
269                     "IOException when getting input stream from [" + xmlFile
270                             + "]: returning null document root");
271             return null;
272         }
273 
274         inputSource = new InputSource(xmlInputStream);
275 
276         try {
277             factory = DocumentBuilderFactory.newInstance();
278             parser = factory.newDocumentBuilder();
279             document = parser.parse(inputSource);
280         } catch (Exception e) {
281             LOG.warning("Unable to parse xml file [" + xmlFile + "]."
282                     + "Reason is [" + e + "]");
283             return null;
284         }
285 
286         return document;
287     }
