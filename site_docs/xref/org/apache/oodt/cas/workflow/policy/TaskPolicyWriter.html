50      public static void writeTask(PolicyAwareWorkflowTask updateTask, Map taskMap) {
51          String taskXmlFileFullPath = ensureEndingSlash(updateTask
52                  .getPolicyDirPath())
53                  + TASKS_XML_FILE_NAME;
54  
55          // do the update
56          if (taskMap.containsKey(updateTask.getTaskId())) {
57              taskMap.put(updateTask.getTaskId(), updateTask);
58          }
59  
60          // subset the task map list to only be tasks from the given
61          // updateTask.getPolicyDirPath
62          Map taskSubsetMap = subsetByPolicyDirPath(taskMap, updateTask
63                  .getPolicyDirPath());
64  
65          try {
66              XMLUtils.writeXmlToStream(getTaskXmlDocument(taskSubsetMap),
67                      new FileOutputStream(new File(taskXmlFileFullPath)));
68              LOG.log(Level.INFO, "Successfully updated task policy file: ["
69                      + taskXmlFileFullPath + "]");
70          } catch (FileNotFoundException e) {
71              e.printStackTrace();
72              LOG.log(Level.WARNING, "Unable to write task policy file: ["
73                      + taskXmlFileFullPath + "]: file not found");
74          }
75  
76      }

78      private static Document getTaskXmlDocument(Map policyAwareTasks) {
79          DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
80          factory.setNamespaceAware(true);
81          Document document = null;
82  
83          try {
84              DocumentBuilder builder = factory.newDocumentBuilder();
85              document = builder.newDocument();
86  
87              Element root = (Element) document
88                      .createElement(CAS_TASKS_OUTER_ELEM);
89              root.setAttribute(CAS_XML_NS_DESC, CAS_NS);
90              document.appendChild(root);
91  
92              if (policyAwareTasks != null && policyAwareTasks.size() > 0) {
93                  for (Iterator i = policyAwareTasks.keySet().iterator(); i
94                          .hasNext();) {
95                      String taskId = (String) i.next();
96                      PolicyAwareWorkflowTask task = (PolicyAwareWorkflowTask) policyAwareTasks
97                              .get(taskId);
98  
99                      Element taskElem = (Element) document
100                             .createElement(TASK_ELEM);
101                     taskElem.setAttribute(TASK_ID_ATTR, task.getTaskId());
102                     taskElem.setAttribute(TASK_NAME_ATTR, task.getTaskName());
103                     taskElem.setAttribute(TASK_INST_CLASS_ATTR, task
104                             .getTaskInstanceClassName());
105 
106                     if (task.getConditions() != null
107                             && task.getConditions().size() > 0) {
108                         Element conditionsElem = (Element) document
109                                 .createElement(TASK_CONDITIONS_ELEM);
110                         for (Iterator j = task.getConditions().iterator(); j
111                                 .hasNext();) {
112                             WorkflowCondition cond = (WorkflowCondition) j
113                                     .next();
114                             Element condElem = (Element) document
115                                     .createElement(TASK_COND_ELEM);
116                             condElem.setAttribute(TASK_COND_ID_ATTR, cond
117                                     .getConditionId());
118                             conditionsElem.appendChild(condElem);
119                         }
120 
121                         taskElem.appendChild(conditionsElem);
122                     }
123 
124                     if (task.getRequiredMetFields() != null
125                             && task.getRequiredMetFields().size() > 0) {
126                         Element reqMetFieldsElem = (Element) document
127                                 .createElement(TASK_REQ_MET_FIELDS_ELEM);
128 
129                         for (Iterator j = task.getRequiredMetFields()
130                                 .iterator(); j.hasNext();) {
131                             String metField = (String) j.next();
132                             Element reqMetFieldElem = (Element) document
133                                     .createElement(TASK_REQ_MET_FIELD_ELEM);
134                             reqMetFieldElem.setAttribute(
135                                     TASK_REQ_MET_FIELD_NAME_ATTR, metField);
136                             reqMetFieldsElem.appendChild(reqMetFieldElem);
137                         }
138 
139                         taskElem.appendChild(reqMetFieldsElem);
140                     }
141                     if (task.getTaskConfig() != null
142                             && task.getTaskConfig().getProperties().keySet()
143                                     .size() > 0) {
144                         Element taskConfigElem = (Element) document
145                                 .createElement(TASK_CONFIG_ELEM);
146                         for (Iterator j = task.getTaskConfig().getProperties()
147                                 .keySet().iterator(); j.hasNext();) {
148                             String propName = (String) j.next();
149                             String propVal = task.getTaskConfig().getProperty(
150                                     propName);
151 
152                             EnvSavingConfiguration config = (EnvSavingConfiguration) task
153                                     .getTaskConfig();
154                             Element configPropElem = (Element) document
155                                     .createElement(PROPERTY_ELEM);
156                             if (config.isReplace(propName)) {
157                                 configPropElem.setAttribute(
158                                         PROPERTY_ELEM_ENVREPLACE_ATTR, String
159                                                 .valueOf(true));
160                             }
161 
162                             configPropElem.setAttribute(
163                                     PROPERTY_ELEM_NAME_ATTR, propName);
164                             configPropElem.setAttribute(
165                                     PROPERTY_ELEM_VALUE_ATTR, propVal);
166                             taskConfigElem.appendChild(configPropElem);
167                         }
168 
169                         taskElem.appendChild(taskConfigElem);
170                     }
171 
172                     root.appendChild(taskElem);
173                 }
174             }
175 
176             return document;
177         } catch (ParserConfigurationException pce) {
178             LOG.log(Level.WARNING, "Error generating tasks xml document!: "
179                     + pce.getMessage());
180         }
181 
182         return null;
183     }

185     private static String ensureEndingSlash(String path) {
186         String fixedPath = path;
187 
188         if (!fixedPath.endsWith("/")) {
189             fixedPath += "/";
190         }
191 
192         return fixedPath;
193     }

195     private static Map subsetByPolicyDirPath(Map origMap, String policyDirPath) {
196         Map newMap = new TreeMap(new Comparator() {
197 
198             public int compare(Object o1, Object o2) {
199                 String t1 = (String) o1;
200                 String t2 = (String) o2;
201 
202                 return t1.compareTo(t2);
203             }
204 
205         });
206 
207         if (origMap != null && origMap.keySet() != null
208                 && origMap.keySet().size() > 0) {
209             for (Iterator i = origMap.keySet().iterator(); i.hasNext();) {
210                 String taskId = (String) i.next();
211                 PolicyAwareWorkflowTask task = (PolicyAwareWorkflowTask) origMap
212                         .get(taskId);
213                 if (task.getPolicyDirPath().equals(policyDirPath)) {
214                     newMap.put(taskId, task);
215                 }
216             }
217         }
218 
219         return newMap;
220     }
