67     public void testAddNode() throws MalformedURLException {
68        int capacity = 10;
69        String nodeId = "TestNodeId";
70        String ipAddr = "http://localhost:9999";
71        cmdLineUtility.run(("--url http://localhost:9000"
72              + " --operation --addNode --capacity " + capacity + " --nodeId "
73              + nodeId + " --ipAddr " + ipAddr).split(" "));
74        MethodCallDetails methodCallDetails = client.getLastMethodCallDetails();
75        assertEquals("addNode", methodCallDetails.getMethodName());
76        ResourceNode actualNode = (ResourceNode) methodCallDetails.getArgs().get(0);
77        ResourceNode expectedNode = new ResourceNode(nodeId, new URL(ipAddr), capacity);
78        assertEquals(expectedNode.getIpAddr(), actualNode.getIpAddr());
79        assertEquals(expectedNode.getCapacity(), actualNode.getCapacity());
80        assertEquals(expectedNode.getNodeId(), actualNode.getNodeId());
81     }

230    public void testSubmitJob() throws MalformedURLException {
231       String jobDefinitionFile = "src/main/resources/examples/jobs/exJob.xml";
232       cmdLineUtility.run(("--url http://localhost:9000"
233             + " --operation --submitJob --def "
234             + jobDefinitionFile).split(" "));
235       JobSpec spec = JobBuilder.buildJobSpec(jobDefinitionFile);
236       MethodCallDetails methodCallDetails = client.getLastMethodCallDetails();
237       assertEquals("submitJob", methodCallDetails.getMethodName());
238       Job actualJob = (Job) methodCallDetails.getArgs().get(0);
239       assertEquals(spec.getJob().getJobInstanceClassName(), actualJob.getJobInstanceClassName());
240       assertEquals(spec.getJob().getJobInputClassName(), actualJob.getJobInputClassName());
241       assertEquals(spec.getJob().getQueueName(), actualJob.getQueueName());
242       assertEquals(spec.getJob().getLoadValue(), actualJob.getLoadValue());
243       JobInput actualJobInput = (JobInput) methodCallDetails.getArgs().get(1);
244       assertEquals(spec.getIn().getClass(), actualJobInput.getClass());
245       assertEquals(2, methodCallDetails.getArgs().size());
246 
247       OptionPropertyRegister.clearRegister();
248 
249       String url = "http://localhost:9000";
250       cmdLineUtility.run(("--url http://localhost:9000"
251             + " --operation --submitJob --def "
252             + jobDefinitionFile + " --nodeUrl " + url).split(" "));
253       methodCallDetails = client.getLastMethodCallDetails();
254       assertEquals("submitJob", methodCallDetails.getMethodName());
255       actualJob = (Job) methodCallDetails.getArgs().get(0);
256       assertEquals(spec.getJob().getJobInstanceClassName(), actualJob.getJobInstanceClassName());
257       assertEquals(spec.getJob().getJobInputClassName(), actualJob.getJobInputClassName());
258       assertEquals(spec.getJob().getQueueName(), actualJob.getQueueName());
259       assertEquals(spec.getJob().getLoadValue(), actualJob.getLoadValue());
260       actualJobInput = (JobInput) methodCallDetails.getArgs().get(1);
261       assertEquals(spec.getIn().getClass(), actualJobInput.getClass());
262       assertEquals(new URL(url), methodCallDetails.getArgs().get(2));
263       assertEquals(3, methodCallDetails.getArgs().size());
264    }
