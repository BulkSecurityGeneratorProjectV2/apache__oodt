42  /**
43   * Test class for {@link DumpMetadataCliAction}.
44   * 
45   * @author bfoster (Brian Foster)
46   */
47  public class TestDumpMetadataCliAction extends TestCase {
48  
49     private static final String PRODUCT_ID = "TestProductId";
50     private static final String FILE_NAME = "data.dat";
51  
52     private File tmpFile;
53  
54     @Override
55     public void setUp() throws Exception {
56        File bogusFile = File.createTempFile("bogus", "bogus");
57        tmpFile = new File(bogusFile.getParentFile(), "MetadataDump");
58        tmpFile.mkdirs();
59        bogusFile.delete();
60     }
61  
62     @Override
63     public void tearDown() throws Exception {
64        FileUtils.forceDelete(tmpFile);
65     }
66  
67     public void testDataFlow() throws CmdLineActionException, IOException {
68        MockDumpMetadataCliAction cliAction = new MockDumpMetadataCliAction();
69        cliAction.setProductId(PRODUCT_ID);
70        ActionMessagePrinter printer = new ActionMessagePrinter();
71        cliAction.execute(printer);
72        assertEquals(
73              "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"no\"?>\n"
74            + "<cas:metadata xmlns:cas=\"http://oodt.jpl.nasa.gov/1.0/cas\">\n"
75            + "<keyval type=\"vector\">\n"
76            + "<key>Filename</key>\n"
77            + "<val>data.dat</val>\n"
78            + "</keyval>\n"
79            + "</cas:metadata>\n",
80              printer.getPrintedMessages().get(0));
81  
82        cliAction.setOutputDir(tmpFile);
83        cliAction.execute(printer);
84        assertTrue(new File(tmpFile, FILE_NAME + ".met").exists());
85        assertEquals(
86              "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"no\"?>\n"
87              + "<cas:metadata xmlns:cas=\"http://oodt.jpl.nasa.gov/1.0/cas\">\n"
88              + "<keyval type=\"vector\">\n"
89              + "<key>Filename</key>\n"
90              + "<val>data.dat</val>\n"
91              + "</keyval>\n"
92              + "</cas:metadata>\n",
93              FileUtils.readFileToString(new File(tmpFile, FILE_NAME + ".met")));
94     }
95  
96     public class MockDumpMetadataCliAction extends DumpMetadataCliAction {
97        @Override
98        public XmlRpcFileManagerClient getClient() throws MalformedURLException,
99              ConnectionException {
100          return new XmlRpcFileManagerClient(new URL("http://localhost:9000"),
101                false) {
102             @Override
103             public Product getProductById(String productId) {
104                Product p = new Product();
105                p.setProductId(productId);
106                return p;
107             }
108             @Override
109             public Metadata getMetadata(Product product) {
110                Metadata m = new Metadata();
111                m.addMetadata(FILENAME, FILE_NAME);
112                return m;
113             }
114          };
115       }
116    }
