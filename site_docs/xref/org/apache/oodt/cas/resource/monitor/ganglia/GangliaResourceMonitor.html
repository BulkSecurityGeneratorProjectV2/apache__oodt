54  	/**
55  	 * Make a new GangliaResourceMonitor that reads information from a ganglia
56  	 * meta daemon.
57  	 * 
58  	 * @param loadCalculator
59  	 *            LoadCalculator
60  	 *            {@link org.apache.oodt.cas.resource.monitor.ganglia.loadcalc.LoadCalculator}
61  	 *            to calculate load
62  	 */
63  	public GangliaResourceMonitor(LoadCalculator loadCalculator,
64  			String gmetadHost, int gmetadPort) {
65  		this.loadCalculator = loadCalculator;
66  		this.loadMap = new ConcurrentHashMap<String, Integer>();
67  		this.gmetaNodes = new ConcurrentHashMap<String, Map<String, String>>();
68  		this.gmetaAdapters = new ConcurrentHashMap<String, GangliaAdapter>();
69  		try {
70  			this.initGmetaNodes(gmetadHost, gmetadPort);
71  		} catch (Exception e) {
72  			LOG.log(Level.SEVERE, e.getMessage());
73  			LOG.log(Level.WARNING,
74  					"URL exception initializing gmetad nodes: [" + gmetadHost
75  							+ ":" + gmetadPort + "]: Message: "
76  							+ e.getMessage());
77  		}
78  
79  	}

81  	@Override
82  	public int getLoad(ResourceNode node) throws MonitorException {
83  		Map<String, String> nodeProperties;
84  		String nodeId = node.getNodeId();
85  		nodeProperties = this.locateNode(nodeId);
86  		if (nodeProperties == null) {
87  			throw new MonitorException(
88  					"GangliaMonitor: not tracking requested node: [" + nodeId
89  							+ "]");
90  		}
91  
92  		// calculate load
93  		double calcLoad = this.loadCalculator.calculateLoad(nodeProperties);
94  		System.out.println(calcLoad);
95  		int load = Long.valueOf(Math.round(calcLoad)).intValue();
96  		System.out.println("LOAD is: "+load);
97  		return load;
98  	}

100 	@Override
101 	public boolean assignLoad(ResourceNode node, int loadValue)
102 			throws MonitorException {
103 		// technically this method should simply do nothing, since
104 		// putting a job onto a node should cause Ganglia to detect
105 		// for now we'll simply track what the current perceived load
106 		// on a node is - we may want to factor this into the weighting
107 		// in load calculator later
108 		String nodeId = node.getNodeId();
109 		if (loadMap.containsKey(nodeId)) {
110 			int currLoad = loadMap.get(nodeId);
111 			currLoad += loadValue;
112 			loadMap.put(nodeId, currLoad);
113 		} else {
114 			loadMap.put(nodeId, loadValue);
115 		}
116 		return true;
117 	}

133 	@Override
134 	public List getNodes() throws MonitorException {
135 		List<ResourceNode> nodes = new Vector<ResourceNode>();
136 		if (this.gmetaAdapters != null) {
137 			for (GangliaAdapter adapter : this.gmetaAdapters.values()) {
138 				Map<String, Map<String, String>> aNodes = adapter
139 						.getResourceNodeStatus();
140 				for (Map<String, String> map : aNodes.values()) {
141 					try {
142 						nodes.add(this.nodeFromMap(map));
143 					} catch (MalformedURLException e) {
144 						LOG.log(Level.SEVERE, e.getMessage());
145 						throw new MonitorException(e.getMessage());
146 					}
147 				}
148 			}
149 		}
150 
151 		return nodes;
152 	}

164 	@Override
165 	public ResourceNode getNodeByURL(URL ipAddr) throws MonitorException {
166 		if (this.gmetaAdapters != null) {
167 			for (GangliaAdapter adapter : this.gmetaAdapters.values()) {
168 				Map<String, Map<String, String>> aNodes = adapter
169 						.getResourceNodeStatus();
170 				for (Map.Entry<String, Map<String, String>> aNodeId : aNodes.entrySet()) {
171 					String host = ipAddr.getHost();
172 					int port = ipAddr.getPort();
173 					Map<String, String> nodeProps = aNodeId.getValue();
174 					if (aNodeId.getKey().equals(host)
175 							&& nodeProps.get(DEFAULT_PORT).equals(
176 									String.valueOf(port))) {
177 						try {
178 							return this.nodeFromMap(aNodeId.getValue());
179 						} catch (MalformedURLException e) {
180 							LOG.log(Level.SEVERE, e.getMessage());
181 							throw new MonitorException(e.getMessage());
182 						}
183 					}
184 				}
185 			}
186 		}
187 
188 		return null;
189 	}

191 	@Override
192 	public boolean reduceLoad(ResourceNode node, int loadValue)
193 			throws MonitorException {
194 		String nodeId = node.getNodeId();
195 		if (this.loadMap.containsKey(nodeId)) {
196 			int currLoad = loadMap.get(nodeId);
197 			currLoad = Math.min(0, currLoad - loadValue);
198 			this.loadMap.put(nodeId, currLoad);
199 		} else {
200 			this.loadMap.put(nodeId, 0);
201 		}
202 		
203 		return true;
204 	}

206 	private Map<String, String> locateNode(String nodeId) {
207 		if (this.gmetaAdapters != null && this.gmetaAdapters.size() > 0) {
208 			for (Map.Entry<String, GangliaAdapter> nId : this.gmetaAdapters.entrySet()) {
209 				GangliaAdapter adapter = nId.getValue();
210 				try {
211 					System.out.println("Querying gmetad: ["+adapter.getUrlString()+"]");
212 					Map<String, Map<String, String>> nodeStatus = adapter
213 							.getResourceNodeStatus();
214 					System.out.println("Looking for nodeid: ["+nodeId+"]");
215 					if (nodeStatus.containsKey(nodeId)) {
216 						System.out.println("NODE met: "+nodeStatus.get(nodeId));
217 						return nodeStatus.get(nodeId);
218 					}
219 				} catch (MonitorException e) {
220 					LOG.log(Level.WARNING,
221 							"MonitorException contacting Ganglia: ["
222 									+ adapter.getUrlString() + "]");
223 					LOG.log(Level.SEVERE, e.getMessage());
224 				}
225 			}
226 
227 		}
228 
229 		return null;
230 	}

249 	private GangliaAdapter createAdapter(Map<String, String> node) {
250 		return new GangliaAdapter(node.get("host"), Integer.valueOf(node
251 				.get("port")));
252 	}
253 
254 	private void addGmetadNode(String host, int port) {
255 		Map<String, String> rootNode = new ConcurrentHashMap<String, String>();
256 		rootNode.put("host", host);
257 		rootNode.put("port", String.valueOf(port));
258 		this.gmetaNodes.put(host, rootNode);
259 		this.gmetaAdapters.put(host, this.createAdapter(rootNode));
260 	}
261 
262 	private void removeGmetadNode(String host) {
263 		if (this.gmetaNodes.containsKey(host)
264 				&& this.gmetaAdapters.containsKey(host)) {
265 			LOG.log(Level.FINE,
266 					"Removing gmetad node: ["
267 							+ gmetaAdapters.get(host).getUrlString() + "]");
268 			this.gmetaAdapters.remove(host);
269 			this.gmetaNodes.remove(host);
270 		}
271 	}
