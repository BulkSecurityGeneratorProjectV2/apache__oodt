16  package org.apache.oodt.commons.io;
17  
18  import java.io.FilterInputStream;
19  import java.io.InputStream;
20  import java.io.IOException;
21  import org.apache.oodt.commons.util.Base64;
22  
23  /** An input stream that decodes its data from the RFC-1512 base 64 format.
24   *
25   * Wrap this input stream around another input stream, and all the bytes will be converted
26   * from their base-64 format when you read from it.
27   *
28   * @author Kelly
29   */
30  public class Base64DecodingInputStream extends FilterInputStream {
31  	/** Construct a base-64 decoding input stream.
32  	 *
33  	 * @param inputStream The input stream to decode.
34  	 */
35  	public Base64DecodingInputStream(InputStream inputStream) {
36  		super(inputStream);
37  	}
38  
39  	/** Read the next byte.
40  	 *
41  	 * Decode more base-64 data and return the next decoded byte.
42  	 *
43  	 * @return The byte, or -1 on end of stream.
44  	 * @throws IOException If an I/O error occurs.
45  	 */
46  	public int read() throws IOException {
47  		if (in == null) {
48  		  throw new IOException("Can't read from a closed stream");
49  		}
50  
51  		// If we've used up the decoded data buffer, read 4 more bytes and decode 'em.
52  		if (buffer == null || index == buffer.length) {
53  			byte[] streamBuf = new byte[4];
54  			int toRead = 4;
55  			int atIndex = 0;
56  			int actuallyGot;
57  			boolean firstRead = true;
58  			while (toRead > 0) {
59  				actuallyGot = in.read(streamBuf, atIndex, toRead);
60  				if (actuallyGot == -1) {
61  					if (firstRead) {
62  					  return -1;
63  					} else {
64  					  break;
65  					}
66  				}
67  				firstRead = false;
68  				atIndex += actuallyGot;
69  				toRead -= actuallyGot;
70  			}
71  			buffer = Base64.decode(streamBuf);
72  			if (buffer.length == 0) {
73  				buffer = null;
74  				return -1;
75  			}
76  			index = 0;
77  		}
78  		return buffer[index++] & 0xff;
79  	}
80  
81  	/** Read a bunch of bytes.
82  	 *
83  	 * This decodes base-64 data from the underlying stream and puts the result into
84  	 * the given array.
85  	 *
86  	 * @param b The buffer to fill with decoded base-64 data.
87  	 * @param offset Where in the buffer to start filling.
88  	 * @param length How many bytes to fill.
89  	 * @return The actual number of decoded bytes.
90  	 * @throws IOException If an I/O error occurs.
91  	 */
92  	public int read(byte[] b, int offset, int length) throws IOException {
93  		if (b == null) {
94  		  throw new IllegalArgumentException("Can't read data into a null array");
95  		}
96  		if (offset < 0 || offset >= b.length) {
97  		  throw new IndexOutOfBoundsException("Can't read data into an array with indexes 0.." + (b.length - 1)
98  											  + " at index " + offset);
99  		}
100 		if (length < 0) {
101 		  throw new IllegalArgumentException("Can't read a negative amount of data");
102 		}
103 		if (offset + length > b.length) {
104 		  throw new IndexOutOfBoundsException("Can't read data past the right edge of an array");
105 		}
106 		if (in == null) {
107 		  throw new IOException("Can't read from a closed stream");
108 		}
109 
110 		int c = read();
111 		if (c == -1) {
112 		  return -1;
113 		}
114 		b[offset] = (byte) c;
115 		int i = 1;
116 		try {
117 			for (; i < length; ++i) {
118 				c = read();
119 				if (c == -1) {
120 				  break;
121 				}
122 				b[offset + i] = (byte) c;
123 			}
124 		} catch (IOException ignore) {}
125 		return i;
126 	}
127 
128 	/** Skip bytes.
129 	 *
130 	 * This method skips and discards <var>n</var> decoded bytes on the input stream.
131 	 *
132 	 * @param n Number of bytes to skip.
133 	 * @return Actual number of bytes skipped.
134 	 * @throws IOException If an I/O error occurs.
135 	 */
136 	public long skip(long n) throws IOException {
137 		if (in == null) {
138 		  throw new IOException("Can't skip past data on a closed stream");
139 		}
140 		int actuallySkipped = 0;
141 		while (n > 0) {
142 			if (read() == -1) {
143 			  return actuallySkipped;
144 			}
145 			--n;
146 			++actuallySkipped;
147 		}
148 		return actuallySkipped;
149 	}
150 
151 	/** Return bytes available for reading or skipping without blocking.
152 	 *
153 	 * @return The number of bytes that can be read from this stream or skipped over
154 	 * on the stream without blocking.
155 	 * @throws IOException If an I/O error occurs.
156 	 */
157 	public int available() throws IOException {
158 		if (in == null) {
159 		  throw new IOException("Can't see how many bytes are available on a closed stream");
160 		}
161 		if (buffer != null && index < buffer.length) {
162 		  return buffer.length - index;
163 		}
164 		return in.available() >= 4? 1 : 0;
165 	}
166 
167 	/** Close this stream.
168 	 *
169 	 * @throws IOException If an I/O error occurs.
170 	 */
171 	public void close() throws IOException {
172 		if (in == null) {
173 		  throw new IOException("Can't close a closed stream");
174 		}
175 		in.close();
176 		in = null;
177 		buffer = null;
178 	}
179 
180 	/** Buffer for decoded data.
181 	 */
182 	private byte[] buffer;
183 
184 	/** Where we'll next read out of the buffer.
185 	 *
186 	 * Since we always read 4 bytes at a time (a base-64 block), we can decode that
187 	 * into as many as 3 bytes, so start out the index in an invalid location.
188 	 */
189 	private int index = 3;
190 }
