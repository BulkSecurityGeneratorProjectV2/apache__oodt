16  package org.apache.oodt.commons.util;
17  
18  import java.io.*;
19  import java.util.*;
20  import org.apache.oodt.commons.io.*;
21  import org.apache.oodt.commons.util.*;
22  
23  /**
24  	The <code>FileLogger</code> class is intended to be used with the
25  	{@link Log} class in order to setup a log file as a {@link LogListener}.
26  
27  	@author S. Hardman
28  	@version $Revision: 1.1 $
29  */
30  public class FileLogger extends WriterLogger {
31  
32  	/**
33  		Constructor given no arguments.
34  
35  		This constructor will create a new log file or append to an existing
36  		log file if the specified file exists. The constructor will then call
37  		the other constructor with the output stream argument. This constructor
38  		utilizes two system properties for specifying the path and name of the
39  		log file. They are as follows:
40  
41  		<p>org.apache.oodt.commons.util.FileLogger.path - If specified, this is the directory where
42  			the log file will be found or created, otherwise the local directory
43  			will be utilized.
44  
45  		<p>org.apache.oodt.commons.util.FileLogger.name - If specified, this is the name of the log
46  			file to be found or created, otherwise the name will be as follows:
47  			"eda_yyyymmddhhmmssSSS.log".
48  
49  		@throws FileNotFoundException If the file cannot be opened for any
50  			reason.
51  	*/
52  	public FileLogger () throws FileNotFoundException {
53  		this(new FileOutputStream(System.getProperty("org.apache.oodt.commons.util.FileLogger.path", ".") + "/" + System.getProperty("org.apache.oodt.commons.util.FileLogger.name", "eda_" + DateConvert.tsFormat(new Date()) + ".log"), true));
54  	}
55  
56  
57  	/**
58  		Constructor given an output stream.
59  
60  		This constructor calls the {@link WriterLogger} constructor with the
61  		same signature.
62  
63  		@param outputStream The output stream representing the log destination.
64  	*/
65  	public FileLogger (OutputStream outputStream) {
66  		super(outputStream);
67  	}
68  }
