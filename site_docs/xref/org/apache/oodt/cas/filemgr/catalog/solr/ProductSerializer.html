17  package org.apache.oodt.cas.filemgr.catalog.solr;
18  
19  import java.util.List;
20  
21  import org.apache.oodt.cas.filemgr.structs.Product;
22  import org.apache.oodt.cas.filemgr.structs.Reference;
23  import org.apache.oodt.cas.filemgr.structs.exceptions.CatalogException;
24  import org.apache.oodt.cas.metadata.Metadata;

26  /**
27   * API for serializing CAS objects (Products, Metadata and References) into documents to be indexed by Solr,
28   * and vice-versa deserializing Solr response documents into CAS objects.
29   * 
30   * Implementing classes are responsible for converting the CAS fields
31   * into corresponding Solr fields as defined in their specific Solr schema.xml, 
32   * and vice-versa for transforming the Solr fields into corresponding CAS fields.
33   * 
34   * @author Luca Cinquini
35   *
36   */
37  public interface ProductSerializer {
38  	
39  	/**
40  	 * Method to transform a CAS Product into one or more documents to be indexed by Solr.
41  	 * 
42  	 * @param product : the product to be indexed
43  	 * @param create: true to create a new product, false to update an existing one.
44  	 * @return
45  	 */
46  	List<String> serialize(Product product, boolean create);
47  	
48  	/**
49  	 * Method to transform a CAS Metadata object into one or more Solr "update" documents
50  	 * (i.e. documents containing attributes of an existing product).
51  	 * 
52  	 * @param productId : the product unique identifier
53  	 * @param metadata : the CAS metadata hash table
54  	 * @param replace : true to override the existing metadata values, false to add new values to the existing ones.
55  	 *                  Existing metadata fields not contained in the hash are left unchanged.
56  	 *                  Note: the special value Parameters.NULL is used to indicate that the corresponding key must be removed
57  	 *                  (as the Metadata object cannot store null or empty lists).
58  	 * @return
59  	 */
60  	List<String> serialize(String productId, Metadata metadata, boolean replace);
61  	
62  	/**
63  	 * Method to transform a CAS Product References object into one or more Solr "update" documents
64  	 * (i.e. documents containing attributes of an existing product).
65  	 * 
66  	 * @param productId
67  	 * @param rootReference : the product root reference, may be null
68  	 * @param references : list of product references, may be empty
69  	 * @param replace : true to replace the existing references, false to add new references to the existing ones
70  	 * @return
71  	 */
72  	List<String> serialize(String productId, Reference rootReference, List<Reference> references, boolean replace);
73  	
74  	/**
75  	 * Method to parse a full XML response document into a list of Product, References and Metadata objects.
76  	 * 
77  	 * @param xml : the Solr XML document to parse (input)
78  	 * @return
79  	 * @throws Exception
80  	 */
81  	QueryResponse deserialize(String xml) throws CatalogException;
82  	
83  	/**
84  	 * Mime type of the documents generated by the specific implementation (XML, JSON etc.)
85  	 * @return
86  	 */
87  	String getMimeType();
88  
89  }
