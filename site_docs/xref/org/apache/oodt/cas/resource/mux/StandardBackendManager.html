17  package org.apache.oodt.cas.resource.mux;

20  import org.apache.oodt.cas.resource.monitor.Monitor;

24  import java.util.HashMap;

29  /**
30   * This manager keeps track of the mux-able backends for the resource manager.
31   * It effectively maps a queue to the backend that this queue feeds.
32   *
33   * It uses a private BackendSet to keep track of everything.
34   *
35   * For reference, a backend is a set of the following:
36   *    1. Batch manger, responsible for running jobs
37   *    2. Scheduler, responsible for scheduling a job to run
38   *    3. Monitor, responsible for managing nodes
39   *
40   * @author starchmd
41   */
42  public class StandardBackendManager implements BackendManager {
43      Map<String,BackendSet> queueToBackend = new HashMap<String,BackendSet>();
44  
45      /**
46       * Add in a backend set to this manager.
47       * @param queue - queue that maps to the given monitor, batchmgr, and scheduler
48       * @param monitor - monitor used for this set
49       * @param batchmgr - batch manager for this set
50       * @param scheduler - scheduler for this set
51       */
52      public void addSet(String queue,Monitor monitor, Batchmgr batchmgr, Scheduler scheduler) {
53          queueToBackend.put(queue, new BackendSet(monitor,batchmgr,scheduler));
54      }
55      /**
56       * Return monitor for the given queue.
57       * @param queue - queue to check
58       * @return montior
59       * @throws QueueManagerException when queue does not exist
60       */
61      public Monitor getMonitor(String queue) throws QueueManagerException {
62          BackendSet set = queueToBackend.get(queue);
63          if (set == null) {
64              throw new QueueManagerException("Queue '" + queue + "' does not exist");
65          }
66          return set.monitor;
67      }
68      /**
69       * Return batch manager for the given queue.
70       * @param queue - queue to check
71       * @return batchmgr
72       * @throws QueueManagerException when queue does not exist
73       */
74      public Batchmgr getBatchmgr(String queue) throws QueueManagerException {
75          BackendSet set = queueToBackend.get(queue);
76          if (set == null) {
77              throw new QueueManagerException("Queue '" + queue + "' does not exist");
78          }
79          return set.batchmgr;
80      }
81      /**
82       * Return scheduler for the given queue.
83       * @param queue - queue to check
84       * @return scheduler
85       * @throws QueueManagerException when queue does not exist
86       */
87      public Scheduler getScheduler(String queue) throws QueueManagerException {
88          BackendSet set = queueToBackend.get(queue);
89          if (set == null) {
90              throw new QueueManagerException("Queue '" + queue + "' does not exist");
91          }
92          return set.scheduler;
93      }
94      /**
95       * Return a list of all monitors.
96       * @return list of all monitors
97       */
98      public List<Monitor> getMonitors() {
99          List<Monitor> monitors = new LinkedList<Monitor>();
100         for (BackendSet set : queueToBackend.values()) {
101             monitors.add(set.monitor);
102         }
103         return monitors;
104     }
105     /**
106      * Class that holds a set of the three backend pieces.
107      * Private class, because no accessor/modifiers have been
108      * created(public members). Acts like a struct.
109      *
110      * @author starchmd
111      */
112     private class BackendSet {
113         public Monitor monitor = null;
114         public Batchmgr batchmgr = null;
115         public Scheduler scheduler = null;
116 
117         public BackendSet(Monitor monitor, Batchmgr batchmgr, Scheduler scheduler) {
118             this.monitor = monitor;
119             this.batchmgr = batchmgr;
120             this.scheduler = scheduler;
121         }
122     }
123 }
