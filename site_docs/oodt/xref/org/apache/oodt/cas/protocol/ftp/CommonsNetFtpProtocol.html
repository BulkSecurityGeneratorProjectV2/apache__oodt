36  /**
37   * This class is responsible for FTP transfers. It is built as a wrapper around
38   * Apache's FTPClient class in order to connect it into the Crawler's Protocol
39   * infrastructure.
40   *
41   * @author bfoster
42   *
43   */
44  public class CommonsNetFtpProtocol implements Protocol {
45  
46  	private final FTPClient ftp;
47  	private String homeDir;
48  
49  	/**
50  	 * Creates a new FtpClient
51  	 */
52  	public CommonsNetFtpProtocol() {
53  		ftp = new FTPClient();
54  	}
55  
56  	/**
57  	 * {@inheritDoc}
58  	 */
59  	public void connect(String host, Authentication auth)
60  			throws ProtocolException {
61  		// server cannot be null
62  		if (host == null) {
63  			throw new ProtocolException("Tried to connect to server == NULL");
64  		}
65  
66  		try {
67  			ftp.connect(host);
68  			ftp.enterLocalPassiveMode();
69  		} catch (Exception e) {
70  			throw new ProtocolException("Failed to connect to server : "
71  					+ e.getMessage());
72  		}
73  
74  		try {
75  			// try logging in
76  			if (!ftp.login(auth.getUser(), auth.getPass())) {
77  				throw new ProtocolException("Failed logging into host " + host
78  						+ " as user " + auth.getUser());
79  			}
80  
81  			// set file type to binary
82  			ftp.setFileType(FTPClient.BINARY_FILE_TYPE);
83  
84  			homeDir = ftp.printWorkingDirectory();
85  		} catch (Exception e) {
86  			// login failed
87  			throw new ProtocolException("Exception thrown while logging into host "
88  					+ host + " as user " + auth.getUser());
89  		}
90  	}
91  
92  	/**
93  	 * {@inheritDoc}
94  	 */
95  	public ProtocolFile pwd() throws ProtocolException {
96  		try {
97  			return new ProtocolFile(ftp.printWorkingDirectory(), true);
98  		} catch (Exception e) {
99  			throw new ProtocolException("Failed to pwd : " + e.getMessage());
100 		}
101 	}
102 
103 	public List<ProtocolFile> ls() throws ProtocolException {
104 		try {
105 			String path = this.pwd().getPath();
106 			FTPFile[] files = ftp.listFiles();
107 			List<ProtocolFile> returnFiles = new LinkedList<ProtocolFile>();
108 		  for (FTPFile file : files) {
109 			if (file == null) {
110 			  continue;
111 			}
112 			returnFiles.add(new ProtocolFile(path + "/" + file.getName(), file
113 				.isDirectory()));
114 		  }
115 			return returnFiles;
116 		} catch (Exception e) {
117 			throw new ProtocolException("Failed to get file list : " + e.getMessage());
118 		}
119 	}
120 
121 	/**
122 	 * {@inheritDoc}
123 	 */
124 	public List<ProtocolFile> ls(ProtocolFileFilter filter)
125 			throws ProtocolException {
126 		try {
127 			FTPFile[] files = ftp.listFiles();
128 			List<ProtocolFile> returnFiles = new LinkedList<ProtocolFile>();
129 		  for (FTPFile file : files) {
130 			if (file == null) {
131 			  continue;
132 			}
133 			String path = this.pwd().getPath();
134 			ProtocolFile pFile = new ProtocolFile(path + "/" + file.getName(), file
135 				.isDirectory());
136 			if (filter.accept(pFile)) {
137 			  returnFiles.add(pFile);
138 			}
139 		  }
140 			return returnFiles;
141 		} catch (Exception e) {
142 			throw new ProtocolException("Failed to get file list : " + e.getMessage());
143 		}
144 	}

146 	/**
147 	 * {@inheritDoc}
148 	 */
149 	public void get(ProtocolFile fromFile, File toFile) throws ProtocolException {
150 		// file or toLocalFile cannot be null
151 		if (fromFile == null || toFile == null) {
152 			throw new ProtocolException(
153 					"Can't download file -> ProtocolFile == null || toLocalFile == null");
154 		}
155 
156 		// download file
157 		OutputStream os = null;
158 		try {
159 			os = new FileOutputStream(toFile);
160 			if (!ftp.retrieveFile(fromFile.getName(), os)) {
161 				throw new ProtocolException("Failed to download file "
162 						+ fromFile.getName());
163 			}
164 		} catch (Exception e) {
165 			// download failed
166 			toFile.delete();
167 			throw new ProtocolException("FAILED to download: " + fromFile.getName()
168 					+ " : " + e.getMessage(), e);
169 		} finally {
170 			// close output stream
171 			if (os != null) {
172 			  try {
173 				os.close();
174 			  } catch (Exception e) {
175 				toFile.delete();
176 				throw new ProtocolException("Failed to close outputstream : "
177 											+ e.getMessage(), e);
178 			  }
179 			}
180 		}
181 	}

183 	/**
184 	 * {@inheritDoc}
185 	 */
186 	public void put(File fromFile, ProtocolFile toFile) throws ProtocolException {
187 		try {
188 			ftp.storeFile(toFile.getPath(), new FileInputStream(fromFile));
189 		} catch (Exception e) {
190 			throw new ProtocolException("Failed to put file '" + fromFile + "' : "
191 					+ e.getMessage(), e);
192 		}
193 	}
194 
195 	/**
196 	 * {@inheritDoc}
197 	 */
198 	public void cd(ProtocolFile file) throws ProtocolException {
199 		try {
200 			if (!ftp.changeWorkingDirectory(file.getPath())) {
201 			  throw new Exception("Directory change method returned false");
202 			}
203 		} catch (Exception e) {
204 			throw new ProtocolException("Failed to cd to " + file.getPath() + " : "
205 					+ e.getMessage());
206 		}
207 	}
208 
209 	public void cdRoot() throws ProtocolException {
210 		cd(new ProtocolFile(ProtocolFile.SEPARATOR, true));
211 	}
212 
213 	public void cdHome() throws ProtocolException {
214 		cd(new ProtocolFile(homeDir, true));
215 	}
216 
217 	/**
218 	 * {@inheritDoc}
219 	 */
220 	public void close() throws ProtocolException {
221 		try {
222 			ftp.disconnect();
223 		} catch (Exception e) {
224 			throw new ProtocolException("Failed to disconnect from server");
225 		}
226 	}
227 
228 	/**
229 	 * {@inheritDoc}
230 	 */
231 	public boolean connected() {
232 		return ftp.isConnected();
233 	}
234 
235 	/**
236 	 * {@inheritDoc}
237 	 */
238 	public void delete(ProtocolFile file) throws ProtocolException {
239 		try {
240 			ftp.deleteFile(file.getPath());
241 		} catch (Exception e) {
242 			throw new ProtocolException("Failed to delete file '" + file.getPath()
243 					+ "' : " + e.getMessage(), e);
244 		}
245 	}
