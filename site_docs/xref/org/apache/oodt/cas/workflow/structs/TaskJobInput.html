95    public void read(Object in) {
96      if (!(in instanceof Map)) {
97        return;
98      }
99  
100     Map inHash = (Map) in;
101 
102     this.taskConfig = XmlRpcStructFactory
103         .getWorkflowTaskConfigurationFromXmlRpc((Map) inHash
104             .get("task.config"));
105     this.dynMetadata.addMetadata((Map) inHash.get("task.metadata"));
106     this.workflowTaskInstanceClassName = (String)inHash.get("task.instance.class");
107 
108   }

115   public Object write() {
116     // need to create a Map with the task metadata and the task config
117     Map outHash = new ConcurrentHashMap();
118     outHash.put("task.config", XmlRpcStructFactory
119         .getXmlRpcWorkflowTaskConfiguration(this.taskConfig));
120     outHash.put("task.metadata", this.dynMetadata.getMap());
121     outHash.put("task.instance.class", this.workflowTaskInstanceClassName);
122     return outHash;
123   }

154   /*
155    * (non-Javadoc)
156    * 
157    * @see org.apache.oodt.cas.resource.util.Configurable#configure(java.util.Properties)
158    */
159   public void configure(Properties props) {
160     // looking for dyn.metadata.file
161     // looking for static.config.file
162     String staticConfigFile = props.getProperty("static.config.file");
163     String dynMetadataFile = props.getProperty("dyn.metadata.file");
164 
165     if (staticConfigFile != null) {
166       try {
167         this.taskConfig.getProperties().load(
168             new FileInputStream(new File(staticConfigFile)));
169       } catch (RuntimeException ex) {
170         throw ex;
171       } catch (Exception ignore) {  
172       }
173     }
174 
175     if (dynMetadataFile != null) {
176       InputStream in = null;
177       try {
178         this.dynMetadata = new Metadata();
179         Properties fileProps = new Properties();
180         in = new BufferedInputStream(new FileInputStream(new File(dynMetadataFile)));
181         fileProps.load(in);
182         for (Object key: fileProps.keySet()){
183           String keyStr = (String)key;
184           this.dynMetadata.addMetadata(keyStr, fileProps.getProperty(keyStr));
185         }
186       } catch (RuntimeException ex) {
187         throw ex;
188       } catch (Exception ignore) {
189       } finally {
190         if (in != null) {
191           try {
192             in.close();
193           } catch (IOException ignore) {
194           }
195         }
196       }
197     }
198 
199   }

201   @Override
202   public Map<String, Vector<String>> getMetadata() {
203     //combine the workflow metadata, and config
204     Map<String, Vector<String>> met = new ConcurrentHashMap<String, Vector<String>>();
205     
206     if(this.taskConfig != null && this.taskConfig.getProperties() != null && 
207         this.taskConfig.getProperties().keySet() != null && 
208         this.taskConfig.getProperties().keySet().size() > 0){
209       for(Object prop: this.taskConfig.getProperties().keySet()){
210         String propName = String.valueOf(prop);
211         Vector<String> vals = new Vector<String>();
212         String propVal = this.taskConfig.getProperty(propName);
213         if (propVal != null){
214           vals.add(propVal);
215           met.put(propName, vals);
216         }
217       }
218     }
219     
220     //NOTE: this implies that dynMetadata will have precedence over config metadata
221     //as returned by the Resmgr Job i/f
222     if(this.dynMetadata != null && this.dynMetadata.getAllKeys() != null && 
223         this.dynMetadata.getAllKeys().size() > 0){
224       for(String metName: this.dynMetadata.getAllKeys()){
225         Vector<String> vals = new Vector<String>();
226         vals.addAll(this.dynMetadata.getAllValues(metName));
227         if (vals.size() > 0){
228           met.put(metName, vals);
229         }
230       }
231     }
232     
233     return met;
234   }
