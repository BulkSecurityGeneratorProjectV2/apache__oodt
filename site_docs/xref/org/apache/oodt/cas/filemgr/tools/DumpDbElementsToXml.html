33  /**
34   * @author mattmann
35   * @version $Revision$
36   * 
37   * <p>
38   * A Simple program to dump XML elements out of a DataSourceValidationLayer and
39   * into an XML file.
40   * </p>
41   * 
42   */
43  public final class DumpDbElementsToXml {
44  
45      private DumpDbElementsToXml() throws InstantiationException {
46          throw new InstantiationException(
47                  "Don't construct private constructors!");
48      }
49  
50      /**
51       * @param args
52       */
53      public static void main(String[] args) throws IOException, ValidationLayerException {
54          String propFile = null, outXmlFile = null;
55          String usage = "DumpDbElementsToXml --propFile </path/to/propFile> --out </path/to/xml/file>\n";
56  
57          for (int i = 0; i < args.length; i++) {
58              if (args[i].equals("--propFile")) {
59                  propFile = args[++i];
60              } else if (args[i].equals("--out")) {
61                  outXmlFile = args[++i];
62              }
63          }
64  
65          if (propFile == null || outXmlFile == null) {
66              System.err.println(usage);
67              System.exit(1);
68          }
69  
70          System.out.println("Loading properties file: [" + propFile + "]");
71          System.getProperties().load(new FileInputStream(new File(propFile)));
72  
73          ValidationLayer dbLayer = new DataSourceValidationLayerFactory()
74                  .createValidationLayer();
75  
76          List elementList = dbLayer.getElements();
77          XmlStructFactory.writeElementXmlDocument(elementList, outXmlFile);
78  
79      }

81  }
