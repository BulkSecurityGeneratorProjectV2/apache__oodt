47  /**
48   * 
49   * The ThreadPooling portion of the WorkflowEngine. This class is meant to be an
50   * extension point for WorkflowEngines that want to implement ThreadPooling.
51   * This WorkflowEngine provides everything needed to manage a ThreadPool using
52   * Doug Lea's wonderful java.util.concurrent package that made it into JDK5.
53   * 
54   * @author mattmann
55   * @version $Revsion$
56   * 
57   */
58  public class ThreadPoolWorkflowEngine implements WorkflowEngine, WorkflowStatus {

60    /* our thread pool */
61    private PooledExecutor pool = null;

64    private ConcurrentHashMap workerMap = null;

67    private static final Logger LOG = Logger
68        .getLogger(ThreadPoolWorkflowEngine.class.getName());

71    private WorkflowInstanceRepository instRep = null;

74    private XmlRpcResourceManagerClient rClient = null;

77    private URL wmgrUrl = null;

79    /**
80     * Default Constructor.
81     * 
82     * @param instRep
83     *          The WorkflowInstanceRepository to be used by this engine.
84     * @param queueSize
85     *          The size of the queue that the workflow engine should use
86     *          (irrelevant if unlimitedQueue is set to true)
87     * @param maxPoolSize
88     *          The minimum thread pool size.
89     * @param minPoolSize
90     *          The maximum thread pool size.
91     * @param threadKeepAliveTime
92     *          The amount of minutes that each thread in the pool should be kept
93     *          alive.
94     * @param unlimitedQueue
95     *          Whether or not to use a queue whose bounds are dictated by the
96     *          physical memory of the underlying hardware.
97     * @param resUrl
98     *          A URL pointer to a resource manager. If this is set Tasks will be
99     *          wrapped as Resource Manager {@link Job}s and sent through the
100    *          Resource Manager. If this parameter is not set, local execution
101    *          (the default) will be used
102    */
103   public ThreadPoolWorkflowEngine(WorkflowInstanceRepository instRep,
104       int queueSize, int maxPoolSize, int minPoolSize,
105       long threadKeepAliveTime, boolean unlimitedQueue, URL resUrl) {
106 
107     this.instRep = instRep;
108     Channel c;
109     if (unlimitedQueue) {
110       c = new LinkedQueue();
111     } else {
112       c = new BoundedBuffer(queueSize);
113     }
114 
115     pool = new PooledExecutor(c, maxPoolSize);
116     pool.setMinimumPoolSize(minPoolSize);
117     pool.setKeepAliveTime(1000 * 60 * threadKeepAliveTime);
118 
119     workerMap = new ConcurrentHashMap();
120 
121     if (resUrl != null) {
122       rClient = new XmlRpcResourceManagerClient(resUrl);
123     }
124   }

133   public synchronized void pauseWorkflowInstance(String workflowInstId) {
134     // okay, try and look up that worker thread in our hash map
135     IterativeWorkflowProcessorThread worker = (IterativeWorkflowProcessorThread) workerMap
136         .get(workflowInstId);
137     if (worker == null) {
138       LOG.log(Level.WARNING,
139           "WorkflowEngine: Attempt to pause workflow instance id: "
140               + workflowInstId
141               + ", however, this engine is not tracking its execution");
142       return;
143     }
144 
145     // otherwise, all good
146     worker.pause();
147 
148   }

159     IterativeWorkflowProcessorThread worker = (IterativeWorkflowProcessorThread) workerMap
160         .get(workflowInstId);
161     if (worker == null) {
162       LOG.log(Level.WARNING,
163           "WorkflowEngine: Attempt to resume workflow instance id: "
164               + workflowInstId + ", however, this engine is "
165               + "not tracking its execution");
166       return;
167     }
168 
169     // also check to make sure that the worker is currently paused
170     // only can resume WorkflowInstances that are paused, right?
171     if (!worker.isPaused()) {
172       LOG.log(Level.WARNING,
173           "WorkflowEngine: Attempt to resume a workflow that "
174               + "isn't paused currently: instance id: " + workflowInstId);
175       return;
176     }
177 
178     // okay, all good
179     worker.resume();
180 
181   }

183   /*
184    * (non-Javadoc)
185    * 
186    * @see
187    * org.apache.oodt.cas.workflow.engine.WorkflowEngine#startWorkflow(org.apache
188    * .oodt.cas.workflow.structs.Workflow, org.apache.oodt.cas.metadata.Metadata)
189    */
190   public synchronized WorkflowInstance startWorkflow(Workflow workflow,
191       Metadata metadata) throws EngineException {
192     // to start the workflow, we create a default workflow instance
193     // populate it
194     // persist it
195     // add it to the worker map
196     // start it
197 
198     WorkflowInstance wInst = new WorkflowInstance();
199     wInst.setWorkflow(workflow);
200     wInst.setCurrentTaskId(((WorkflowTask) workflow.getTasks().get(0))
201         .getTaskId());
202     wInst.setSharedContext(metadata);
203     wInst.setStatus(CREATED);
204     persistWorkflowInstance(wInst);
205 
206     IterativeWorkflowProcessorThread worker = new IterativeWorkflowProcessorThread(
207         wInst, instRep, this.wmgrUrl);
208     worker.setRClient(rClient);
209     workerMap.put(wInst.getId(), worker);
210 
211     wInst.setStatus(QUEUED);
212     persistWorkflowInstance(wInst);
213 
214     try {
215       pool.execute(worker);
216     } catch (InterruptedException e) {
217       throw new EngineException(e);
218     }
219 
220     return wInst;
221 
222   }

241   public synchronized boolean updateMetadata(String workflowInstId, Metadata met) {
242     // okay, try and look up that worker thread in our hash map
243     IterativeWorkflowProcessorThread worker = (IterativeWorkflowProcessorThread) workerMap
244         .get(workflowInstId);
245     if (worker == null) {
246       LOG.log(Level.WARNING,
247           "WorkflowEngine: Attempt to update metadata context "
248               + "for workflow instance id: " + workflowInstId
249               + ", however, this engine is " + "not tracking its execution");
250       return false;
251     }
252 
253     worker.getWorkflowInstance().setSharedContext(met);
254     try {
255       persistWorkflowInstance(worker.getWorkflowInstance());
256     } catch (Exception e) {
257       LOG.log(
258           Level.WARNING,
259           "Exception persisting workflow instance: ["
260               + worker.getWorkflowInstance().getId() + "]: Message: "
261               + e.getMessage());
262       return false;
263     }
264 
265     return true;
266   }

322     IterativeWorkflowProcessorThread worker = (IterativeWorkflowProcessorThread) workerMap
323         .get(workflowInstId);
324     if (worker == null) {
325       // try and get the metadata
326       // from the workflow instance repository (as it was persisted)

328         WorkflowInstance inst = instRep.getWorkflowInstanceById(workflowInstId);
329         return inst.getSharedContext();
330       } catch (InstanceRepositoryException e) {
331         LOG.log(Level.FINEST, "WorkflowEngine: Attempt to get metadata "
332             + "for workflow instance id: " + workflowInstId
333             + ", however, this engine is "
334             + "not tracking its execution and the id: [" + workflowInstId
335             + "] " + "was never persisted to " + "the instance repository");
336         LOG.log(Level.SEVERE, e.getMessage());
337         return new Metadata();
338       }
339     }
340 
341     return worker.getWorkflowInstance().getSharedContext();
342   }

357   protected static double getWallClockMinutes(WorkflowInstance inst) {
358     if (inst == null) {
359       return 0.0;
360     }
361 
362     Date currentDateOrStopTime = (inst.getEndDateTimeIsoStr() != null
363         && !inst.getEndDateTimeIsoStr().equals("") && !inst
364         .getEndDateTimeIsoStr().equals("null")) ? safeDateConvert(inst
365         .getEndDateTimeIsoStr()) : new Date();
366 
367     Date workflowStartDateTime;
368 
369     if (inst.getStartDateTimeIsoStr() == null || ((inst
370                                                        .getStartDateTimeIsoStr().equals("") || inst
371                                                        .getStartDateTimeIsoStr().equals("null")))) {
372       return 0.0;
373     }
374 
375     try {
376       workflowStartDateTime = DateConvert.isoParse(inst
377           .getStartDateTimeIsoStr());
378     } catch (ParseException e) {
379       return 0.0;
380     }
381 
382     long diffMs = currentDateOrStopTime.getTime()
383         - workflowStartDateTime.getTime();
384     double diffSecs = (diffMs * 1.0 / 1000.0);
385     return diffSecs / 60.0;
386 
387   }

389   protected static double getCurrentTaskWallClockMinutes(WorkflowInstance inst) {
390     if (inst == null) {
391       return 0.0;
392     }
393 
394     Date currentDateOrStopTime = (inst.getCurrentTaskEndDateTimeIsoStr() != null
395         && !inst.getCurrentTaskEndDateTimeIsoStr().equals("") && !inst
396         .getCurrentTaskEndDateTimeIsoStr().equals("null")) ? safeDateConvert(inst
397         .getCurrentTaskEndDateTimeIsoStr()) : new Date();
398 
399     Date workflowTaskStartDateTime;
400 
401     if (inst.getCurrentTaskStartDateTimeIsoStr() == null || ((inst
402                                                                   .getCurrentTaskStartDateTimeIsoStr().equals("")
403                                                               || inst
404                                                                   .getCurrentTaskStartDateTimeIsoStr().equals("null")))) {
405       return 0.0;
406     }
407 
408     try {
409       workflowTaskStartDateTime = DateConvert.isoParse(inst
410           .getCurrentTaskStartDateTimeIsoStr());
411     } catch (ParseException e) {
412       return 0.0;
413     }
414 
415     // should never be in this state, so return 0
416     if (workflowTaskStartDateTime.after(currentDateOrStopTime)) {
417       LOG.log(
418           Level.WARNING,
419           "Start date time: ["
420               + DateConvert.isoFormat(workflowTaskStartDateTime)
421               + " of workflow inst [" + inst.getId() + "] is AFTER "
422               + "End date time: ["
423               + DateConvert.isoFormat(currentDateOrStopTime)
424               + "] of workflow inst.");
425       return 0.0;
426     }
427 
428     long diffMs = currentDateOrStopTime.getTime()
429         - workflowTaskStartDateTime.getTime();
430     double diffSecs = (diffMs * 1.0 / 1000.0);
431     return diffSecs / 60.0;
432   }

434   private synchronized void persistWorkflowInstance(WorkflowInstance wInst)
435       throws EngineException {
436 
437     try {
438       if (wInst.getId() == null || (wInst.getId().equals(""))) {
439         // we have to persist it by adding it
440         // rather than updating it
441         instRep.addWorkflowInstance(wInst);
442 
443       } else {
444         // persist by update
445         instRep.updateWorkflowInstance(wInst);
446       }
447     } catch (InstanceRepositoryException e) {
448       LOG.log(Level.SEVERE, e.getMessage());
449       throw new EngineException(e.getMessage());
450     }
451 
452   }

457     } catch (Exception e) {
