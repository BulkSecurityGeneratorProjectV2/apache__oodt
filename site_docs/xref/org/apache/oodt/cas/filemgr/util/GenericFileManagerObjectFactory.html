64      /**
65       * <p>
66       * Constructs a new {@link DataTransfer} from the specified
67       * <code>serviceFactory</code>.
68       * </p>
69       * 
70       * @param serviceFactory
71       *            The Service Factory class name that will be instantiated to
72       *            provide DataTransfer objects.
73       * @return A newly instantiated {@link DataTransfer} object.
74       */
75      @SuppressWarnings("unchecked")
76      public static DataTransfer getDataTransferServiceFromFactory(
77              String serviceFactory) {
78          DataTransferFactory dataTransferFactory;
79          Class<DataTransferFactory> dataTransferFactoryClass;
80  
81          try {
82              dataTransferFactoryClass = (Class<DataTransferFactory>) Class.forName(serviceFactory);
83              dataTransferFactory = dataTransferFactoryClass.newInstance();
84              return dataTransferFactory.createDataTransfer();
85          } catch (ClassNotFoundException e) {
86              LOG.log(Level.SEVERE, e.getMessage());
87              LOG.log(Level.WARNING,
88                      "ClassNotFoundException when loading data transfer factory class "
89                              + serviceFactory + " Message: " + e.getMessage());
90          } catch (InstantiationException e) {
91              LOG.log(Level.SEVERE, e.getMessage());
92              LOG.log(Level.WARNING,
93                      "InstantiationException when loading data transfer factory class "
94                              + serviceFactory + " Message: " + e.getMessage());
95          } catch (IllegalAccessException e) {
96              LOG.log(Level.SEVERE, e.getMessage());
97              LOG.log(Level.WARNING,
98                      "IllegalAccessException when loading data transfer factory class "
99                              + serviceFactory + " Message: " + e.getMessage());
100         }
101 
102         return null;
103     }
