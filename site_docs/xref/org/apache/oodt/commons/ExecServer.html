63    /** Start a server.
64  	 *
65  	 * The command-line should have two arguments:
66  	 *
67  	 * <ol>
68  	 *   <li>The fully-qualified class name of the server to execute.</li>
69  	 *   <li>The object name of the server to register with the naming service.</li>
70  	 * </ol>
71  	 *
72  	 * @param argv The command-line arguments
73  	 */
74  	public static void main(String[] argv) {
75  		if (argv.length < 2) {
76  			System.err.println("Usage: class-name-of-server object-name");
77  			System.exit(1);
78  		}
79  
80  		String className = argv[0];
81  		String name = argv[1];
82  
83  		// Enable support of our special URLs, like stdin:
84  		System.setProperty("java.protocol.handler.pkgs", "org.apache.oodt.commons.net.protocol");
85  
86  		try {
87  			// Get the configuration.
88  			configuration = Configuration.getConfiguration();
89  			configuration.mergeProperties(System.getProperties());
90  
91  			// Set up the logger.
92  			LogInit.init(System.getProperties(), name);
93  
94  			// Run initializers
95  			try {
96  				runInitializers();
97  			} catch (EDAException ex) {
98  				ex.printStackTrace();
99  				System.exit(1);
100 			}
101 
102 			// Create it.
103 			final ExecServer server = new ExecServer(name, className);
104 
105 			// Print it.
106 			if (Boolean.getBoolean(PRINT_IOR_PROPERTY)) {
107 				if (server.getServant() instanceof RemoteObject) {
108 					RemoteObject remoteObject = (RemoteObject) server.getServant();
109 					RemoteStub remoteStub = (RemoteStub) RemoteObject.toStub(remoteObject);
110 					RemoteRef ref = remoteStub.getRef();
111 					System.out.print("RMI:");
112 					System.out.flush();
113 					ObjectOutputStream objOut
114 						= new ObjectOutputStream(new Base64EncodingOutputStream(System.out));
115 					objOut.writeObject(ref);
116 					objOut.flush();
117 					System.out.println();
118 				} else {
119 					org.omg.PortableServer.Servant servant=(org.omg.PortableServer.Servant)server.getServant();
120 					org.omg.CORBA.ORB orb = servant._orb();
121 					System.out.println(orb.object_to_string(servant._this_object(orb)));
122 				}
123 				System.out.flush();
124 			}
125 
126 			// Bind it.
127 			if (!Boolean.getBoolean(DISABLE_BINDING)) {
128 				binder = new Binder(name, server);
129 				binder.start();
130 			}
131 
132 			// Prepare for the inevitable
133 			Runtime.getRuntime().addShutdownHook(new Thread() {
134 				public void run() {
135 					server.shutdown0();
136 				}
137 			});
138 
139 			// We're done here.
140 			for (;;) {
141 			  try {
142 				Thread.currentThread().join();
143 			  } catch (InterruptedException ignore) {
144 			  }
145 			}
146 		} catch (IOException ex) {
147 			System.err.println("I/O error during initialization: " + ex.getMessage());
148 			ex.printStackTrace();
149 		} catch (SAXParseException ex) {
150 			System.err.println("Error in the configuration file at line " + ex.getLineNumber() + ", column "
151 				+ ex.getColumnNumber() + ": " + ex.getMessage());
152 		} catch (SAXException ex) {
153 			System.err.println("Error " + ex.getClass().getName() + " while attempting to parse the configuration"
154 				+ " file: " + ex.getMessage());
155 		} catch (java.lang.reflect.InvocationTargetException ex) {
156 			Throwable target = ex.getTargetException();
157 			System.err.println("Constructor for \"" + className + "\" threw " + target.getClass().getName() + ": "
158 				+ ex.getMessage());
159 			target.printStackTrace();
160 		} catch (RuntimeException ex) {
161 			throw ex;
162 		} catch (Exception ex) {
163 			System.err.println("Exception " + ex.getClass().getName() + " initializing server \"" + name
164 				+ "\" with class \"" + className + "\": " + ex.getMessage());
165 			ex.printStackTrace();
166 		} 
167 		System.exit(1);
168 	}

175 	 *
176 	 * @param name Name of the server
177 	 * @param className Name of class that implements the server.
178 	 * @throws ClassNotFoundException If the class for <var>className</var> can't be found.
179 	 * @throws NoSuchMethodException If the constructor for <var>className</var> taking a single <code>ExecServer</code>
180 	 *         can't be found.
181 	 * @throws InstantiationException If the class for <var>className</var> is abstract or is an interface.
182 	 * @throws IllegalAccessException If the class for <var>className</var> isn't public.
183 	 * @throws InvocationTargetException If an exception occurs in the constructor for <var>className</var>.
184 	 * @throws DOMException If the server's status document can't be created.
185 	 * @throws UnknownHostException If the local host name can't be determined.
186 	 */
187 	public ExecServer(String name, String className) throws ClassNotFoundException, NoSuchMethodException,
188 		InstantiationException, IllegalAccessException, InvocationTargetException, DOMException, UnknownHostException {
189 		this.name = name;
190 
191 		// Find the class and the required constructor.
192 		Class clazz = Class.forName(className);
193 		Constructor ctor = clazz.getConstructor(new Class[]{ExecServer.class});
194 
195 		// Invoke the constructor to create the servant.
196 		servant = ctor.newInstance(new Object[]{this});
197 		Date startDate = new Date();
198 
199 		// Create the XML-RPC interface to this server.
200 		xmlrpcServer = new XmlRpcServer();
201 		xmlrpcServer.addHandler("server", this);
202 
203 		// Create the server status document.
204 		DocumentType docType = XML.getDOMImplementation().createDocumentType("server", STATUS_FPI, STATUS_URL);
205 		statusDocument = XML.getDOMImplementation().createDocument(/*namespaceURI*/null, "server", docType);
206 		Element serverElement = statusDocument.getDocumentElement();
207 		XML.add(serverElement, "name", name);
208 		XML.add(serverElement, "class", className);
209 		XML.add(serverElement, "state", "up");
210 		Element startElement = statusDocument.createElement("start");
211 		serverElement.appendChild(startElement);
212 		Element userElement = statusDocument.createElement("user");
213 		startElement.appendChild(userElement);
214 		XML.add(userElement, "name", System.getProperty("user.name", "UNKNOWN"));
215 		XML.add(userElement, "cwd", System.getProperty("user.dir", "UNKNOWN"));
216 		XML.add(userElement, "home", System.getProperty("user.home", "UNKNOWN"));
217 		Element dateElement = statusDocument.createElement("date");
218 		startElement.appendChild(dateElement);
219 		dateElement.setAttribute("ms", String.valueOf(startDate.getTime()));
220 		dateElement.appendChild(statusDocument.createTextNode(startDate.toString()));
221 		XML.add(startElement, "config", System.getProperty("org.apache.oodt.commons.Configuration.url", "UNKNOWN"));
222 		Element hostElement = statusDocument.createElement("host");
223 		serverElement.appendChild(hostElement);
224 		XML.add(hostElement, "name", InetAddress.getLocalHost().getHostName());
225 		Element osElement = statusDocument.createElement("os");
226 		hostElement.appendChild(osElement);
227 		XML.add(osElement, "name", System.getProperty("os.name", "UNKNOWN"));
228 		XML.add(osElement, "version", System.getProperty("os.version", "UNKNOWN"));
229 		XML.add(osElement, "arch", System.getProperty("os.arch", "UNKNOWN"));
230 		Element vmElement = statusDocument.createElement("vm");
231 		serverElement.appendChild(vmElement);
232 		XML.add(vmElement, "name", System.getProperty("java.vm.name", "UNKNOWN"));
233 		XML.add(vmElement, "version", System.getProperty("java.version", "UNKNOWN"));
234 		XML.add(vmElement, "classpath", System.getProperty("java.class.path", "UNKNOWN"));
235 		XML.add(vmElement, "extdirs", System.getProperty("java.ext.dirs", "UNKNOWN"));
236 		logElement = statusDocument.createElement("log");
237 		serverElement.appendChild(logElement);
238 	}

275 	 * @return Its status.
276 	 */
277 	public String getServerStatus() {
278 		// Update the status document with the current log.
279 	  for (Object o : LogInit.MEMORY_LOGGER.getMessages()) {
280 		String message = (String) o;
281 		Element messageElement = statusDocument.createElement("message");
282 		messageElement.setAttribute("xml:space", "preserve");
283 		messageElement.appendChild(statusDocument.createTextNode(message));
284 		logElement.appendChild(messageElement);
285 	  }
286 
287 		// Serialize the document.
288 		String rc = XML.serialize(statusDocument);
289 
290 		// Remove all the log messages from the document.
291 		NodeList children = logElement.getChildNodes();
292 		for (int i = 0; i < children.getLength(); ++i) {
293 		  logElement.removeChild(children.item(i));
294 		}
295 
296 		// Return the serialized form, which included the log messages.
297 		System.err.println(rc);
298 		return rc;
299 	}

383 	/**
384 	 * Binding thread.
385 	 */
386 	private static class Binder extends Thread {
387 		public Binder(String name, ExecServer server) {
388 			super("Binder for " + name);
389 			setDaemon(true);
390 			this.name = name;
391 			this.server = server;
392 			keepBinding = true;
393 		}
394 		public void run() {
395 			while (shouldKeepBinding()) {
396 			  try {
397 				Context objectContext = configuration.getObjectContext();
398 				objectContext.rebind(name, server.getServant());
399 				objectContext.close();
400 			  } catch (Exception ex) {
401 				System.err.println("Exception binding at " + new Date() + "; will keep trying...");
402 				ex.printStackTrace();
403 			  } finally {
404 				try {
405 				  Thread.sleep(REBIND_PERIOD);
406 				} catch (InterruptedException ignore) {
407 				}
408 			  }
409 			}
410 		}
411 		public synchronized void stopBinding() {
412 			keepBinding = false;
413 		}
414 		private synchronized boolean shouldKeepBinding() {
415 			return keepBinding;
416 		}
417 		private boolean keepBinding;
418 		private String name;
419 		private ExecServer server;
420 	}

422 	/**
423 	 * Run all initializers.
424 	 *
425 	 * This instantiates and calls the {@link Initializer#initialize} method of each
426 	 * initializer specified by class name in a comma separated list of classes in the
427 	 * system properties.  The property name is <code>org.apache.oodt.commons.initializers</code>, or
428 	 * if not defined, <code>org.apache.oodt.commons.ExecServer.initializers</code>, or if not
429 	 * defined, <code>initializers</code>.  And if that one's not defined, then none
430 	 * are run.
431 	 *
432 	 * @throws EDAException if an error occurs.
433 	 */
434 	public static void runInitializers() throws EDAException {
435 		String initList = System.getProperty("org.apache.oodt.commons.initializers",
436 			System.getProperty("org.apache.oodt.commons.ExecServer.initializers", System.getProperty("initializers", "")));
437 		for (Iterator i = org.apache.oodt.commons.util.Utility.parseCommaList(initList); i.hasNext();) {
438 			String iname = (String) i.next();
439 			try {
440 				Class initClass = Class.forName(iname);
441 				Initializer init = (Initializer) initClass.newInstance();
442 				init.initialize();
443 			} catch (ClassNotFoundException ex) {
444 				System.err.println("Initializer \"" + iname + "\" not found; aborting");
445 				throw new EDAException(ex);
446 			} catch (InstantiationException ex) {
447 				System.err.println("Initializer \"" + iname + "\" is abstract; aborting");
448 				throw new EDAException(ex);
449 			} catch (IllegalAccessException ex) {
450 				System.err.println("Initializer \"" + iname + "\" isn't public; aborting");
451 				throw new EDAException(ex);
452 			}
453 		}
454 	}
