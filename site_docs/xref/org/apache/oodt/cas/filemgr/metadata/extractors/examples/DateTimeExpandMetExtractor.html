17  package org.apache.oodt.cas.filemgr.metadata.extractors.examples;

19  // OODT imports
20  import org.apache.oodt.cas.filemgr.metadata.extractors.AbstractFilemgrMetExtractor;
21  import org.apache.oodt.cas.filemgr.structs.Product;
22  import org.apache.oodt.cas.metadata.Metadata;
23  import org.apache.oodt.cas.metadata.exceptions.MetExtractionException;
24  import org.apache.oodt.cas.metadata.util.PathUtils;

26  // Google imports
27  import com.google.common.base.Strings;

29  /**
30   * Expands a given metadata date time field out into multiple metadata fields (i.e. year, month,
31   * day, hour, minute, and second).
32   * 
33   * @author bfoster@apache.org (Brian Foster)
34   */
35  public class DateTimeExpandMetExtractor extends AbstractFilemgrMetExtractor {
36  
37    private static final String FULL_DATE_TIME_KEY = "FullDateTimeKey";
38    private static final String FULL_DATE_TIME_FORMAT = "FullDateTimeFormat";
39  
40    private static final String YEAR_KEY = "YearKey";
41    private static final String MONTH_KEY = "MonthKey";
42    private static final String DAY_KEY = "DayKey";
43    private static final String HOUR_KEY = "HourKey";
44    private static final String MINUTE_KEY = "MinuteKey";
45    private static final String SECOND_KEY = "SecondKey";
46  
47    private String fullDateTimeKey;
48    private String fullDateTimeFormat;
49    private String yearKey;
50    private String monthKey;
51    private String dayKey;
52    private String hourKey;
53    private String minuteKey;
54    private String secondKey;
55  
56    @Override
57    public void doConfigure() {
58      fullDateTimeKey = getKey(FULL_DATE_TIME_KEY);
59      fullDateTimeFormat = getKey(FULL_DATE_TIME_FORMAT);
60      yearKey = getKey(YEAR_KEY);
61      monthKey = getKey(MONTH_KEY);
62      dayKey = getKey(DAY_KEY);
63      hourKey = getKey(HOUR_KEY);
64      minuteKey = getKey(MINUTE_KEY);
65      secondKey = getKey(SECOND_KEY);
66    }
67  
68    @Override
69    public Metadata doExtract(Product product, Metadata metadata) throws MetExtractionException {
70      String fullDateTime = getFullDateTime(metadata);
71  
72      createDateField(metadata, yearKey, fullDateTime, fullDateTimeFormat, "yyyy");
73      createDateField(metadata, monthKey, fullDateTime, fullDateTimeFormat, "MM");
74      createDateField(metadata, dayKey, fullDateTime, fullDateTimeFormat, "dd");
75      createDateField(metadata, hourKey, fullDateTime, fullDateTimeFormat, "HH");
76      createDateField(metadata, minuteKey, fullDateTime, fullDateTimeFormat, "mm");
77      createDateField(metadata, secondKey, fullDateTime, fullDateTimeFormat, "ss");
78      return metadata;
79    }
80  
81    private String getKey(String key) {
82      if (configuration.containsKey(key)) {
83        return configuration.getProperty(key);
84      } else {
85        return null;
86      }
87    }
88  
89    private String getFullDateTime(Metadata metadata)
90         throws MetExtractionException {
91      if (!Strings.isNullOrEmpty(fullDateTimeKey)) {
92        return metadata.getMetadata(fullDateTimeKey);
93      } else {
94        throw new MetExtractionException("Failed to find DateTimeKey " + fullDateTimeKey);
95      }
96    }
97  
98    private void createDateField(Metadata metadata, String fieldKey, String fullDateTime,
99        String fullDateTimeFormat, String fieldFormat) throws MetExtractionException {
100     if (!Strings.isNullOrEmpty(fieldKey)) {
101       try {
102         metadata.addMetadata(fieldKey, PathUtils.doDynamicReplacement(String.format(
103             "[FORMAT(%s,%s,%s)]", fullDateTimeFormat, fullDateTime, fieldFormat)));
104       } catch (Exception e) {
105         throw new MetExtractionException(String.format(
106             "Failed to create field for key %s from fullDateTime %s and fullDateTimeFormat %s",
107             fieldKey, fullDateTime, fullDateTimeFormat), e);
108       }
109     }
110   }
