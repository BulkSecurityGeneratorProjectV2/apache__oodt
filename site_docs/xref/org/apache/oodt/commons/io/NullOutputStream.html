16  package org.apache.oodt.commons.io;
17  
18  import java.io.*;
19  
20  /** A null output stream.
21   *
22   * This output stream throws away all data it gets.
23   *
24   * @author Kelly
25   */
26  public class NullOutputStream extends OutputStream {
27  	/** Construct a null output stream.
28  	 */
29  	public NullOutputStream() {
30  		open = true;
31  	}
32  
33  	/** Write a byte to the output stream, which is thrown away.
34  	 *
35  	 * @param b The byte to toss.
36  	 * @throws IOException If the stream is closed.
37  	 */
38  	public void write(int b) throws IOException {
39  		checkOpen();
40  	}
41  
42  	/** Write a byte array to the output stream, which is thrown away.
43  	 *
44  	 * @param a The array to write.
45  	 * @param offset Where in the array to ignore bytes to write.
46  	 * @param length How many bytes to ignore.
47  	 * @throws IOException If the stream is closed.
48  	 */
49  	public void write(byte[] a, int offset, int length) throws IOException {
50  		if (a == null) {
51  		  throw new NullPointerException("Can't write a null array");
52  		} else if ((offset < 0) || (offset > a.length) || (length < 0) || ((offset + length) > a.length)
53  			|| ((offset + length) < 0)) {
54  		  throw new IndexOutOfBoundsException("Offset " + offset + " and length " + length
55  											  + " not in array of length " + a.length);
56  		}
57  		checkOpen();
58  	}
59  
60  	/** Flush an output stream, which does nothing.
61  	 *
62  	 * @throws IOException If the stream is closed.
63  	 */
64  	public void flush() throws IOException {
65  		checkOpen();
66  	}
67  
68  	/** Close an output stream.
69  	 *
70  	 * @throws IOException If the stream is already closed.
71  	 */
72  	public void close() throws IOException {
73  		checkOpen();
74  		open = false;
75  	}
76  
77  	/** Check if we're open.
78  	 *
79  	 * @throws IOException If we're not open.
80  	 */
81  	private void checkOpen() throws IOException {
82  		if (!open) {
83  		  throw new IOException("Stream closed");
84  		}
85  	}
86  
87  	/** Is the output stream open? */
88  	private boolean open;
89  }
