71      /**
72       * Convert the configuration to XML.
73       *
74       * @param owner Owning document.
75       * @return XML representation of this configuration.
76       */
77      public Node toXML(Document owner) {
78          Element elem = owner.createElement("configuration");                    // <configuration>
79          elem.setAttribute("xmlns", NS);                                         // Set default namespace
80          elem.setAttribute("https", httpsRequired? "true" : "false");            // Add https attribute
81          elem.setAttribute("localhost", localhostRequired? "true" : "false");    // Add localhost attribute
82  
83          if (password != null && password.length > 0)                            // If we have a password
84          {
85              elem.setAttribute("password", encode(password));                    // Add passowrd attribute
86          }
87  
88          for (Object productServer : productServers) {            // For each product server
89              ProductServer ps = (ProductServer) productServer;                        // Get the product server
90              elem.appendChild(ps.toXML(owner));                                  // And add it under <configuration>
91          }
92  
93          for (Object profileServer : profileServers) {            // For each profile server
94              ProfileServer ps = (ProfileServer) profileServer;                        // Get the profile server
95              elem.appendChild(ps.toXML(owner));                                  // And add it under the <configuration>
96          }
97  
98          if (!codeBases.isEmpty()) {                                             // Got any code bases?
99              Element cbs = owner.createElement("codeBases");                     // Boo yah.  Make a parent for 'em
100             elem.appendChild(cbs);                                              // Add parent
101             for (Object codeBase : codeBases) {             // Then, for each code base
102                 URL url = (URL) codeBase;                                       // Get the URL to it
103                 Element cb = owner.createElement("codeBase");                   // And make a <codeBase> for it
104                 cb.setAttribute("url", url.toString());                         // And an "url" attribute
105                 cbs.appendChild(cb);                                            // Add it
106             }
107         }
108 
109         if (!properties.isEmpty()) {                                            // If we have any properties>
110             Element props = owner.createElement("properties");                  // Add <properties> under <configuration>
111             props.setAttribute("xml:space", "preserve");                        // And make sure space is properly preserved
112             elem.appendChild(props);                                            // Add the space attribute
113             for (Map.Entry<Object, Object> objectObjectEntry : properties.entrySet()) { // For each property
114                 Map.Entry entry =
115                     (Map.Entry) objectObjectEntry;                         // Get the property key/value pair
116                 String key = (String) entry.getKey();                           // Key is always a String
117                 String value = (String) entry.getValue();                       // So is the value
118                 Element prop = owner.createElement("property");                 // Create a <property> element
119                 props.appendChild(prop);                                        // Add it under the <properties>
120                 prop.setAttribute("key", key);                                  // Set the key as an attribute
121                 Text text = owner.createTextNode(value);                        // Make text to hold the value
122                 prop.appendChild(text);                                         // Add it under the <property>
123             }
124         }
125         return elem;
126     }

221     /**
222      * Save the configuration.
223      *
224      * @throws IOException if an I/O error occurs.
225      */
226     public synchronized void save() throws IOException {
227         BufferedWriter writer = null;                           // Start w/no writer
228         try {                                                   // Then try ...
229             writer = new BufferedWriter(new FileWriter(file));  // Create a writer
230             Document doc;                                       // As for the doc...
231             synchronized (DOCUMENT_BUILDER) {                   // Using the document builder...
232                 doc = DOCUMENT_BUILDER.newDocument();           // Create an empty document
233             }
234             Node root = toXML(doc);                             // Convert this config to XML
235             doc.appendChild(root);                              // Add it to the doc
236             DOMSource source = new DOMSource(doc);              // Use the source Luke
237             StreamResult result = new StreamResult(writer);     // And serialize it to the writer
238             TRANSFORMER.transform(source, result);              // Serialize
239         } catch (TransformerException ex) {
240             throw new IllegalStateException("Unexpected TransformerException: " + ex.getMessage());
241         } finally {
242             if (writer != null) {
243                 try {                           // And if we got a writer, try ...
244                     writer.close();                                 // to close it
245                 } catch (IOException ignore) {
246                 }                     // Ignoring any error
247             }
248         }
249     }

251     /**
252      * Parse a serialized configuration document.
253      *
254      * @param file File to parse
255      * @throws IOException if an I/O error occurs.
256      * @throws SAXException if a parse error occurs.
257      */
258     private void parse(File file) throws IOException, SAXException {
259         Document doc;                                                               // Start with a doc ...
260         synchronized (DOCUMENT_BUILDER) {                                           // And using the DOCUMENT_BUILDER
261             doc = DOCUMENT_BUILDER.parse(file);                                     // Try to parse the file
262         }
263         Element root = doc.getDocumentElement();                                    // Assume the root element is <configuration>
264 
265         String httpsAttr = root.getAttribute("https");                              // Get the https attribute
266         httpsRequired = httpsAttr != null && "true".equals(httpsAttr);              // See if it's "true"
267 
268         String localhostAttr = root.getAttribute("localhost");                      // Get the localhost attribute
269         localhostRequired = localhostAttr != null && "true".equals(localhostAttr);  // See if it's "true"
270 
271         String passwordAttr = root.getAttribute("password");                        // Get the password attribute
272         if (passwordAttr != null && passwordAttr.length() > 0)                      // If it's there, and non-empty
273         {
274             password = decode(passwordAttr);                                        // Then decode it
275         }
276 
277         NodeList children = root.getChildNodes();                                   // Get the child nodes
278         for (int i = 0; i < children.getLength(); ++i) {                            // For each child node
279             Node child = children.item(i);                                          // Get the child
280             if (child.getNodeType() == Node.ELEMENT_NODE) {                         // An element?
281                 if ("server".equals(child.getNodeName())) {                         // A <server>?
282                     Server server = Server.create(this, (Element) child);           // Create the correct server
283 
284                     // Keep these in separate sets?
285                     if (server instanceof ProductServer)                            // Is a product server?
286                     {
287                         productServers.add(server);                                 // Add to product servers
288                     } else if (server instanceof ProfileServer)                       // Is a profile server?
289                     {
290                         profileServers.add(server);                                 // Add to profile servers
291                     } else {
292                         throw new IllegalArgumentException("Unexpected server type " + server + " in " + file);
293                     }
294                 } else if ("properties".equals(child.getNodeName())) {              // Is it a <properties>?
295                     NodeList props = child.getChildNodes();                         // Get its children
296                     for (int j = 0; j < props.getLength(); ++j) {                   // For each child
297                         Node node = props.item(j);                                  // Get the chld
298                         if (node.getNodeType() == Node.ELEMENT_NODE                 // And element?
299                             && "property".equals(node.getNodeName())) {             // And it's <property>?
300                             Element propNode = (Element) node;                      // Great, use it as an element
301                             String key = propNode.getAttribute("key");              // Get its key attribute
302                             if (key == null || key.length() == 0)                   // Make sure it's there
303                             {
304                                 throw new SAXException("Required 'key' attribute missing from "
305                                                        + "<property> element");
306                             }
307                             properties.setProperty(key, text(propNode));            // And set it
308                         }
309                     }
310                 } else if ("codeBases".equals(child.getNodeName())) {               // And yadda
311                     NodeList cbs = child.getChildNodes();                           // yadda
312                     for (int j = 0; j < cbs.getLength(); ++j) {                     // yadda.
313                         Node node = cbs.item(j);
314                         if (node.getNodeType() == Node.ELEMENT_NODE
315                             && "codeBase".equals(node.getNodeName())) {
316                             Element cbNode = (Element) node;
317                             String u = cbNode.getAttribute("url");
318                             if (u == null || u.length() == 0) {
319                                 throw new SAXException("Required 'url' attribute missing from "
320                                                        + "<codeBase> element");
321                             }
322                             try {
323                                 codeBases.add(new URL(u));
324                             } catch (MalformedURLException ex) {
325                                 throw new SAXException("url attribute " + u + " isn't a valid URL");
326                             }
327                         }
328                     }
329                 }
330             }
331         }
332     }

367     /**
368      * Get the text from an XML node into a StringBuffer. 
369      *
370      * @param node a <code>Node</code> value.
371      * @param b a <code>StringBuffer</code> value.
372      */
373     private static void text0(Node node, StringBuffer b) {
374         NodeList children = node.getChildNodes();
375         for (int i = 0; i < children.getLength(); ++i) {
376             text0(children.item(i), b);
377         }
378         short type = node.getNodeType();
379         if (type == Node.CDATA_SECTION_NODE || type == Node.TEXT_NODE) {
380             b.append(node.getNodeValue());
381         }

384     public boolean equals(Object obj) {
385         if (obj == this) {
386             return true;
387         }
388         if (obj instanceof Configuration) {
389             Configuration rhs = (Configuration) obj;
390             return codeBases.equals(rhs.codeBases) && productServers.equals(rhs.productServers)
391                 && profileServers.equals(rhs.profileServers) && Arrays.equals(password, rhs.password)
392                 && httpsRequired == rhs.httpsRequired && localhostRequired == rhs.localhostRequired
393                 && properties.equals(rhs.properties);
394         }
395         return false;
396     }
