45  public class DaemonLauncher implements DaemonLauncherMBean {
46  
47      /* our log stream */
48      private static final Logger LOG = Logger.getLogger(DaemonLauncher.class
49              .getName());
50  
51      private Config config;
52  
53      private RemoteSpecs rs;
54  
55      private int rmiRegistryPort;
56  
57      private int nextDaemonId;
58  
59      private DaemonManager dm;
60  
61      private LinkedList<Daemon> activeDaemonList;
62  
63      private LinkedList<File> sitesFiles;
64  
65      private File propertiesFile;
66  
67      private MBeanServer mbs;
68  
69      public DaemonLauncher(int rmiRegistryPort, File propertiesFile,
70              LinkedList<File> sitesFiles) {
71          this.rmiRegistryPort = rmiRegistryPort;
72          nextDaemonId = 0;
73          dm = new DaemonManager();
74          activeDaemonList = new LinkedList<Daemon>();
75          this.sitesFiles = sitesFiles;
76          this.propertiesFile = propertiesFile;
77          registerJMX();
78          this.configure();
79      }
80  
81      private void registerJMX() {
82          try {
83              LocateRegistry.createRegistry(this.rmiRegistryPort);
84              mbs = ManagementFactory.getPlatformMBeanServer();
85              ObjectName name = new ObjectName(
86                      "org.apache.oodt.cas.pushpull.daemon:type=DaemonLauncher"
87                              + this.rmiRegistryPort);
88              mbs.registerMBean(this, name);
89          } catch (Exception e) {
90              LOG.log(Level.SEVERE,
91                      "Failed to register DaemonLauncher as a MBean Object : "
92                              + e.getMessage());
93          }
94      }
95  
96      private synchronized void configure() {
97          try {
98              LOG.log(Level.INFO, "Configuring DaemonLauncher. . .");
99              config = new Config();
100             config.loadConfigFile(propertiesFile);
101             rs = new RemoteSpecs();
102             for (File sitesFile : sitesFiles) {
103                 LOG
104                         .log(Level.INFO, "Loading SiteInfo file '" + sitesFile
105                                 + "'");
106                 rs.loadRemoteSpecs(sitesFile);
107             }
108         } catch (Exception e) {
109             LOG.log(Level.SEVERE, "Failed to configure DaemonLauncher : "
110                     + e.getMessage());
111         }
112     }
113 
114     public synchronized void launchDaemons() {
115         LOG.log(Level.INFO, "Launching Daemons . . .");
116         for (DaemonInfo daemonInfo : rs.getDaemonInfoList()) {
117             int curDaemonID = this.getNextDaemonId();
118             try {
119                 Daemon daemon = new Daemon(this.rmiRegistryPort, curDaemonID,
120                         this.config, daemonInfo, rs.getSiteInfo());
121                 LOG.log(Level.INFO, "Creating Daemon with ID = " + curDaemonID);
122                 daemon.registerDaemonListener(dm);
123                 daemon.startDaemon();
124                 activeDaemonList.add(daemon);
125             } catch (Exception e) {
126                 LOG.log(Level.SEVERE, "Failed to create Daemon with ID = "
127                         + curDaemonID + " : " + e.getMessage());
128             }
129         }
130     }
131 
132     private synchronized int getNextDaemonId() {
133         while (this.dm.getUsedIDs().contains(++this.nextDaemonId)) {
134 
135         }
136         return this.nextDaemonId;
137     }
138 
139     public void refreshDaemons() {
140         this.killAllDaemons();
141 
142         LOG.log(Level.INFO, "Refreshing Daemons . . .");
143         this.configure();
144         launchDaemons();
145     }
146 
147     public synchronized void killAllDaemons() {
148         LOG.log(Level.INFO, "Killing current Daemons . . .");
149         this.nextDaemonId = 0;
150         for (Daemon daemon : this.activeDaemonList) {
151             if (!daemon.getHasBeenToldToQuit()) {
152                 daemon.quit();
153             }
154         }
155         activeDaemonList.clear();
156     }
157 
158     public String[] viewDaemonWaitingList() {
159         return this.dm.getQueueList();
160     }
161 
162     public synchronized void quit() {
163         this.killAllDaemons();
164         this.notify();
165     }

167     public static void main(String[] args) {
168 
169         int rmiRegPort = -1;
170         File propertiesFile = null;
171         LinkedList<File> sitesFiles = new LinkedList<File>();
172 
173         for (int i = 0; i < args.length; ++i) {
174             if (args[i].equals("--rmiRegistryPort")) {
175                 rmiRegPort = Integer.parseInt(args[++i]);
176             } else if (args[i].equals("--propertiesFile")) {
177                 propertiesFile = new File(args[++i]);
178             } else if (args[i].equals("--remoteSpecsFile")) {
179                 sitesFiles.add(new File(args[++i]));
180             }
181         }
182 
183         DaemonLauncher daemonLauncher = new DaemonLauncher(rmiRegPort,
184                 propertiesFile, sitesFiles);
185 
186         daemonLauncher.launchDaemons();
187         synchronized (daemonLauncher) {
188             try {
189                 daemonLauncher.wait();
190             } catch (Exception ignored) {
191             }
192         }
193     }
