184         try{
185             buffReader = new BufferedReader (new FileReader(path));
186             String line = buffReader.readLine();
187             StringBuilder build = new StringBuilder();

281 	private File createPubKeyForPort(int port) throws IOException {
282 		PrintStream ps = null;
283 		BufferedReader br = null;
284 		try {
285 			File publicKeyFile = new File(publicKeysDir, "sample-dsa.pub");
286 			br = new BufferedReader(new FileReader(new File("src/test/resources/sample-dsa.pub").getAbsoluteFile()));
287 			ps = new PrintStream(new FileOutputStream(publicKeyFile));
288 			String nextLine;
289 			while ((nextLine = br.readLine()) != null) {
290 				ps.println(nextLine.replace("2022", Integer.toString(port)));
291 			}
292 			return publicKeyFile;
293 		} finally {
294 			try { ps.close(); } catch (Exception ignored) {}
295 			try { br.close(); } catch (Exception ignored) {}
