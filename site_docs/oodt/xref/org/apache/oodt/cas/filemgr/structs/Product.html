109     public Product(InputStream is) throws InstantiationException {
110         if (is == null) {
111             throw new InstantiationException(
112                     "Unable to parse product stream: stream not set!");
113         }
114 
115         try {
116             DocumentBuilderFactory factory = DocumentBuilderFactory
117                     .newInstance();
118             factory.setNamespaceAware(true);
119             DocumentBuilder parser = factory.newDocumentBuilder();
120             references = new Vector<Reference>();
121             parse(parser.parse(new InputSource(is)));
122         } catch (Exception e) {
123             throw new InstantiationException(
124                     "Unable to parse metadata stream.[" + e.getMessage() + "]");
125         }
126     }

277     public Document toXML() throws UnsupportedEncodingException, FileManagerException {
278         DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
279         factory.setNamespaceAware(true);
280         Document doc;
281 
282         try {
283             DocumentBuilder builder = factory.newDocumentBuilder();
284             doc = builder.newDocument();
285 
286             Element root = (Element) doc.createElement("cas:product");
287             XMLUtils.addAttribute(doc, root, "xmlns:cas",
288                     "http://oodt.jpl.nasa.gov/1.0/cas");
289             XMLUtils.addAttribute(doc, root, "id", this.productId);
290             XMLUtils.addAttribute(doc, root, "name", URLEncoder.encode(
291                     this.productName, "UTF-8"));
292             doc.appendChild(root);
293 
294             XMLUtils.addNode(doc, root, "structure", this.productStructure);
295             XMLUtils.addNode(doc, root, "transferStatus", this.transferStatus);
296             XMLUtils.addNode(doc, root, "type",
297                     this.productType != null ? this.productType.getName() : "");
298 
299             if (this.getProductReferences() != null
300                     && this.getProductReferences().size() > 0) {
301                 Element refsElem = XMLUtils.addNode(doc, root, "references");
302                 for (Reference r : this.getProductReferences()) {
303                     Element refElem = XMLUtils.addNode(doc, refsElem,
304                         "reference");
305                     XMLUtils.addAttribute(doc, refElem, "orig", r
306                         .getOrigReference());
307                     XMLUtils.addAttribute(doc, refElem, "dataStore", r
308                         .getDataStoreReference());
309                     XMLUtils.addAttribute(doc, refElem, "size", String
310                         .valueOf(r.getFileSize()));
311 
312                 }
313             }
314 
315         } catch (ParserConfigurationException pce) {
316             LOG.log(Level.WARNING, "Error generating product xml file!: "
317                     + pce.getMessage());
318             throw new FileManagerException("Error generating product xml file!: "
319                     + pce.getMessage());
320         }
321 
322         return doc;
323     }

325     private void parse(Document doc) {
326         Element root = doc.getDocumentElement();
327 
328         this.productId = root.getAttribute("id");
329         try {
330             this.productName = URLDecoder.decode(root.getAttribute("name"),
331                     "UTF-8");
332         } catch (UnsupportedEncodingException e) {
333             LOG.log(Level.WARNING,
334                     "Unable to set product name: error decoding encoded string: Message: "
335                             + e.getMessage());
336 
337         }
338 
339         this.productStructure = XMLUtils.getElementText("structure", root);
340         this.productType = new ProductType();
341         this.productType.setName(XMLUtils.getElementText("type", root));
342         this.transferStatus = XMLUtils.getElementText("transferStatus", root);
343 
344         Element refsElem = XMLUtils.getFirstElement("references", root);
345 
346         if (refsElem != null) {
347             NodeList refNodeList = refsElem.getElementsByTagName("reference");
348 
349             if (refNodeList != null && refNodeList.getLength() > 0) {
350                 for (int i = 0; i < refNodeList.getLength(); i++) {
351                     Element refElem = (Element) refNodeList.item(i);
352                     Reference r = new Reference();
353                     r.setOrigReference(refElem.getAttribute("orig"));
354                     r.setDataStoreReference(refElem.getAttribute("dataStore"));
355                     r.setFileSize(Long.valueOf(refElem.getAttribute("size")));
356                     this.references.add(r);
357                 }
358             }
359         }
