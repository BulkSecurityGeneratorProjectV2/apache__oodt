77    public void testOrdering() {
78      Product testProduct = getTestProduct();
79      Metadata testMet = getTestMetadata("test");
80  
81      try {
82        myCat.addProduct(testProduct);
83        myCat.addMetadata(testMet, testProduct);
84      } catch (Exception e) {
85        LOG.log(Level.SEVERE, e.getMessage());
86        fail(e.getMessage());
87      }
88  
89      Product retProduct;
90      try {
91        Metadata retMet = myCat.getMetadata(testProduct);
92        assertNotNull(retMet);
93        assertNotNull(retMet.getAllMetadata("CAS.ProductName"));
94        assertEquals(3, retMet.getAllMetadata("CAS.ProductName").size());
95        assertEquals("test", retMet.getAllMetadata("CAS.ProductName").get(0));
96        assertEquals("test2", retMet.getAllMetadata("CAS.ProductName").get(1));
97        assertEquals("test3", retMet.getAllMetadata("CAS.ProductName").get(2));
98      } catch (Exception e) {
99        fail(e.getMessage());
100     }
