41  /**
42   * @author mattmann
43   * @version $Revision$
44   * 
45   * <p>
46   * A tool to move products based on their metadata attributes.
47   * </p>
48   * 
49   */
50  public class MetadataBasedProductMover {
51  
52      public static final double DOUBLE = 1000.0;
53      /*
54               * the metadata path sepc string, e.g.,
55               * /path/to/final/loc/[LocalDay]/[Filename]
56               */
57      private String pathSpec = null;
58  
59      /* the client to the file manager */
60      private XmlRpcFileManagerClient fmgrClient = null;
61  
62      /* our log stream */
63      private static final Logger LOG = Logger
64              .getLogger(MetadataBasedProductMover.class.getName());
65  
66      /**
67       * Default constructor.
68       * 
69       * @param pathSpec
70       *            A path specification based on the product metadata for the
71       *            final data store location.
72       * @param fmUrl
73       *            A string URL to the file manager.
74       * @throws InstantiationException
75       *             If the passed in url is malformed.
76       */
77      public MetadataBasedProductMover(String pathSpec, String fmUrl)
78              throws InstantiationException {
79          this.pathSpec = pathSpec;
80          try {
81              this.fmgrClient = new XmlRpcFileManagerClient(new URL(fmUrl));
82          } catch (MalformedURLException e) {
83              throw new InstantiationException(e.getMessage());
84          } catch (ConnectionException e) {
85              throw new InstantiationException(e.getMessage());
86          }
87      }
88  
89      public void moveProducts(ProductType type) throws CatalogException, URISyntaxException, DataTransferException {
90          // paginate through the product list
91  
92          ProductPage page = fmgrClient.getFirstPage(type);
93  
94          for (int i = 0; i < page.getTotalPages(); i++) {
95              if (page.getPageProducts() != null
96                      && page.getPageProducts().size() > 0) {
97                  for (Product p : page.getPageProducts()) {
98                      p.setProductReferences(fmgrClient.getProductReferences(p));
99                      Metadata met = fmgrClient.getMetadata(p);
100                     Reference r = ((Reference) p.getProductReferences().get(0));
101                     String newLocPath = PathUtils.replaceEnvVariables(
102                         this.pathSpec, met);
103 
104                     if (locationsMatch(r.getDataStoreReference(), newLocPath)) {
105                         LOG.log(Level.INFO,
106                             "Current and New locations match. " + p.getProductName() + " was not moved.");
107                         continue;
108                     }
109 
110                     LOG.log(Level.INFO, "Moving product: ["
111                                         + p.getProductName() + "] from: ["
112                                         + new File(new URI(r.getDataStoreReference()))
113                                         + "] to: [" + newLocPath + "]");
114                     long timeBefore = System.currentTimeMillis();
115                     fmgrClient.moveProduct(p, newLocPath);
116                     long timeAfter = System.currentTimeMillis();
117                     double seconds = ((timeAfter - timeBefore) * 1.0) / DOUBLE;
118                     LOG.log(Level.INFO, "Product: [" + p.getProductName()
119                                         + "] move successful: took: [" + seconds
120                                         + "] seconds");
121                 }
122 
123                 if (!page.isLastPage()) {
124                     page = fmgrClient.getNextPage(type, page);
125                 }
126             }
127         }
128     }
129 
130     private boolean locationsMatch(String currentLocation, String newLocation) throws java.net.URISyntaxException {
131     	String currentLocationURI = new URI(currentLocation).getSchemeSpecificPart();
132     	String newLocationURI = new URI(newLocation).getSchemeSpecificPart();
133 
134         return currentLocationURI.equals(newLocationURI);
135     	
136     }
137     
138     public void moveProducts(String typeName)
139         throws RepositoryManagerException, CatalogException, DataTransferException, URISyntaxException {
140         moveProducts(fmgrClient.getProductTypeByName(typeName));
141     }
142 
143     /**
144      * @param args
145      */
146     public static void main(String[] args)
147         throws URISyntaxException, CatalogException, RepositoryManagerException, DataTransferException,
148         InstantiationException {
149         String typeName = null, pathSpec = null, fmUrlStr = null;
150         String usage = "MetadataBasedProductMover [options]\n"
151                 + "--typeName <product type>\n"
152                 + "--fileManagerUrl <url to file manager>\n"
153                 + "--pathSpec <path spec using '[' and ']' to delimit met fields>\n";
154 
155         for (int i = 0; i < args.length; i++) {
156             if (args[i].equals("--typeName")) {
157                 typeName = args[++i];
158             } else if (args[i].equals("--pathSpec")) {
159                 pathSpec = args[++i];
160             } else if (args[i].equals("--fileManagerUrl")) {
161                 fmUrlStr = args[++i];
162             }
163         }
164 
165         if (typeName == null || pathSpec == null || fmUrlStr == null) {
166             System.err.println(usage);
167             System.exit(1);
168         }
169 
170         MetadataBasedProductMover mover = new MetadataBasedProductMover(
171                 pathSpec, fmUrlStr);
172         mover.moveProducts(typeName);
173     }
