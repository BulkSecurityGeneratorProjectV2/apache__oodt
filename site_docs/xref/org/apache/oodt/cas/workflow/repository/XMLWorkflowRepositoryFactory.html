19  package org.apache.oodt.cas.workflow.repository;
20  
21  //JDK imports
22  import java.io.File;
23  import java.net.URI;
24  import java.net.URISyntaxException;
25  import java.util.ArrayList;
26  import java.util.List;
27  import java.util.Arrays;
28  import java.util.logging.Level;
29  import java.util.logging.Logger;
30  
31  import org.apache.oodt.cas.metadata.util.PathUtils;
32  import org.apache.oodt.commons.io.DirectorySelector;

35  /**
36   * @author mattmann
37   * @author luca
38   * @version $Revision$
39   * 
40   * <p>A Factory class for creating {@link XMLWorkflowRepository}s.</p>
41   */
42  public class XMLWorkflowRepositoryFactory implements WorkflowRepositoryFactory {
43  
44  	/* list of dir uris specifying file paths to workflow directories */
45  	private List<String> workflowDirList = null;
46  	
47  	/* our log stream */
48  	private static Logger LOG = Logger.getLogger(XMLWorkflowRepositoryFactory.class.getName());
49  	
50  	/**
51  	 * <p>Default Constructor</p>.
52  	 */
53  	public XMLWorkflowRepositoryFactory() {
54  		
55  		String workflowDirUris = System.getProperty("org.apache.oodt.cas.workflow.repo.dirs");
56  		
57          // only returns true if org.apache.oodt.cas.workflow.repo.dirs.recursive=true
58          boolean recursive = Boolean.parseBoolean( 
59          		System.getProperty("org.apache.oodt.cas.workflow.repo.dirs.recursive") );
60  		
61  		if(workflowDirUris != null){
62  			/* do env var replacement */
63  			workflowDirUris = PathUtils.replaceEnvVariables(workflowDirUris);
64  			String [] dirUris = workflowDirUris.split(",");
65  			
66  			// recursive directory listing
67  			if (recursive) {
68  				
69              	// empty list
70  				workflowDirList = new ArrayList<String>();
71              	
72              	// loop over specified root directories,
73              	// add directories and sub-directories that contain all workflow related XML files
74              	for (String rootDir : dirUris) {
75              		try {
76              			
77              			DirectorySelector dirsel = new DirectorySelector(
78              					Arrays.asList( 
79              							new String[] {"events.xml", "tasks.xml", "conditions.xml"} ));
80              			workflowDirList.addAll( dirsel.traverseDir(new File(new URI(rootDir))) );
81              			
82              		} catch (URISyntaxException e) {
83              			LOG.log(Level.WARNING, "URISyntaxException when traversing directory: "+rootDir);
84              		}
85              	}        
86  			
87  			// non-recursive directory listing
88  			} else {
89  				workflowDirList = Arrays.asList(dirUris);
90  			}
91  			
92              LOG.log(Level.FINE,"Collecting XML workflows from the following directories:");
93              for (String pdir : workflowDirList) {
94              	LOG.log(Level.FINE, pdir);
95              }
96  			
97  		}
98  		
99  	}
100 
101 	/* (non-Javadoc)
102 	 * @see org.apache.oodt.cas.workflow.repository.WorkflowRepositoryFactory#createRepository()
103 	 */
104 	public WorkflowRepository createRepository() {
105 		if(workflowDirList != null){
106 			return new XMLWorkflowRepository(workflowDirList);
107 		}
108 		else{
109 			LOG.log(Level.WARNING, "Cannot create XML Workflow Repository: no workflow dir uris specified: value: "+System.getProperty("org.apache.oodt.cas.workflow.repo.dirs"));
110 			return null;
111 		}
112 	}
