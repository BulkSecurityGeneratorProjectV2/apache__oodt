63      public static WorkflowInstanceMetMap parseMetMapFile(String mapFilePath)
64          throws CommonsException, WorkflowException {
65          Document doc = getDocumentRoot(mapFilePath);
66          Element rootElem = doc.getDocumentElement();
67          WorkflowInstanceMetMap map = new WorkflowInstanceMetMap();
68  
69          // make sure that there is a default tag
70          Element defaultElem = DOMUtil.getFirstElement(rootElem,
71                  DEFAULT_WORKFLOW_MAP);
72  
73          if (defaultElem == null) {
74              throw new WorkflowException("file: [" + mapFilePath
75                      + "] must specify a default workflow to field map!");
76          }
77  
78          map.addDefaultFields(readFields(defaultElem));
79  
80          NodeList workflowMapNodes = rootElem
81                  .getElementsByTagName(WORKFLOW_TAG_NAME);
82  
83          if (workflowMapNodes != null && workflowMapNodes.getLength() > 0) {
84              for (int i = 0; i < workflowMapNodes.getLength(); i++) {
85                  Element workflowMapElem = (Element) workflowMapNodes.item(i);
86                  String workflowId = workflowMapElem
87                          .getAttribute(WORKFLOW_TAG_ID_ATTR);
88                  List workflowFields = readFields(workflowMapElem);
89                  map.addWorkflowToMap(workflowId, workflowFields);
90              }
91          }
92          
93          return map;
94  
95      }
96  
97      private static List readFields(Element rootFldElem) {
98          NodeList defaultFldNodes = rootFldElem.getElementsByTagName(FIELD_TAG);
99          List fields = null;
100 
101         if (defaultFldNodes != null && defaultFldNodes.getLength() > 0) {
102             fields = new Vector(defaultFldNodes.getLength());
103             for (int i = 0; i < defaultFldNodes.getLength(); i++) {
104                 Element defaultFldElem = (Element) defaultFldNodes.item(i);
105                 fields.add(defaultFldElem.getAttribute(FIELD_TAG_NAME_ATTR));
106             }
107         }
108 
109         return fields;
110     }

112     private static Document getDocumentRoot(String xmlFile) {
113         // open up the XML file
114         DocumentBuilderFactory factory;
115         DocumentBuilder parser;
116         Document document;
117         InputSource inputSource;
118 
119         InputStream xmlInputStream;
120 
121         try {
122             xmlInputStream = new File(xmlFile).toURL().openStream();
123         } catch (IOException e) {
124             LOG.log(Level.WARNING,
125                     "IOException when getting input stream from [" + xmlFile
126                             + "]: returning null document root");
127             return null;
128         }
129 
130         inputSource = new InputSource(xmlInputStream);
131 
132         try {
133             factory = DocumentBuilderFactory.newInstance();
134             parser = factory.newDocumentBuilder();
135             document = parser.parse(inputSource);
136         } catch (Exception e) {
137             LOG.warning("Unable to parse xml file [" + xmlFile + "]."
138                     + "Reason is [" + e + "]");
139             return null;
140         }
141 
142         return document;
143     }
