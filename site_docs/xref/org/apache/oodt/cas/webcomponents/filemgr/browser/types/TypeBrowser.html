223   private void refreshProductPage() {
224     Query query = new Query();
225     System.out.println("CALLING REFRESH PRODUCT PAGE, CRITERIA:");
226     for (TermQueryCriteria crit : this.criteria) {
227       System.out.println(crit);
228     }
229     query.getCriteria().addAll(this.criteria);
230     try {
231       this.productPage = fm.getFm().pagedQuery(query, type, this.pageNum);
232     } catch (CatalogException e) {
233       LOG.log(Level.SEVERE, "Unable to obtain page products: type: ["
234           + type.getName() + "]: Reason: " + e.getMessage());
235     }
236   }

238   private void computeStartEndIdx() {
239     if (this.productPage.getTotalPages() == 1) {
240       this.totalProducts = this.productPage.getPageProducts().size();
241       this.pageNum = 1;
242     } else if (productPage.getTotalPages() == 0) {
243       this.totalProducts = 0;
244       this.pageNum = 1;
245     } else {
246       this.totalProducts = (productPage.getTotalPages() - 1) * PAGE_SIZE;
247       this.pageNum = this.productPage.getPageNum();
248 
249       // get the last page
250       ProductPage lastPage;
251       Query query = new Query();
252       query.getCriteria().addAll(this.criteria);
253 
254       try {
255         lastPage = fm.getFm().pagedQuery(query, this.type,
256             this.productPage.getTotalPages());
257         this.totalProducts += lastPage.getPageProducts().size();
258       } catch (Exception ignore) {
259       }
260     }
261     this.endIdx = this.totalProducts != 0 ? Math.min(this.totalProducts,
262         (PAGE_SIZE) * (this.pageNum)) : 0;
263     this.startIdx = this.totalProducts != 0 ? ((this.pageNum - 1) * PAGE_SIZE) + 1
264         : 0;
265   }

303   class AddCriteriaForm extends Form<ElementCrit> {
304 
305     private static final long serialVersionUID = -4543355158252106121L;
306 
307     /**
308      * @param id
309      *          The wicket:id component ID of this form.
310      */
311     public AddCriteriaForm(String id) {
312       super(id, new CompoundPropertyModel<ElementCrit>(new ElementCrit()));
313       List<Element> ptypeElements = fm.safeGetElementsForProductType(type);
314       Collections.sort(ptypeElements, new Comparator<Element>() {
315         public int compare(Element e1, Element e2) {
316           return e1.getElementName().compareTo(e2.getElementName());
317         }
318       });
319 
320       add(new DropDownChoice<Element>("criteria_list", new PropertyModel(
321           getDefaultModelObject(), "elem"), new ListModel<Element>(
322           ptypeElements), new ChoiceRenderer<Element>("elementName",
323           "elementId")));
324       add(new TextField<TermQueryCriteria>(
325           "criteria_form_add_element_value",
326           new PropertyModel<TermQueryCriteria>(getDefaultModelObject(), "value")));
327       add(new Button("criteria_elem_add"));
328     }
329 
330     /*
331      * (non-Javadoc)
332      * 
333      * @see org.apache.wicket.markup.html.form.Form#onSubmit()
334      */
335     @Override
336     protected void onSubmit() {
337       ElementCrit tc = (ElementCrit) getDefaultModelObject();
338       for (TermQueryCriteria tqc : criteria) {
339         if (tqc.getElementName().equals(tc.getElem().getElementName())) {
340           return;
341         }
342       }
343       criteria.add(new TermQueryCriteria(tc.getElem().getElementName(), tc
344           .getValue()));
345       ((FMBrowserSession) getSession()).setCriteria(criteria);
346       refreshProductPage();
347       computeStartEndIdx();
348       PageParameters parameters = new PageParameters();
349       parameters.add("name", type.getName());
350       parameters.add("pageNum", String.valueOf(pageNum));
351       setResponsePage(getPage().getClass(), parameters);
352     }
353   }

399   class TermQueryCriteriaRemoveButton extends Button {
400 
401     private TermQueryCriteria crit;
402 
403     public TermQueryCriteriaRemoveButton(String id, TermQueryCriteria crit) {
404       super(id);
405       this.crit = crit;
406     }
407 
408     /*
409      * (non-Javadoc)
410      * 
411      * @see org.apache.wicket.markup.html.form.Button#onSubmit()
412      */
413     @Override
414     public void onSubmit() {
415       for (int i = 0; i < criteria.size(); i++) {
416         TermQueryCriteria cr = criteria.get(i);
417         if (cr.getElementName().equals(crit.getElementName())) {
418           criteria.remove(i);
419           ((FMBrowserSession) getSession()).setCriteria(criteria);
420           refreshProductPage();
421           computeStartEndIdx();
422           PageParameters parameters = new PageParameters();
423           parameters.add("name", type.getName());
424           parameters.add("pageNum", String.valueOf(pageNum));
425           setResponsePage(getPage().getClass(), parameters);
426           break;
427         }
428       }
429     }
