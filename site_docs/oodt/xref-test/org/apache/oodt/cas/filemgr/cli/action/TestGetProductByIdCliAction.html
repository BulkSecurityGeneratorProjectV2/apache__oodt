77     public void testDataFlow() throws CmdLineActionException {
78        ActionMessagePrinter printer = new ActionMessagePrinter();
79        MockGetProductByIdCliAction cliAction = new MockGetProductByIdCliAction();
80        cliAction.setProductId(PRODUCT_ID);
81        cliAction.execute(printer);
82        assertEquals(20, printer.getPrintedMessages().size());
83        assertEquals("Product:", printer.getPrintedMessages().get(0));
84        assertEquals("\n", printer.getPrintedMessages().get(1));
85        assertEquals(" - ID: " + PRODUCT_ID, printer.getPrintedMessages().get(2));
86        assertEquals("\n", printer.getPrintedMessages().get(3));
87        assertEquals(" - Name: " + PRODUCT_NAME, printer.getPrintedMessages()
88              .get(4));
89        assertEquals("\n", printer.getPrintedMessages().get(5));
90        assertEquals(" - ProductType: " + PRODUCT_TYPE_NAME, printer
91              .getPrintedMessages().get(6));
92        assertEquals("\n", printer.getPrintedMessages().get(7));
93        assertEquals(" - Structure: " + PRODUCT_STRUCTURE, printer
94              .getPrintedMessages().get(8));
95        assertEquals("\n", printer.getPrintedMessages().get(9));
96        assertEquals(" - Status: " + PRODUCT_STATUS, printer.getPrintedMessages()
97              .get(10));
98        assertEquals("\n", printer.getPrintedMessages().get(11));
99        assertEquals(" - RootRef: " + ROOT_REF,
100             printer.getPrintedMessages().get(12));
101       assertEquals("\n", printer.getPrintedMessages().get(13));
102       assertEquals(" - References: ", printer.getPrintedMessages().get(14));
103       assertEquals("\n", printer.getPrintedMessages().get(15));
104       assertEquals("    - " + DS_REF_1 + " (" + FILE_SIZE_REF_1 + ")", printer
105             .getPrintedMessages().get(16));
106       assertEquals("\n", printer.getPrintedMessages().get(17));
107       assertEquals("    - " + DS_REF_2 + " (" + FILE_SIZE_REF_2 + ")", printer
108             .getPrintedMessages().get(18));
109       assertEquals("\n", printer.getPrintedMessages().get(19));
110    }

112    public class MockGetProductByIdCliAction extends GetProductByIdCliAction {
113       public XmlRpcFileManagerClient getClient() throws MalformedURLException,
114             ConnectionException {
115          return new XmlRpcFileManagerClient(new URL("http://localhost:9000"),
116                false) {
117             public Product getProductById(String productId) {
118                Product p = new Product();
119                p.setProductId(productId);
120                p.setProductName(PRODUCT_NAME);
121                ProductType pt = new ProductType();
122                pt.setName(PRODUCT_TYPE_NAME);
123                p.setProductType(pt);
124                p.setProductStructure(PRODUCT_STRUCTURE);
125                p.setTransferStatus(PRODUCT_STATUS);
126                p.setRootRef(new Reference("file:/dummy/path", ROOT_REF, 2));
127                p.setProductReferences(Lists.newArrayList(new Reference(
128                      ORIG_REF_1, DS_REF_1, FILE_SIZE_REF_1), new Reference(
129                      ORIG_REF_2, DS_REF_2, FILE_SIZE_REF_2)));
130                return p;
131             }
132          };
133       }
