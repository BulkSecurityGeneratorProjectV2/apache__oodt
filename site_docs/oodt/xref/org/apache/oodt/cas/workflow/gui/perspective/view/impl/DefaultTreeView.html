78    public TreePath getTreePath(DefaultMutableTreeNode node, ModelGraph graph) {
79      if (node.getUserObject().equals(graph)) {
80        return new TreePath(node.getPath());
81      } else {
82        for (int i = 0; i < node.getChildCount(); i++) {
83          // System.out.println("i: " + ((DefaultMutableTreeNode)
84          // node.getChildAt(i)).getUserObject());
85          TreePath treePath = this.getTreePath(
86              (DefaultMutableTreeNode) node.getChildAt(i), graph);
87          if (treePath != null) {
88            return treePath;
89          }
90        }
91        return null;
92      }
93    }

95    private TreePath getTreePath(TreePath currentPath, ViewState state) {
96      String lookingForPath = state.getCurrentMetGroup();
97      Stack<DefaultMutableTreeNode> stack = new Stack<DefaultMutableTreeNode>();
98      DefaultMutableTreeNode baseNode = (DefaultMutableTreeNode) currentPath
99          .getLastPathComponent();
100     for (int i = 0; i < baseNode.getChildCount(); i++) {
101       stack.push((DefaultMutableTreeNode) baseNode.getChildAt(i));
102     }
103     while (!stack.empty()) {
104       DefaultMutableTreeNode node = stack.pop();
105       if (node.getUserObject().equals("static-metadata")) {
106         for (int i = 0; i < node.getChildCount(); i++) {
107           stack.push((DefaultMutableTreeNode) node.getChildAt(i));
108         }
109       } else if (node.getUserObject() instanceof ConcurrentHashMap) {
110         String key = (String) ((ConcurrentHashMap<String, String>) node.getUserObject())
111             .keySet().iterator().next();
112         if (lookingForPath.equals(key)) {
113           return new TreePath(node.getPath());
114         } else if (lookingForPath.startsWith(key + "/")) {
115           lookingForPath = lookingForPath
116               .substring(lookingForPath.indexOf("/") + 1);
117           stack.clear();
118           for (int i = 0; i < node.getChildCount(); i++) {
119             stack.add((DefaultMutableTreeNode) node.getChildAt(i));
120           }
121         }
122       }
123     }
124     return currentPath;
125   }

134   public void refreshView(final ViewState state) {
135     Rectangle visibleRect = null;
136     if (this.tree != null) {
137       visibleRect = this.tree.getVisibleRect();
138     }
139 
140     this.removeAll();
141 
142     this.actionsMenu = this.createPopupMenu(state);
143 
144     DefaultMutableTreeNode root = new DefaultMutableTreeNode("WORKFLOWS");
145     for (ModelGraph graph : state.getGraphs()) {
146       root.add(this.buildTree(graph, state));
147     }
148     tree = new JTree(root);
149     tree.setShowsRootHandles(true);
150     tree.setRootVisible(false);
151     tree.add(this.actionsMenu);
152 
153     if (state.getSelected() != null) {
154       TreePath treePath = this.getTreePath(root, state.getSelected());
155       if (state.getCurrentMetGroup() != null) {
156         treePath = this.getTreePath(treePath, state);
157       } else if (Boolean.parseBoolean(state
158           .getFirstPropertyValue(EXPAND_STATIC_METADATA))) {
159         DefaultMutableTreeNode baseNode = (DefaultMutableTreeNode) treePath
160             .getLastPathComponent();
161         for (int i = 0; i < baseNode.getChildCount(); i++) {
162           if (((DefaultMutableTreeNode) baseNode.getChildAt(i)).getUserObject()
163               .equals("static-metadata")) {
164             treePath = new TreePath(
165                 ((DefaultMutableTreeNode) baseNode.getChildAt(i)).getPath());
166             break;
167           }
168         }
169       } else if (Boolean.parseBoolean(state
170           .getFirstPropertyValue(EXPAND_PRECONDITIONS))) {
171         if (treePath == null) {
172           treePath = this.getTreePath(root, state.getSelected()
173                                                  .getPreConditions());
174         }
175         DefaultMutableTreeNode baseNode = (DefaultMutableTreeNode) treePath
176             .getLastPathComponent();
177         for (int i = 0; i < baseNode.getChildCount(); i++) {
178           if (((DefaultMutableTreeNode) baseNode.getChildAt(i)).getUserObject()
179               .equals("pre-conditions")) {
180             treePath = new TreePath(
181                 ((DefaultMutableTreeNode) baseNode.getChildAt(i)).getPath());
182             break;
183           }
184         }
185       } else if (Boolean.parseBoolean(state
186           .getFirstPropertyValue(EXPAND_POSTCONDITIONS))) {
187         if (treePath == null) {
188           treePath = this.getTreePath(root, state.getSelected()
189                                                  .getPostConditions());
190         }
191         DefaultMutableTreeNode baseNode = (DefaultMutableTreeNode) treePath
192             .getLastPathComponent();
193         for (int i = 0; i < baseNode.getChildCount(); i++) {
194           if (((DefaultMutableTreeNode) baseNode.getChildAt(i)).getUserObject()
195               .equals("post-conditions")) {
196             treePath = new TreePath(
197                 ((DefaultMutableTreeNode) baseNode.getChildAt(i)).getPath());
198             break;
199           }
200         }
201       }
202       this.tree.expandPath(treePath);
203       this.tree.setSelectionPath(treePath);
204     }

208       public void valueChanged(TreeSelectionEvent e) {

218               || node.getUserObject().equals("pre-conditions")

220             state.setSelected((ModelGraph) ((DefaultMutableTreeNode) node
221                 .getParent()).getUserObject());
222             state.setCurrentMetGroup(null);
223             state.setProperty(EXPAND_STATIC_METADATA, Boolean.toString(node
224                 .getUserObject().equals("static-metadata")));
225             state.setProperty(EXPAND_PRECONDITIONS,
226                 Boolean.toString(node.getUserObject().equals("pre-conditions")));
227             state.setProperty(EXPAND_POSTCONDITIONS, Boolean.toString(node
228                 .getUserObject().equals("post-conditions")));
229             DefaultTreeView.this.notifyListeners();
230           } else if (node.getUserObject() instanceof ConcurrentHashMap) {
231             DefaultMutableTreeNode metNode = null;
232             String group = null;
233             Object[] path = e.getPath().getPath();
234             for (int i = path.length - 1; i >= 0; i--) {
235               if (((DefaultMutableTreeNode) path[i]).getUserObject() instanceof ModelGraph) {
236                 metNode = (DefaultMutableTreeNode) path[i];
237                 break;
238               } else if (((DefaultMutableTreeNode) path[i]).getUserObject() instanceof ConcurrentHashMap) {
239                 if (group == null) {
240                   group = (String) ((ConcurrentHashMap<String, String>) ((DefaultMutableTreeNode) path[i])
241                       .getUserObject()).keySet().iterator().next();
242                 } else {
243                   group = (String) ((ConcurrentHashMap<String, String>) ((DefaultMutableTreeNode) path[i])
244                       .getUserObject()).keySet().iterator().next()
245                           + "/"
246                           + group;
247                 }
248               }
249             }
250             ModelGraph graph = (ModelGraph) metNode.getUserObject();
251             state.setSelected(graph);
252             state.setCurrentMetGroup(group);
253             DefaultTreeView.this.notifyListeners();
254           } else {
255             state.setSelected(null);
256             DefaultTreeView.this.notifyListeners();
257           }
258         }
259       }

355   private DefaultMutableTreeNode buildTree(ModelGraph graph, ViewState state) {
356     DefaultMutableTreeNode node = new DefaultMutableTreeNode(graph);
357     DefaultMutableTreeNode metadataNode = new DefaultMutableTreeNode(
358         "static-metadata");
359     Metadata staticMetadata = new Metadata();
360     if (graph.getInheritedStaticMetadata(state) != null) {
361       staticMetadata.replaceMetadata(graph.getInheritedStaticMetadata(state));
362     }
363     if (graph.getModel().getStaticMetadata() != null) {
364       staticMetadata.replaceMetadata(graph.getModel().getStaticMetadata());
365     }
366     this.addMetadataNodes(metadataNode, staticMetadata);
367     if (!metadataNode.isLeaf()) {
368       node.add(metadataNode);
369     }
370 
371     if (graph.getPreConditions() != null) {
372       DefaultMutableTreeNode preConditions = new DefaultMutableTreeNode(
373           "pre-conditions");
374       List<ModelGraph> leafNodes = graph.getPreConditions().getLeafNodes();
375       for (ModelGraph cond : leafNodes) {
376         DefaultMutableTreeNode condNode = new DefaultMutableTreeNode(cond);
377         preConditions.add(condNode);
378       }
379       node.add(preConditions);
380     }
381     if (graph.getPostConditions() != null) {
382       DefaultMutableTreeNode postConditions = new DefaultMutableTreeNode(
383           "post-conditions");
384       List<ModelGraph> leafNodes = graph.getPostConditions().getLeafNodes();
385       for (ModelGraph cond : leafNodes) {
386         DefaultMutableTreeNode condNode = new DefaultMutableTreeNode(cond);
387         postConditions.add(condNode);
388       }
389       node.add(postConditions);
390     }
391     for (ModelGraph child : graph.getChildren()) {
392       if (!GuiUtils.isDummyNode(child.getModel())) {
393         node.add(this.buildTree(child, state));
394       }
395     }
396     return node;
397   }

399   private void addMetadataNodes(DefaultMutableTreeNode metadataNode,
400       Metadata staticMetadata) {
401     for (String group : staticMetadata.getGroups()) {
402       Object userObj;
403       if (staticMetadata.getMetadata(group) != null) {
404         ConcurrentHashMap<String, String> map = new ConcurrentHashMap<String, String>();
405         map.put(group,
406             StringUtils.join(staticMetadata.getAllMetadata(group), ","));
407         userObj = map;
408       } else {
409         ConcurrentHashMap<String, String> map = new ConcurrentHashMap<String, String>();
410         map.put(group, null);
411         userObj = map;
412       }
413       DefaultMutableTreeNode groupNode = new DefaultMutableTreeNode(userObj);
414       metadataNode.add(groupNode);
415       this.addMetadataNodes(groupNode, staticMetadata.getSubMetadata(group));
416     }

419   private PopupMenu createPopupMenu(final ViewState state) {
420     final String ACTIONS_POP_MENU_NAME = "Actions";
421     final String VIEW_CONDITION_MAP = "View...";
422     PopupMenu actionsMenu = new PopupMenu(ACTIONS_POP_MENU_NAME);
423     actionsMenu.add(new MenuItem(VIEW_CONDITION_MAP));
424     actionsMenu.addActionListener(new ActionListener() {
425 
426       public void actionPerformed(ActionEvent e) {
427         if (e.getActionCommand().equals(VIEW_CONDITION_MAP)) {
428           DefaultMutableTreeNode node = (DefaultMutableTreeNode) tree
429               .getSelectionPath().getLastPathComponent();
430           ModelGraph graphToFocus = null;
431           if (Boolean.parseBoolean(state
432               .getFirstPropertyValue(EXPAND_PRECONDITIONS))
433               || Boolean.parseBoolean(state
434                   .getFirstPropertyValue(EXPAND_POSTCONDITIONS))) {
435             ModelGraph graph = state.getSelected();
436             if (Boolean.parseBoolean(state
437                 .getFirstPropertyValue(EXPAND_PRECONDITIONS))) {
438               graphToFocus = graph.getPreConditions();
439             } else {
440               graphToFocus = graph.getPostConditions();
441             }
442           } else if (node.getUserObject() instanceof ModelGraph) {
443             graphToFocus = (ModelGraph) node.getUserObject();
444           }
445           DefaultTreeView.this.notifyListeners(new ViewChange.NEW_VIEW(
446               graphToFocus, DefaultTreeView.this));
447         }
448       }

452     final String ORDER_SUB_POP_MENU_NAME = "Order";
453     final String TO_FRONT_ITEM_NAME = "Move To Front";
454     final String TO_BACK_ITEM_NAME = "Move To Back";
455     final String FORWARD_ITEM_NAME = "Move Forward";
456     final String BACKWARDS_ITEM_NAME = "Move Backwards";
457     actionsMenu.add(orderSubMenu = new PopupMenu(ORDER_SUB_POP_MENU_NAME));
458     orderSubMenu.add(new MenuItem(TO_FRONT_ITEM_NAME));
459     orderSubMenu.add(new MenuItem(TO_BACK_ITEM_NAME));
460     orderSubMenu.add(new MenuItem(FORWARD_ITEM_NAME));
461     orderSubMenu.add(new MenuItem(BACKWARDS_ITEM_NAME));
462     orderSubMenu.addActionListener(new ActionListener() {
463 
464       public void actionPerformed(ActionEvent e) {
465         ModelGraph graph = state.getSelected();
466         ModelGraph parent = graph.getParent();
467         if (e.getActionCommand().equals(TO_FRONT_ITEM_NAME)) {
468           if (parent.getChildren().remove(graph)) {
469             parent.getChildren().add(0, graph);
470           }
471         } else if (e.getActionCommand().equals(TO_BACK_ITEM_NAME)) {
472           if (parent.getChildren().remove(graph)) {
473             parent.getChildren().add(graph);
474           }
475         } else if (e.getActionCommand().equals(FORWARD_ITEM_NAME)) {
476           int index = parent.getChildren().indexOf(graph);
477           if (index != -1) {
478             parent.getChildren().remove(index);
479             parent.getChildren().add(Math.max(0, index + 1), graph);
480           }
481         } else if (e.getActionCommand().equals(BACKWARDS_ITEM_NAME)) {
482           int index = parent.getChildren().indexOf(graph);
483           if (index != -1) {
484             parent.getChildren().remove(index);
485             parent.getChildren().add(Math.max(0, index - 1), graph);
486           }
487         }
488         DefaultTreeView.this.notifyListeners();
489         DefaultTreeView.this.refreshView(state);
490       }
491 
492     });
493     return actionsMenu;
494   }
