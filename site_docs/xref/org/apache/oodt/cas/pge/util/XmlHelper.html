67  	public static List<Pair<String, String>> getImports(Element elem,
68  			Metadata metadata) throws PGEException {
69  		List<Pair<String, String>> imports = Lists.newArrayList();
70  		NodeList nodeList = elem.getElementsByTagName(IMPORT_TAG);
71  		for (int i = 0; i < nodeList.getLength(); i++) {
72  			Element importElem = (Element) nodeList.item(i);
73  			String namespace = getNamespace(importElem, metadata);
74  			String file = getFile(importElem, metadata);
75  			imports.add(new Pair<String, String>(namespace, file));
76  		}
77  		return imports;
78  	}

123 	public static String getMetadataKey(Element metElem, Metadata metadata)
124 		throws PGEException {
125 		String key = metElem.getAttribute(KEY_ATTR);
126 
127 		// no key attr, so check for key_gen attr.
128 		if (Strings.isNullOrEmpty(key)) {
129 			key = fillIn(metElem.getAttribute(KEY_GEN_ATTR), metadata);
130 		}
131 
132 		// if still no key value, then fail.
133 		if (Strings.isNullOrEmpty(key)) {
134 			throw new PGEException("Must specify either metadata attr '"
135 					+ KEY_ATTR + "' or '" + KEY_GEN_ATTR + "'");
136 		}
137 
138 		// else success!
139 		return key;
140 	}

167 	public static List<String> getMetadataValues(Element elem, Metadata metadata)
168 		throws PGEException {
169 		List<String> values = Lists.newArrayList();
170 
171 		// Read val attr.
172 		String value = elem.getAttribute(VAL_ATTR);
173 
174 		// Check if val tag was not specified see if value was given as element
175 		// text.
176 		if (Strings.isNullOrEmpty(value)) {
177 			value = elem.getTextContent();
178 		}
179 
180 		// If value was found.
181 		if (!Strings.isNullOrEmpty(value)) {
182 
183 			// Is multi-value so split up value.
184 			if (isMultiValue(elem, metadata)) {
185 				for (String v : Splitter.on(",").split(value)) {
186 
187 					// Check for envReplace and perform met replacement on value
188 					// if set.
189 					if (isEnvReplaceNoRecur(elem, metadata)) {
190 						values.add(fillIn(v, metadata, false));
191 					} else if (isEnvReplace(elem, metadata)) {
192 						values.add(fillIn(v, metadata));
193 					}
194 				}
195 
196 				// Is scalar
197 			} else {
198 
199 				// Check for envReplace and perform met replacement on value if
200 				// set.
201 				if (isEnvReplaceNoRecur(elem, metadata)) {
202 					value = fillIn(value, metadata, false);
203 				} else if (isEnvReplace(elem, metadata)) {
204 					value = fillIn(value, metadata);
205 				}
206 				values.add(value);
207 			}
208 
209 		}
210 		return values;
211 	}

260 	public static List<DynamicConfigFile> getDynamicConfigFiles(Element elem,
261 			Metadata metadata) throws PGEException {
262 		List<DynamicConfigFile> dynamicConfigFiles = Lists.newArrayList();
263 		NodeList nodeList = elem.getElementsByTagName(DYN_INPUT_FILES_TAG);
264 
265 		// Check if dynInput element exists.
266 		if (nodeList.getLength() > 0) {
267 			Element dynamicConfigFilesElem = (Element) nodeList.item(0);
268 			nodeList = dynamicConfigFilesElem.getElementsByTagName(FILE_TAG);
269 
270 			// Load each dynamic input file information.
271 			for (int i = 0; i < nodeList.getLength(); i++) {
272 				Element fileElem = (Element) nodeList.item(i);
273 				String path = getPath(fileElem, metadata);
274 				String writer = getWriter(fileElem, metadata);
275 				List<String> args = getArgs(fileElem, metadata);
276 				dynamicConfigFiles.add(new DynamicConfigFile(path, writer, args
277 					.toArray(new String[args.size()])));
278 			}
279 		}
280 		return dynamicConfigFiles;
281 	}

388 	public static void getRegExpOutputFiles(Element elem, Metadata metadata,
389 			OutputDir outputDir) throws PGEException {
390 		NodeList fileList = elem.getElementsByTagName(FILES_TAG);
391 		for (int j = 0; j < fileList.getLength(); j++) {
392 			Element fileElement = (Element) fileList.item(j);
393 			String outputFile = fileElement.getAttribute(REGEX_ATTR);
394 			if (outputFile.equals("")){
395 				outputFile = fillIn(fileElement.getAttribute(NAME_ATTR),
396 						metadata);
397 			}
398 			PathUtilsNamingConvention renamingConvention = null;
399 			NodeList renamingConvNodes = fileElement
400 					.getElementsByTagName(RENAMING_CONV_TAG);
401 
402 			if (renamingConvNodes.getLength() > 0) {
403 				Element renamingElement = (Element) renamingConvNodes.item(0);
404 				String namingExpr = renamingElement
405 						.getAttribute(NAMING_EXPR_ATTR);
406 				if (renamingElement.getAttribute(ENV_REPLACE_ATTR)
407 						.toLowerCase().equals("true")){
408 					namingExpr = fillIn(namingExpr, metadata, false);
409 				}
410 				else if (!renamingElement.getAttribute(ENV_REPLACE_ATTR)
411 						.toLowerCase().equals("false")){
412 					namingExpr = fillIn(namingExpr, metadata);
413 				}
414 				renamingConvention = new PathUtilsNamingConvention();
415 				renamingConvention.setNamingConv(namingExpr);
416 				NodeList metadataNodes = renamingElement
417 						.getElementsByTagName(METADATA_TAG);
418 				for (int k = 0; k < metadataNodes.getLength(); k++) {
419 					renamingConvention.addTmpReplaceMet(((Element) metadataNodes
420 							.item(k)).getAttribute(KEY_ATTR), Arrays
421 							.asList(((Element) metadataNodes.item(k))
422 									.getAttribute(VAL_ATTR).split(",")));
423 				}
424 			}
425 
426 			outputDir
427 					.addRegExprOutputFiles(new RegExprOutputFiles(outputFile,
428 							fillIn(fileElement
429 									.getAttribute(MET_FILE_WRITER_CLASS_ATTR),
430 									metadata), renamingConvention, (Object[]) fillIn(
431 									fileElement.getAttribute(ARGS_ATTR),
432 									metadata).split(",")));
433 		}

442 	public static String fillIn(String value, Metadata inputMetadata,
443 			boolean envReplaceRecur) throws PGEException {
444 		try {
445 			while ((value = PathUtils
446 					.doDynamicReplacement(value, inputMetadata)).contains("[")
447 					&& envReplaceRecur) {
448 			}
449 			if (value.toUpperCase().matches(
450 					"^\\s*SQL\\s*\\(.*\\)\\s*\\{.*\\}\\s*$")) {
451 			  value = QueryUtils
452 				  .getQueryResultsAsString(new XmlRpcFileManagerClient(
453 					  new URL(inputMetadata
454 						  .getMetadata(QUERY_FILE_MANAGER_URL
455 							  .getName())))
456 					  .complexQuery(SqlParser
457 						  .parseSqlQueryMethod(value)));
458 			}
459 			return value;
460 		} catch (Exception e) {
461 			throw new PGEException("Failed to parse value: " + value, e);
462 		}
