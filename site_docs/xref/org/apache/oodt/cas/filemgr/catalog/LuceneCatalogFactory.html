39  /**
40   * @author mattmann

43   * <p>
44   * A Factory for creating {@link Lucene}-based {@link Catalog}s.
45   * </p>
46   * 
47   */
48  public class LuceneCatalogFactory implements CatalogFactory {
49  
50    public static final int VAL = 20;
51    public static final int VAL1 = 60;
52    public static final int VAL2 = 60;
53    public static final int VAL3 = 20;
54    /* path to the index directory for lucene catalogs */
55  	private String indexFilePath = null;
56  	private IndexWriterConfig config = new IndexWriterConfig(new StandardAnalyzer());
57  
58  	/* our validation layer */
59  	private ValidationLayer validationLayer = null;
60  	
61  	/* the page size for pagination */
62  	private int pageSize = -1;
63  	
64  	/* the write lock timeout */
65  	private long writeLockTimeOut = -1L;
66  	
67  	/* the commit lock timeout */
68  	private long commitLockTimeOut = -1L;
69  	
70  	/* the merge factor */
71  	private int mergeFactor = -1;
72  	
73  	/* Whether or not to enforce strict definition of metadata fields:
74  	 * 'lenient=false' means that all metadata fields need to be explicitly defined in the XML configuration file */
75  	private boolean lenientFields = false;
76  	
77  	/* our log stream */
78      private static final Logger LOG = Logger.getLogger(LuceneCatalogFactory.class.getName());
79      
80  	/**
81  	 * 
82  	 */
83  	public LuceneCatalogFactory() throws IllegalArgumentException {
84  		indexFilePath = System
85  				.getProperty("org.apache.oodt.cas.filemgr.catalog.lucene.idxPath");
86  		if (indexFilePath == null) {
87  			throw new IllegalArgumentException(
88  					"error initializing lucene catalog: "
89  							+ "[org.apache.oodt.cas.filemgr.catalog.lucene.idxPath="
90  							+ null);
91  		}
92  
93  		//do env var replacement
94  		indexFilePath = PathUtils.replaceEnvVariables(indexFilePath);
95  		
96  		// instantiate validation layer, unless catalog is explicitly configured for lenient fields
97  		lenientFields = Boolean.parseBoolean( System.getProperty("org.apache.oodt.cas.filemgr.catalog.lucene.lenientFields", "false") );
98  		if (!lenientFields) {
99  			String validationLayerFactoryClass = System
100 				.getProperty("filemgr.validationLayer.factory",
101 						"org.apache.oodt.cas.filemgr.validation.XMLValidationLayerFactory");
102 			validationLayer = GenericFileManagerObjectFactory
103 				.getValidationLayerFromFactory(validationLayerFactoryClass);
104 		}
105 		
106 		pageSize = Integer.getInteger("org.apache.oodt.cas.filemgr.catalog.lucene.pageSize", VAL);
107 		
108 		commitLockTimeOut = Long
109 			.getLong(
110 				"org.apache.oodt.cas.filemgr.catalog.lucene.commitLockTimeout.seconds",
111 				VAL1);
112 		writeLockTimeOut = Long
113 			.getLong(
114 				"org.apache.oodt.cas.filemgr.catalog.lucene.writeLockTimeout.seconds",
115 				VAL2);
116 		mergeFactor = Integer.getInteger(
117 			"org.apache.oodt.cas.filemgr.catalog.lucene.mergeFactor", VAL3);
118 	}
119 
120 	/*
121 	 * (non-Javadoc)
122 	 * 
123 	 * @see org.apache.oodt.cas.filemgr.catalog.CatalogFactory#createCatalog()
124 	 */
125 	public Catalog createCatalog() {
126 	    File indexDir = new File(indexFilePath);
127 	    // Create the index if it does not already exist
128 	    IndexWriter writer = null;
129 		config.setOpenMode(IndexWriterConfig.OpenMode.CREATE_OR_APPEND);
130 	    if (!indexDir.exists()) {
131 	        try {
132 				try {
133 					Directory indexDir2 = FSDirectory.open(new File( indexFilePath ).toPath());
134 					writer = new IndexWriter(indexDir2, config);
135 				} catch (IOException e) {
136 					e.printStackTrace();
137 				}
138 
139 	        } catch (Exception e) {
140 	            LOG.severe("Unable to create index: " + e.getMessage());
141 	        } finally {
142 	            if (writer != null) {
143 	                try {
144 	                    writer.close();
145 	                } catch (Exception e) {
146 	                    LOG.severe("Unable to close index: " + e.getMessage());
147 	                }
148 	            }
149 	        }
150 	    }
151 		return new LuceneCatalog(indexFilePath, validationLayer, pageSize,
152 				commitLockTimeOut, writeLockTimeOut, mergeFactor);
153 	}
