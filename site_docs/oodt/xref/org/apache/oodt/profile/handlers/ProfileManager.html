16  package org.apache.oodt.profile.handlers;
17  
18  import org.apache.oodt.profile.Profile;
19  import org.apache.oodt.profile.ProfileException;
20  
21  import java.util.Collection;
22  import java.util.Iterator;
23  
24  /**
25   * Interface of a profile manager.
26   *
27   * A profile manager can manage profiles by adding, removing, and manipulating them---in
28   * addition to searching them.
29   *
30   * @author Kelly
31   * @version $Revision: 1.1.1.1 $
32   */
33  public interface ProfileManager extends ProfileHandler {
34  	/**
35  	 * Add a profile into the set of profiles managed by this profile server.
36  	 *
37  	 * @param profile The profile to add or replace.
38  	 * @throws ProfileException If any error occurs.
39  	 */
40  	void add(Profile profile) throws ProfileException;
41  
42  	/**
43  	 * Add all of the profiles in the given collection to the set managed by this
44  	 * server.
45  	 *
46  	 * Any profiles that already exist (as identified by their profile IDs) are replaced.
47  	 *
48  	 * @param collection Collection of {@link Profile}s to add.
49  	 * @throws ProfileException If any error occurs.
50  	 */
51  	void addAll(Collection collection) throws ProfileException;
52  
53  	/**
54  	 * Clear all profiles.
55  	 *
56  	 * This irrevocably removes all profiles from the server, leaving it with none.
57  	 *
58  	 * @throws ProfileException If any error occurs.
59  	 */
60  	void clear() throws ProfileException;
61  
62  	/**
63  	 * Tell if the given profile is managed by this server.
64  	 *
65  	 * @param profile The profile to check.
66  	 * @return True if <var>profile</var> is present in the server, false otherwise.
67  	 * @throws ProfileException If any error occurs.
68  	 */
69  	boolean contains(Profile profile);
70  
71  	/**
72  	 * Tell if the given collection of profiles are managed by this server.
73  	 *
74  	 * @param collection The collection of {@link Profile}s to check.
75  	 * @return True if every {@link Profile} in <var>collection</var> are present in
76  	 * the server, false otherwise.
77  	 * @throws ProfileException If any error occurs.
78  	 */
79  	boolean containsAll(Collection collection);
80  
81  	/**
82  	 * Get all profiles.
83  	 *
84  	 * @return  A collection of profiles
85  	 * @throws ProfileException If any error occurs.
86  	 */
87  	Collection getAll();
88  
89  	/**
90  	 * Tell if the set of profiles managed by this server is empty.
91  	 *
92  	 * @return True if there are no profiles in this server.
93  	 * @throws ProfileException If any error occurs.
94  	 */
95  	boolean isEmpty();
96  
97  	/**
98  	 * Iterate over the available profiles.
99  	 *
100 	 * Each call to <code>next</code> yields a copy of the next {@link Profile} object
101 	 * managed by this server.  Updates to profile objects are not backed by the
102 	 * server; make your updates and then call {@link #add}.
103 	 *
104 	 * @return An iterator over {@link Profile}s.
105 	 * @throws ProfileException If any error occurs.
106 	 */
107 	Iterator iterator();
108 
109 	/**
110 	 * Remove the profile with the given ID.
111 	 *
112 	 * This removes forever the given profile from the set of profiles managed by this
113 	 * server.
114 	 *
115 	 * @param profID What profile to remove.
116 	 * @param version What version of the the profile identified by <var>profID</var> to remove.
117 	 * @return True if thie profile was actually present and removed; false if it wasn't present.
118 	 * @throws ProfileException If any error occurs.
119 	 */
120 	boolean remove(String profID, String version) throws ProfileException;
121 
122 	/**
123 	 * Remove the profile with the given ID.
124 	 *
125 	 * This removes forever the given profile from the set of profiles managed by this
126 	 * server.
127 	 *
128 	 * @param profID ID of profile to remove.
129 	 * @return True if thie profile was actually present and removed; false if it wasn't present.
130 	 * @throws ProfileException If any error occurs.
131 	 */
132 	boolean remove(String profID) throws ProfileException;
133 
134 	/**
135 	 * Get the size of the set of profiles managed by this server.
136 	 *
137 	 * @return The number of available profiles.
138 	 * @throws ProfileException If any error occurs.
139 	 */
140 	int size();
141 
142 	/**
143 	 * Replace a profile into the set of profiles managed by this profile server.
144 	 *
145 	 * @param profile The profile to add or replace.
146 	 * @throws ProfileException If any error occurs.
147 	 */
148 	void replace(Profile profile) throws ProfileException;
149 }
