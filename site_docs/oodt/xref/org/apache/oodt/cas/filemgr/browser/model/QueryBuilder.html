20  import org.apache.lucene.index.Term;
21  import org.apache.lucene.queryparser.classic.QueryParser;
22  import org.apache.lucene.search.*;
23  import org.apache.lucene.util.BytesRef;
24  import org.apache.oodt.cas.filemgr.structs.Query;
25  import org.apache.oodt.cas.filemgr.structs.RangeQueryCriteria;
26  import org.apache.oodt.cas.filemgr.structs.TermQueryCriteria;
27  import org.apache.oodt.cas.filemgr.tools.CASAnalyzer;
28  
29  import java.util.List;
30  import java.util.logging.Level;
31  import java.util.logging.Logger;

33  public class QueryBuilder {
34  
35    private CasDB database;
36    private static Logger LOG = Logger.getLogger(QueryBuilder.class.getName());
37    public QueryBuilder(CasDB db) {
38      database = db;
39    }
40  
41    public Query ParseQuery(String query) {
42      // note that "__FREE__" is a control work for free text searching
43      QueryParser parser = new QueryParser("__FREE__", new CASAnalyzer());
44  
45      org.apache.lucene.search.Query luceneQ = null;
46      org.apache.oodt.cas.filemgr.structs.Query casQ = new org.apache.oodt.cas.filemgr.structs.Query();
47  
48      try {
49        luceneQ = parser.parse(query);
50      } catch (org.apache.lucene.queryparser.classic.ParseException e) {
51        // TODO Auto-generated catch block
52        LOG.log(Level.SEVERE, e.getMessage());
53      }
54  
55      System.out.println(luceneQ != null ? luceneQ.toString() : null);
56      GenerateCASQuery(casQ, luceneQ);
57  
58      return casQ;
59    }
60  
61    public void GenerateCASQuery(org.apache.oodt.cas.filemgr.structs.Query casQ,
62        org.apache.lucene.search.Query luceneQ) {
63      if (luceneQ instanceof TermQuery) {
64        Term t = ((TermQuery) luceneQ).getTerm();
65        if (!t.field().equals("__FREE__")) {
66          String element = database.getElementID(t.field());
67          if (!element.equals("") && !t.text().equals("")) {
68  
69            casQ.addCriterion(new TermQueryCriteria(element, t.text()));
70          }
71        }
72      } else if (luceneQ instanceof PhraseQuery) {
73        Term[] t = ((PhraseQuery) luceneQ).getTerms();
74        if (!t[0].field().equals("__FREE__")) {
75          for (Term aT : t) {
76            String element = database.getElementID(aT.field());
77            if (!element.equals("") && !aT.text().equals("")) {
78              casQ.addCriterion(new TermQueryCriteria(element, aT.text()));
79            }
80          }
81        }
82      } else if (luceneQ instanceof TermRangeQuery) {
83        BytesRef startT = ((TermRangeQuery) luceneQ).getLowerTerm();
84        BytesRef endT = ((TermRangeQuery) luceneQ).getUpperTerm();
85        String element = database.getElementID(((TermRangeQuery) luceneQ).getField());
86        if (!element.equals("") && !startT.utf8ToString().equals("")
87            && !endT.utf8ToString().equals("")) {
88          casQ.addCriterion(new RangeQueryCriteria(element, startT.utf8ToString(), endT
89              .utf8ToString()));
90        }
91      } else if (luceneQ instanceof BooleanQuery) {
92        List<BooleanClause> clauses = ((BooleanQuery) luceneQ).clauses();
93        for (BooleanClause clause : clauses) {
94          GenerateCASQuery(casQ, (clause).getQuery());
95        }
96      } else {
97        System.out.println("Error Parsing Query");
98        System.exit(-1);
99      }
100   }
