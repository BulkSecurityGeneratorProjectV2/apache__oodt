72      /*
73       * (non-Javadoc)
74       * 
75       * @see org.apache.oodt.cas.filemgr.validation.ValidationLayer#addElement(org.apache.oodt.cas.filemgr.structs.Element)
76       */
77      public void addElement(Element element) throws ValidationLayerException {
78          Connection conn = null;
79          Statement statement = null;
80          ResultSet rs = null;
81  
82          try {
83              conn = dataSource.getConnection();
84              conn.setAutoCommit(false);
85              statement = conn.createStatement();
86  
87              String addMetaElemSql = "INSERT INTO elements (element_name, dc_element, element_description) VALUES ('"
88                      + element.getElementName()
89                      + ", '"
90                      + element.getDCElement()
91                      + "', '" + element.getDescription() + "')";
92  
93              LOG.log(Level.FINE, "addMetadataElement: Executing: "
94                      + addMetaElemSql);
95              statement.execute(addMetaElemSql);
96  
97              String elementId = "";
98  
99              String getMetaIdSql = "SELECT MAX(element_id) AS max_id FROM elements";
100             LOG.log(Level.FINE, "addElement: Executing: " + getMetaIdSql);
101             rs = statement.executeQuery(getMetaIdSql);
102 
103             while (rs.next()) {
104                 elementId = String.valueOf(rs.getInt("max_id"));
105             }
106 
107             element.setElementId(elementId);
108             conn.commit();
109 
110         } catch (Exception e) {
111             LOG.log(Level.SEVERE, e.getMessage());
112             LOG
113                     .log(Level.WARNING, "Exception adding element "
114                             + element.getElementName() + ". Message: "
115                             + e.getMessage());
116             try {
117                 if (conn != null) {
118                     conn.rollback();
119                 }
120             } catch (SQLException e2) {
121                 LOG.log(Level.SEVERE,
122                         "Unable to rollback addElement transaction. Message: "
123                                 + e2.getMessage());
124             }
125             throw new ValidationLayerException(e);
126         } finally {
127 
128             if (rs != null) {
129                 try {
130                     rs.close();
131                 } catch (SQLException ignore) {
132                 }
133 
134             }
135 
136             if (statement != null) {
137                 try {
138                     statement.close();
139                 } catch (SQLException ignore) {
140                 }
141 
142             }
143 
144             if (conn != null) {
145                 try {
146                     conn.close();
147 
148                 } catch (SQLException ignore) {
149                 }
150 
151             }
152         }
153 
154     }

523     /*
524      * (non-Javadoc)
525      * 
526      * @see org.apache.oodt.cas.filemgr.validation.ValidationLayer#getElements(org.apache.oodt.cas.filemgr.structs.ProductType)
527      */
528     public List<Element> getElements(ProductType type)
529             throws ValidationLayerException {
530         Connection conn = null;
531         Statement statement = null;
532         ResultSet rs = null;
533         List<Element> elements;
534 
535         elements = new Vector<Element>();
536 
537         String currProduct = type.getProductTypeId();
538         while (currProduct != null) {
539 
540             try {
541                 conn = dataSource.getConnection();
542                 statement = conn.createStatement();
543 
544                 String elementSql = "SELECT elements.* from elements, product_type_element_map WHERE product_type_element_map.product_type_id = ";
545 
546                 if (quoteFields) {
547                     elementSql += "'" + currProduct + "'";
548                 } else {
549                     elementSql += currProduct;
550                 }
551 
552                 elementSql += " AND product_type_element_map.element_id = elements.element_id";
553 
554                 LOG.log(Level.FINE, "getElements: Executing: " + elementSql);
555                 rs = statement.executeQuery(elementSql);
556 
557                 while (rs.next()) {
558                     Element element = DbStructFactory.getElement(rs);
559                     elements.add(element);
560                 }
561 
562             } catch (Exception e) {
563                 LOG.log(Level.SEVERE, e.getMessage());
564                 LOG.log(Level.WARNING, "Exception reading elements. Message: "
565                         + e.getMessage());
566                 throw new ValidationLayerException(e);
567             } finally {
568 
569                 if (rs != null) {
570                     try {
571                         rs.close();
572                     } catch (SQLException ignore) {
573                     }
574 
575                     rs = null;
576                 }
577 
578                 if (statement != null) {
579                     try {
580                         statement.close();
581                     } catch (SQLException ignore) {
582                     }
583 
584                     statement = null;
585                 }
586 
587                 if (conn != null) {
588                     try {
589                         conn.close();
590 
591                     } catch (SQLException ignore) {
592                     }
593 
594                     conn = null;
595                 }
596             }
597 
598             // advance to the product parent
599             try {
600                 conn = dataSource.getConnection();
601                 statement = conn.createStatement();
602 
603                 String getParentSql = "SELECT parent_id from sub_to_super_map where product_type_id = ";
604                 if (quoteFields) {
605                     getParentSql += "'" + currProduct + "'";
606                 } else {
607                     getParentSql += currProduct;
608                 }
609 
610                 LOG.log(Level.FINE, "getElements: Executing: " + getParentSql);
611                 rs = statement.executeQuery(getParentSql);
612 
613                 currProduct = null;
614                 while (rs.next()) {
615                     currProduct = DbStructFactory.getParent(rs);
616                 }
617 
618             } catch (Exception e) {
619                 LOG.log(Level.SEVERE, e.getMessage());
620                 LOG.log(Level.WARNING,
621                         "Exception reading product parent. Message: "
622                                 + e.getMessage());
623                 throw new ValidationLayerException(e);
624             } finally {
625                 if (rs != null) {
626                     try {
627                         rs.close();
628                     } catch (SQLException ignore) {
629                     }
630                     rs = null;
631                 }
632                 if (statement != null) {
633                     try {
634                         statement.close();
635                     } catch (SQLException ignore) {
636                     }
637                     statement = null;
638                 }
639                 if (conn != null) {
640                     try {
641                         conn.close();
642                     } catch (SQLException ignore) {
643                     }
644                     conn = null;
645                 }
646             }
647 
648         }
649 
650         return elements;
651     }
