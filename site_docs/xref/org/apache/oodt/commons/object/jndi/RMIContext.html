86  	public Object lookup(String name) throws NamingException {
87  		checkName(name);
88  		name = toRMIName(name);
89  		if (name.length() == 0) {
90  		  return new RMIContext(environment);
91  		}
92  		Registry registry = getRegistry();
93  		try {
94  			return registry.lookup(name);
95  		} catch (NotBoundException ex) {
96  			throw new NameNotFoundException(name + " not found in RMI registry " + registry);
97  		} catch (RemoteException ex) {
98  			throw new NamingException("Remote exception: " + ex.getMessage());
99  		}
100 	}

162 	public NamingEnumeration list(String name) throws NamingException {
163 		if (name.length() > 0) {
164 		  throw new NotContextException("Subcontexts not supported");
165 		}
166 				
167 		final Iterator i = getCurrentBindings().iterator();
168 		return new NamingEnumeration() {
169 			public void close() {}
170 			public boolean hasMore() {
171 				return i.hasNext();
172 			}
173 			public Object next() throws NamingException {
174 				String n = "urn:eda:rmi:" + (String) i.next();
175 				org.apache.oodt.commons.Service server = (org.apache.oodt.commons.Service) lookup(n);
176 				try {
177 					return new NameClassPair(n, server.getServerInterfaceName());
178 				} catch (RemoteException ex) {
179 					throw new CommunicationException(ex.getMessage());
180 				}
181 			}
182 			public boolean hasMoreElements() {
183 				return hasMore();
184 			}
185 			public Object nextElement() {
186 				Object next = null;
187 				try {
188 					next = next();
189 				} catch (RuntimeException ex) {
190 					throw ex;
191 				} catch (Exception ignore) {}
192 				return next;
193 			}
194 		};
195 	}

307 	 * Get the current bindings in the RMI registry.
308 	 *
309 	 * @return A list of the current bindings, as simple string names.

311 	private List getCurrentBindings() throws NamingException {
312 		List names;
313 		try {
314 			Registry registry = getRegistry();

322 	/**
323 	 * Convert the context object name into an rmiregistry name.
324 	 *
325 	 * @param name Context name.
326 	 * @return rmiregistry name.
327 	 */
328 	private String toRMIName(String name) {
329 		if (name == null) {
330 		  return "";
331 		}
332 		if (name.startsWith("urn:eda:rmi:")) {
333 		  return name.substring(12);
334 		}
335 		if (name.startsWith("rmi:")) {
336 		  return name.substring(4);
337 		}
338 		return name;
339 	}

341 	/**
342 	 * Get the RMI registry.
343 	 *
344 	 * @return a <code>Registry</code> value.
345 	 * @throws NamingException if an error occurs.
346 	 */
347 	private Registry getRegistry() throws NamingException {
348 		if (registry != null) {
349 		  return registry;
350 		}
351 		try {
352 			String host = environment.containsKey("host")? (String) environment.get("host") : "localhost";
353 			int port = environment.containsKey("port")? (Integer) environment.get("port")
354 				: Registry.REGISTRY_PORT;
355 
356 			
357 			registry = LocateRegistry.getRegistry(host, port);
358 		} catch (RemoteException ex) {
359 			throw new NamingException("Remote exception locating registry: " + ex.getMessage());
360 		}
361 		return registry;
362 	}

364 	/**
365 	 * Ensure the name is an RMI object context name.
366 	 *
367 	 * RMI object context names are URNs in the <code>eda</code> namespace, in the
368 	 * <code>rmi</code> subnamespace.
369 	 *
370 	 * @param name Name to check.
371 	 * @throws InvalidNameException If <var>name</var>'s not an RMI object context name.
372 	 */
373 	private void checkName(String name) throws InvalidNameException {
374 		if (name == null) {
375 		  throw new IllegalArgumentException("Can't check a null name");
376 		}
377 		if (name.length() == 0) {
378 		  throw new InvalidNameException("Name's length is zero");
379 		}
380 		if (!name.startsWith("urn:eda:rmi:")) {
381 		  throw new InvalidNameException("Not an RMI name; try urn:eda:rmi:yadda-yadda");
382 		}
383 	}
