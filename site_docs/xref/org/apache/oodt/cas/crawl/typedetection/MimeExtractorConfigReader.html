41  /**
42   * Static reader class for {@link MimeExtractor}s.
43   *
44   * @author mattmann (Chris Mattmann)
45   * @author bfoster (Brian Foster)
46   */
47  public final class MimeExtractorConfigReader implements
48          MimeExtractorConfigMetKeys {
49    private static Logger LOG = Logger.getLogger(MimeExtractorConfigReader.class.getName());
50      private MimeExtractorConfigReader() throws InstantiationException {
51          throw new InstantiationException("Don't construct reader classes!");
52      }
53  
54      public static MimeExtractorRepo read(String mapFilePath)
55          throws ClassNotFoundException, FileNotFoundException, MetExtractionException, InstantiationException,
56          IllegalAccessException, CrawlerActionException {
57          try {
58              Document doc = XMLUtils.getDocumentRoot(new FileInputStream(
59                      mapFilePath));
60              Element root = doc.getDocumentElement();
61  
62              MimeExtractorRepo extractorRepo = new MimeExtractorRepo();
63              extractorRepo.setMagic(Boolean.valueOf(
64                  root.getAttribute(MAGIC_ATTR)));
65              String mimeTypeFile = PathUtils.replaceEnvVariables(root
66                    .getAttribute(MIME_REPO_ATTR));
67              if (!mimeTypeFile.startsWith("/")) {
68                 mimeTypeFile = new File(new File(mapFilePath).getParentFile(),
69                       mimeTypeFile).getAbsolutePath();
70              }
71              extractorRepo.setMimeRepoFile(mimeTypeFile);
72  
73              Element defaultExtractorElem = XMLUtils.getFirstElement(
74                      DEFAULT_EXTRACTOR_TAG, root);
75              if (defaultExtractorElem != null) {
76                  NodeList defaultExtractorElems = defaultExtractorElem
77                          .getElementsByTagName(EXTRACTOR_TAG);
78                  LinkedList<MetExtractorSpec> defaultExtractorSpecs = new LinkedList<MetExtractorSpec>();
79                  for (int i = 0; i < defaultExtractorElems.getLength(); i++) {
80                      Element extractorElem = (Element) defaultExtractorElems
81                              .item(i);
82                      Element preCondsElem = XMLUtils.getFirstElement(
83                            PRECONDITION_COMPARATORS_TAG, extractorElem);
84                      LinkedList<String> preCondComparatorIds = new LinkedList<String>();
85                      if (preCondsElem != null) {
86                         NodeList preCondComparators = 
87                            preCondsElem.getElementsByTagName(PRECONDITION_COMPARATOR_TAG);
88                         for (int k = 0; k < preCondComparators.getLength(); k++) {
89                           preCondComparatorIds.add(((Element) preCondComparators
90                               .item(k)).getAttribute(ID_ATTR));
91                         }
92                      }
93                      // This seems wrong, so added support for CLASS_ATTR while still
94                      //  supporting EXTRACTOR_CLASS_TAG as an attribute for specifying
95                      //  extractor class.
96                      String extractorClass = extractorElem.getAttribute(CLASS_ATTR);
97                      if (Strings.isNullOrEmpty(extractorClass)) {
98                         extractorClass = extractorElem.getAttribute(EXTRACTOR_CLASS_TAG);
99                      }
100                     String extractorConfigFile = getFilePathFromElement(
101                           extractorElem, EXTRACTOR_CONFIG_TAG);
102                     if (extractorConfigFile != null && !extractorConfigFile.startsWith("/")) {
103                        extractorConfigFile = new File(new File(mapFilePath).getParentFile(),
104                              extractorConfigFile).getAbsolutePath();
105                     }
106                     defaultExtractorSpecs
107                             .add(new MetExtractorSpec(extractorClass,
108                                     extractorConfigFile,
109                                     preCondComparatorIds));
110                 }
111                 extractorRepo
112                         .setDefaultMetExtractorSpecs(defaultExtractorSpecs);
113                 extractorRepo.setDefaultNamingConventionId(
114                       getNamingConventionId(defaultExtractorElem));
115             }
116 
117             NodeList mimeElems = root.getElementsByTagName(MIME_TAG);
118             for (int i = 0; i < mimeElems.getLength(); i++) {
119                 Element mimeElem = (Element) mimeElems.item(i);
120                 String mimeType = mimeElem.getAttribute(MIME_TYPE_ATTR);
121                 LinkedList<MetExtractorSpec> specs = new LinkedList<MetExtractorSpec>();
122 
123                 // Load naming convention class.
124                 extractorRepo.setNamingConventionId(mimeType,
125                       getNamingConventionId(mimeElem));
126 
127                 NodeList extractorSpecElems = mimeElem
128                         .getElementsByTagName(EXTRACTOR_TAG);
129                 if (extractorSpecElems != null
130                         && extractorSpecElems.getLength() > 0) {
131                     for (int j = 0; j < extractorSpecElems.getLength(); j++) {
132                         Element extractorSpecElem = (Element) extractorSpecElems
133                                 .item(j);
134                         MetExtractorSpec spec = new MetExtractorSpec();
135                         spec.setMetExtractor(extractorSpecElem
136                                 .getAttribute(CLASS_ATTR));
137 
138                         // get config file if specified
139                         String configFilePath = getFilePathFromElement(
140                                 extractorSpecElem, EXTRACTOR_CONFIG_TAG);
141                         if (configFilePath != null) {
142                            if (!configFilePath.startsWith("/")) {
143                               configFilePath = new File(new File(mapFilePath).getParentFile(),
144                                     configFilePath).getAbsolutePath();
145                            }
146                            spec.setExtractorConfigFile(configFilePath);
147                         }
148 
149                         // get preconditions file if specified
150                         Element preCondsElem = XMLUtils.getFirstElement(
151                               PRECONDITION_COMPARATORS_TAG, extractorSpecElem);
152                         if (preCondsElem != null) {
153                            NodeList preCondComparators = preCondsElem
154                                  .getElementsByTagName(PRECONDITION_COMPARATOR_TAG);
155                            LinkedList<String> preCondComparatorIds = new LinkedList<String>();
156                            for (int k = 0; k < preCondComparators.getLength(); k++) {
157                              preCondComparatorIds
158                                  .add(((Element) preCondComparators.item(k))
159                                      .getAttribute(ID_ATTR));
160                            }
161                            spec.setPreConditionComparatorIds(preCondComparatorIds);
162                         }
163 
164                         specs.add(spec);
165                     }
166                 }
167                 extractorRepo.addMetExtractorSpecs(mimeType, specs);
168             }
169             return extractorRepo;
170         } catch (IllegalAccessException e) {
171           LOG.log(Level.SEVERE, e.getMessage());
172           throw e;
173         } catch (InstantiationException e) {
174           LOG.log(Level.SEVERE, e.getMessage());
175           throw e;
176         } catch (MetExtractionException e) {
177           LOG.log(Level.SEVERE, e.getMessage());
178           throw e;
179         } catch (FileNotFoundException e) {
180           LOG.log(Level.SEVERE, e.getMessage());
181           throw e;
182         } catch (ClassNotFoundException e) {
183           LOG.log(Level.SEVERE, e.getMessage());
184           throw e;
185         } catch (CrawlerActionException e) {
186           LOG.log(Level.SEVERE, e.getMessage());
187           throw e;
188         }
189     }

191     private static String getNamingConventionId(Element parent) throws CrawlerActionException {
192        NodeList namingConventions = parent
193              .getElementsByTagName(NAMING_CONVENTION_TAG);
194        if (namingConventions != null && namingConventions.getLength() > 0) {
195           if (namingConventions.getLength() > 1) {
196              throw new CrawlerActionException("Can only have 1 '"
197                    + NAMING_CONVENTION_TAG + "' tag per mimetype");
198           }
199           Element namingConvention = (Element) namingConventions.item(0);
200           return namingConvention.getAttribute(ID_ATTR);
201        }
202        return null;
203     }
204 
205     private static String getFilePathFromElement(Element root, String elemName) {
206         String filePath = null;
207         Element elem = XMLUtils.getFirstElement(elemName, root);
208         if (elem != null) {
209             filePath = elem.getAttribute(FILE_ATTR);
210             if (Boolean.valueOf(elem.getAttribute(ENV_REPLACE_ATTR))) {
211               filePath = PathUtils.replaceEnvVariables(filePath);
212             }
213         }
214         return filePath;
215     }
