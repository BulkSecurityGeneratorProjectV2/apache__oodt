19  package org.apache.oodt.cas.resource.scheduler;
20  
21  //JAVA imports
22  import java.util.logging.Level;
23  import java.util.logging.Logger;
24  
25  //OODT imports
26  import org.apache.oodt.cas.resource.queuerepo.XmlQueueRepositoryFactory;
27  import org.apache.oodt.cas.resource.util.GenericResourceManagerObjectFactory;
28  import org.apache.oodt.cas.resource.jobqueue.JobQueue;
29  import org.apache.oodt.cas.resource.monitor.Monitor;
30  import org.apache.oodt.cas.resource.batchmgr.Batchmgr;
31  
32  /**
33   * @author woollard
34   * @author bfoster
35   * @version $Revision$
36   * 
37   * <p>
38   * A Least recently used scheduler factory interface.
39   * </p>
40   * 
41   */
42  public class LRUSchedulerFactory implements SchedulerFactory {
43  
44  	private static final Logger LOG = Logger
45  			.getLogger(LRUSchedulerFactory.class.getName());
46  
47  	/*
48  	 * a list of URIs pointing to directories that have the
49  	 * node-to-queue-mapping.xml files
50  	 */
51  	private LRUQueueManager queueManager;
52  
53  	/* our monitor */
54  	private Monitor mon = null;
55  
56  	/* our batchmgr */
57  	private Batchmgr batcher = null;
58  
59  	/* our job queue */
60  	private JobQueue queue = null;
61  
62  	public LRUSchedulerFactory() {
63  		String queueRepoFactoryClassStr = System.getProperty(
64  				"org.apache.oodt.cas.resource.queues.repo.factory",
65  				XmlQueueRepositoryFactory.class.getCanonicalName());
66  		String batchmgrClassStr = System
67  				.getProperty("resource.batchmgr.factory",
68  						"org.apache.oodt.cas.resource.batchmgr.XmlRpcBatchmgrFactory");
69  		String monitorClassStr = System
70  				.getProperty("resource.monitor.factory",
71  						"org.apache.oodt.cas.resource.monitor.AssignmentMonitorFactory");
72  
73  		String jobQueueClassStr = System
74  				.getProperty("resource.jobqueue.factory",
75  						"org.apache.oodt.cas.resource.jobqueue.JobStackJobQueueFactory");
76  
77  		try {
78  			queueManager = new LRUQueueManager(
79  					GenericResourceManagerObjectFactory
80  							.getQueueRepositoryFromFactory(
81  									queueRepoFactoryClassStr).loadQueues());
82  		} catch (Exception e) {
83  			LOG.log(Level.SEVERE, "Failed to create queue manager : "
84  					+ e.getMessage(), e);
85  			queueManager = null;
86  		}
87  		batcher = GenericResourceManagerObjectFactory
88  				.getBatchmgrServiceFromFactory(batchmgrClassStr);
89  		mon = GenericResourceManagerObjectFactory
90  				.getMonitorServiceFromFactory(monitorClassStr);
91  
92  		queue = GenericResourceManagerObjectFactory
93  				.getJobQueueServiceFromFactory(jobQueueClassStr);
94  
95  		// set the monitor for this batcher
96  		batcher.setMonitor(mon);
97  
98  		// set the job repo for this batcher
99  		batcher.setJobRepository(queue.getJobRepository());
100 
101 	}
102 
103 	public Scheduler createScheduler() {
104 		return new LRUScheduler(mon, batcher, queue, queueManager);
105 	}
106 
107 }
