18  package org.apache.oodt.cas.metadata.preconditions;
19  
20  //JDK imports
21  import org.apache.oodt.cas.metadata.exceptions.PreconditionComparatorException;
22  
23  import java.io.File;
24  import java.util.logging.Level;
25  import java.util.logging.Logger;
26  
27  /**
28   * A {@link PreConditionComparator} that checks for the existence of an arbitrary file (a "sub-product") 
29   * within the product directory.
30   * If the product is not a directory, or the filePath field is not specified, this class behaves
31   * exactly like the superclass {@link ExistanceCheckComparator}.
32   * 
33   * @author luca (Luca Cinquini)
34   *
35   */
36  public class SubProductExistenceCheckComparator extends
37  		ExistanceCheckComparator {
38  	
39  	private static final Logger LOG = Logger.getLogger(ExistanceCheckComparator.class.getName());
40  	
41      private String filePath;
42  
43  	public String getFilePath() {
44  		return filePath;
45  	}
46  
47  	/**
48  	 * Sets the sub-product file path (must be relative to the product absolute path).
49  	 */
50  	public void setFilePath(String filePath) {
51  		this.filePath = filePath;
52  	}
53  	
54      @Override
55      protected int performCheck(File product, Boolean compareItem)
56              throws PreconditionComparatorException {
57      	
58      	if (product.isDirectory() && filePath!=null) {
59      		File file = new File(product.getAbsolutePath()+"/"+filePath);
60      		LOG.log(Level.INFO, "Checking existence of file="+file.getAbsolutePath());
61      		return Boolean.valueOf(file.exists()).compareTo(compareItem);
62      	} else {
63      		return Boolean.valueOf(product.exists()).compareTo(compareItem);
64      	}
65      }
66  	
67  }
