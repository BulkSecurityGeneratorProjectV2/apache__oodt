103    @Override
104    public void run(Metadata metadata, WorkflowTaskConfiguration config)
105          throws WorkflowTaskInstanceException {
106       try {
107          // Initialize CAS-PGE.
108          pgeMetadata = createPgeMetadata(metadata, config);
109          pgeConfig = createPgeConfig();
110          runPropertyAdders();
111          wm = createWorkflowManagerClient();
112          workflowInstId = getWorkflowInstanceId();
113          logger = createLogger(); // use workflow ID specific logger from now on 
114 
115          // Write out PgeMetadata.
116          dumpMetadataIfRequested();
117 
118          // Setup the PGE.
119          createExeDir();
120          createOuputDirsIfRequested();
121          updateStatus(CONF_FILE_BUILD.getWorkflowStatusName());
122          createDynamicConfigFiles();
123          updateStatus(STAGING_INPUT.getWorkflowStatusName());
124          stageFiles();
125 
126          // Run the PGE.
127          runPge();
128 
129          // Ingest products.
130          runIngestCrawler(createProductCrawler());
131 
132          // Commit dynamic metadata.
133          updateDynamicMetadata();
134       } catch (Exception e) {
135          logger.log(Level.SEVERE, "PGETask FAILED!!! : " + e.getMessage(), e);
136          throw new WorkflowTaskInstanceException("PGETask FAILED!!! : "
137                + e.getMessage(), e);
138       }
139    }

175    protected PgeMetadata createPgeMetadata(Metadata dynMetadata,
176          WorkflowTaskConfiguration config) {
177       logger.info("Converting workflow configuration to static metadata...");
178       Metadata staticMetadata = new Metadata();
179       for (Object objKey : config.getProperties().keySet()) {
180          String key = (String) objKey;
181          PgeTaskMetKeys metKey = PgeTaskMetKeys.getByName(key);
182          if (metKey != null && metKey.isVector()) {
183             List<String> values = Lists.newArrayList(
184                   Splitter.on(",").trimResults()
185                   .omitEmptyStrings()
186                   .split(config.getProperty(key)));
187             logger.finest("Adding static metadata: key = [" + key
188                   + "] value = " + values);
189             staticMetadata.addMetadata(key, values);
190          } else {
191             String value = config.getProperty(key);
192             logger.finest("Adding static metadata: key = [" + key
193                   + "] value = [" + value + "]");
194             staticMetadata.addMetadata(key, value);
195          }
196       }
197       logger.info("Loading workflow context metadata...");
198       for (String key : dynMetadata.getAllKeys()) {
199          logger.finest(
200                "Adding dynamic metadata: key = [" + key + "] value = "
201                      + dynMetadata.getAllMetadata(key));
202       }
203       return new PgeMetadata(staticMetadata, dynMetadata);
204    }

396    protected void runPge() throws PGEException, XmlRpcException {
397       ScriptFile sf = null;
398       try {
399          long startTime = System.currentTimeMillis();
400          logger.info("PGE start time [" + new Date(startTime) + "]");
401 
402          // create script to run
403          sf = buildPgeRunScript();
404          sf.writeScriptFile(getScriptPath().getAbsolutePath());
405 
406          // run script and evaluate whether success or failure
407          updateStatus(RUNNING_PGE.getWorkflowStatusName());
408          logger.info("Starting execution of PGE...");
409          if (!wasPgeSuccessful(ExecUtils.callProgram(
410                pgeConfig.getShellType() + " " + getScriptPath(), logger,
411                new File(pgeConfig.getExeDir()).getAbsoluteFile()))) {
412             throw new RuntimeException("Pge didn't finish successfully");
413          } else {
414             logger.info(
415                   "Successfully completed running: '" + sf.getCommands() + "'");
416          }
417 
418          long endTime = System.currentTimeMillis();
419          logger.info("PGE end time [" + new Date(startTime) + "]");
420 
421          long runTime = endTime - startTime;
422          logger.info("PGE runtime in millis [" + runTime + "]");
423 
424          pgeMetadata.replaceMetadata(PGE_RUNTIME, Long.toString(runTime));
425 
426       } catch (WorkflowException e) {
427         throw new PGEException("Exception when executing PGE commands '" + (sf.getCommands()) + "' : " + e.getMessage(), e);
428       } catch (IOException e) {
429         throw new PGEException("Exception when executing PGE commands '" + (sf.getCommands()) + "' : " + e.getMessage(), e);
430       }
431    }

437    protected void processOutput() throws IOException {
438      for (final OutputDir outputDir : this.pgeConfig.getOuputDirs()) {
439          File[] createdFiles = new File(outputDir.getPath()).listFiles();
440        if (createdFiles != null) {
441          for (File createdFile : createdFiles) {
442              Metadata outputMetadata = new Metadata();
443              for (RegExprOutputFiles regExprFiles : outputDir
444                      .getRegExprOutputFiles()) {
445                  if (Pattern.matches(regExprFiles.getRegExp(), createdFile
446                          .getName())) {
447                      try {
448                          PcsMetFileWriter writer = (PcsMetFileWriter) Class
449                                  .forName(regExprFiles.getConverterClass())
450                                  .newInstance();
451                          outputMetadata.replaceMetadata(this.getMetadataForFile(
452                  (regExprFiles.getRenamingConv() != null)
453                ? createdFile = this.renameFile(createdFile, regExprFiles.getRenamingConv())
454                : createdFile, writer, regExprFiles.getArgs()));
455                      } catch (Exception e) {
456                          logger.severe(
457                                  "Failed to create metadata file for '"
458                                          + createdFile + "' : "
459                                          + e.getMessage());
460                      }
461                  }
462              }
463              if (outputMetadata.getAllKeys().size() > 0) {
464                this.writeFromMetadata(outputMetadata, createdFile.getAbsolutePath()
465                                                       + "." + this.pgeMetadata.getMetadata(MET_FILE_EXT));
466              }
467          }
468        }
469      }

493 	protected ProductCrawler createProductCrawler()
494         throws MalformedURLException, IllegalAccessException, CrawlerActionException, MetExtractionException,
495         InstantiationException, FileNotFoundException, ClassNotFoundException {
496      /* create a ProductCrawler based on whether or not the output dir specifies a MIME_EXTRACTOR_REPO */
497       logger.info("Configuring ProductCrawler...");
498       ProductCrawler crawler;
499       if (pgeMetadata.getMetadata(MIME_EXTRACTOR_REPO) != null && 
500     		  !pgeMetadata.getMetadata(MIME_EXTRACTOR_REPO).equals("")){
501           crawler = new AutoDetectProductCrawler();
502           ((AutoDetectProductCrawler)crawler).
503             setMimeExtractorRepo(pgeMetadata.getMetadata(MIME_EXTRACTOR_REPO));    	  
504       }
505       else{
506     	  crawler = new StdProductCrawler();
507       }
508 
509       crawler.setClientTransferer(pgeMetadata
510             .getMetadata(INGEST_CLIENT_TRANSFER_SERVICE_FACTORY));
511       crawler.setFilemgrUrl(pgeMetadata.getMetadata(INGEST_FILE_MANAGER_URL));
512       String crawlerConfigFile = pgeMetadata.getMetadata(CRAWLER_CONFIG_FILE);
513       if (!Strings.isNullOrEmpty(crawlerConfigFile)) {
514          crawler.setApplicationContext(
515                new FileSystemXmlApplicationContext(crawlerConfigFile));
516          List<String> actionIds = pgeMetadata.getAllMetadata(ACTION_IDS);
517          if (actionIds != null) {
518             crawler.setActionIds(actionIds);
519          }
520       }
521       crawler.setRequiredMetadata(pgeMetadata.getAllMetadata(REQUIRED_METADATA));
522       crawler.setCrawlForDirs(Boolean.parseBoolean(pgeMetadata
523             .getMetadata(CRAWLER_CRAWL_FOR_DIRS)));
524       crawler.setNoRecur(!Boolean.parseBoolean(
525             pgeMetadata.getMetadata(CRAWLER_RECUR)));
526       logger.fine(
527             "Passing Workflow Metadata to CAS-Crawler as global metadata . . .");
528       crawler.setGlobalMetadata(pgeMetadata.asMetadata(PgeMetadata.Type.DYNAMIC));
529       logger.fine("Created ProductCrawler ["
530             + crawler.getClass().getCanonicalName() + "]");
531       return crawler;
532    }

534    protected void runIngestCrawler(ProductCrawler crawler) throws PGEException, IOException, XmlRpcException {
535       // Determine if we need to create Metadata files
536 	   if (crawler instanceof StdProductCrawler){
537 		   this.processOutput();
538 	   }
539 	   
540 	   // Determine directories to crawl.
541       List<File> crawlDirs = new LinkedList<File>();
542       for (OutputDir outputDir : pgeConfig.getOuputDirs()) {
543          crawlDirs.add(new File(outputDir.getPath()));
544       }
545 
546       // Start crawlin...
547       updateStatus(CRAWLING.getWorkflowStatusName());
548       boolean attemptIngestAll = Boolean.parseBoolean(pgeMetadata
549             .getMetadata(ATTEMPT_INGEST_ALL));
550       for (File crawlDir : crawlDirs) {
551          logger.info("Crawling for products in [" + crawlDir + "]");
552          crawler.crawl(crawlDir);
553          if (!attemptIngestAll) {
554             verifyIngests(crawler);
555          }
556       }
557       if (attemptIngestAll) {
558          verifyIngests(crawler);
559       }
560    }

562    protected void verifyIngests(ProductCrawler crawler) throws PGEException {
563       logger.info("Verifying ingests successful...");
564       boolean ingestsSuccess = true;
565       String exceptionMsg = "";
566       for (IngestStatus status : crawler.getIngestStatus()) {
567          if (status.getResult().equals(IngestStatus.Result.FAILURE)) {
568             exceptionMsg += (exceptionMsg.equals("") ? "" : " : ")
569                   + "Failed to ingest product [file='"
570                   + status.getProduct().getAbsolutePath() + "',result='"
571                   + status.getResult() + "',msg='" + status.getMessage() + "']";
572             ingestsSuccess = false;
573          } else if (!status.getResult().equals(IngestStatus.Result.SUCCESS)) {
574             logger.warning("Product was not ingested [file='"
575                   + status.getProduct().getAbsolutePath() + "',result='"
576                   + status.getResult() + "',msg='" + status.getMessage() + "']");
577          }
578       }
579       if (!ingestsSuccess) {
580          throw new PGEException(exceptionMsg);
581       } else {
582          logger.info("Ingests were successful");
583       }
584    }
