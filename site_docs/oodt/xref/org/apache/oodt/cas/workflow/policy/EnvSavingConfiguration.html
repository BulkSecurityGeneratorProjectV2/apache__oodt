6   package org.apache.oodt.cas.workflow.policy;
7   
8   //JDK imports
9   import java.util.Properties;
10  
11  //OODT imports
12  import org.apache.oodt.cas.workflow.structs.WorkflowTaskConfiguration;
13  
14  /**
15   * @author mattmann
16   * @version $Revision$
17   * 
18   * <p>
19   * A {@link WorkflowTaskConfiguration} that preserves whether or not a task config
20   * property was envReplace'd or not.
21   * </p>.
22   */
23  public class EnvSavingConfiguration extends WorkflowTaskConfiguration {
24  
25      private Properties envReplaceMap;
26  
27      public EnvSavingConfiguration() {
28          super();
29          envReplaceMap = new Properties();
30      }
31  
32      /*
33       * (non-Javadoc)
34       * 
35       * @see gov.nasa.jpl.oodt.cas.workflow.structs.WorkflowTaskConfiguration#addConfigProperty(java.lang.String,
36       *      java.lang.String)
37       */
38      public void addConfigProperty(String propName, String propVal) {
39          addConfigProperty(propName, propVal, false);
40  
41      }
42  
43      /**
44       * Adds a configuration property, along with its envReplace
45       * information.
46       * 
47       * @param propName The workflow config property name.
48       * @param propVal The workflow config property val.
49       * @param isReplace Whether or not the property should be envReplaced.
50       */
51      public void addConfigProperty(String propName, String propVal,
52              boolean isReplace) {
53          super.addConfigProperty(propName, propVal);
54          envReplaceMap.put(propName, String.valueOf(isReplace));
55      }
56  
57      public boolean isReplace(String propName) {
58          return Boolean.valueOf((String) this.envReplaceMap.get(propName))
59                  .booleanValue();
60      }
61  
62  }
