69  	public void newTable(String[][] data){
70  		this.removeAll();
71  		if(data.length>1){
72  			header = new HeaderRow(listener,data[0].length);
73  			for(int j=0;j<data[0].length;j++){
74  				header.setText(j, data[0][j]);
75  			}
76  			this.add(header);
77  			
78  			for(int j=1;j<data.length;j++){
79  				Row r = new Row(data[0].length);

82  				}

84  			}
85  		}

88  	}

94  	public void sortRows(int col, String sortType){
95  		int length = this.getComponentCount();
96  		int i,j;
97  
98  	    for (i=length; --i >=1;) {
99  	       for (j=1; j<i;j++) {
100 	    	   Row jThRow = (Row)this.getComponent(j);
101 	    	   Row iThRow = (Row)this.getComponent(i);
102 	    	   if(sortType.equals("Accending")){
103 	    		   if(jThRow.getText(col).compareTo(iThRow.getText(col))>0){
104 	    			   swapRows(j,j+1);
105 	    		   }
106 	    	   } else {
107 	    		   if(jThRow.getText(col).compareTo(iThRow.getText(col))<0){
108 	    			   swapRows(j,j+1);
109 	    		   }
110 	    	   }
111 	       }
112 	    }
113 	    this.validate();
114 	    this.repaint();
115 	    
116 	}

119 		Component c1 = this.getComponent(index1);
120 		Component c2 = this.getComponent(index2);
121 		this.add(c1, index2);
122 		this.add(c2, index1);
