32  /**
33   * Query servlet provides common query functionality for profile queries and product
34   * queries.  It treats GETs as POSTs, retrieves the complete XMLQuery (if the
35   * <code>xmlq</code> request parameter appears) or constructs a new XMLQuery (using the
36   * <code>q</code> request parameter, which contains just a query expression), updates
37   * system properties for the handlers, instantiates any new handlers, and then runs the
38   * query.
39   *
40   */
41  public abstract class QueryServlet extends GridServlet {
42  	/**
43  	 * Get a list of {@link Server}s that will provide handlers to handle the query.
44  	 * Subclasses implement this by returning a list of either {@link ProductServer}s
45  	 * or {@link ProfileServer}s.
46  	 *
47  	 * @param config a <code>Configuration</code> value.
48  	 * @return a <code>List</code> value of {@link Server}s of some kind.
49  	 */
50  	protected abstract List getServers(Configuration config);
51  
52  	/**
53  	 * Handle the query.  Subclasses implement this by parsing the query and returning
54  	 * some results.
55  	 *
56  	 * @param query The query to handle.
57  	 * @param handlers A list of either product <code>QueryHandler</code>s or profile <code>ProfileHandler</code>s.
58  	 * @param req a <code>HttpServletRequest</code> value.
59  	 * @param res a <code>HttpServletResponse</code> value.
60  	 * @throws IOException if an I/O error occurs.
61  	 * @throws ServletException if a servlet error occurs.
62  	 */
63  	protected abstract void handleQuery(XMLQuery query, List handlers, HttpServletRequest req, HttpServletResponse res)
64  		throws IOException;
65  
66  	/**
67  	 * Treat GETs as POSTs.
68  	 *
69  	 * @param req a <code>HttpServletRequest</code> value.
70  	 * @param res a <code>HttpServletResponse</code> value.
71  	 * @throws IOException if an error occurs.
72  	 * @throws ServletException if an error occurs.
73  	 */
74  	public void doGet(HttpServletRequest req, HttpServletResponse res) throws IOException, ServletException {
75  		doPost(req, res);
76  	}

78  	/**
79  	 * Handle the query.
80  	 *
81  	 * @param req a <code>HttpServletRequest</code> value.
82  	 * @param res a <code>HttpServletResponse</code> value.
83  	 * @throws IOException if an error occurs.
84  	 * @throws ServletException if an error occurs.
85  	 */
86  	public void doPost(HttpServletRequest req, HttpServletResponse res) throws IOException, ServletException {
87  		try {
88  			XMLQuery query = getQuery(req, res);			       // Get the query
89  			if (query == null) {
90  			  return;                       // No query? My favorite case, right here!
91  			}
92  
93  			Configuration config = getConfiguration();		       // Get the current configuration.
94  			updateProperties(config);				       // Using it, update the system properties
95  			updateHandlers(getServers(config));			       // And any servers.
96  
97  			List handlerObjects = new ArrayList(handlers.size());	       // Start with no handlers.
98  		  for (Object handler : handlers) {           // For each server
99  			InstantiedHandler ih = (InstantiedHandler) handler;   // Get its handler
100 			handlerObjects.add(ih.getHandler());               // Add its handler
101 		  }
102 			handleQuery(query, handlerObjects, req, res);		       // Handlers: handle query, please
103 		} catch (RuntimeException ex) {
104 			throw ex;
105 		} catch (Exception ex) {
106 			throw new ServletException(ex);
107 		}
108 	}

110 	/**
111 	 * Get the query from an HTTP request.  The request must have either a
112 	 * <code>xmlq</code> parameter (which is given priority) or a <code>q</code>
113 	 * parameter.  We expect <code>xmlq</code> to contain the XML text format of an
114 	 * XMLQuery object.  <code>q</code> can contain just the query expression, from
115 	 * which we'll construct a fresh <code>XMLQuery</code> with reasonable defaults.
116 	 * If the user specifies the <code>q</code> paramater, we'll treat it as a parsed
117 	 * query, letting the XMLQuery class parse it and build its various expression
118 	 * stacks according to the DIS syntax.  The user can specify the <code>unp</code>
119 	 * parameter to control this behavior, though.  Set to the string
120 	 * <code>true</code> and we'll treat the query as <em>unparsed</em>, and the
121 	 * XMLQuery class should leave it alone.  Otherwise, an other value (or
122 	 * unspecified) will be interpreted as <code>false</code>, meaning the query will
123 	 * be parsed.
124 	 *
125 	 * @param req a <code>HttpServletRequest</code> value.
126 	 * @param res a <code>HttpServletResponse</code> value.
127 	 * @return a <code>XMLQuery</code> value.
128 	 * @throws IOException if an error occurs.
129 	 */
130 	protected XMLQuery getQuery(HttpServletRequest req, HttpServletResponse res) throws IOException {
131 		String xmlq = req.getParameter("xmlq");				       // Grab any xmlq
132 		String q = req.getParameter("q");				       // Grab any q
133 		String unp = req.getParameter("unp");				       // And grab any unp (pronounced "unp")
134 		if (xmlq == null) {
135 		  xmlq = "";                           // No xmlq?  Use epsilon
136 		}
137 		if (q == null) {
138 		  q = "";                               // No q?  Use lambda
139 		}
140 		if (unp == null) {
141 		  unp = "";                           // Use some other greek letter for empty str
142 		}
143 		String[] mimes = req.getParameterValues("mime");		       // Grab any mimes
144 		if (mimes == null) {
145 		  mimes = EMPTY_STRING_ARRAY;                   // None?  Use empty array
146 		}
147 
148 		if (xmlq.length() > 0) {
149 		  try {                           // Was there an xmlq?
150 			return new XMLQuery(xmlq);                       // Use it in its entirety, ignoring the rest
151 		  } catch (SAXException ex) {                           // Can't parse it?
152 			res.sendError(HttpServletResponse.SC_BAD_REQUEST,           // Then that's a bad ...
153 				"cannot parse xmlq: " + ex.getMessage());           // ... request, which I hate
154 			return null;                               // so flag it with a null
155 		  }
156 		} else if (q.length() > 0) {					       // Was there a q?
157 			boolean unparsed = "true".equals(unp);			       // If so, was there also an unp?
158 			return new XMLQuery(q, "wgq", "Web Grid Query",		       // Use it to make an XMLQuery
159 				"Query from Web-Grid", /*ddID*/null,                   // And all of these extra
160 				/*resultModeId*/null, /*propType*/null,                // parameters really annoy
161 				/*propLevels*/null, /*maxResults*/Integer.MAX_VALUE,   // the poop out of me
162 				Arrays.asList(mimes), !unparsed);                      // It's just a query for /sbin/fsck sake!
163 		}
164 
165 		res.sendError(HttpServletResponse.SC_BAD_REQUEST, "xmlq or q parameters required");
166 		return null;
167 	}

169 	/**
170 	 * Update the query handlers instantiated.
171 	 *
172 	 * @param servers a <code>List</code> of {@link Server}s.
173 	 * @throws ClassNotFoundException if a class can't be found.
174 	 * @throws InstantiationException if a class can't be instantiated.
175 	 * @throws IllegalAccessException if a constructor isn't public.
176 	 */
177 	private synchronized void updateHandlers(List servers) throws ClassNotFoundException, InstantiationException,
178 		IllegalAccessException {
179 		eachServer:
180 	  for (Object server1 : servers) {               // For each server
181 		Server server = (Server) server1;                   // Grab the server
182 		for (Object handler1 : handlers) {           // For each handler
183 		  InstantiedHandler handler = (InstantiedHandler) handler1; // Grab the handler
184 		  if (handler.getServer().equals(server))               // Have we already instantiated?
185 		  {
186 			continue eachServer;                   // Yes, try the next server
187 		  }
188 		}
189 		InstantiedHandler handler                       // No.  Create ...
190 			= new InstantiedHandler(server, server.createHandler()); // ... a fresh handler
191 		handlers.add(handler);                           // Save it
192 	  }
193 
194 		for (Iterator i = handlers.iterator(); i.hasNext();) {		       // Now, for each handler
195 			InstantiedHandler handler = (InstantiedHandler) i.next();      // Grab the handler
196 			if (!servers.contains(handler.getServer()))		       // Does its server still exist?
197 			{
198 			  i.remove();                           // If not, remove the handler
199 			}
200 		}
201 	}

203 	/**
204 	 * Update system properties used by query handlers, if any have changed.
205 	 *
206 	 * @param config a <code>Configuration</code> value.
207 	 */
208 	private synchronized void updateProperties(Configuration config) {
209 		if (properties != null) {					       // Any old properties?
210 			if (properties.equals(config.getProperties())) {
211 			  return;           // Yes, any changes?  No?  Then done.
212 			}
213 		  for (Object o : properties.keySet()) {
214 			System.getProperties().remove(o);           // and remove it.
215 		  }
216 		}
217 		properties = (Properties) config.getProperties().clone();	       // Now copy the new settings
218 		System.getProperties().putAll(properties);			       // And set them!
219 	}

236 		/**

243 			this.server = server;
244 			this.handler = handler;
245 		}

249 		 *

251 		 */

254 		}

258 		 *
259 		 * @return a <code>ProfileHandler</code> or a product <code>QueryHandler</code>.

263 		}

266 		private Server server;
