95  			/*children*/Collections.EMPTY_LIST, "1", /*revNotes*/Collections.EMPTY_LIST);
96  		p.setProfileAttributes(pa);
97  		ResourceAttributes ra = new ResourceAttributes(p, "id", "title", /*formats*/Collections.EMPTY_LIST, "description",
98  			/*creators*/Collections.EMPTY_LIST, /*subjects*/Collections.EMPTY_LIST, /*publishers*/Collections.EMPTY_LIST,
99  			/*contributors*/Collections.EMPTY_LIST, /*dates*/Collections.EMPTY_LIST, /*types*/Collections.EMPTY_LIST,
100 			/*sources*/Collections.EMPTY_LIST, /*languages*/Collections.EMPTY_LIST, /*relations*/Collections.EMPTY_LIST,
101 			/*coverages*/Collections.EMPTY_LIST, /*rights*/Collections.EMPTY_LIST, Collections.singletonList("context"),
102 			"granule", "grainy", Collections.singletonList("file:/dev/null"));
103 		p.setResourceAttributes(ra);
104 		EnumeratedProfileElement e = new EnumeratedProfileElement(p, "mode", "mode", "Mode", "string", "mode",
105 			/*synonyms*/Collections.EMPTY_LIST, /*obligation*/false, /*maxOccurrence*/1, "No comment",
106 			Collections.singletonList("The current\n mode setting\n  is set to indent\n\n   a\n\n"
107 				+ "    number of increasing\n     times."));
108 		p.getProfileElements().put("mode", e);
109 
110 		Profile q = new Profile(p.toString());
111 		e = (EnumeratedProfileElement) q.getProfileElements().values().iterator().next();
112 		assertEquals("The current\n mode setting\n  is set to indent\n\n   a\n\n"
113 			+ "    number of increasing\n     times.", e.getValues().get(0));
114 	}
