37  /**
38   * @author mattmann
39   * @version $Revsion$
40   * 
41   * <p>
42   * A simple {@link WorkflowInstanceRepository} that does not require the use of
43   * a database to track the status of the Workflow executions it manages.
44   * </p>
45   */
46  public class MemoryWorkflowInstanceRepository extends
47          AbstractPaginatibleInstanceRepository {
48  
49      /* our workflow instance map: maps workfllowInstId to WorkflowInstance */
50      private ConcurrentHashMap workflowInstMap = null;
51  
52      /* our log stream */
53      private static final Logger LOG = Logger
54              .getLogger(MemoryWorkflowInstanceRepository.class.getName());
55  
56      /**
57       * <p>
58       * Default Constructor
59       * </p>
60       */
61      public MemoryWorkflowInstanceRepository(int pageSize) {
62          workflowInstMap = new ConcurrentHashMap();
63          this.pageSize = pageSize;
64      }
65  
66      /*
67       * (non-Javadoc)
68       * 
69       * @see org.apache.oodt.cas.workflow.instrepo.WorkflowInstanceRepository#addWorkflowInstance(org.apache.oodt.cas.workflow.structs.WorkflowInstance)
70       */
71      public synchronized void addWorkflowInstance(WorkflowInstance wInst)
72              throws InstanceRepositoryException {
73          String instId = "urn:" + DateConvert.isoFormat(new Date());
74          wInst.setId(instId);
75          workflowInstMap.put(instId, wInst);
76      }
77  
78      /*
79       * (non-Javadoc)
80       * 
81       * @see org.apache.oodt.cas.workflow.instrepo.WorkflowInstanceRepository#removeWorkflowInstance(org.apache.oodt.cas.workflow.structs.WorkflowInstance)
82       */
83      public synchronized void removeWorkflowInstance(WorkflowInstance wInst)
84              throws InstanceRepositoryException {
85          workflowInstMap.remove(wInst.getId());
86      }
87  
88      /*
89       * (non-Javadoc)
90       * 
91       * @see org.apache.oodt.cas.workflow.instrepo.WorkflowInstanceRepository#updateWorkflowInstance(org.apache.oodt.cas.workflow.structs.WorkflowInstance)
92       */
93      public synchronized void updateWorkflowInstance(WorkflowInstance wInst)
94              throws InstanceRepositoryException {
95          WorkflowInstance inst = (WorkflowInstance) workflowInstMap.get(wInst
96                  .getId());
97  
98          if (inst == null) {
99              LOG
100                     .log(
101                             Level.WARNING,
102                             "Attempt to update workflow instance id: "
103                                     + wInst.getId()
104                                     + " workflow instance is not being tracked by this engine!");
105             return;
106         }
107 
108         workflowInstMap.put(wInst.getId(), wInst);
109     }

130     /*
131      * (non-Javadoc)
132      * 
133      * @see org.apache.oodt.cas.workflow.instrepo.WorkflowInstanceRepository#getWorkflowInstancesByStatus(java.lang.String)
134      */
135     public List getWorkflowInstancesByStatus(String status)
136             throws InstanceRepositoryException {
137         List instances = new Vector();
138 
139         for (Object o : workflowInstMap.keySet()) {
140             String workflowInstId = (String) o;
141             WorkflowInstance inst = (WorkflowInstance) workflowInstMap
142                 .get(workflowInstId);
143             if (inst.getStatus().equals(status)) {
144                 instances.add(inst);
145             }
146         }
147 
148         return instances;
149     }

151     /*
152      * (non-Javadoc)
153      * 
154      * @see org.apache.oodt.cas.workflow.instrepo.AbstractPaginatibleInstanceRepository#paginateWorkflows(int,
155      *      java.lang.String)
156      */
157     protected List paginateWorkflows(int pageNum, String status)
158             throws InstanceRepositoryException {
159         // first sort insts by startDateTime
160         List allInsts = Arrays.asList(this.workflowInstMap.keySet().toArray());
161         Collections.sort(allInsts, new Comparator() {
162 
163             public int compare(Object o1, Object o2) {
164                 WorkflowInstance inst1 = (WorkflowInstance) o1;
165                 WorkflowInstance inst2 = (WorkflowInstance) o2;
166 
167                 return inst1.getStartDateTimeIsoStr().compareTo(
168                         inst2.getStartDateTimeIsoStr());
169             }
170 
171         });
172 
173         int startNum = (pageNum - 1) * pageSize;
174         if (startNum > allInsts.size()) {
175             startNum = 0;
176         }
177 
178         List instIds = new Vector(pageSize);
179 
180         for (int i = startNum; i < Math.min(allInsts.size(),
181                 (startNum + pageSize)); i++) {
182             WorkflowInstance wInst = (WorkflowInstance)allInsts.get(i);
183             instIds.add(wInst.getId());
184         }
185 
186         return instIds;
187     }

198     /*
199      * (non-Javadoc)
200      * 
201      * @see org.apache.oodt.cas.workflow.instrepo.WorkflowInstanceRepository#getNumWorkflowInstancesByStatus(java.lang.String)
202      */
203     public int getNumWorkflowInstancesByStatus(String status)
204             throws InstanceRepositoryException {
205         int cnt = 0;
206 
207         if (this.workflowInstMap != null && this.workflowInstMap.keySet().size() > 0) {
208             for (Object o : this.workflowInstMap.keySet()) {
209                 String wInstId = (String) o;
210                 WorkflowInstance inst = (WorkflowInstance) this.workflowInstMap
211                     .get(wInstId);
212                 if (inst.getStatus().equals(status)) {
213                     cnt++;
214                 }
215             }
216         }
217 
218         return cnt;
219     }
