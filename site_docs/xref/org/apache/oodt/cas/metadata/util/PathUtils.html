62      public static String replaceEnvVariables(String origPath,
63              Metadata metadata, boolean expand) {
64          StringBuilder finalPath = new StringBuilder();
65  
66          for (int i = 0; i < origPath.length(); i++) {
67              if (origPath.charAt(i) == '[') {
68                  VarData data = readEnvVarName(origPath, i);
69                  String var;
70                  if (metadata != null
71                          && metadata.getMetadata(data.getFieldName()) != null) {
72                      List valList = metadata.getAllMetadata(data.getFieldName());
73                      var = (String) valList.get(0);
74                      if (expand) {
75                          for (int j = 1; j < valList.size(); j++) {
76                              var += DELIMITER + (String) valList.get(j);
77                          }
78                      }
79                  } else {
80                      var = EnvUtilities.getEnv(data.getFieldName());
81                  }
82                  finalPath.append(var);
83                  i = data.getEndIdx();
84              } else {
85                  finalPath.append(origPath.charAt(i));
86              }
87          }
88  
89          return finalPath.toString();
90      }

114     public static String doDynamicDateReplacement(String string,
115             Metadata metadata) throws CasMetadataException, CommonsException {
116         Pattern datePattern = Pattern
117                 .compile("\\[\\s*DATE\\s*(?:[+-]{1}[^\\.]{1,}?){0,1}\\.\\s*(?:(?:DAY)|(?:MONTH)|(?:YEAR)|(?:UTC)|(?:TAI)){1}\\s*\\]");
118         Matcher dateMatcher = datePattern.matcher(string);
119         while (dateMatcher.find()) {
120             String dateString = string.substring(dateMatcher.start(),
121                     dateMatcher.end());
122             GregorianCalendar gc = new GregorianCalendar();
123 
124             // roll the date if roll days was specfied
125             int plusMinusIndex;
126             if ((plusMinusIndex = dateString.indexOf('-')) != -1
127                     || (plusMinusIndex = dateString.indexOf('+')) != -1) {
128                 int dotIndex;
129                 if ((dotIndex = dateString.indexOf('.', plusMinusIndex)) != -1) {
130                     int rollDays = Integer.parseInt(PathUtils
131                             .replaceEnvVariables(
132                                     dateString.substring(plusMinusIndex,
133                                             dotIndex), metadata).replaceAll(
134                                     "[\\+\\s]", ""));
135                     gc.add(GregorianCalendar.DAY_OF_YEAR, rollDays);
136                 } else {
137                     throw new CasMetadataException(
138                         "Malformed dynamic date replacement specified (no dot separator) - '"
139                         + dateString + "'");
140                 }
141             }
142 
143             // determine type and make the appropriate replacement
144             String[] splitDate = dateString.split("\\.");
145             if (splitDate.length < 2) {
146                 throw new CasMetadataException("No date type specified - '" + dateString
147                                                + "'");
148             }
149             String dateType = splitDate[1].replaceAll("[\\[\\]\\s]", "");
150             String replacement;
151             if (dateType.equals("DAY")) {
152                 replacement = StringUtils.leftPad(gc
153                         .get(GregorianCalendar.DAY_OF_MONTH)
154                         + "", 2, "0");
155             } else if (dateType.equals("MONTH")) {
156                 replacement = StringUtils.leftPad((gc
157                         .get(GregorianCalendar.MONTH) + 1)
158                         + "", 2, "0");
159             } else if (dateType.equals("YEAR")) {
160                 replacement = gc.get(GregorianCalendar.YEAR) + "";
161             } else if (dateType.equals("UTC")) {
162                 replacement = DateUtils.toString(DateUtils.toUtc(gc));
163             } else if (dateType.equals("TAI")) {
164                 replacement = DateUtils.toString(DateUtils.toTai(gc));
165             } else {
166                 throw new CasMetadataException("Invalid date type specified '"
167                         + dateString + "'");
168             }
169 
170             string = StringUtils.replace(string, dateString, replacement);
171             dateMatcher = datePattern.matcher(string);
172         }
173         return string;
174     }

176     /**
177      * usage format: [DATE_ADD(<date>,<date-format>,<add-amount>,<hr | min | sec | day | mo | yr>)]
178      * example: [DATE_ADD(2009-12-31, yyyy-MM-dd, 1, day)] . . . output will be: 2010-01-01
179      * - dynamic replacement is allowed for the <date> as well, for example: 
180      *  [DATE_ADD([DATE.UTC], yyyy-MM-dd'T'HH:mm:ss.SSS'Z', 1, day)] will add one day to the 
181      *  current UTC time
182      */
183     public static String doDynamicDateRollReplacement(String string,
184             Metadata metadata) throws ParseException, CasMetadataException, CommonsException {
185         Pattern dateFormatPattern = Pattern
186                 .compile("\\[\\s*DATE_ADD\\s*\\(.{1,}?,.{1,}?,.{1,}?,.{1,}?\\)\\s*\\]");
187         Matcher dateFormatMatcher = dateFormatPattern.matcher(string);
188         while (dateFormatMatcher.find()) {
189             String dateFormatString = string.substring(dateFormatMatcher
190                     .start(), dateFormatMatcher.end());
191 
192             // get arguments
193             Matcher argMatcher = Pattern.compile("\\(.*\\)").matcher(
194                     dateFormatString);
195             argMatcher.find();
196             String argsString = dateFormatString.substring(argMatcher.start() + 1,
197                     argMatcher.end() - 1); 
198             argsString = doDynamicReplacement(argsString, metadata);
199             String[] args = argsString.split(",");
200             String dateString = args[0].trim();
201             dateString = doDynamicReplacement(dateString, metadata);
202             String dateFormat = args[1].trim();
203             int addAmount = Integer.parseInt(args[2].trim());
204             String addUnits = args[3].trim().toLowerCase();
205 
206             // reformat date
207             Date date = new SimpleDateFormat(dateFormat).parse(dateString);
208             Calendar calendar = (Calendar) Calendar.getInstance().clone();
209             calendar.setTime(date);
210             if (addUnits.equals("hr") || addUnits.equals("hour")) {
211                 calendar.add(Calendar.HOUR_OF_DAY, addAmount);
212             } else if (addUnits.equals("min") || addUnits.equals("minute")) {
213                 calendar.add(Calendar.MINUTE, addAmount);
214             } else if (addUnits.equals("sec") || addUnits.equals("second")) {
215                 calendar.add(Calendar.SECOND, addAmount);
216             } else if (addUnits.equals("day")) {
217                 calendar.add(Calendar.DAY_OF_YEAR, addAmount);
218             } else if (addUnits.equals("mo") || addUnits.equals("month")) {
219                 calendar.add(Calendar.MONTH, addAmount);
220             } else if (addUnits.equals("yr") || addUnits.equals("year")) {
221                 calendar.add(Calendar.YEAR, addAmount);
222             }
223             
224             String newDateString = new SimpleDateFormat(dateFormat).format(calendar.getTime());
225             
226             // swap in date string
227             string = StringUtils.replace(string, dateFormatString,
228                     newDateString);
229             dateFormatMatcher = dateFormatPattern.matcher(string);
230         }
231 
232         return string;
233     }

269     /**
270      * Replaces String method of format [UTC_TO_TAI(<utc-string format: "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'">)]
271      * with TAI time with format: "yyyy-MM-dd'T'HH:mm:ss.SSS-0000<leapSecs>"
272      */
273     public static String doDynamicUtcToTaiDateReplacement(String string,
274             Metadata metadata) throws ParseException, CommonsException, CasMetadataException {
275         Pattern utcToTaiPattern = Pattern.compile("\\[\\s*UTC_TO_TAI\\s*\\(.{1,}?\\)\\s*\\]");
276         Matcher matcher = utcToTaiPattern.matcher(string);
277         while (matcher.find()) {
278             String utcToTaiString = string.substring(matcher.start(), matcher.end());
279             Matcher argMatcher = Pattern.compile("\\(.*\\)").matcher(utcToTaiString);
280             argMatcher.find();
281             String utcDateString = 
282                 utcToTaiString.substring(argMatcher.start() + 1, argMatcher.end() - 1).trim();
283             utcDateString = doDynamicReplacement(utcDateString, metadata);
284             string = StringUtils.replace(string, utcToTaiString, 
285                     DateUtils.toString(DateUtils.toTai(DateUtils.toCalendar(utcDateString, 
286                             DateUtils.FormatType.UTC_FORMAT))));
287             matcher = utcToTaiPattern.matcher(string);
288         }
289         return string;
290     }

292     /**
293      * Replaces String method of format [DATE_TO_SECS(<date-string>,<DateUtils.FormatType>,<epoch-date format: "yyyy-MM-dd">)]
294      * with seconds between <epoch-date> and <date-string> 
295      */
296     public static String doDynamicDateToSecsReplacement(String string,
297             Metadata metadata) throws CommonsException, ParseException, CasMetadataException {
298         Pattern utcToTaiPattern = Pattern.compile("\\[\\s*DATE_TO_SECS\\s*\\(.{1,}?\\,.{1,}?,.{1,}?\\)\\s*\\]");
299         Matcher matcher = utcToTaiPattern.matcher(string);
300         while (matcher.find()) {
301             String dateToSecsString = string.substring(matcher.start(), matcher.end());
302             Matcher argMatcher = Pattern.compile("\\(.*\\)").matcher(dateToSecsString);
303             argMatcher.find();
304             String argsString = dateToSecsString.substring(argMatcher.start() + 1,
305                     argMatcher.end() - 1);
306             argsString = doDynamicReplacement(argsString, metadata);
307             String[] args = argsString.split(",");
308             String dateString = args[0].trim();
309             String dateType = args[1].trim();
310             String epochString = args[2].trim();
311             Calendar date = DateUtils.toCalendar(dateString, DateUtils.FormatType.valueOf(dateType));
312             Calendar epoch = DateUtils.toLocalCustomFormatCalendar(epochString, "yyyy-MM-dd");
313             String seconds = DateUtils.toString(DateUtils.getTimeInSecs(date, epoch));
314             string = StringUtils.replace(string, dateToSecsString, seconds);
315             matcher = utcToTaiPattern.matcher(string);
316         }
317         return string;
318     }

348     private static VarData readEnvVarName(String origPathStr, int startIdx) {
349         StringBuilder varName = new StringBuilder();
350         int idx = startIdx + 1;
351 
352         do {
353             varName.append(origPathStr.charAt(idx));
354             idx++;
355         } while (origPathStr.charAt(idx) != ']');
356 
357         VarData data = new PathUtils().new VarData();
358         data.setFieldName(varName.toString());
359         data.setEndIdx(idx);
360         return data;
361 
362     }
