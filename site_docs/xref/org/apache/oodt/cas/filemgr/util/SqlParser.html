42  /**
43   * 
44   * @author bfoster
45   * @version $Revision$
46   * 
47   *          <p>
48   *          A fairly robust SQL parser, based on the Shunting yard
49   *          algorithm
50   *          </p>
51   *          
52   *          <p>
53   *          Evaluates SQL like string statements contained in a string. The SQL
54   *          statement should be enclosed within: SQL ({@literal <sql_arguments>}
55   *          ) { {@literal <sql_statement>} . the {@literal <sql_arguments>} can
56   *          be either FORMAT, SORT_BY, or FILTER. Syntax: SQL (FORMAT='<metadata
57   *          formated output>',SORT_BY='<metadata name>', FILTER='<start_time
58   *          metadata element>, <end_time metadata element>, <priority metadata
59   *          element>, <filter type>') { SELECT
60   *          <list-of-comma-segregated-metadata-elements-to-query-on> FROM
61   *          <productTypes-comma-segregated> WHERE <metadata-boolean-expressions>
62   *          } Here is an example SQL statement: SQL (FORMAT='FileLocation/Filename',
63   *          SORT_BY='FileSize',FILTER=<StartDateTime,EndDateTime,DataVersion,
64   *          TakeHighestPriority) { SELECT FileLocation,Filename,FileSize FROM IASI_L1C 
65   *          WHERE ProductionDateTime >= '2007-12-01T00:00:00.000000Z' } This example
66   *          would query the cas-filemgr for metadata values:
67   *          FileLocation,Filename,FileSize for any data file where the
68   *          ProductType == IASI_L1C and the ProductionDateTime >=
69   *          2007-12-01T00:00:00.000000Z. It would then combine the return data
70   *          files metadata via the specified FORMAT. Each data file's metadata
71   *          will be formated to a string representation of (with the actual
72   *          values replaced in the location of the metadata keys):
73   *          FileLocation/Filename. They will be concatenated together, in
74   *          FileSize order.
75   *          </p>
76   */

83      public static ComplexQuery parseSqlQueryMethod(String sqlStringQueryMethod)
84              throws QueryFormulationException {
85          if (!Pattern.matches("((?:SQL)|(?:sql))\\s*(.*)\\s*\\{\\s*SELECT.*FROM.*(?:WHERE.*){0,1}\\}", sqlStringQueryMethod)) {
86              throw new QueryFormulationException("Malformed SQL method");
87          }
88          
89          try {
90              ComplexQuery complexQuery = parseSqlQuery(stripOutSqlDefinition(sqlStringQueryMethod));
91              
92              for (Expression expr : getSqlStatementArgs(sqlStringQueryMethod)) {
93                  if (expr.getKey().toUpperCase().equals("FORMAT")) {
94                      complexQuery.setToStringResultFormat(expr.getValue());
95                  } else if (expr.getKey().toUpperCase().equals("SORT_BY")) {
96                      complexQuery.setSortByMetKey(expr.getValue());
97                  } else if (expr.getKey().toUpperCase().equals("FILTER")) {
98                      complexQuery.setQueryFilter(createFilter(expr));
99                  }
100             }
101             
102             return complexQuery;
103         }catch (Exception e) {
104             LOG.log(Level.SEVERE, e.getMessage());
105             throw new QueryFormulationException("Failed to parse SQL method : " + e.getMessage());
106         }
107     }

109     public static ComplexQuery parseSqlQuery(String sqlStringQuery)
110             throws QueryFormulationException {
111         String[] splitSqlStatement = sqlStringQuery
112                 .split("((?:SELECT)|(?:FROM)|(?:WHERE))");
113         String[] selectValues = (splitSqlStatement[1].trim() + ",").split(",");
114         String[] fromValues = (splitSqlStatement[2].trim() + ",").split(",");
115         ComplexQuery sq = new ComplexQuery();
116         List<String> selectValuesList = Arrays.asList(selectValues);
117         if (!selectValuesList.contains("*")) {
118             sq.setReducedMetadata(Arrays.asList(selectValues));
119         }
120         List<String> fromValuesList = Arrays.asList(fromValues);
121         if (!fromValuesList.contains("*")) {
122             sq.setReducedProductTypeNames(fromValuesList);
123         }
124         
125         if (splitSqlStatement.length > 3) {
126             sq.addCriterion(parseStatement(toPostFix(splitSqlStatement[3]
127                 .trim())));
128         }
129         return sq;
130     }

137     public static String unparseSqlQuery(ComplexQuery complexQuery) throws QueryFormulationException {
138         LinkedList<String> outputArgs = new LinkedList<String>();
139         if (complexQuery.getToStringResultFormat() != null) {
140             outputArgs.add("FORMAT = '" + complexQuery.getToStringResultFormat() + "'");
141         }
142         if (complexQuery.getSortByMetKey() != null) {
143             outputArgs.add("SORT_BY = '" + complexQuery.getSortByMetKey() + "'");
144         }
145         if (complexQuery.getQueryFilter() != null) {
146             String filterString = "FILTER = '"
147                     + complexQuery.getQueryFilter().getStartDateTimeMetKey() + ","
148                     + complexQuery.getQueryFilter().getEndDateTimeMetKey() + ","
149                     + complexQuery.getQueryFilter().getPriorityMetKey() + ","
150                     + complexQuery.getQueryFilter().getFilterAlgor().getClass().getCanonicalName() + ","
151                     + complexQuery.getQueryFilter().getFilterAlgor().getEpsilon();
152             outputArgs.add(filterString + "'");
153         }
154         String sqlQueryString = getInfixCriteriaString(complexQuery.getCriteria());
155         if (sqlQueryString != null && sqlQueryString.startsWith("(") && sqlQueryString.endsWith(")")) {
156             sqlQueryString = sqlQueryString.substring(1, sqlQueryString.length() - 1);
157         }
158         return "SQL ("
159                 + listToString(outputArgs)
160                 + ") { SELECT " + listToString(complexQuery.getReducedMetadata())
161                 + " FROM " + (complexQuery.getReducedProductTypeNames() != null ? listToString(complexQuery.getReducedProductTypeNames()) : "*")
162                 + (sqlQueryString != null ? " WHERE " + sqlQueryString : "") + " }";
163     }

165     public static String getInfixCriteriaString(List<QueryCriteria> criteriaList) throws QueryFormulationException {
166         if (criteriaList.size() > 1) {
167             return getInfixCriteriaString(new BooleanQueryCriteria(criteriaList, BooleanQueryCriteria.AND));
168         } else if (criteriaList.size() == 1) {
169             return getInfixCriteriaString(criteriaList.get(0));
170         } else {
171             return null;
172         }

175     public static String getInfixCriteriaString(QueryCriteria criteria) {
176         StringBuilder returnString = new StringBuilder();
177         if (criteria instanceof BooleanQueryCriteria) {
178             BooleanQueryCriteria bqc = (BooleanQueryCriteria) criteria;
179             List<QueryCriteria> terms = bqc.getTerms();
180             switch(bqc.getOperator()){
181             case 0:
182                 returnString.append("(").append(getInfixCriteriaString(terms.get(0)));
183                 for (int i = 1; i < terms.size(); i++) {
184                     returnString.append(" AND ").append(getInfixCriteriaString(terms.get(i)));
185                 }
186                 returnString.append(")");
187                 break;
188             case 1:
189                 returnString.append("(").append(getInfixCriteriaString(terms.get(0)));
190                 for (int i = 1; i < terms.size(); i++) {
191                     returnString.append(" OR ").append(getInfixCriteriaString(terms.get(i)));
192                 }
193                 returnString.append(")");
194                 break;
195             case 2:
196                 QueryCriteria qc = bqc.getTerms().get(0);
197                 if (qc instanceof TermQueryCriteria) {
198                     TermQueryCriteria tqc = (TermQueryCriteria) qc;
199                     returnString.append(tqc.getElementName()).append(" != '").append(tqc.getValue()).append("'");
200                 }else {
201                     returnString.append("NOT(").append(getInfixCriteriaString(qc)).append(")");
202                 }
203                 break;
204             }
205         }else if (criteria instanceof RangeQueryCriteria) {
206             RangeQueryCriteria rqc = (RangeQueryCriteria) criteria;
207             String opString = rqc.getInclusive() ? "=" : "";
208             if (rqc.getStartValue() != null) {
209                 opString = ">" + opString + " '" + rqc.getStartValue() + "'";
210             }else {
211                 opString = "<" + opString + " '" + rqc.getEndValue() + "'";
212             }
213             returnString.append(rqc.getElementName()).append(" ").append(opString);
214         }else if (criteria instanceof TermQueryCriteria) {
215             TermQueryCriteria tqc = (TermQueryCriteria) criteria;
216             returnString.append(tqc.getElementName()).append(" == '").append(tqc.getValue()).append("'");
217         }
218         return returnString.toString();
219     }

225     private static List<Expression> getSqlStatementArgs(String sqlStringQueryMethod) throws QueryFormulationException {
226         boolean inExpr = false;
227         int startArgs = 0;
228         for (int i = 0; i < sqlStringQueryMethod.length(); i++) {
229             char curChar = sqlStringQueryMethod.charAt(i);
230             switch (curChar) {
231             case '(':
232                 startArgs = i + 1;
233                 break;
234             case ')':
235                 if (!inExpr) {
236                     String[] args = sqlStringQueryMethod.substring(startArgs, i).trim().split("'\\s*,");
237                     LinkedList<Expression> argsList = new LinkedList<Expression>();
238                     for (String arg : args) {
239                         argsList.add(new Expression((arg = arg.trim()).endsWith("'") ? arg : (arg + "'")));
240                     }
241                     return argsList;
242                 } else {
243                     break;
244                 }
245             case '\'':
246                 inExpr = !inExpr;
247                 break;
248             }
249         }
250         throw new QueryFormulationException("Failed to read in args");
251     }

253     private static QueryFilter createFilter(Expression expr) throws InstantiationException, IllegalAccessException, ClassNotFoundException {
254         String[] filterArgs = expr.getValue().split(",");
255         FilterAlgor filterAlgor = (FilterAlgor) Class.forName(filterArgs[3]).newInstance();
256         QueryFilter qf = new QueryFilter(filterArgs[0], filterArgs[1], filterArgs[2], filterAlgor);
257         filterAlgor.setEpsilon(Integer.parseInt(filterArgs[4]));
258         return qf;
259     }

261     /**
262      * Uses "Shunting yard algorithm" (see:
263      * http://en.wikipedia.org/wiki/Shunting_yard_algorithm)
264      */
265     private static LinkedList<String> toPostFix(String statement) {
266         LinkedList<String> postFix = new LinkedList<String>();
267         Stack<String> stack = new Stack<String>();
268 
269         for (int i = 0; i < statement.length(); i++) {
270             char curChar = statement.charAt(i);
271             switch (curChar) {
272             case '(':
273                 stack.push("(");
274                 break;
275             case ')':
276                 String value;
277                 while (!(value = stack.pop()).equals("(")) {
278                     postFix.add(value);
279                 }
280                 if (stack.peek().equals("NOT")) {
281                     postFix.add(stack.pop());
282                 }
283                 break;
284             case ' ':
285                 break;
286             default:
287                 if (statement.substring(i, i + 3).equals("AND")) {
288                     while (!stack.isEmpty()
289                             && (stack.peek().equals("AND"))) {
290                         postFix.add(stack.pop());
291                     }
292                     stack.push("AND");
293                     i += 2;
294                 } else if (statement.substring(i, i + 2).equals("OR")) {
295                     while (!stack.isEmpty()
296                             && (stack.peek().equals("AND") || stack.peek()
297                                     .equals("OR"))) {
298                         postFix.add(stack.pop());
299                     }
300                     stack.push("OR");
301                     i += 1;
302                 } else if (statement.substring(i, i + 3).equals("NOT")) {
303                     stack.push("NOT");
304                     i += 2;
305                 } else {
306                     int endIndex = statement.indexOf('\'', statement.indexOf(
307                             '\'', i) + 1) + 1;
308                     postFix.add(statement.substring(i, endIndex));
309                     i = endIndex - 1;
310                 }
311             }
312         }
313 
314         while (!stack.isEmpty()) {
315             postFix.add(stack.pop());
316         }
317 
318         return postFix;
319     }

321     private static QueryCriteria parseStatement(LinkedList<String> postFixStatement)
322             throws QueryFormulationException {
323         Stack<QueryCriteria> stack = new Stack<QueryCriteria>();
324         for (String item : postFixStatement) {
325             if (item.equals("AND")) {
326                 BooleanQueryCriteria bQC = new BooleanQueryCriteria();
327                 bQC.addTerm(stack.pop());
328                 bQC.addTerm(stack.pop());
329                 stack.push(bQC);
330             } else if (item.equals("OR")) {
331                 BooleanQueryCriteria bQC = new BooleanQueryCriteria();
332                 bQC.setOperator(BooleanQueryCriteria.OR);
333                 bQC.addTerm(stack.pop());
334                 bQC.addTerm(stack.pop());
335                 stack.push(bQC);
336             } else if (item.equals("NOT")) {
337                 BooleanQueryCriteria bQC = new BooleanQueryCriteria();
338                 bQC.setOperator(BooleanQueryCriteria.NOT);
339                 bQC.addTerm(stack.pop());
340                 stack.push(bQC);
341             } else {
342                 stack.push(new Expression(item).convertToQueryCriteria());
343             }
344         }
345         return stack.pop();
346     }

348     private static String listToString(List<String> list) {
349         StringBuilder arrayString = new StringBuilder();
350         if (list.size() > 0) {
351             arrayString.append(list.get(0));
352             for (int i = 1; i < list.size(); i++) {
353                 arrayString.append(",").append(list.get(i));
354             }
355         }
356         return arrayString.toString();
357     }

360     private static class Expression {
361 
362         public static final short GREATER_THAN = 12;
363 
364         public static final short LESS_THAN = 3;
365 
366         public static final short EQUAL_TO = 9;
367 
368         public static final short NOT_EQUAL_TO = 15;
369 
370         public static final short GREATER_THAN_OR_EQUAL_TO = 13;
371 
372         public static final short LESS_THAN_OR_EQUAL_TO = 11;
373 
374         public static final short NOT = 6;
375 
376         private String[] stringValues = new String[] { "`", "`", "`", "<", "`",
377                 "`", "!", "`", "`", "=", "`", "<=", ">", ">=", "`", "!=" };
378 
379         private String expression;
380 
381         private String key;
382 
383         private String val;
384 
385         private int op;
386 
387         public Expression(String expression) {
388             this.parseExpression(this.expression = expression);
389         }
390 
391         public Expression(String key, int op, String val) {
392             this.key = key.trim();
393             this.op = op;
394             this.val = this.removeTickBounds(val.trim());
395         }
396 
397         private void parseExpression(String expression) {
398             Matcher matcher = Pattern.compile("((?:>=)|(?:<=)|(?:==)|(?:!=)|(?:=)|(?:>)|(?:<))").matcher(expression);
399             matcher.find();
400             this.key = expression.substring(0, matcher.start()).trim();
401             this.val = this.removeTickBounds(expression.substring(matcher.end()).trim());
402             String opString = matcher.group();
403             for (char c : opString.toCharArray()) {
404                 this.op = this.op | this.getShortValueForOp(c);
405             }
406         }
407 
408         private String removeTickBounds(String value) {
409             if (value.startsWith("'") && value.endsWith("'")) {
410                 value = value.substring(1, value.length() - 1);
411             }
412             return value;
413         }
414 
415         private int getShortValueForOp(char op) {
416             switch (op) {
417             case '>':
418                 return GREATER_THAN;
419             case '<':
420                 return LESS_THAN;
421             case '=':
422                 return EQUAL_TO;
423             case '!':
424                 return NOT;
425             default:
426                 return 0;
427             }
428         }
429 
430         public QueryCriteria convertToQueryCriteria()
431                 throws QueryFormulationException {
432             switch (this.op) {
433             case GREATER_THAN:
434                 return new RangeQueryCriteria(this.key, this.val, null, false);
435             case LESS_THAN:
436                 return new RangeQueryCriteria(this.key, null, this.val, false);
437             case EQUAL_TO:
438                 return new TermQueryCriteria(this.key, this.val);
439             case NOT_EQUAL_TO:
440                 BooleanQueryCriteria notEqBQC = new BooleanQueryCriteria();
441                 notEqBQC.setOperator(BooleanQueryCriteria.NOT);
442                 notEqBQC.addTerm(new TermQueryCriteria(this.key, this.val));
443                 return notEqBQC;
444             case GREATER_THAN_OR_EQUAL_TO:
445                 return new RangeQueryCriteria(this.key, this.val, null, true);
446             case LESS_THAN_OR_EQUAL_TO:
447                 return new RangeQueryCriteria(this.key, null, this.val, true);
448             }
449             throw new QueryFormulationException(
450                     "Was not able to form query . . . probably an invalid operator -- "
451                             + this.toString());
452         }
453 
454         public String getKey() {
455             return this.key;
456         }
457 
458         public String getValue() {
459             return this.val;
460         }
461 
462         public int getOp() {
463             return this.op;
464         }
465 
466         public String getExpression() {
467             return this.expression;
468         }
469 
470         public String toString() {
471             return this.key + " " + this.stringValues[this.op] + " " + this.val;
472         }
