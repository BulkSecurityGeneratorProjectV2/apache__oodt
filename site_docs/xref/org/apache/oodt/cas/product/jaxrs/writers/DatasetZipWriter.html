20  import java.io.File;
21  import java.io.FileInputStream;
22  import java.io.IOException;
23  import java.io.OutputStream;
24  import java.lang.annotation.Annotation;
25  import java.lang.reflect.Type;
26  
27  import javax.ws.rs.Produces;
28  import javax.ws.rs.WebApplicationException;
29  import javax.ws.rs.core.MediaType;
30  import javax.ws.rs.core.MultivaluedMap;
31  import javax.ws.rs.ext.MessageBodyWriter;
32  import javax.ws.rs.ext.Provider;
33  
34  import org.apache.commons.io.IOUtils;
35  import org.apache.oodt.cas.product.jaxrs.resources.DatasetResource;
36  
37  /**
38   * A {@link Provider} that writes {@link DatasetResource dataset resources} to
39   * output streams for HTTP responses with "application/zip" content-type.
40   * @author rlaidlaw
41   * @version $Revision$
42   */
43  @Provider
44  @Produces("application/zip")
45  public class DatasetZipWriter implements MessageBodyWriter<DatasetResource>
46  {
47    @Override
48    public long getSize(DatasetResource resource, Class<?> type, Type genericType,
49      Annotation[] annotations, MediaType mediaType)
50    {
51      return -1;
52    }
53  
54  
55  
56    @Override
57    public boolean isWriteable(Class<?> type, Type genericType,
58      Annotation[] annotations, MediaType mediaType)
59    {
60      return true;
61    }
62  
63  
64  
65    @Override
66    public void writeTo(DatasetResource resource, Class<?> type, Type genericType,
67      Annotation[] annotations, MediaType mediaType,
68      MultivaluedMap<String, Object> httpHeaders, OutputStream entityStream)
69        throws IOException, WebApplicationException
70    {
71      // Create a zip file for the dataset resource.
72      File zipFile = new DatasetZipper().createZipFile(resource);
73  
74      // Add the zip file to the HTTP response entity stream.
75      httpHeaders.add("Content-Type", "application/zip");
76      httpHeaders.add("Content-Disposition",
77        "attachment; filename=\"" + zipFile.getName() + "\"");
78      FileInputStream fis = new FileInputStream(zipFile);
79      IOUtils.copy(fis, entityStream);
80      fis.close();
81    }
82  }
