41  /**
42   * @author cecilia
43   * @version $Revision$
44   * 
45   * <p>
46   * A tool that creates a job and send it to the Resource Manager. The list of
47   * jobs created has a 1-to-1 correspondence to the list of directory names
48   * </p>.
49   */
50  public final class RunDirJobSubmitter {
51  
52      /* our log stream */
53      private static final Logger LOG = Logger.getLogger(RunDirJobSubmitter.class
54              .getName());
55  
56      /* our res mgr client */
57      private XmlRpcResourceManagerClient client = null;
58  
59      public RunDirJobSubmitter(URL rUrl) {
60          client = new XmlRpcResourceManagerClient(rUrl);
61      }
62  
63      public void submitRunDirJobFile(String jobFname, String inputFname)
64              throws JobExecutionException {
65  
66          // -------------------------------------------------------------
67          // check validity of given job filename and input filename
68          // -------------------------------------------------------------
69  
70          File jobFile = new File(jobFname);
71          if (!jobFile.exists()) {
72              // file doesn't exist
73              throw new JobExecutionException("RunDirJobSubmitter: input file "
74                      + jobFname + " does not exist.");
75          } else if (!jobFile.isFile()) {
76              // file is a directory
77              throw new JobExecutionException("RunDirJobSubmitter: input file "
78                      + jobFname + " is not a file.");
79          }
80  
81          File f = new File(inputFname);
82          if (!f.exists()) {
83              // file doesn't exist
84              throw new JobExecutionException("RunDirJobSubmitter: input file "
85                      + inputFname + " does not exist.");
86          } else if (!f.isFile()) {
87              // file is a directory
88              throw new JobExecutionException("RunDirJobSubmitter: input file "
89                      + inputFname + " is not a file.");
90          }
91  
92          // ----------------------------------------------------------------
93          // create a default JobSpec
94          // ----------------------------------------------------------------
95          JobSpec spec = JobBuilder.buildJobSpec(jobFile.getAbsolutePath());
96          Job job = spec.getJob();
97          NameValueJobInput jobInput = (NameValueJobInput) spec.getIn();
98  
99          // ----------------------------------------------------------------
100         // open the file to read. traverse through the list of directories
101         // name given & override the default Job's runDirName value with the
102         // directory name. then submit the Job.
103         // ----------------------------------------------------------------
104 
105         try {
106             BufferedReader in = new BufferedReader(new FileReader(inputFname));
107             if (!in.ready()) {
108                 throw new IOException();
109             }
110 
111             String line;
112             String jobId;
113             while ((line = in.readLine()) != null) {
114 
115                 // overwrite the runDirName
116                 jobInput.setNameValuePair("runDirName", line);
117 
118                 jobId = submitJob(job, jobInput);
119                 LOG.log(Level.INFO, "Job Submitted: id: [" + jobId + "]");
120             }
121 
122             in.close();
123         } catch (IOException e) {
124             throw new JobExecutionException("RunDirJobSubmitter: " + e);
125         }
126 
127     }
128 
129     public static void main(String[] args) throws JobExecutionException, MalformedURLException {
130         String resMgrUrlStr = null;
131         String jobFileName = null;
132         String runDirFileName = null;
133 
134         String usage = "RunDirJobSubmitter --rUrl <resource mgr url> "
135                 + "--jobFile <input job file> "
136                 + "--runDirFile <input running directories file> \n";
137 
138         for (int i = 0; i < args.length; i++) {
139             if (args[i].equals("--rUrl")) {
140                 resMgrUrlStr = args[++i];
141             } else if (args[i].equals("--jobFile")) {
142                 jobFileName = args[++i];
143             } else if (args[i].equals("--runDirFile")) {
144                 runDirFileName = args[++i];
145             }
146         }
147 
148         if (resMgrUrlStr == null || jobFileName == null
149                 || runDirFileName == null) {
150             System.err.println(usage);
151             System.exit(1);
152         }
153 
154         RunDirJobSubmitter submitter = new RunDirJobSubmitter(new URL(
155                 resMgrUrlStr));
156         submitter.submitRunDirJobFile(jobFileName, runDirFileName);
157     }
158 
159     private String submitJob(Job job, JobInput jobInput)
160             throws JobExecutionException {
161         return client.submitJob(job, jobInput);
162     }
