35  /**
36   * A JAX-RS resource representing a {@link Product}.
37   * @author rlaidlaw
38   * @version $Revision$
39   */
40  @XmlRootElement(name = "product")
41  @XmlType(propOrder = {"productId", "productName", "productStructure",
42    "productTypeName", "transferStatus", "metadataResource",
43    "referenceResources"})
44  @XmlAccessorType(XmlAccessType.NONE)
45  public class ProductResource
46  {
47    private Product product;
48    private MetadataResource metadataResource;
49    private List<ReferenceResource> referenceResources =
50      new ArrayList<ReferenceResource>();
51  
52    // The file manager's working directory for this resource, used for example
53    // when creating zip archives.
54    private File workingDir;
55  
56  
57  
58    /**
59     * Default constructor required by JAXB.
60     */
61    public ProductResource()
62    {
63    }
64  
65  
66  
67    /**
68     * Constructor that sets the product, metadata, references and
69     * working directory for the resource.
70     * @param product the product associated with the resource
71     * @param metadata the metadata associated with the resource
72     * @param references the references associated with the resource
73     * @param workingDir the working directory for creating temporary files to
74     * attach to responses
75     */
76    public ProductResource(Product product, Metadata metadata,
77      List<Reference> references, File workingDir)
78    {
79      this.product = product;
80      this.metadataResource = new MetadataResource(metadata);
81      for (int index = 0; index < references.size(); index++)
82      {
83        referenceResources.add(new ReferenceResource(product.getProductId(),
84          index, references.get(index), workingDir));
85      }
86      this.workingDir = workingDir;
87    }
88  
89

91    /**
92     * Gets the working directory for the product.
93     * @return the working directory
94     */
95    public File getWorkingDir()
96    {
97      return workingDir;
98    }
