45      public static void chmodFile(File file, String perms){
46          Chmoder chmod = new Chmoder();
47          chmod.setFile(file);
48          chmod.setPerm(perms);
49          chmod.execute();
50      }
51  
52      public static void deleteAllFilesAndDir(File startDir) {
53          Deleter delete = new Deleter();
54          delete.setDir(startDir);
55          delete.setIncludeEmptyDirs(true);
56          delete.setVerbose(true);
57          delete.setFollowSymlinks(true);
58          delete.execute();
59      }
60  
61      public static void untarFile(File tarFile, File destDir) throws IOException {
62          Untarer untar = new Untarer();
63          Untar.UntarCompressionMethod compMethod = new UntarCompressionMethod();
64          compMethod.setValue("gzip");
65          untar.setCompression(compMethod);
66          untar.setDest(destDir);
67          untar.setSrc(tarFile);
68          untar.execute();
69  
70          // now that the work is done, use file utils to
71          // move everything out of destDir/tarFileNoExt
72          // into destDir
73          // then delete destDir/tarFileNoExt
74          String wrongInstallDir = destDir.getCanonicalPath() + File.separator
75                  + getFileNameNoExt(tarFile.getName());
76          FileUtils.copyDirectory(new File(wrongInstallDir), destDir, true);
77  
78          // grrr java IO, love it
79          // because it sucks, we have to CHMOD everything in destDir/bin
80          File binDir = new File(destDir.getCanonicalPath() + File.separator
81                  + "bin");
82          Chmoder chmod = new Chmoder();
83          chmod.setDir(binDir);
84          chmod.setPerm("ugo+rx");
85          chmod.setIncludes("*");
86          chmod.execute();
87  
88          deleteAllFilesAndDir(new File(wrongInstallDir));
89  
90      }
91  
92      final static class Untarer extends Untar {
93          public Untarer() {
94              project = new Project();
95              project.init();
96              taskType = "untar";
97              taskName = "untar";
98              target = new Target();
99          }
100 
101     }

103     final static class Deleter extends Delete {
104         public Deleter() {
105             project = new Project();
106             project.init();
107             taskType = "delete";
108             taskName = "delete";
109             target = new Target();
110         }
111 
112     }

114     final static class Chmoder extends Chmod {
115         public Chmoder() {
116             project = new Project();
117             project.init();
118             taskType = "chmod";
119             taskName = "chmod";
120             target = new Target();
121         }
122     }

124     private static String getFileNameNoExt(String filename) {
125         return filename.substring(0, filename.indexOf(".tar"));
