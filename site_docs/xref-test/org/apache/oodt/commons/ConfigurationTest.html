24  /** Unit test the {@link Configuration} class.
25   *
26   * @author Kelly
27   */ 
28  public class ConfigurationTest extends TestCase {
29  	/** Construct the test case for the {@link Configuration} class. */
30  	public ConfigurationTest(String name) {
31  		super(name);
32  	}
33  
34  	protected void setUp() throws Exception {
35  		// Create a temporary test configuration file.
36  		tmpFile = File.createTempFile("conf", ".xml");
37  		BufferedOutputStream out = new BufferedOutputStream(new FileOutputStream(tmpFile));
38  		byte[] doc = TEST_DOC.getBytes();
39  		out.write(doc, 0, doc.length);
40  		out.close();
41  		oldValue = System.getProperty(Configuration.ENTITY_DIRS_PROP);
42  		System.setProperty(Configuration.ENTITY_DIRS_PROP, "/dir/1,/dir/2");
43  	}
44  
45  	protected void tearDown() throws Exception {
46  		// Nuke the temporary test configuration file.
47  		tmpFile.delete();
48  		if (oldValue == null)
49  			System.getProperties().remove(Configuration.ENTITY_DIRS_PROP);
50  		else
51  			System.setProperty(Configuration.ENTITY_DIRS_PROP, oldValue);
52  	}
53  
54  	/** Test the various property methods. */
55  	public void testConfiguration() throws IOException, SAXException {
56  		Configuration c = new Configuration(tmpFile.toURL());
57  		Properties props = new Properties();
58  		props.setProperty("globalKey1", "preset-value");
59  		c.mergeProperties(props);
60  		assertEquals("preset-value", props.getProperty("globalKey1"));
61  		assertEquals("globalValue2", props.getProperty("globalKey2"));
62  		assertEquals("http://testhost.test.domain:12345", c.getWebServerBaseURL());
63  		assertEquals("http://testhost.test.domain:12345/ns.ior", c.getNameServerURL());
64  		assertEquals(42, c.getNameServerStateFrequency());
65  		assertEquals(4, c.getEntityRefDirs().size());
66  		assertTrue(c.getEntityRefDirs().contains("/dir/one"));
67  		assertTrue(c.getEntityRefDirs().contains("/dir/two"));
68  		assertTrue(c.getEntityRefDirs().contains("/dir/1"));
69  		assertTrue(c.getEntityRefDirs().contains("/dir/2"));
70  		assertEquals("/dir/1,/dir/2,/dir/one,/dir/two", System.getProperty(Configuration.ENTITY_DIRS_PROP));
71  		Collection servers = c.getExecServerConfigs();
72  		assertEquals(2, servers.size());
73  	  for (Object server : servers) {
74  		ExecServerConfig esc = (ExecServerConfig) server;
75  		if (esc.getClassName().equals("test.Class1")) {
76  		  assertEquals("Name1", esc.getObjectKey());
77  		  assertEquals(1, esc.getProperties().size());
78  		} else if (esc.getClassName().equals("test.Class2")) {
79  		  assertEquals("Name2", esc.getObjectKey());
80  		  assertEquals(3, esc.getProperties().size());
81  		  for (Map.Entry<Object, Object> objectObjectEntry : esc.getProperties().entrySet()) {
82  			Map.Entry entry = (Map.Entry) objectObjectEntry;
83  			if (entry.getKey().equals("localKey1")) {
84  			  assertEquals("localKey2", entry.getValue());
85  			} else if (entry.getKey().equals("globalKey2")) {
86  			  assertEquals("local-override", entry.getValue());
87  			} else if (entry.getKey().equals("org.apache.oodt.commons.Configuration.url")) {
88  			  ; // This one's OK.
89  			} else {
90  			  fail("Unknown local property \"" + entry.getKey() + "\" in exec server");
91  			}
92  		  }
93  		} else {
94  		  fail("Unknown ExecServerConfig \"" + esc.getClassName() + "\" in servers from Configuration");
95  		}
96  	  }
97  	}
