47  public class EventToWorkflowViewer extends Panel {
48  
49    private static final long serialVersionUID = -5120511045763328237L;
50  
51    private final WorkflowMgrConn wm;
52  
53    public EventToWorkflowViewer(String id, String workflowUrlStr, final Class<? extends WebPage> viewerPage) {
54      super(id);
55      this.wm = new WorkflowMgrConn(workflowUrlStr);
56      WebMarkupContainer wTable = new WebMarkupContainer("wtable");
57      wTable.setVisible(false);
58      PropertyModel<List<Workflow>> workflowsModel = new PropertyModel<List<Workflow>>(this, "workflows");
59      ListView<Workflow> workflowView = new ListView<Workflow>("workflow_list", workflowsModel) {
60        private static final long serialVersionUID = 5894604290395257941L;
61  
62        @Override
63        protected void populateItem(ListItem<Workflow> item) {
64          Link<String> wLink = new Link<String>("workflow_link", new Model(item.getModelObject().getId())){
65            
66             /* (non-Javadoc)
67             * @see org.apache.wicket.markup.html.link.Link#onClick()
68             */
69            @Override
70            public void onClick() {
71              PageParameters params = new PageParameters();
72              params.add("id", getModelObject());
73              setResponsePage(viewerPage, params);
74            }
75          };
76          
77          wLink.add(new Label("workflow_name", item.getModelObject().getName()));
78          item.add(wLink);
79        }
80      };
81  
82      EventWorkflowForm form = 
83        new EventWorkflowForm("event_workflow_frm", workflowsModel, wTable);
84  
85      wTable.add(workflowView);
86      add(wTable);
87      add(form);
88    }
89    
90    class EventWorkflowForm extends Form<List<Workflow>>{
91  
92      private static final long serialVersionUID = -3209079518977783358L;
93      
94      private String selectedEvent;
95      
96      private WebMarkupContainer wTable;
97  
98      public EventWorkflowForm(String id, IModel<List<Workflow>> model, WebMarkupContainer wTable) {
99        super(id, model);
100       add(new DropDownChoice<String>("event_list", new PropertyModel(this, "selectedEvent"), 
101           wm.safeGetRegisteredEvents()));
102       this.wTable = wTable;
103       
104     }
105 
106     /* (non-Javadoc)
107      * @see org.apache.wicket.markup.html.form.Form#onSubmit()
108      */
109     @Override
110     protected void onSubmit() {
111        List<Workflow> workflows = wm.safeGetWorkflowsByEvent(this.selectedEvent);
112        setDefaultModelObject(workflows);
113        if(workflows != null && workflows.size() > 0){
114          this.wTable.setVisible(true);
115          this.wTable.add(new Label("selectedEvent", this.selectedEvent));
116        }
117     }
118   
119     public String getSelectedEvent(){
120       return this.selectedEvent;
121     }

123   }
