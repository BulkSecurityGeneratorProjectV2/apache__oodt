43  /**
44   * A {@link Provider} that writes {@link ReferenceResource reference resources}
45   * to output streams for HTTP responses with "application/rss+xml" content-type.
46   * @author rlaidlaw
47   * @version $Revision$
48   */
49  @Provider
50  @Produces("application/rss+xml")
51  public class ReferenceRssWriter extends RssWriter
52    implements MessageBodyWriter<ReferenceResource>
53  {
54    private static final Logger LOGGER = Logger.getLogger(ReferenceRssWriter.class
55      .getName());
56  
57  
58  
59    @Override
60    public long getSize(ReferenceResource resource, Class<?> type,
61      Type genericType, Annotation[] annotations, MediaType mediaType)
62    {
63      return -1;
64    }
65  
66  
67  
68    @Override
69    public boolean isWriteable(Class<?> type, Type genericType,
70      Annotation[] annotations, MediaType mediaType)
71    {
72      return true;
73    }
74  
75  
76  
77    @Override
78    public void writeTo(ReferenceResource resource, Class<?> type,
79      Type genericType, Annotation[] annotations, MediaType mediaType,
80      MultivaluedMap<String, Object> httpHeaders, OutputStream entityStream)
81        throws IOException, WebApplicationException
82    {
83      try
84      {
85        String productId = resource.getProductId();
86        int refIndex = resource.getRefIndex();
87  
88        String base = getBaseUri() + "reference";
89        String query = "?productId=" + productId + "&refIndex=" + refIndex;
90        String idLink = base + query;
91        String fileLink = base + ".file" + query;
92        String selfLink = base + ".rss" + query;
93        String currentDate = dateFormatter.format(new Date());
94  
95        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
96        factory.setNamespaceAware(true);
97        Document doc = factory.newDocumentBuilder().newDocument();
98        Element rss = XMLUtils.addNode(doc, doc, "rss");
99        XMLUtils.addAttribute(doc, rss, "version", "2.0");
100       XMLUtils.addAttribute(doc, rss, "xmlns:atom",
101         "http://www.w3.org/2005/Atom");
102 
103       Element channel = XMLUtils.addNode(doc, rss, "channel");
104       XMLUtils.addNode(doc, channel, "title", "Reference(" + refIndex +
105         ") from Product " + productId);
106 
107       Element atomLink = XMLUtils.addNode(doc, channel, "atom:link");
108       XMLUtils.addAttribute(doc, atomLink, "href", selfLink);
109       XMLUtils.addAttribute(doc, atomLink, "rel", "self");
110       XMLUtils.addAttribute(doc, atomLink, "type", "application/rss+xml");
111 
112       XMLUtils.addNode(doc, channel, "link", idLink);
113       XMLUtils.addNode(doc, channel, "description", "reference");
114 
115       XMLUtils.addNode(doc, channel, "language", LANGUAGE);
116       XMLUtils.addNode(doc, channel, "copyright", COPYRIGHT);
117       XMLUtils.addNode(doc, channel, "pubDate", currentDate);
118       XMLUtils.addNode(doc, channel, "category", "Reference");
119       XMLUtils.addNode(doc, channel, "generator", GENERATOR);
120       XMLUtils.addNode(doc, channel, "lastBuildDate", currentDate);
121 
122       Element item = XMLUtils.addNode(doc, channel, "item");
123       XMLUtils.addNode(doc, item, "title", "reference (" + refIndex +
124         ") for product " + productId);
125       XMLUtils.addNode(doc, item, "link", fileLink);
126       XMLUtils.addNode(doc, item, "description",
127         "A file manager reference entity");
128       XMLUtils.addNode(doc, item, "guid", idLink);
129 
130       XMLUtils.writeXmlToStream(doc, entityStream);
131     }
132     catch (ParserConfigurationException e)
133     {
134       String message = "Unable to create RSS XML document for RSS response.";
135       LOGGER.log(Level.WARNING, message, e);
136       throw new InternalServerErrorException(message);
137     }
138   }
