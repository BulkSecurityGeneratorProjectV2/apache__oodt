19  
20  //JDK imports
21  import org.apache.oodt.cas.workflow.engine.processor.WorkflowProcessor;
22  
23  import java.util.Collections;
24  import java.util.Comparator;
25  import java.util.List;
26  import java.util.logging.Level;
27  import java.util.logging.Logger;
28  
29  //OODT imports
30  
31  /**
32   * 
33   * Sorts based on highest {@link Priority} for a {@link WorkflowProcessor} using
34   * a particular {@link #boostAmount} provided by a calling party. The
35   * {@link #boostAmount} is computed based on a function that allows
36   * {@link #boostAmount} to grow over time depending on
37   * {@link #secondsBetweenBoosts} and a maximum {@link #boostCap}.
38   * 
39   * @author bfoster
40   * @author mattmann
41   * @version $Revision$
42   * 
43   */
44  public class HighestFIFOPrioritySorter implements PrioritySorter {
45  
46    public static final double DOUBLE = 1000.0;
47    private int secondsBetweenBoosts;
48    private double boostAmount;
49    private double boostCap;
50    private static final Logger LOG = Logger
51        .getLogger(HighestFIFOPrioritySorter.class.getName());
52  
53    public HighestFIFOPrioritySorter(int secondsBetweenBoosts,
54        double boostAmount, double boostCap) {
55      this.secondsBetweenBoosts = secondsBetweenBoosts;
56      this.boostAmount = boostAmount;
57      this.boostCap = boostCap;
58    }
59  
60    /*
61     * (non-Javadoc)
62     * 
63     * @see
64     * org.apache.oodt.cas.workflow.structs.PrioritySorter#sort(java.util.List)
65     */
66    @Override
67    public synchronized void sort(List<WorkflowProcessor> candidates) {
68  
69      Collections.sort(candidates, new Comparator<WorkflowProcessor>() {
70        public int compare(WorkflowProcessor o1, WorkflowProcessor o2) {
71          return calculatePriority(o2).compareTo(calculatePriority(o1));
72        }
73      });
74    }
75  
76    private Double calculatePriority(WorkflowProcessor processorStub) {
77      double aliveTime;
78  
79      try {
80        aliveTime = (double) (System.currentTimeMillis() - processorStub
81            .getWorkflowInstance().getCreationDate().getTime());
82      } catch (Exception e) {
83        LOG.log(Level.SEVERE, e.getMessage());
84        LOG.log(Level.WARNING,
85            "Unable to compute aliveTime for computing FIFO priority: Reason: ["
86                + e.getMessage() + "]");
87        aliveTime = 0.0;
88      }
89  
90      double boostPercentage = aliveTime / DOUBLE
91          / (double) this.secondsBetweenBoosts;
92      return Math.max(
93          processorStub.getWorkflowInstance().getPriority().getValue(),
94          Math.min(
95              this.boostCap,
96              processorStub.getWorkflowInstance().getPriority()
97                           .getValue()
98              + (boostPercentage * this.boostAmount)));
99    }
