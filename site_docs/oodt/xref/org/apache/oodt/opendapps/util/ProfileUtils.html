105   public static ResourceAttributes getResourceAttributes(OpendapConfig conf,
106       String opendapUrl, DConnect dConn, Metadata datasetMet) {
107     ResourceAttributes resAttr = new ResourceAttributes();
108     for (ConstantSpec spec : conf.getConstSpecs()) {
109       if (spec.getType().equals(RES_ATTR_SPEC_TYPE)) {
110         try {        	
111         	      	        	
112         	// first process expanded '[@...]' instructions
113         	List<String> values = multipleEnvVariablesReplacement(spec.getValue(), datasetMet);
114         	
115         	// then process standard '[...]' instructions
116         	for (String value : values) {
117           	String _value = PathUtils.replaceEnvVariables(value, datasetMet, true);          		        		
118         		if (StringUtils.hasText(_value)) {
119         			setResourceAttributes(resAttr, spec.getName(), _value);
120         		}
121         		
122         	}
123  
124         } catch (Exception e) {
125           LOG.log(Level.SEVERE, e.getMessage());
126           LOG.log(Level.WARNING, "Error setting field: [" + spec.getName()
127               + "] in resource attributes: Message: " + e.getMessage());
128         }
129       }
130     }
131 
132 
133     return resAttr;
134   }

136   /**
137    * Utility method to process environment replacement instructions of the form '[@key]'
138    * resulting in as many output values as there are values for the environment variable 'key'.
139    * Note that currently only one such pattern '[@key']' can be processed.
140    * 
141    * @param value
142    * @param metadata
143    * @return
144    */
145   private static List<String> multipleEnvVariablesReplacement(String value, Metadata metadata) {
146   	
147   	List<String> newValues = new ArrayList<String>();
148 
149   	// regexp matching found > replace values
150   	int start = value.indexOf("[@");
151   	if (start>=0) {
152   		
153   			int end = value.indexOf("]",start+2);
154   			// remove '[@',']' to obtain environment variable key
155   			String envKey = value.substring(start+2,end);
156     		List<String> envValues = metadata.getAllMetadata(envKey);
157     		if (envValues!=null) {
158       		for (String envValue : envValues) {
159       			// create new metadata value for this environment replacement
160       			String newValue = value.replaceAll("\\[@"+envKey+"\\]", envValue);
161       			newValues.add(newValue);
162       		}
163   		}
164   		
165     // regexp matching not found > return original value
166   	} else {
167   		newValues.add(value);
168   	}
169   	
170   	return newValues;
171   	
172   }

187   public static Map<String, ProfileElement> getProfileElements(
188       OpendapConfig conf, DConnect dConn, Metadata datasetMet, Profile profile) throws Exception {
189   	
190     OpendapProfileElementExtractor pe = new OpendapProfileElementExtractor(conf);
191     Map<String, ProfileElement> profElements = new ConcurrentHashMap<String, ProfileElement>();
192 
193     // extracts all variables defined in DDS
194     try {
195       	
196     		DDS dds = dConn.getDDS();
197       	      	
198       	// loop over all variables found
199       	Enumeration variables = dds.getVariables();
200       	while (variables.hasMoreElements()) {
201       		
202       		BaseType variable = (BaseType)variables.nextElement();
203       		String varName = variable.getName();
204       		if (variable instanceof DArray) {
205       			LOG.log(Level.FINE, "Extracting Darray variable: "+varName);
206       		} else if (variable instanceof DGrid) {
207       			LOG.log(Level.FINE, "Extracting Dgrid variable: "+varName);
208       		}     		
209 
210       		RewriteSpec spec = getProfileElementSpec(varName, conf);
211       		if (spec!=null) {
212       			// use configuration to set variable re-name and type
213       			String peName = spec.getRename() != null && !spec.getRename().equals("") ? spec.getRename() : spec.getOrigName();
214             if (spec.getElementType().equals(RANGED_ELEMENT_TYPE)) {
215               profElements.put(peName, pe.extractRangedProfileElement(peName, spec.getOrigName(), profile, dConn.getDAS()));
216             } else if (spec.getElementType().equals(ENUM_ELEMENT_TYPE)) {
217               profElements.put(peName, pe.extractEnumeratedProfileElement(peName, spec.getOrigName(), profile, dConn.getDAS()));
218             }
219       		} else {
220       			// if not explicitly configured, assume variable if of RANGED_ELEMENT_TYPE
221       			profElements.put(varName, pe.extractRangedProfileElement(varName, varName, profile, dConn.getDAS()));
222       		}
223       		
224       	}
225       	
226     } catch(Exception e) {
227       LOG.log(Level.SEVERE, e.getMessage());
228       LOG.log(Level.WARNING, "Error extracting metadata from DDS ("+dConn.URL()+") :"  +e.getMessage());
229       // rethrow the exception so that this dataset is not harvested
230       throw e;
231     }
232 
233     // add profile elements from <datasetMetadata> specification
234     if (datasetMet != null) {
235       for (DatasetMetElem datasetSpec : conf.getDatasetMetSpecs()) {
236       	// retrieve values from metadata container
237         List<String> values = datasetMet.getAllMetadata(datasetSpec.getValue());
238       	addValuesToEnumeratedProfileElement(datasetSpec.getProfileElementName(), values, profile, profElements);
239       }
240     }
241     
242     // add profile elements from <constants> specification
243     for (ConstantSpec spec : conf.getConstSpecs()) {
244       if (spec.getType().equals(PROF_ELEM_SPEC_TYPE)) {
245       	// retrieve value from XML configuration file, replace with value from metadata container if required,
246       	// split according to delimiter
247         String replaceVal = PathUtils.replaceEnvVariables(spec.getValue(), datasetMet);
248         List<String> values = Arrays.asList(replaceVal.split(PathUtils.DELIMITER));
249       	addValuesToEnumeratedProfileElement(spec.getName(), values, profile, profElements);
250       }
251     }
252 
253     return profElements;
254 
255   }

257   /**
258    * Method to add one or more values to an EnumeratedProfileElement, creating the element if not existing already.
259    * The supplied values are added only if valid.
260    */
261   private static void addValuesToEnumeratedProfileElement(final String name, final List<String> values, Profile profile, Map<String, ProfileElement> profElements) {
262   	
263   	// try retrieve existing profile element
264   	ProfileElement epe = profElements.get(name);
265   	// or create a new one if not found
266   	if (epe==null) {
267   		 epe = new EnumeratedProfileElement(profile);
268   	   epe.setName(name);
269   	} 
270   	if (values!=null) {
271       for (String value : values) {
272       	if (StringUtils.hasText(value) && !value.equalsIgnoreCase("null")) {
273       		epe.getValues().add(value);
274       	}
275       }
276   	}
277     // only save profile elements with one or more values
278     if (epe.getValues().size()>0) {
279       profElements.put(name, epe);
280     }
281   	
282   }

308   private static void setResourceAttributes(ResourceAttributes resAttr,

310     if (StringUtils.hasText(value) && !value.equalsIgnoreCase("null")) {
311       if (propName.equals("Identifier")) {
312         resAttr.setIdentifier(value);
313       } else if (propName.equals("Title")) {
314         resAttr.setTitle(value);

317       } else if (propName.equals("Description")) {
318         resAttr.setDescription(value);
319       } else if (propName.equals("Creator")) {
320         resAttr.getCreators().addAll( parseValues(value) );
321       } else if (propName.equals("Subject")) {
322         resAttr.getSubjects().addAll( parseValues(value) );
323       } else if (propName.equals("Publisher")) {
324         resAttr.getPublishers().addAll( parseValues(value) );
325       } else if (propName.equals("Contributor")) {
326         resAttr.getContributors().addAll( parseValues(value) );
327       } else if (propName.equals("Date")) {
328         resAttr.getDates().addAll( parseValues(value) );
329       } else if (propName.equals("Type")) {
330         resAttr.getTypes().addAll( parseValues(value) );
331       } else if (propName.equals("Source")) {
332         resAttr.getSources().addAll( parseValues(value) );

367    * Utility method to split a metadata field value according to the known delimiter.
368    * @param value
369    * @return
370    */
371   public static List<String> parseValues(String value) {
372   	List<String> values = new ArrayList<String>();
373   	for (String val : value.split(PathUtils.DELIMITER)) {
374   		if (StringUtils.hasText(val) && !val.equalsIgnoreCase("null")) {
375   			values.add(val);
376   		}
377   	}
378   	
379   	return values;
380   }

382   /**
383    * Method to add a (name,value) pair to the metadata container if the value is not null or empty,
384    * and doesn't exist already.
385    * @param met
386    * @param value
387    */
388   public static void addIfNotNull(Metadata met, String key, String value) {
389   	// do not add a null value
390   	if (StringUtils.hasText(value) && !value.equalsIgnoreCase("null")) {
391   		// do not add the same value twice for the same metadata key
392   		if (!met.containsKey(key) || !met.getAllMetadata(key).contains(value)) {
393   			met.addMetadata(key, value);
394   		}
395     }
396   }

398 	/**
399 	 * Method to add multiple (key, value) pairs to the metadata container if not existing already.
400      */
401 	public static void addIfNotExisting(Metadata metadata, String key, Enumeration<String> values) {
402 		if (StringUtils.hasText(key) && !metadata.containsKey(key)) {
403 			while (values.hasMoreElements()) {
404 				String value = values.nextElement();
405 				if (StringUtils.hasText(value) && !value.equalsIgnoreCase("null")) {
406 					metadata.addMetadata(key,value);
407 				}
408 			}
409 		}
410 	}
