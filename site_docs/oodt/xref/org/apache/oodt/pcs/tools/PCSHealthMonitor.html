280   private List getCrawlerStatus() {
281     List crawlers = this.crawlProps.getCrawlers();
282     String crawlHost = this.crawlProps.getCrawlHost();
283     List statuses = new Vector();
284 
285     if (crawlers != null && crawlers.size() > 0) {
286       Collections.sort(crawlers, new Comparator() {
287         public int compare(Object o1, Object o2) {
288           CrawlInfo c1 = (CrawlInfo) o1;
289           CrawlInfo c2 = (CrawlInfo) o2;
290 
291           return c1.getCrawlerName().compareTo(c2.getCrawlerName());
292         }
293 
294       });
295 
296       for (Object crawler : crawlers) {
297         CrawlInfo info = (CrawlInfo) crawler;
298         String crawlerUrlStr = "http://" + crawlHost + ":"
299                                + info.getCrawlerPort();
300         CrawlerStatus status = new CrawlerStatus();
301         status.setInfo(info);
302         status.setStatus(printUp(getCrawlerUp(crawlerUrlStr)));
303         status.setCrawlHost(crawlHost);
304         statuses.add(status);
305       }
306     }
307 
308     return statuses;
309 
310   }

320   private List getJobStatusHealth() {
321     if (!getWmUp()) {
322       return new Vector();
323     }
324 
325     List statuses = new Vector();
326     List states = this.statesFile.getStates();
327 
328     if (states != null && states.size() > 0) {
329       for (Object state1 : states) {
330         String state = (String) state1;
331         int numPipelines = this.wm.safeGetNumWorkflowInstancesByStatus(state);
332         if (numPipelines == -1) {
333           numPipelines = 0;
334         }
335 
336         JobHealthStatus jobStatus = new JobHealthStatus();
337         jobStatus.setStatus(state);
338         jobStatus.setNumPipelines(numPipelines);
339         statuses.add(jobStatus);
340       }
341     }
342 
343     return statuses;

346   private List getIngestHealth() {
347     if (this.crawlProps.getCrawlers().size()==0) {
348       return new Vector();
349     }
350 
351     List statuses = new Vector();
352 
353     for (Object o : this.crawlProps.getCrawlers()) {
354       CrawlInfo info = (CrawlInfo) o;
355       String crawlUrlStr = "http://" + this.crawlProps.getCrawlHost() + ":"
356                            + info.getCrawlerPort();
357       try {
358         CrawlDaemonController controller = new CrawlDaemonController(
359             crawlUrlStr);
360         CrawlerHealth health = new CrawlerHealth();
361         health.setCrawlerName(info.getCrawlerName());
362         health.setNumCrawls(controller.getNumCrawls());
363         health
364             .setAvgCrawlTime((double) (controller.getAverageCrawlTime() / DOUBLE));
365         statuses.add(health);
366 
367       } catch (Exception e) {
368         CrawlerHealth health = new CrawlerHealth();
369         health.setCrawlerName(info.getCrawlerName());
370         health.setNumCrawls(CRAWLER_DOWN_INT);
371         health.setAvgCrawlTime(CRAWLER_DOWN_DOUBLE);
372         statuses.add(health);
373       }
374 
375     }
376 
377     return statuses;
378   }

575   private String getBiggestString(List crawlInfos) {
576     int biggestStrSz = Integer.MIN_VALUE;
577     String biggestStr = null;
578 
579     for (Object crawlInfo : crawlInfos) {
580       CrawlInfo info = (CrawlInfo) crawlInfo;
581       String crawlInfoName = info.getCrawlerName();
582       if (crawlInfoName.length() > biggestStrSz) {
583         biggestStr = crawlInfoName;
584         biggestStrSz = biggestStr.length();
585       }
586     }
587 
588     return biggestStr;
589   }

591   private String getStrPadding(String initString, String compareString) {
592     int sizeInitStr = initString.length();
593     int sizeCompareStr = compareString.length();
594 
595     int diff = Math.abs(sizeInitStr - sizeCompareStr);
596     StringBuilder buf = new StringBuilder();
597     for (int i = 0; i < diff; i++) {
598       buf.append(" ");
599     }
600 
601     return buf.toString();
602   }
