32   * A Free Text TermQueryExpression
33   * </p>

36  public class FreeTextQueryExpression extends TermQueryExpression {
37  	
38  	private HashSet<String> noiseWordHash = new HashSet<String>(
39  		Arrays.asList(new String[] { "a", "all", "am", "an", "and",
40          "any", "are", "as", "at", "be", "but", "can", "did", "do", "does",
41          "for", "from", "had", "has", "have", "here", "how", "i", "if",
42          "in", "is", "it", "no", "not", "of", "on", "or", "so", "that",
43          "the", "then", "there", "this", "to", "too", "up", "use", "what",
44          "when", "where", "who", "why", "you" }));
45  
46      /**
47       * A method for adding unparsed free text to the FreeTextCriteria. Free text
48       * entered as a string is tokenized and punctuation and common words are
49       * dropped before the values are added to the query. In order to query for
50       * pre-parsed keywords, see the setValues method of this class.
51       * 
52       * @param text
53       *            The free text to be parsed and searched on.
54       */
55      public void addFreeText(String text) {
56          // remove punctuation from the text
57          text = text.replaceAll("\\p{Punct}+", "");
58  
59          // tokenize string using default delimiters
60          StringTokenizer tok = new StringTokenizer(text);
61          String token;
62  
63          // filter noise words and add to values vector
64          List<String> values = new Vector<String>();
65          while (tok.hasMoreElements()) {
66              token = tok.nextToken();
67              if (!noiseWordHash.contains(token)) {
68                values.add(token);
69              }
70          }
71          if (values.size() > 0) {
72          	values.addAll(this.term.getValues());
73          	this.term.setValues(values);
74          }
75      }
76  
77      /**
78       * Implementation of the abstract method inherited from QueryCriteria for
79       * generating a human-parsable string version of the query criteria. Note
80       * that the returned String follows the Lucene query language.
81       * 
82       * @return The query as a String.
83       */
84      public String toString() {
85          StringBuilder serial = new StringBuilder();
86          serial.append("({").append(this.bucketNames).append("} ").append(this.term.getName()).append(" :|");
87          for (String value : this.term.getValues()) {
88              serial.append("+").append(value);
89              serial.append("|: )");
90          }
91          return serial.toString();
92      }
93      
94  	@Override
95  	public FreeTextQueryExpression clone() {
96  		FreeTextQueryExpression ftQE = new FreeTextQueryExpression();
97  		ftQE.noiseWordHash = new HashSet<String>(this.noiseWordHash);
98  		ftQE.setTerm(this.term.clone());
99  		ftQE.setBucketNames(this.bucketNames);
100 		return ftQE;
101 	}
