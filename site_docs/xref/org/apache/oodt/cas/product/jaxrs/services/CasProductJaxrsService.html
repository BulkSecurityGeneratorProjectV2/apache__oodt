67    /**
68     * Gets an HTTP response that represents a {@link Reference} from a {@link
69     * Product} from the file manager.
70     * @param productId the ID of the product that the reference belongs to
71     * @param refIndex the index of the reference within the product's list of
72     * references
73     * @return an HTTP response that represents a {@link Reference} from a {@link
74     * Product} from the file manager
75     */
76    @GET
77    @Path("reference")
78    @Produces({"application/octet-stream", "application/xml", "application/json",
79      "application/atom+xml", "application/rdf+xml", "application/rss+xml",
80      "application/zip"})
81    public ReferenceResource getReference(
82      @QueryParam("productId") String productId,
83      @QueryParam("refIndex") int refIndex)
84    {
85      if (productId == null || productId.trim().equals(""))
86      {
87        throw new BadRequestException("This URL requires a productId query "
88          + "parameter with a product ID value, "
89          + "e.g. /reference?productId=1787a257-df87-11e2-8a2d-e3f6264e86c5");
90      }
91  
92      try
93      {
94        XmlRpcFileManagerClient client = getContextClient();
95        Product product = client.getProductById(productId);
96        List<Reference> references = client.getProductReferences(product);
97  
98        return new ReferenceResource(productId, refIndex,
99          references.get(refIndex), getContextWorkingDir());
100     }
101     catch (Exception e)
102     {
103       String message = "Unable to find the requested resource.";
104       LOGGER.log(Level.FINE, message, e);
105       throw new NotFoundException(message + " " + e.getMessage());
106     }
107   }

154   /**
155    * Gets an HTTP response that represents a set of {@link Product products}
156    * from the file manager.
157    * @param productTypeId the ID of the {@link ProductType} for the data set or
158    * "ALL" to denote all product types
159    * @return an HTTP response that represents a set of {@link Product products}
160    * from the file manager
161    */
162   @GET
163   @Path("dataset")
164   @Produces({"application/xml", "application/json", "application/atom+xml",
165    "application/rdf+xml", "application/rss+xml", "application/zip"})
166   public DatasetResource getDataset(
167     @QueryParam("productTypeId") String productTypeId)
168   {
169     if (productTypeId == null || productTypeId.trim().equals(""))
170     {
171       throw new BadRequestException("This URL requires a productTypeId query "
172         + "parameter and either a product type ID value or 'ALL' for all "
173         + "product types.");
174     }
175 
176     try
177     {
178       XmlRpcFileManagerClient client = getContextClient();
179 
180       String datasetId;
181       String datasetName;
182       Metadata datasetMetadata;
183 
184       List<ProductType> productTypes = new Vector<ProductType>();
185       if (productTypeId.equals("ALL"))
186       {
187         productTypes = client.getProductTypes();
188         datasetId = productTypeId;
189         datasetName = productTypeId;
190         datasetMetadata = new Metadata();
191         datasetMetadata.addMetadata("ProductType", productTypeId);
192       }
193       else
194       {
195         ProductType productType = client.getProductTypeById(productTypeId);
196         productTypes.add(productType);
197         datasetId = productType.getProductTypeId();
198         datasetName = productType.getName();
199         datasetMetadata = productType.getTypeMetadata();
200       }
201 
202       String productDirPath = getContextWorkingDir().getCanonicalPath()
203         + "/" + datasetName;
204 
205       DatasetResource resource = new DatasetResource(datasetId, datasetName,
206         datasetMetadata, getContextWorkingDir());
207 
208       // Add all products of the chosen type(s) to the dataset.
209       for (ProductType productType : productTypes)
210       {
211         for (Product product : client.getProductsByProductType(productType))
212         {
213           product.setProductReferences(client.getProductReferences(product));
214           resource.addProductResource(new ProductResource(product,
215             client.getMetadata(product), product.getProductReferences(),
216             new File(productDirPath)));
217         }
218       }
219       return resource;
220     }
221     catch (Exception e)
222     {
223       String message = "Unable to find the requested resource.";
224       LOGGER.log(Level.FINE, message, e);
225       throw new NotFoundException(message + " " + e.getMessage());
226     }
227   }

231   /**
232    * Gets an HTTP response that represents the status of a currently active
233    * file transfer for the file manager.
234    * @param dataStoreRef the data store reference for the file being transferred
235    * @return an HTTP response that represents the status of a currently active
236    * file transfer for the file manager
237    */
238   @GET
239   @Path("transfer")
240   @Produces({"application/xml", "application/json", "application/atom+xml",
241     "application/rdf+xml", "application/rss+xml"})
242   public TransferResource getTransfer(
243     @QueryParam("dataStoreRef") String dataStoreRef)
244   {
245     if (dataStoreRef == null || dataStoreRef.trim().equals(""))
246     {
247       throw new BadRequestException("This URL requires a dataStoreRef query "
248         + "parameter and a data store reference value, "
249         + "e.g. /transfer?dataStoreRef=file:/repository/test.txt/test.txt");
250     }
251 
252     try
253     {
254       XmlRpcFileManagerClient client = getContextClient();
255       for (FileTransferStatus status : client.getCurrentFileTransfers())
256       {
257         Reference reference = status.getFileRef();
258         if (dataStoreRef.equals(reference.getDataStoreReference()))
259         {
260           Product product = status.getParentProduct();
261           Metadata metadata = client.getMetadata(product);
262           return new TransferResource(product, metadata, status);
263         }
264       }
265 
266       throw new Exception("Unable to find a current file transfer status for"
267         + "data store reference: " + dataStoreRef);
268     }
269     catch (Exception e)
270     {
271       String message = "Unable to find the requested resource.";
272       LOGGER.log(Level.FINE, message, e);
273       throw new NotFoundException(message + " " + e.getMessage());
274     }
275   }

279   /**
280    * Gets an HTTP response that represents the statuses of all currently active
281    * file transfers for the file manager, optionally filtered by product ID.
282    * @param productId the ID of a product or ALL to denote all products
283    * @return an HTTP response that represents the statuses of all currently
284    * active file transfers for the file manager
285    */
286   @GET
287   @Path("transfers")
288   @Produces({"application/xml", "application/json", "application/atom+xml",
289     "application/rdf+xml", "application/rss+xml"})
290   public TransfersResource getTransfers(
291     @QueryParam("productId") String productId)
292   {
293     if (productId == null || productId.trim().equals(""))
294     {
295       throw new BadRequestException("This URL requires a productId query "
296         + "parameter and either a valid product ID value or 'ALL' for all "
297         + "products.");
298     }
299 
300     try
301     {
302       List<TransferResource> transferResources =
303         new ArrayList<TransferResource>();
304       XmlRpcFileManagerClient client = getContextClient();
305       for (FileTransferStatus status : client.getCurrentFileTransfers())
306       {
307         Product product = status.getParentProduct();
308         if(productId.equals("ALL") || productId.equals(product.getProductId()))
309         {
310           Metadata metadata = client.getMetadata(product);
311           transferResources.add(
312             new TransferResource(product, metadata, status));
313         }
314       }
315       return new TransfersResource(productId, transferResources);
316     }
317     catch (Exception e)
318     {
319       String message = "Unable to find the requested resource.";
320       LOGGER.log(Level.FINE, message, e);
321       throw new NotFoundException(message + " " + e.getMessage());
322     }
323   }

327   /**
328    * Gets the file manager's working directory from the servlet context.
329    * @return the file manager working directory
330    * @throws Exception if an object cannot be retrieved from the context
331    * attribute
332    */
333   public File getContextWorkingDir() throws CasProductException {
334     Object workingDirObject = context.getAttribute("workingDir");
335     if (workingDirObject != null && workingDirObject instanceof File)
336     {
337       return (File) workingDirObject;
338     }
339 
340     String message = "Unable to get the file manager's working "
341       + "directory from the servlet context.";
342     LOGGER.log(Level.WARNING, message);
343     throw new CasProductException(message);
344   }
