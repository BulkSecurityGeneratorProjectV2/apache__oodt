74       * @see org.apache.oodt.cas.workflow.engine.WorkflowInstanceRepository#addWorkflowInstance(org.apache.oodt.cas.workflow.structs.WorkflowInstance)

76      public synchronized void addWorkflowInstance(WorkflowInstance wInst)
77              throws InstanceRepositoryException {
78          Connection conn = null;
79          Statement statement = null;
80          ResultSet rs = null;
81  
82          try {
83              conn = dataSource.getConnection();
84              conn.setAutoCommit(false);
85              statement = conn.createStatement();
86  
87              String startWorkflowSql;
88              String taskIdField;
89              String workflowIdField;
90  
91              if (quoteFields) {
92                  taskIdField = "'"
93                          + wInst.getWorkflow().getTasks().get(0)
94                                  .getTaskId() + "'";
95                  workflowIdField = "'" + wInst.getWorkflow().getId() + "'";
96              } else {
97                  taskIdField = wInst.getWorkflow().getTasks()
98                          .get(0).getTaskId();
99                  workflowIdField = wInst.getWorkflow().getId();
100             }
101 
102             startWorkflowSql = "INSERT INTO workflow_instances "
103                     + "(workflow_instance_status, workflow_id, current_task_id,"
104                     + "start_date_time, end_date_time, current_task_start_date_time,"
105                     + "current_task_end_date_time, priority, times_blocked) " + "VALUES ('"
106                     + wInst.getStatus() + "', " + workflowIdField + ","
107                     + taskIdField + ", '" + wInst.getStartDateTimeIsoStr()
108                     + "','" + wInst.getEndDateTimeIsoStr() + "','"
109                     + wInst.getCurrentTaskStartDateTimeIsoStr() + "','"
110                     + wInst.getCurrentTaskEndDateTimeIsoStr() + "', "+wInst.getPriority().getValue()+", "
111                     + wInst.getTimesBlocked() + ")";
112 
113             LOG.log(Level.FINE, "sql: Executing: " + startWorkflowSql);
114             statement.execute(startWorkflowSql);
115 
116             String workflowInstId = "";
117 
118             synchronized (workflowInstId) {
119                 String getWorkflowInstIdSql = "SELECT MAX(workflow_instance_id) "
120                         + "AS max_id FROM workflow_instances";
121 
122                 rs = statement.executeQuery(getWorkflowInstIdSql);
123 
124                 while (rs.next()) {
125                     workflowInstId = String.valueOf(rs.getInt("max_id"));
126                 }
127             }
128 
129             conn.commit();
130             wInst.setId(workflowInstId);
131 
132             // now add its metadata
133             addWorkflowInstanceMetadata(wInst);
134         } catch (Exception e) {
135             LOG.log(Level.SEVERE, e.getMessage());
136             LOG.log(Level.WARNING, "Exception starting workflow. Message: "
137                     + e.getMessage());
138             try {
139                 if (conn != null) {
140                     conn.rollback();
141                 }
142             } catch (SQLException e2) {
143                 LOG.log(Level.SEVERE,
144                         "Unable to rollback startWorkflow transaction. Message: "
145                                 + e2.getMessage());
146             }
147             throw new InstanceRepositoryException(e.getMessage());
148         } finally {
149 
150             if (rs != null) {
151                 try {
152                     rs.close();
153                 } catch (SQLException ignore) {
154                 }
155 
156             }
157 
158             if (statement != null) {
159                 try {
160                     statement.close();
161                 } catch (SQLException ignore) {
162                 }
163 
164             }
165 
166             if (conn != null) {
167                 try {
168                     conn.close();
169 
170                 } catch (SQLException ignore) {
171                 }
172 
173             }
174         }
175 
176     }

707     protected List paginateWorkflows(int pageNum, String status)
708             throws InstanceRepositoryException {
709         Connection conn = null;
710         Statement statement = null;
711         ResultSet rs = null;
712 
713         List wInstIds = null;
714         int numResults;
715 
716         if (status == null || (status.equals(""))) {
717             numResults = getNumWorkflowInstances();
718         } else {
719             numResults = getNumWorkflowInstancesByStatus(status);
720         }
721 
722         try {
723             conn = dataSource.getConnection();
724             statement = conn.createStatement(ResultSet.TYPE_SCROLL_INSENSITIVE,
725                     ResultSet.CONCUR_READ_ONLY);
726 
727             String getWorkflowSql = "SELECT workflow_instance_id FROM workflow_instances ";
728             if (status != null && !status.equals("")) {
729                 getWorkflowSql += "WHERE workflow_instance_status = '" + status
730                         + "'";
731 
732             }
733 
734             getWorkflowSql += "ORDER BY workflow_instance_id DESC ";
735 
736             LOG.log(Level.FINE, "workflow instance paged query: executing: "
737                     + getWorkflowSql);
738 
739             rs = statement.executeQuery(getWorkflowSql);
740             wInstIds = new Vector();
741 
742             int startNum = (pageNum - 1) * pageSize;
743 
744             if (startNum > numResults) {
745                 startNum = 0;
746             }
747 
748             // must call next first, or else no relative cursor
749             if (rs.next()) {
750                 // grab the first one
751                 int numGrabbed;
752                 if(pageNum == 1){
753                     numGrabbed = 1;
754                     wInstIds.add(rs.getString("workflow_instance_id"));                    
755                 }
756                 else{
757                     numGrabbed = 0;
758                 }
759 
760                 if(pageNum != 1){
761                     // now move the cursor to the correct position
762                     rs.relative(startNum);                    
763                 }
764 
765                 // grab the rest
766                 while (rs.next() && numGrabbed < pageSize) {
767                     String wInstId = rs.getString("workflow_instance_id");
768                     wInstIds.add(wInstId);
769                     numGrabbed++;
770                 }
771             }
772 
773             if (wInstIds.size() == 0) {
774                 wInstIds = null;
775             }
776 
777         } catch (Exception e) {
778             LOG.log(Level.SEVERE, e.getMessage());
779             LOG.log(Level.WARNING, "Exception performing query. Message: "
780                     + e.getMessage());
781             try {
782                 if (conn != null) {
783                     conn.rollback();
784                 }
785             } catch (SQLException e2) {
786                 LOG.log(Level.SEVERE,
787                         "Unable to rollback query transaction. Message: "
788                                 + e2.getMessage());
789             }
790             throw new InstanceRepositoryException(e.getMessage());
791         } finally {
792 
793             if (rs != null) {
794                 try {
795                     rs.close();
796                 } catch (SQLException ignore) {
797                 }
798 
799             }
800 
801             if (statement != null) {
802                 try {
803                     statement.close();
804                 } catch (SQLException ignore) {
805                 }
806 
807             }
808 
809             if (conn != null) {
810                 try {
811                     conn.close();
812 
813                 } catch (SQLException ignore) {
814                 }
815 
816             }
