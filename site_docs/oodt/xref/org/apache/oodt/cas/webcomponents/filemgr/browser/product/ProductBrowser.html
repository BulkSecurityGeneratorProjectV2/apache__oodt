18  package org.apache.oodt.cas.webcomponents.filemgr.browser.product;
19  
20  //JDK imports
21  import java.io.File;
22  import java.net.URI;
23  import java.text.NumberFormat;
24  import java.util.Collections;
25  import java.util.List;
26  import java.util.logging.Level;
27  import java.util.logging.Logger;
28  
29  //OODT imports
30  import org.apache.oodt.cas.filemgr.structs.Product;
31  import org.apache.oodt.cas.filemgr.structs.Reference;
32  import org.apache.oodt.cas.filemgr.structs.exceptions.DataTransferException;
33  import org.apache.oodt.cas.metadata.Metadata;
34  import org.apache.oodt.cas.webcomponents.filemgr.FileManagerConn;
35  
36  //Wicket imports
37  import org.apache.wicket.markup.html.basic.Label;
38  import org.apache.wicket.markup.html.list.ListItem;
39  import org.apache.wicket.markup.html.list.ListView;
40  import org.apache.wicket.markup.html.panel.Panel;
41  import org.apache.wicket.model.util.ListModel;
42  
43  /**
44   * 
45   * This page is mounted to:
46   * 
47   * <code>/browser/product/[ProductId]</code>
48   * 
49   * A {@link Product} web page component to display 
50   * all of its metadata and reference information.
51   * 
52   * @author mattmann
53   * @version $Revision$
54   * 
55   */
56  public class ProductBrowser extends Panel {
57  
58    private static final long serialVersionUID = -1204491609423677032L;
59  
60    private static final Logger LOG = Logger.getLogger(ProductBrowser.class
61        .getName());
62    
63    private FileManagerConn fm;
64  
65    public ProductBrowser(String componentId, String fmUrlStr, String productId) {
66      super(componentId);
67      this.fm = new FileManagerConn(fmUrlStr);
68      Product product = fm.safeGetProductById(productId);
69      try {
70        String productPctTransferred = NumberFormat.getPercentInstance().format(
71            fm.getFm().getProductPctTransferred(
72                product));
73        add(new Label("product_pct_transfer", productPctTransferred));
74      } catch (DataTransferException e) {
75        LOG.log(Level.WARNING, "Unable to determine pct transfer for product: ["
76            + product.getProductName() + "]: Reason: " + e.getMessage());
77      }
78  
79      add(new Label("product_name", product.getProductName()));
80      final Metadata met = fm.getMetadata(product);
81      if (met != null) {
82        add(new Label("no_prod_met_display") {
83          /*
84           * (non-Javadoc)
85           * 
86           * @see org.apache.wicket.Component#isVisible()
87           */
88          @Override
89          public boolean isVisible() {
90            return false;
91          }
92        });
93  
94        List<String> metKeys = met.getAllKeys();
95        Collections.sort(metKeys);
96  
97        add(new ListView<String>("met_elem", new ListModel<String>(metKeys)) {
98          /*
99           * (non-Javadoc)
100          * 
101          * @see
102          * org.apache.wicket.markup.html.list.ListView#populateItem(org.apache
103          * .wicket.markup.html.list.ListItem)
104          */
105         @Override
106         protected void populateItem(ListItem<String> item) {
107           item.add(new Label("met_elem_name", item.getModelObject()));
108 
109           item
110               .add(new ListView<String>("met_values_list",
111                   new ListModel<String>(met.getAllMetadata(item
112                       .getModelObject()))) {
113                 /*
114                  * (non-Javadoc)
115                  * 
116                  * @see
117                  * org.apache.wicket.markup.html.list.ListView#populateItem(
118                  * org.apache.wicket.markup.html.list.ListItem)
119                  */
120                 @Override
121                 protected void populateItem(ListItem<String> item) {
122                   item.add(new Label("met_value", item.getModelObject()));
123                 }
124               });
125 
126         }
127       });
128     } else {
129       add(new Label("no_prod_met_display", "No Product Metadata!"));
130     }
131 
132     List<Reference> refs = fm
133         .getProductReferences(product);
134     if (refs != null && refs.size() > 0) {
135       add(new Label("no_prod_ref_display") {
136         /*
137          * (non-Javadoc)
138          * 
139          * @see org.apache.wicket.Component#isVisible()
140          */
141         @Override
142         public boolean isVisible() {
143           return false;
144         }
145       });
146 
147       add(new ListView<Reference>("ref_list", new ListModel<Reference>(refs)) {
148         /*
149          * (non-Javadoc)
150          * 
151          * @see
152          * org.apache.wicket.markup.html.list.ListView#populateItem(org.apache
153          * .wicket.markup.html.list.ListItem)
154          */
155         @Override
156         protected void populateItem(ListItem<Reference> refItem) {
157           Reference r = refItem.getModelObject();
158           String filePath = null;
159           try {
160             filePath = new File(new URI(r.getDataStoreReference()))
161                 .getAbsolutePath();
162           } catch (Exception ignore) {
163           }
164 
165           refItem.add(new Label("ref_file_path", filePath));
166           refItem.add(new Label("ref_file_size", String
167               .valueOf(r.getFileSize())));
168           try {
169             refItem.add(new Label("ref_pct_transferred", NumberFormat
170                 .getPercentInstance().format(
171                    fm.getFm()
172                         .getRefPctTransferred(r))));
173           } catch (DataTransferException e) {
174             LOG.log(Level.WARNING,
175                 "Unable to determine product reference size: Reason: "
176                     + e.getMessage());
177             refItem.add(new Label("ref_pct_transferred", "N/A"));
178           }
179 
180         }
181       });
182     } else {
183       add(new Label("no_prod_ref_display", "No Product References!"));
184     }
185 
186   }
