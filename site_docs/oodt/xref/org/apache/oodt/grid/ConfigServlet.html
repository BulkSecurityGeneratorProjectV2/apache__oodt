37     * Handle updates to a configuration by saving them and directing back to the
38     * config page.
39     * 
40     * @param req
41     *          a <code>HttpServletRequest</code> value.
42     * @param res
43     *          a <code>HttpServletResponse</code> value.
44     * @throws ServletException
45     *           if an error occurs.
46     * @throws IOException
47     *           if an error occurs.
48     */
49    public void doPost(HttpServletRequest req, HttpServletResponse res)
50        throws ServletException, IOException {
51      Configuration config = getConfiguration(); // Get the singleton
52                                                 // configuration
53      if (!approveAccess(config, req, res)) {
54        return; // Check if the user can access this page
55      }
56  
57      ConfigBean cb = getConfigBean(req); // Get the bean
58      cb.setMessage(""); // Clear out any message
59      if (!cb.isAuthentic()) {
60        throw new ServletException(new AuthenticationRequiredException());
61      }
62      boolean needSave = false; // Assume no changes for now
63  
64      String newPass = req.getParameter("password"); // See if she wants to change
65                                                     // password
66      if (newPass != null && newPass.length() > 0) { // Got a password, and it's
67                                                     // not empty?
68        config.setPassword(newPass.getBytes()); // Yes, set its bytes
69        needSave = true; // We need to save
70      }
71  
72      boolean https = config.isHTTPSrequired(); // Are we currently using https?
73      String httpsParam = req.getParameter("https"); // Get her https param
74      boolean newHttps = httpsParam != null && "on".equals(httpsParam); // See if
75                                                                        // she
76                                                                        // wants
77                                                                        // to use
78                                                                        // https
79                                                                        // or not
80      if (https != newHttps) { // Different?
81        config.setHTTPSrequired(newHttps); // Yes, set the new state
82        needSave = true; // We need to save
83      }
84  
85      boolean local = config.isLocalhostRequired(); // Are we requiring localhost?
86      String localhostParam = req.getParameter("localhost"); // Get her localhost
87                                                             // param
88      boolean newLocal = localhostParam != null && "on".equals(localhostParam); // See
89                                                                                // if
90                                                                                // she
91                                                                                // wants
92                                                                                // to
93                                                                                // require
94                                                                                // localhost
95      if (local != newLocal) { // Different?
96        config.setLocalhostRequired(newLocal); // Yes, set the new state
97        needSave = true; // We need to save
98      }
99  
100     String newKey = req.getParameter("newkey"); // See if she's got a new
101                                                 // property
102     if (newKey != null && newKey.length() > 0) { // And make sure it's nonempty
103       String newVal = req.getParameter("newval"); // Got one, get its value
104       if (newVal == null) {
105         newVal = ""; // Make sure it's at least an empty string
106       }
107       config.getProperties().setProperty(newKey, newVal); // Set the new
108                                                           // property
109       needSave = true; // We need to save
110     }
111 
112     needSave |= updateProperties(config.getProperties(), // Make any updates to
113                                                          // existing property
114         req.getParameterMap()); // values, and see note if we need to save
115     try {
116       needSave |= updateServers(config, req.getParameterMap(), 'd'); // Same
117                                                                      // goes for
118                                                                      // product
119                                                                      // servers
120       needSave |= updateServers(config, req.getParameterMap(), 'm'); // And
121                                                                      // profile
122                                                                      // servers
123       needSave |= updateCodeBases(config, req.getParameterMap()); // And for
124                                                                   // code bases
125     } catch (MalformedURLException ex) { // Make sure code base URLs are OK
126       cb.setMessage("Code bases must be valid URLs (" // Not OK?
127           + ex.getMessage() + "); no changes made"); // Let user know via a
128                                                      // message
129       req.getRequestDispatcher("config.jsp").forward(req, res); // And make no
130                                                                 // changes until
131                                                                 // she fixes it
132       return;
133     }
134 
135     if (needSave) { // Do we need to save?
136       config.save(); // Then do it already!
137       cb.setMessage("Changes saved."); // And let the user know
138     } else {
139       cb.setMessage("No changes made."); // Oh, no changes were made, let user
140                                          // know
141     }
142     req.getRequestDispatcher("config.jsp").forward(req, res); // Back to the
143                                                               // config page
144   }

146   /**
147    * Update changes to the code bases.
148    * 
149    * @param config
150    *          a <code>Configuration</code> value.
151    * @param params
152    *          a <code>Map</code> value.
153    * @return True if changes need to be saved.
154    * @throws MalformedURLException
155    *           if an error occurs.
156    */
157   private boolean updateCodeBases(Configuration config, Map params)
158       throws MalformedURLException {
159     boolean needSave = false; // Assume no change
160     List codeBases = config.getCodeBases(); // Get the current code bases
161 
162     List toRemove = new ArrayList(); // Hold indexes of code bases to remove
163     for (Object o : params.entrySet()) { // For each
164       // parameter
165       Map.Entry entry = (Map.Entry) o; // Get its entry
166       String key = (String) entry.getKey(); // And its name
167       String value = ((String[]) entry.getValue())[0]; // And its zeroth value
168       if (key.startsWith("delcb-") && "on".equals(value)) { // If it's checked
169         Integer index = Integer.valueOf(key.substring(6)); // Parse out the index
170         toRemove.add(index); // Add it to the list
171       }
172     }
173     if (!toRemove.isEmpty()) { // And if we have any indexes
174       Collections.sort(toRemove); // Sort 'em and put 'em in reverse ...
175       Collections.reverse(toRemove); // ... order so we can safely remove them
176       for (Object aToRemove : toRemove) { // For each index
177         // to remove
178         int index = (Integer) aToRemove; // Get the index value
179         codeBases.remove(index); // And buh-bye.
180       }
181       needSave = true; // Definitely need to save changes now
182     }
183 
184     String[] newCBs = (String[]) params.get("newcb"); // Was there a new code
185                                                       // base specified?
186     if (newCBs != null && newCBs.length == 1) { // And was there exactly one
187                                                 // value?
188       String newCB = newCBs[0]; // Get that exactly one value
189       if (newCB != null && newCB.length() > 0) { // Is it nonnull and nonempty?
190         URL newURL = new URL(newCB); // Treat is as an URL
191         codeBases.add(newURL); // Add it to the list
192         needSave = true; // Ad we gotta save
193       }
194     }
195     return needSave;
196   }

198   /**
199    * Update the list of product/profile servers based on request parameters.
200    * 
201    * @param config
202    *          System configuration
203    * @param params
204    *          Request parameters.
205    * @param type
206    *          <code>d</code> (data) if product servers, <code>m</code>
207    *          (metadata) if profile servers
208    * @return True if any changes were made, false if no changes were made
209    */
210   private boolean updateServers(Configuration config, Map params, char type) {
211     List servers = type == 'd' ? config.getProductServers() : config
212         .getProfileServers();
213     boolean needSave = false; // Assume no changes for now
214 
215     List toRemove = new ArrayList(); // Start with empty list of indexes to
216                                      // remove
217     for (Object o : params.entrySet()) { // Go
218       // through
219       // each
220       // parameter
221       Map.Entry entry = (Map.Entry) o; // Get its key/value
222       String name = (String) entry.getKey(); // The key is a String
223       if (name.startsWith(type + "rm-")) { // Is it an "drm-" or "mrm-"?
224         Integer index = Integer.valueOf(name.substring(4)); // Yes, get it sindex
225         toRemove.add(index); // Add it to the list
226       }
227     }
228 
229     if (!toRemove.isEmpty()) { // Got any to remove?
230       Collections.sort(toRemove); // We need to go through them in reverse ord-
231       Collections.reverse(toRemove); // -er, so that removals don't shift
232                                      // indexes
233       for (Object aToRemove : toRemove) { // For each index
234         int index = (Integer) aToRemove; // Get its int value
235         servers.remove(index); // and buh-bye
236       }
237       needSave = true; // Gotta save after all that, whew.
238     }
239 
240     if (params.containsKey(type + "-newcn")) { // Adding a new server?
241       String[] newClasses = (String[]) params.get(type + "-newcn"); // And the
242                                                                     // new class
243                                                                     // name
244       if (newClasses != null && newClasses.length == 1) { // Are present and
245                                                           // there's only one of
246                                                           // each
247         String newClass = newClasses[0]; // Get the new class
248         if (newClass != null && newClass.length() > 0) { // And nonempty
249           Server server;
250           if (type == 'd') // If it's data
251           {
252             server = new ProductServer(config, newClass); // It's a product
253           }
254 // server
255           else
256             // otherwise it's metadata
257           {
258             server = new ProfileServer(config, newClass); // Which is a profile
259           }
260                                                           // server
261           servers.add(server); // Add it to the set of servers
262           needSave = true; // And after all this, we need to save!
263         }
264       }
265     }
266 
267     return needSave;
268   }

270   /**
271    * Update properties based on request parameters.
272    * 
273    * @param props
274    *          <code>Properties</code> to update
275    * @param params
276    *          Request parameters
277    * @return True if changes need to be saved, false otherwise
278    */
279   private boolean updateProperties(Properties props, Map params) {
280     boolean needSave = false; // Assume no save for now
281     for (Object o : params.entrySet()) { // Go
282       // through
283       // each
284       // request
285       // parameter
286       Map.Entry entry = (Map.Entry) o; // Get the key/value
287       String name = (String) entry.getKey(); // Key is always a string
288       String newValue = ((String[]) entry.getValue())[0]; // Value is String[],
289       // get the zeroth
290       if (name.startsWith("val-")) { // If the param is "val-"
291         String key = name.substring(4); // Then find the key
292         if (props.containsKey(key)) { // If that key exists
293           String value = props.getProperty(key); // Find its value
294           if (value == null || !value.equals(newValue)) {// Are they different?
295             props.setProperty(key, newValue); // Yes, set the new value
296             needSave = true; // And we need to save
297           }
298         }
299       } else if (name.startsWith("del-")) { // If the param is "del-"
300         String key = name.substring(4); // Then find the key
301         if (props.containsKey(key)) { // If that key exists
302           props.remove(key); // Then remove its mapping
303           needSave = true; // And we need to save
304         }
305       }
306     }
307     return needSave;
308   }
