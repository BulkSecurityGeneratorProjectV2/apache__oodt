44  public class MemoryBasedIngestMapper implements IngestMapper {

46  	private static Logger LOG = Logger.getLogger(MemoryBasedIngestMapper.class.getName());
47  	
48  	protected ConcurrentHashMap<String, TransactionIdMapping> catalogServiceTransactionIdKeyMapping;
49  	protected ConcurrentHashMap<String, TransactionIdMapping> catalogInfoKeyMapping;
50  	protected ConcurrentHashMap<String, List<CatalogReceipt>> catalogIdToCatalogReceiptMapping;
51  	
52  	public MemoryBasedIngestMapper() {
53  		this.catalogServiceTransactionIdKeyMapping = new ConcurrentHashMap<>();
54  		this.catalogInfoKeyMapping = new ConcurrentHashMap<>();
55  		this.catalogIdToCatalogReceiptMapping = new ConcurrentHashMap<>();

65  	@Override
66  	public synchronized TransactionId<?> getCatalogServiceTransactionId(
67  			TransactionId<?> catalogTransactionId, String catalogId)
68  			throws CatalogRepositoryException {
69  		LOG.log(Level.INFO, "Looking up CatalogService TransactionId for Catalog TransactionId '" + catalogTransactionId + "' and catalog '" + catalogId + "'");
70  		String key = generateKey(catalogTransactionId.toString(), catalogId);
71  		TransactionIdMapping mapping = this.catalogInfoKeyMapping.get(key);
72  		if (mapping != null) {
73  			return mapping.catalogServiceTransactionId;
74  		}else {
75  			return null;
76  		}
77  	}

86  	@Override
87  	public synchronized TransactionId<?> getCatalogTransactionId(
88  			TransactionId<?> catalogServiceTransactionId, String catalogId)
89  			throws CatalogRepositoryException {
90  		TransactionIdMapping mapping = this.catalogServiceTransactionIdKeyMapping
91  				.get(catalogServiceTransactionId.toString());
92  		if (mapping != null) {
93  		  for (CatalogReceipt receipt : mapping.getCatalogReceipts()) {
94  			if (receipt.getCatalogId().equals(catalogId)) {
95  			  return receipt.getTransactionId();
96  			}
97  		  }
98  		}
99  		return null;
100 	}

109 	@Override
110 	public synchronized Set<TransactionId<?>> getPageOfCatalogTransactionIds(IndexPager indexPager,
111 			String catalogId) throws CatalogRepositoryException {
112 		Set<TransactionId<?>> catalogTransactionIds = new HashSet<>();
113 		List<CatalogReceipt> catalogReceipts = this.catalogIdToCatalogReceiptMapping.get(catalogId);
114 		if (catalogReceipts != null) {
115 		  for (int i = indexPager.getPageNum() * indexPager.getPageSize();
116 			   i < catalogReceipts.size() && i < (indexPager.getPageNum() + 1) * indexPager.getPageSize(); i++) {
117 			catalogTransactionIds.add(catalogReceipts.get(i).getTransactionId());
118 		  }
119 		}
120 		return catalogTransactionIds;
121 	}

127 	@Override
128 	public synchronized void deleteAllMappingsForCatalog(String catalogId)
129 			throws CatalogRepositoryException {
130 		List<CatalogReceipt> catalogReceipts = this.catalogIdToCatalogReceiptMapping.remove(catalogId);
131 		if (catalogReceipts != null) {
132 			for (CatalogReceipt catalogReceipt : catalogReceipts) {
133 				TransactionIdMapping mapping = this.catalogInfoKeyMapping.remove(generateKey(catalogReceipt.getTransactionId().toString(), catalogReceipt.getCatalogId()));
134 				if (mapping != null) {
135 				  mapping.getCatalogReceipts().remove(catalogReceipt);
136 				}
137 			}
138 		}

160 	 * (non-Javadoc)
161 	 * @see org.apache.oodt.cas.catalog.repository.CatalogRepository#deleteTransactionIdMapping(org.apache.oodt.cas.catalog.struct.TransactionId, org.apache.oodt.cas.catalog.struct.TransactionId, java.lang.String)
162 	 */
163 	@Override
164 	public synchronized void deleteTransactionIdMapping(
165 			TransactionId<?> catalogTransactionId, String catalogId)
166 			throws CatalogRepositoryException {
167 		List<CatalogReceipt> catalogReceipts = this.catalogIdToCatalogReceiptMapping.get(catalogId);
168 		for (int i = 0; i < catalogReceipts.size(); i++) {
169 			if (catalogReceipts.get(i).getCatalogId().equals(catalogId) && catalogReceipts.get(i).getTransactionId().equals(catalogTransactionId)) {
170 				catalogReceipts.remove(i);
171 				break;
172 			}
173 		}
174 		TransactionIdMapping mapping = this.catalogInfoKeyMapping.remove(generateKey(catalogTransactionId.toString(), catalogId));
175 		this.catalogServiceTransactionIdKeyMapping.remove(mapping.getCatalogServiceTransactionId().toString());
176 	}

190 	 * (non-Javadoc)

192 	 * @seeorg.apache.oodt.cas.catalog.repository.CatalogRepository#

194 	 * org.apache.oodt.cas.catalog.struct.TransactionId,
195 	 * org.apache.oodt.cas.catalog.struct.TransactionId)
196 	 */
197 	@Override
198 	public synchronized void storeTransactionIdMapping(
199 			TransactionId<?> catalogServiceTransactionId,
200 			TransactionIdFactory catalogServiceTransactionIdFactory,
201 			CatalogReceipt catalogReceipt,
202 			TransactionIdFactory catalogTransactionIdFactory)
203 			throws CatalogRepositoryException { 
204 		TransactionIdMapping mapping = this.catalogServiceTransactionIdKeyMapping
205 				.get(catalogServiceTransactionId.toString());
206 		if (mapping == null) {
207 		  mapping = new TransactionIdMapping(catalogServiceTransactionId);
208 		}
209 		mapping.addCatalogReceipt(catalogReceipt);
210 		this.catalogServiceTransactionIdKeyMapping.put(
211 				catalogServiceTransactionId.toString(), mapping);
212 		this.catalogInfoKeyMapping.put(generateKey(catalogReceipt.getTransactionId().toString(), catalogReceipt.getCatalogId()), mapping);
213 		List<CatalogReceipt> catalogReceipts = this.catalogIdToCatalogReceiptMapping.get(catalogReceipt.getCatalogId());
214 		if (catalogReceipts == null) {
215 		  catalogReceipts = new Vector<CatalogReceipt>();
216 		}
217 		catalogReceipts.add(catalogReceipt);
218 		this.catalogIdToCatalogReceiptMapping.put(catalogReceipt.getCatalogId(), catalogReceipts);
219 	}

245 		TransactionIdMapping mapping = this.catalogServiceTransactionIdKeyMapping.get(catalogServiceTransactionId);
246 		for (CatalogReceipt catalogReceipt : mapping.getCatalogReceipts()) {

254 	private static String generateKey(String catalogTransactionId, String catalogId) {
255 		return catalogTransactionId + ":" + catalogId;

258 	private static class TransactionIdMapping {
259 
260 		private TransactionId<?> catalogServiceTransactionId;
261 		private List<CatalogReceipt> catalogReceipts;
262 
263 		public TransactionIdMapping(TransactionId<?> catalogServiceTransactionId) {
264 			this.catalogServiceTransactionId = catalogServiceTransactionId;
265 			this.catalogReceipts = new Vector<>();
266 		}
267

269 			this.catalogReceipts.add(catalogReceipt);

273 			return this.catalogReceipts;
