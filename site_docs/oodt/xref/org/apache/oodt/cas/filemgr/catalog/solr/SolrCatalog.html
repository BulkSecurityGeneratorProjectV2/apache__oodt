78  	/**
79  	 * This method implementation will remove the specified keys and values from the product metadata,
80  	 * leaving all other metadata keys and values unchanged.
81  	 * Note: ALL occurrences of each specified (key, value) pair will be removed, not just the first.
82  	 */
83  	@Override
84  	public void removeMetadata(Metadata metadata, Product product) throws CatalogException {
85  		
86  		// retrieve full existing metadata
87  		Metadata currentMetadata = getMetadata(product);
88  		
89  		// metadata to be updated
90  		Metadata updateMetadata = new Metadata();
91  		
92  		// loop over keys of metadata be updated
93  		for (String key : metadata.getKeys()) {
94  			
95  			// list of values remaining after removal
96  			List<String> values = new ArrayList<String>();
97  			if (currentMetadata.containsKey(key)) {
98  				for (String value : currentMetadata.getAllMetadata(key)) {
99  					if (!metadata.getAllMetadata(key).contains(value)) {
100 						values.add(value);
101 					}					
102 				}
103 
104 				// add remaining values to updated metadata
105 				if (values.isEmpty()) {
106 					// special value because Metadata will NOT store an empty list
107 					values.add(Parameters.NULL);
108 				} 
109 				updateMetadata.addMetadata(key, values);
110 				
111 			}
112 		}
113 		
114 		// generate Solr update documents
115 		// replace=true to override existing values
116 		List<String> docs = productSerializer.serialize(product.getProductId(), updateMetadata, true);
117 				
118 		// send documents to Solr server
119 		solrClient.index(docs, true, productSerializer.getMimeType());
120 
121 	}

131 		LOG.info("Adding product:"+product.getProductName());
132 		
133 		// generate product identifier if not existing already
134 		if (!StringUtils.hasText(product.getProductId())) {
135 			String productId = this.productIdGenerator.generateId(product);
136 			product.setProductId(productId);
137 		}
138 		
139 		// serialize product for ingestion into Solr
140 		List<String> docs = productSerializer.serialize(product, true); // create=true
141 				
142 		// send records to Solr
143 		solrClient.index(docs, true, productSerializer.getMimeType());
144 		
145 	}

230 	/**
231 	 * Common utility to retrieve a range of products matching the specified {@link Query} and {@link ProductType}.
232 	 * This method transforms the given constraints in a map of HTTP (name, value) pairs and delegates to the following method.
233 	 * 
234 	 * @param query
235 	 * @param type
236 	 * @param offset
237 	 * @param limit
238 	 * @return
239 	 * @throws CatalogException
240 	 */
241 	private QueryResponse getProducts(Query query, ProductType type, int offset, int limit) throws CatalogException {
242 		
243 		// build HTTP request
244 		ConcurrentHashMap<String, String[]> params = new ConcurrentHashMap<String, String[]>();
245 		// product type constraint
246 		params.put("q", new String[]{Parameters.PRODUCT_TYPE_NAME+":"+type.getName()} );
247     // convert filemgr query into a Solr query
248 		List<String> qc = new ArrayList<String>();
249     for (QueryCriteria queryCriteria : query.getCriteria()) {
250     		LOG.info("Query criteria="+queryCriteria.toString());
251     		qc.add(queryCriteria.toString());
252     }
253     params.put("fq", qc.toArray( new String[ qc.size() ] ));
254     // sort
255     params.put("sort", new String[]{ Parameters.PRODUCT_RECEIVED_TIME+" desc"} );
256 
257     return this.getProducts(params, offset, limit);
258 		
259 	}

261 	/**
262 	 * Common utility to retrieve a range of products matching the specified query parameters
263 	 * @param params : HTTP query parameters used for query to Solr
264 	 * @param offset : the index of the first result, starting at 0
265 	 * @param limit : the maximum number of returned results (use -1 to return ALL results)
266 	 * @return
267 	 */
268 	private QueryResponse getProducts(Map<String, String[]> params, int offset, int limit) throws CatalogException {
269 		
270 		// combined results from pagination
271 		QueryResponse queryResponse = new QueryResponse();
272 		queryResponse.setStart(offset);
273 		int start = offset;
274 		while (queryResponse.getCompleteProducts().size()<limit || limit<0) {
275 			
276 			params.put("start", new String[] { ""+start } );
277 			String response = solrClient.query(params, productSerializer.getMimeType());
278 			QueryResponse qr = productSerializer.deserialize(response);
279 			
280 			for (CompleteProduct cp : qr.getCompleteProducts()) {
281 				if (queryResponse.getCompleteProducts().size()<limit) {
282 					queryResponse.getCompleteProducts().add( cp );
283 				}
284 			}
285 			
286 			queryResponse.setNumFound( qr.getNumFound() );
287 			start = offset+queryResponse.getCompleteProducts().size();
288 			if (limit<0) {
289 			  limit = queryResponse.getNumFound(); // retrieve ALL results
290 			}
291 			if (start>=queryResponse.getNumFound()) {
292 			  break; // don't query any longer
293 			}
294 			
295 		}
296 		
297 		LOG.info("Total number of products found="+queryResponse.getNumFound());
298 		LOG.info("Total number of products returned="+queryResponse.getCompleteProducts().size());
299 		return queryResponse;
300 		
301 	}

347 	@Override
348 	public List<String> query(Query query, ProductType type) throws CatalogException {
349 		
350     // execute request for ALL results
351     QueryResponse queryResponse =  this.getProducts(query, type, 0, -1); // get ALL products 
352 
353 		// extract ids from products
354 		List<String> ids = new ArrayList<String>();
355 		for (CompleteProduct cp : queryResponse.getCompleteProducts()) {
356 			ids.add(cp.getProduct().getProductId());
357 		}
358 		
359 		return ids;

522 	private CompleteProduct extractCompleteProduct(String doc) throws CatalogException {
523 		
524 		// deserialize document into Product
525 		LOG.info("Parsing Solr document: "+doc);
526 		QueryResponse queryResponse = productSerializer.deserialize(doc);
527 		int numFound = queryResponse.getNumFound();
528 		if (numFound>1) {
529 			throw new CatalogException("Product query returned "+numFound+" results instead of 1!");
530 		} else if (numFound==0) {
531 			return null; // no product found
532 		} else {
533 			return queryResponse.getCompleteProducts().get(0);
534 		}
535 			
536 	}
