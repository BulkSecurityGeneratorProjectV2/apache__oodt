59  /**
60   * <pre>
61   *    Will crawl external directory structures and will download the files within these structures.
62   *
63   *    This class's settings are set using a java .properties file which can be read in and parsed by Config.java.
64   *    This .properties file should have the following properties set:
65   *
66   *   	{@literal #list of sites to crawl
67   *   	protocol.external.sources=&lt;path-to-xml-file&gt;
68   *
69   *   	#protocol types
70   *   	protocolfactory.types=&lt;list-of-protocols-separated-by-commas&gt; (e.g. ftp,http,https,sftp)
71   *
72   *   	#Protocol factories per types (must have one for each protocol mention in protocolfactory.types -- the property must be name
73   *    	# as such: protocolfactory.&lt;name-of-protocol-type&gt;
74   *   	protocolfactory.ftp=&lt;path-to-java-protocolfactory-class&gt; (e.g. org.apache.oodt.cas.protocol.ftp.FtpClientFactory)
75   *   	protocolfactory.http=&lt;path-to-java-protocolfactory-class&gt;
76   *   	protocolfactory.https=&lt;path-to-java-protocolfactory-class&gt;
77   *   	protocolfactory.sftp=&lt;path-to-java-protocolfactory-class&gt;
78   *
79   *   	#configuration to make java.net.URL accept unsupported protocols -- must exist just as shown
80   *   	java.protocol.handler.pkgs=org.apache.oodt.cas.url.handlers
81   *    }
82   *
83   *    In order to specify which external sites to crawl you must create a XML file which contains the
84   *    the site and necessary information needed to crawl the site, such as username and password.
85   *    protocol.external.sources must contain the path to this file so the crawl knows where to find it.
86   *    You can also train this class on how to crawl each given site.  This is also specified in an XML
87   *    file, whose path must be given in the first mentioned XML file which contians the username and password.
88   *
89   *    Then schema for the external sites XML file is as such:
90   *
91   *    	{@literal &lt;sources&gt;
92   *    	   &lt;source url=&quot;url-of-server&quot;&gt;
93   *    	      &lt;username&gt;username&lt;/username&gt;
94   *    	      &lt;password&gt;password&lt;/password&gt;
95   *    	      &lt;dirstruct&gt;path-to-xml-file&lt;/dirstruct&gt;
96   *    	      &lt;crawl&gt;yes-or-no&lt;/crawl&gt;
97   *    	   &lt;/source&gt;
98   *    	   ...
99   *    	   ...
100  *    	   ...
101  *    	&lt;/sources\&gt;}
102  *
103  *    You may specify as many sources as you would like by specifying multiple {@literal &lt;source&gt;} tags.
104  *    In the {@literal &lt;source&gt;} tag, the parameter 'url' must be specified.  This is the url of the server
105  *    you want the crawler to connect to.  It should be of the following format:
106  *    {@literal &lt;protocol&gt;://&lt;host&gt;} (e.g. sftp://remote.computer.gov)
107  *    If no username and password exist, then these elements can be omitted (they are optional).
108  *    For {@literal &lt;crawl&gt;} place yes or no here.  This is for convenience of being able to keep record of the
109  *    sites and their information in this XML file even if you decide that you no longer need to crawl it
110  *    anymore (just put {@literal &lt;crawl&gt;no&lt;/crawl&gt;} and the crawl will not crawl that site).
111  *    {@literal &lt;dirStruct&gt;} contains a path to another XML file which is documented in DirStruct.java javadoc.  This
112  *    element is optional.  If no {@literal &lt;dirStruct&gt;} is given, then every directory will be crawled on the site
113  *    and every encountered file will be downloaded.
114  * </pre>
115  *
116  * @author bfoster (Brian Foster)
117  */

225     /**
226      * Initializes variables that must be reset when more than one crawl is done
227      *
228      * @throws ThreadEvaluatorException

230     void resetVariables() {
231         numberOfSessions = 0;
232         stagingAreas = new HashSet<File>();
233         avaliableSessions = new Vector<Protocol>();
234         currentlyDownloading = new HashSet<ProtocolFile>();
235         failedDownloadList = new LinkedList<ProtocolFile>();
236         max_allowed_failed_downloads = config.getMaxFailedDownloads();
237         max_sessions = config.getRecommendedThreadCount();
238         threadController = new ThreadPoolExecutor(this.max_sessions,
239                 this.max_sessions, EXTRA_LAZY_SESSIONS_TIMEOUT,
240                 TimeUnit.SECONDS, new LinkedBlockingQueue<Runnable>());
241         if (config.useTracker()) {
242           dtEval = new DownloadThreadEvaluator(this.absMaxAllowedSessions);
243         }
244     }

258     public List<RemoteSiteFile> getNextPage(final RemoteSiteFile dir,
259             final ProtocolFileFilter filter) throws RemoteConnectionException {
260         for (int i = 0; i < 3; i++) {
261             try {
262                 return protocolHandler.nextPage(dir.getSite(), protocolHandler
263                         .getAppropriateProtocol(dir, true, true),
264                         new ProtocolFileFilter() {
265                             @Override
266                            public boolean accept(ProtocolFile file) {
267                                 return filter.accept(file)
268                                         && !FileRetrievalSystem.this
269                                                 .isDownloading(file);
270                             }
271                         });
272             } catch (Exception e) {
273                 LOG.log(Level.WARNING, "Retrying to get next page for " + dir
274                         + " because operation failed : " + e.getMessage(), e);
275             }
276         }
277         throw new RemoteConnectionException("Failed to get next page for "
278                 + dir);
279     }

353     // returns true if download was added to queue. . .false otherwise
354     public boolean addToDownloadQueue(RemoteSite remoteSite, String file,
355             String renamingString, File downloadToDir,
356             String uniqueMetadataElement, boolean deleteAfterDownload, Metadata fileMetadata)
357             throws ToManyFailedDownloadsException, RemoteConnectionException,
358         ProtocolException,
359             AlreadyInDatabaseException, UndefinedTypeException,
360             CatalogException, IOException {
361         if (validate(remoteSite)) {
362             if (!file.startsWith("/")) {
363               file = "/" + file;
364             }
365             return addToDownloadQueue(protocolHandler.getProtocolFileFor(remoteSite,
366                     protocolHandler.getAppropriateProtocolBySite(remoteSite,
367                             true), file, false), renamingString, downloadToDir,
368                     uniqueMetadataElement, deleteAfterDownload, fileMetadata);
369         } else {
370           throw new ProtocolException("Not a valid remote site " + remoteSite);
371         }
372     }

374     public boolean validate(RemoteSite remoteSite) {
375         Preconditions.checkNotNull(remoteSite);
376         LinkedList<RemoteSite> remoteSites = this.siteInfo
377                 .getPossibleRemoteSites(remoteSite.getAlias(), remoteSite
378                         .getURL(), remoteSite.getUsername(), remoteSite
379                         .getPassword());
380         if (remoteSites.size() == 1) {
381             RemoteSite rs = remoteSites.get(0);
382             remoteSite.copy(rs);
383             return true;
384         }
385         return false;
386     }

388     public void waitUntilAllCurrentDownloadsAreComplete()
389             throws ProtocolException {
390         synchronized (this) {
391             for (int i = 0; i < INT; i++) {
392                 try {
393                     if (this.avaliableSessions.size() == this.numberOfSessions) {
394                       return;
395                     } else {
396                       this.wait(TIMEOUT);
397                     }
398                 } catch (Exception ignored) {
399                 }
400             }
401             throw new ProtocolException(
402                     "Downloads appear to be hanging . . . aborting wait . . . waited for 15 minutes");
403         }
404     }

406   public boolean addToDownloadQueue(RemoteSiteFile file,
407                                     String renamingString,
408                                     File downloadToDir,
409                                     String uniqueMetadataElement,
410                                     boolean deleteAfterDownload,
411                                     Metadata fileMetadata) throws ToManyFailedDownloadsException,
412                                                                   RemoteConnectionException,
413                                                                   AlreadyInDatabaseException,
414                                                                   UndefinedTypeException,
415                                                                   CatalogException,
416                                                                   IOException {
417         if (this.failedDownloadList.size() > max_allowed_failed_downloads) {
418           throw new ToManyFailedDownloadsException(
419               "Number of failed downloads exceeds "
420               + max_allowed_failed_downloads
421               + " . . . blocking all downloads from being added to queue . . . "
422               + "reset error flag in order to force allow downloads into queue");
423         }
424         if (this.isDownloading(file)) {
425             LOG.log(Level.WARNING, "Skipping file '" + file
426                     + "' because it is already on the download queue");
427             return false;
428         }
429 
430         RemoteFile remoteFile = new RemoteFile(file);
431         remoteFile.addMetadata(fileMetadata);
432         remoteFile.addMetadata(RemoteFile.RENAMING_STRING, renamingString);
433         remoteFile.addMetadata(RemoteFile.DELETE_AFTER_DOWNLOAD,
434                 deleteAfterDownload + "");
435 
436         if (config.onlyDownloadDefinedTypes()) {
437            String mimeType = this.mimeTypeDetection.getMimeType(file.getName());
438            if (mimeType != null
439                    && !mimeType.equals("application/octet-stream")) {
440                remoteFile.addMetadata(RemoteFile.MIME_TYPE, mimeType);
441                remoteFile.addMetadata(RemoteFile.SUPER_TYPE, this.mimeTypeDetection
442                        .getSuperTypeForMimeType(mimeType));
443                String description = this.mimeTypeDetection
444                        .getDescriptionForMimeType(mimeType);
445                if (!Strings.isNullOrEmpty(description)) {
446                  if(description.contains("&")){
447                    for (String field : description.split("\\&\\&")) {
448                      String[] keyval = field.split("\\=");
449                      remoteFile.addMetadata(keyval[0].trim(), keyval[1].trim());
450                    }
451                  } else{
452                    // it's the ProductType
453                    remoteFile.addMetadata(RemoteFile.PRODUCT_TYPE, description);
454                  }
455                  if (remoteFile.getMetadata(RemoteFile.UNIQUE_ELEMENT) != null) {
456                     uniqueMetadataElement = remoteFile.getMetadata(RemoteFile.UNIQUE_ELEMENT);
457                  }
458                }
459            } else {
460               throw new UndefinedTypeException("File '" + file
461                     + "' is not a defined type");
462            }
463         }
464 
465         downloadToDir = new File(downloadToDir.isAbsolute() ? downloadToDir
466                 .getAbsolutePath() : this.config.getBaseStagingArea() + "/"
467                 + downloadToDir.getPath());
468         if (!this.isStagingAreaInitialized(downloadToDir)) {
469           this.initializeStagingArea(downloadToDir);
470         }
471 
472         remoteFile.addMetadata(RemoteFile.DOWNLOAD_TO_DIR, downloadToDir.getAbsolutePath());
473 
474     	if (remoteFile.getMetadata(RemoteFile.PRODUCT_NAME_GENERATOR) != null) {
475     		remoteFile.addMetadata(RemoteFile.PRODUCT_NAME, RenamingConvention.rename(remoteFile, remoteFile.getMetadata(RemoteFile.PRODUCT_NAME_GENERATOR)));
476     	}else {
477     		remoteFile.setUniqueMetadataElement(uniqueMetadataElement == null ? RemoteFile.FILENAME : uniqueMetadataElement);
478     	}
479 
480         if (!isAlreadyInDatabase(remoteFile)) {
481 
482             // get download location
483             File newFile = getSaveToLoc(remoteFile);
484 
485             // add session to thread pool
486             if (!this.isInStagingArea(newFile)) {
487                 for (int retries = 0;; retries++) {
488                     try {
489                         addSessionToThreadPool(
490                                 getNextAvaliableSession(remoteFile
491                                         .getProtocolFile()), remoteFile,
492                                 newFile);
493                         return true;
494                     } catch (Exception e) {
495                         if (retries < MAX_RETRIES) {
496                             LOG.log(Level.WARNING, "Failed to get session for "
497                                     + file + " . . . retrying in 5 secs");
498                             synchronized (this) {
499                                 try {
500                                     wait(TIMEOUT1);
501                                 } catch (Exception ignored) {
502                                 }
503                             }
504                         } else {
505                             this.failedDownloadList.add(file);
506                             throw new RemoteConnectionException(
507                                     "Failed to get session to download " + file
508                                             + " : " + e.getMessage(), e);
509                         }
510                     }
511                 }
512             } else {
513                 if (deleteAfterDownload) {
514                     try {
515                         protocolHandler
516                                 .delete(protocolHandler.getAppropriateProtocol(
517                                         file, true, true), file);
518                     } catch (Exception e) {
519                         LOG.log(Level.SEVERE,
520                                 "Failed to delete file from server : "
521                                         + e.getMessage());
522                     }
523                 }
524                 LOG.log(Level.WARNING, "Skipping file " + file
525                         + " because it is already in staging area");
526                 return false;
527             }
528         } else {
529           throw new AlreadyInDatabaseException("File " + file
530                                                + " is already the database");
531         }
532     }

538     private boolean isInStagingArea(final File findFile) {
539         return (findFile.exists() || new File(findFile.getParentFile(),
540                 "Downloading_" + findFile.getName()).exists());
541     }

543     private void initializeStagingArea(File stagingArea) throws IOException {
544         LOG.log(Level.INFO, "Preparing staging area " + stagingArea);
545         if (stagingArea.exists()) {
546             File[] failedDownloads = stagingArea.listFiles(new FileFilter() {
547                 @Override
548                public boolean accept(File pathname) {
549                     return pathname.getName().startsWith("Downloading_");
550                 }
551             });
552             if(failedDownloads!=null) {
553               for (File file : failedDownloads) {
554                 LOG.log(Level.INFO, "Removing failed download file "
555                                     + file.getAbsolutePath());
556                 file.delete();
557               }
558             }
559         } else {
560             LOG.log(Level.INFO, "Staging area " + stagingArea.getAbsolutePath()
561                     + " does not exist! -- trying to create it ");
562             if (!stagingArea.mkdirs()) {
563               throw new IOException("Failed to create staging area at "
564                                     + stagingArea.getAbsolutePath());
565             }
566         }
567         this.stagingAreas.add(stagingArea);
568     }

570     File getSaveToLoc(RemoteFile remoteFile) {
571         String renamingString = remoteFile
572                 .getMetadata(RemoteFile.RENAMING_STRING);
573         if (renamingString == null || renamingString.equals("")) {
574             return new File(remoteFile.getMetadata(RemoteFile.DOWNLOAD_TO_DIR)
575                     + "/" + remoteFile.getMetadata(RemoteFile.FILENAME));
576         } else {
577             File newFile = new File(remoteFile
578                     .getMetadata(RemoteFile.DOWNLOAD_TO_DIR)
579                     + "/"
580                     + RenamingConvention.rename(remoteFile, renamingString));
581             if (!newFile.getParentFile().equals(
582                     remoteFile.getMetadata(RemoteFile.DOWNLOAD_TO_DIR))) {
583               newFile.getParentFile().mkdirs();
584             }
585             return newFile;
586         }
587     }

589     Protocol getNextAvaliableSession(RemoteSiteFile file) throws CrawlerException {
590         // wait for available session, then load it
591         Protocol session;
592         while ((session = getSession(file)) == null) {
593             try {
594                 waitMainThread();
595             } catch (InterruptedException ignored) {
596             }
597         }
598         return session;
599     }

628     /**
629      * Gets an available downloading session Protocol. Returns null if none are
630      * available
631      *
632      * @return The found downloading session Protocol
633      * @throws RemoteCommunicationException
634      *             If downloading session Protocol has to be reconnected and
635      *             there is an error communicating with the server
636      */
637     synchronized Protocol getSession(RemoteSiteFile file) throws CrawlerException {
638         try {
639             Protocol session = null;
640 			if (file.getSite().getMaxConnections() < 0
641 					|| file.getSite().getMaxConnections() > this.getCurrentlyDownloadingFiles().size()) {
642 	            if (avaliableSessions.size() > 0) {
643 	                session = modifyAvailableSessionForPath(file);
644 	            } else if (numberOfSessions < max_sessions) {
645 	                session = createNewSessionForPath(file);
646 	                incrementSessions();
647 	            }
648             }
649             return session;
650         } catch (Exception e) {
651             throw new CrawlerException("Failed to get new session : "
652                     + e.getMessage(), e);
653         }

725     void addSessionToThreadPool(final Protocol session,
726             final RemoteFile remoteFile, final File newFile) {
727         this.addToDownloadingList(remoteFile.getProtocolFile());
728         threadController.execute(new Runnable() {
729             @Override
730             public void run() {
731                 boolean successful = false;
732                 int retries = 0;
733                 Protocol curSession = session;
734 
735                 if (FileRetrievalSystem.this.dListener != null) {
736                   FileRetrievalSystem.this.dListener
737                       .downloadStarted(remoteFile.getProtocolFile());
738                 }
739 
740                 // try until successful or all retries have been used
741                 do {
742                     try {
743                         // if thread tracker is to be used
744                         if (config.useTracker()) {
745                             dtEval.startTrackingDownloadRuntimeForFile(newFile);
746                             protocolHandler.download(curSession, remoteFile
747                                     .getProtocolFile(), newFile, remoteFile
748                                     .getMetadata(
749                                             RemoteFile.DELETE_AFTER_DOWNLOAD)
750                                     .equals("true"));
751                             dtEval.fileDownloadComplete(newFile);
752                             threadController
753                                     .setCorePoolSize(max_sessions = dtEval
754                                             .getRecommendedThreadCount());
755                             threadController.setMaximumPoolSize(max_sessions);
756                             // if static number of threads are to be used
757                         } else {
758                             protocolHandler.download(curSession, remoteFile
759                                     .getProtocolFile(), newFile, remoteFile
760                                     .getMetadata(
761                                             RemoteFile.DELETE_AFTER_DOWNLOAD)
762                                     .equals("true"));
763                         }
764 
765                         successful = true;
766                         if (FileRetrievalSystem.this.dListener != null) {
767                           FileRetrievalSystem.this.dListener
768                               .downloadFinished(remoteFile
769                                   .getProtocolFile());
770                         }
771 
772                         remoteFile.addMetadata(RemoteFile.FILE_SIZE, newFile
773                                 .length()
774                                 + "");
775 
776                         // try to create the metadata file
777                         if (config.getWriteMetFile()) {
778 	                        try {
779 	                        	LOG.log(Level.INFO, "Writing metadata file for '" + newFile + "'");
780 	                            remoteFile.addMetadata(RemoteFile.FILE_SIZE,
781 	                                    newFile.length() + "");
782 	                            remoteFile.writeToPropEqValFile(newFile
783 	                                    .getAbsolutePath()
784 	                                    + "." + config.getMetFileExtension(),
785 	                                    config.getListOfMetadataToOutput());
786 	                        } catch (Exception e) {
787 	                            LOG.log(Level.SEVERE,
788 	                                    "Failed to create metadata file for "
789 	                                            + remoteFile.getProtocolFile());
790 	                        }
791                         }
792 
793                     } catch (Exception e) {
794 
795                         // if tracker is being used cancel tracking
796                         if (config.useTracker()) {
797                           dtEval.cancelRuntimeTracking(newFile);
798                         }
799 
800                         // delete any created file from staging area
801                         newFile.delete();
802                         new File(newFile.getAbsolutePath() + "."
803                                 + config.getMetFileExtension()).delete();
804 
805                         // check if a retry is still allowed
806                         if (++retries > MAX_RETRIES) {
807                             FileRetrievalSystem.this.failedDownloadList
808                                     .add(remoteFile.getProtocolFile());
809                             LOG.log(Level.SEVERE, "Failed to download "
810                                     + remoteFile.getProtocolFile() + " : "
811                                     + e.getMessage());
812                             if (FileRetrievalSystem.this.dListener != null) {
813                               FileRetrievalSystem.this.dListener
814                                   .downloadFailed(remoteFile
815                                       .getProtocolFile(), e
816                                       .getMessage());
817                             }
818                             break;
819                         } else if (FileRetrievalSystem.this.failedDownloadList
820                                 .size() < max_allowed_failed_downloads) {
821                             // discard current session and recreate a new
822                             // session to try to download file with
823                             LOG.log(Level.WARNING, "Retrying to download file "
824                                     + remoteFile.getProtocolFile()
825                                     + " because download failed : "
826                                     + e.getMessage(), e);
827                             try {
828                                 protocolHandler.disconnect(curSession);
829                             } catch (Exception ignored) {
830                             }
831                             try {
832                                 curSession = protocolHandler
833                                         .getAppropriateProtocol(remoteFile
834                                                 .getProtocolFile(), false, true);
835                             } catch (Exception exc) {
836                                 LOG.log(Level.SEVERE,
837                                         "Failed to reconnect protocol to retry download of file "
838                                                 + remoteFile.getProtocolFile()
839                                                 + " -- aborting retry : "
840                                                 + e.getMessage(), e);
841                             }
842                         } else {
843                             LOG
844                                     .log(
845                                             Level.SEVERE,
846                                             "Terminating download tries for file "
847                                                     + remoteFile
848                                                             .getProtocolFile()
849                                                     + " do to too many previous download failures : "
850                                                     + e.getMessage(), e);
851                             if (FileRetrievalSystem.this.dListener != null) {
852                               FileRetrievalSystem.this.dListener
853                                   .downloadFailed(remoteFile
854                                       .getProtocolFile(), e
855                                       .getMessage());
856                             }
857                             break;
858                         }
859 
860                     }
861                 } while (!successful);
862 
863                 FileRetrievalSystem.this.removeFromDownloadingList(remoteFile
864                         .getProtocolFile());
865                 determineSessionFate(curSession);
866             }
867         });
