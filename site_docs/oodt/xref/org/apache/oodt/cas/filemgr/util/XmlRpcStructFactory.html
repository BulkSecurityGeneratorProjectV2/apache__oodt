64    public static Map<String, Object> getXmlRpcFileTransferStatus(
65        FileTransferStatus status) {
66      Map<String, Object> statusHash = new Hashtable<String, Object>();
67      statusHash.put("bytesTransferred", Long.toString(status
68          .getBytesTransferred()));
69      statusHash.put("parentProduct", getXmlRpcProduct(status
70          .getParentProduct()));
71      statusHash.put("fileRef", getXmlRpcReference(status.getFileRef()));
72      return statusHash;
73    }

76    public static FileTransferStatus getFileTransferStatusFromXmlRpc(
77        Map<String, Object> statusHash) {
78      FileTransferStatus status = new FileTransferStatus();
79      status.setBytesTransferred(Long.parseLong(statusHash
80          .get("bytesTransferred").toString()));

82      status.setFileRef(getReferenceFromXmlRpc((Hashtable<String, Object>) statusHash.get("fileRef")));
83      return status;

134   public static Map<String, Object> getXmlRpcComplexQuery(ComplexQuery complexQuery) {
135     Map<String, Object> complexQueryHash = getXmlRpcQuery(complexQuery);
136     if (complexQuery.getReducedProductTypeNames() != null) {
137       complexQueryHash
138           .put("reducedProductTypeNames", new Vector<String>(complexQuery.getReducedProductTypeNames()));
139     } else {
140       complexQueryHash.put("reducedProductTypeNames", new Vector<String>());
141     }
142     if (complexQuery.getReducedMetadata() != null) {
143       complexQueryHash.put("reducedMetadata", new Vector<String>(complexQuery.getReducedMetadata()));
144     } else {
145       complexQueryHash.put("reducedMetadata", new Vector<String>());
146     }
147     if (complexQuery.getSortByMetKey() != null) {
148       complexQueryHash.put("sortByMetKey", complexQuery.getSortByMetKey());
149     }
150     if (complexQuery.getToStringResultFormat() != null) {
151       complexQueryHash.put("toStringResultFormat", complexQuery.getToStringResultFormat());
152     }
153     if (complexQuery.getQueryFilter() != null) {
154       complexQueryHash.put("queryFilter", getXmlRpcQueryFilter(complexQuery.getQueryFilter()));
155     }
156     return complexQueryHash;
157   }

159   @SuppressWarnings("unchecked")
160   public static ComplexQuery getComplexQueryFromXmlRpc(Map<String, Object> complexQueryHash) {
161     ComplexQuery complexQuery = new ComplexQuery();
162     complexQuery.setCriteria(getQueryFromXmlRpc(complexQueryHash).getCriteria());
163     if (((Vector<String>) complexQueryHash.get("reducedProductTypeNames")).size() > 0) {
164       complexQuery.setReducedProductTypeNames((Vector<String>) complexQueryHash.get("reducedProductTypeNames"));
165     }
166     if (((Vector<String>) complexQueryHash.get("reducedMetadata")).size() > 0) {
167       complexQuery.setReducedMetadata((Vector<String>) complexQueryHash.get("reducedMetadata"));
168     }
169     complexQuery.setSortByMetKey((String) complexQueryHash.get("sortByMetKey"));
170     complexQuery.setToStringResultFormat((String) complexQueryHash.get("toStringResultFormat"));
171     if (complexQueryHash.get("queryFilter") != null) {
172       complexQuery.setQueryFilter(
173           getQueryFilterFromXmlRpc((Map<String, Object>) complexQueryHash.get("queryFilter")));
174     }
175     return complexQuery;
176   }

178   public static Map<String, Object> getXmlRpcQueryFilter(QueryFilter queryFilter) {
179     Map<String, Object> queryFilterHash = new ConcurrentHashMap<String, Object>();
180     queryFilterHash.put("startDateTimeMetKey", queryFilter.getStartDateTimeMetKey());
181     queryFilterHash.put("endDateTimeMetKey", queryFilter.getEndDateTimeMetKey());
182     queryFilterHash.put("priorityMetKey", queryFilter.getPriorityMetKey());
183     queryFilterHash.put("filterAlgor", getXmlRpcFilterAlgor(queryFilter.getFilterAlgor()));
184     queryFilterHash.put("versionConverterClass", queryFilter.getConverter().getClass().getCanonicalName());
185     return queryFilterHash;
186   }

188   public static QueryFilter getQueryFilterFromXmlRpc(Map<String, Object> queryFilterHash) {
189     String startDateTimeMetKey = (String) queryFilterHash.get("startDateTimeMetKey");
190     String endDateTimeMetKey = (String) queryFilterHash.get("endDateTimeMetKey");
191     String priorityMetKey = (String) queryFilterHash.get("priorityMetKey");
192     @SuppressWarnings("unchecked")
193     FilterAlgor filterAlgor = getFilterAlgorFromXmlRpc((Map<String, Object>) queryFilterHash.get("filterAlgor"));
194     QueryFilter queryFilter = new QueryFilter(startDateTimeMetKey, endDateTimeMetKey, priorityMetKey, filterAlgor);
195     queryFilter.setConverter(GenericFileManagerObjectFactory
196         .getVersionConverterFromClassName((String) queryFilterHash.get("versionConverterClass")));
197     return queryFilter;
198   }

253   public static Map<String, Object> getXmlRpcProduct(Product product) {
254     Map<String, Object> productHash = new Hashtable<String, Object>();
255     if (product.getProductId() != null) {
256       productHash.put("id", product.getProductId());
257     }
258     if (product.getProductName() != null) {
259       productHash.put("name", product.getProductName());
260     }
261     if (product.getProductType() != null) {
262       productHash.put("type", getXmlRpcProductType(product.getProductType()));
263     }
264     if (product.getProductStructure() != null) {
265       productHash.put("structure", product.getProductStructure());
266     }
267     if (product.getTransferStatus() != null) {
268       productHash.put("transferStatus", product.getTransferStatus());
269     }
270     if (product.getProductReferences() != null) {
271       productHash.put("references", getXmlRpcReferences(product
272           .getProductReferences()));
273     }
274     if (product.getRootRef() != null) {
275       productHash.put("rootReference", getXmlRpcReference(product
276           .getRootRef()));
277     }
278     return productHash;
279   }

281   @SuppressWarnings("unchecked")
282   public static Product getProductFromXmlRpc(Map<?, ?> productHash) {
283     Product product = new Product();
284     product.setProductId((String) productHash.get("id"));
285     product.setProductName((String) productHash.get("name"));
286     if (productHash.get("type") != null) {
287       product.setProductType(getProductTypeFromXmlRpc(
288           (Map<String, Object>) productHash.get("type")));
289     }
290     product.setProductStructure((String) productHash.get("structure"));
291     product.setTransferStatus((String) productHash.get("transferStatus"));
292     if (productHash.get("references") != null) {
293       product.setProductReferences(getReferencesFromXmlRpc(
294           (Vector<Map<String, Object>>) productHash
295               .get("references")));
296     }
297     if (productHash.get("rootReference") != null) {
298       product.setRootRef(getReferenceFromXmlRpc(
299           (Map<String, Object>) productHash.get("rootReference")));
300     }
301     return product;
302   }

355   public static Map<String, Object> getXmlRpcProductType(ProductType type) {
356     Map<String, Object> productTypeHash = new Hashtable<String, Object>();
357     // TODO(bfoster): ProductType ID is currently required by XmlRpcFileManager.
358     productTypeHash.put("id", type.getProductTypeId());
359     if (type.getName() != null) {
360       productTypeHash.put("name", type.getName());
361     }
362     if (type.getDescription() != null) {
363       productTypeHash.put("description", type.getDescription());
364     }
365     if (type.getProductRepositoryPath() != null) {
366       productTypeHash.put("repositoryPath", type.getProductRepositoryPath());
367     }
368     if (type.getVersioner() != null) {
369       productTypeHash.put("versionerClass", type.getVersioner());
370     }
371     if (type.getTypeMetadata() != null) {
372       productTypeHash.put("typeMetadata", type.getTypeMetadata().getHashTable());
373     }
374     if (type.getExtractors() != null) {
375       productTypeHash.put("typeExtractors", getXmlRpcTypeExtractors(type.getExtractors()));
376     }
377     if (type.getHandlers() != null) {
378       productTypeHash.put("typeHandlers", getXmlRpcTypeHandlers(type.getHandlers()));
379     }
380     return productTypeHash;
381   }

495   public static Properties getPropertiesFromXmlRpc(Map<String, String> propHash) {
496     Properties props = new Properties();
497 
498     if (propHash != null && propHash.keySet().size() > 0) {
499       for (Map.Entry<String, String> propName : propHash.entrySet()) {
500         String propValue = propName.getValue();
501         props.setProperty(propName.getKey(), propValue);
502       }
503     }
504 
505     return props;
506   }

508   public static Map<String, String> getXmlRpcProperties(Properties props) {
509     Map<String, String> propHash = new Hashtable<String, String>();
510 
511     if (props != null && props.keySet().size() > 0) {
512       for (Object o : props.keySet()) {
513         String propName = (String) o;
514         String propValue = props.getProperty(propName);
515         propHash.put(propName, propValue);
516       }
517     }
518 
519     return propHash;
520   }

547   public static Map<String, Object> getXmlRpcReference(Reference reference) {
548     Map<String, Object> referenceHash = new Hashtable<String, Object>();
549     referenceHash.put("origReference", reference.getOrigReference());
550     referenceHash.put("dataStoreReference", reference
551                                                 .getDataStoreReference() != null ? reference
552                                                 .getDataStoreReference() : "");
553     referenceHash.put("fileSize",
554         Long.toString(reference.getFileSize()));
555     referenceHash.put("mimeType", (reference.getMimeType() == null) ? ""
556                                                                     : reference.getMimeType().getName());
557     return referenceHash;
