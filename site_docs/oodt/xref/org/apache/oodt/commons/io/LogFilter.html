19  import java.util.HashMap;

21  /** A filter for log messages.
22   *
23   * This is a {@link LogListener} that filters out and passes through certain categories of
24   * log messages to another <code>LogListener</code>.  When you construct this listener,
25   * you pass a boolean flag that indicates its pass-through mode: if true, it passes log
26   * messages by default and filters out specified categories; if false, it filters out
27   * messages by default and passes through specified categories.
28   *
29   * <p>Note that this filter only affects events sent to {@link LogListener#messageLogged}.
30   * Events sent to {@link LogListener#streamStarted} and {@link LogListener#streamStopped}
31   * are passed through regardless.
32   *
33   * <p>Categories used by this filter should implement their {@link Object#hashCode} and
34   * {@link Object#equals} methods.
35   *
36   * @see Log
37   * @author Kelly
38   */
39  public class LogFilter implements LogListener {
40  	/** Create a log filter.
41  	 *
42  	 * The log filter passes through and filters out messages before sending them onto
43  	 * the given log listener.
44  	 *
45  	 * @param listener What object will get filtered log messages.
46  	 * @param passThrough If true, pass messages through by default and filter out the
47  	 * specified <var>categories</var> of messages. If false, filter out messages by
48  	 * default and pass through the specified <var>categories</var>.
49  	 * @param categories Categories of messages to filter out (if
50  	 * <var>passThrough</var> is true) or to pass through (if <var>passThrough</var>
51  	 * is false).
52  	 */
53  	public LogFilter(LogListener listener, boolean passThrough, Object[] categories) {
54  		if (listener == null) {
55  		  throw new IllegalArgumentException("Can't filter messages to a null listener");
56  		}
57  		this.listener = listener;
58  		this.passThrough = passThrough;
59  		if (categories == null) {
60  		  return;
61  		}
62  	  for (Object category : categories) {
63  		this.categories.put(category, DUMMY);
64  	  }
65  	}
66  
67  	/** Create a log filter.
68  	 *  
69  	 * The log filter passes through and filters out messages before sending them onto
70  	 * the given log listener. The filter starts out empty (with no categories).
71  	 *
72  	 * @param listener What object will get filtered log messages.
73  	 * @param passThrough If true, pass messages through by default. If false, filter out messages by default.
74  	 */
75  	public LogFilter(LogListener listener, boolean passThrough) {
76  		this(listener, passThrough, /*categories*/null);
77  	}
78  
79  	/** Add an additional category.
80  	 *
81  	 * If the filter is in pass-through mode, messages in this category will be
82  	 * filtered out. If the filter is in filter mode, messages in this category will
83  	 * be passed through.
84  	 *
85  	 * @param category The category to add.
86  	 */
87  	public void addCategory(Object category) {
88  		categories.put(category, DUMMY);
89  	}
90  
91  	/** Remove a category.
92  	 *
93  	 * If the category isn't in the filter, nothing happens.
94  	 *
95  	 * @param category The category to remove.
96  	 */
97  	public void removeCategory(Object category) {
98  		categories.remove(category);
99  	}
100 
101 	/** Pass on the event unmodified to the registered listener.
102 	 *
103 	 * @param event The event to pass.
104 	 */
105 	public void streamStarted(LogEvent event) {
106 		listener.streamStarted(event);
107 	}
108 
109 	/** Pass on the event unmodified to the registered listener.
110 	 *
111 	 * @param event The event to pass.
112 	 */
113 	public void streamStopped(LogEvent event) {
114 		listener.streamStopped(event);
115 	}
116 
117 	/** Filter the event, and possibly pass it onto the registered listener.
118 	 *
119 	 * @param event The event to filter.
120 	 */
121 	public void messageLogged(LogEvent event) {
122 		boolean found = categories.containsKey(event.getCategory());
123 		if ((passThrough && !found) || (!passThrough && found)) {
124 		  listener.messageLogged(event);
125 		}
126 	}
127 
128 	/** Ignore this event.
129 	 */
130 	public void propertyChange(java.beans.PropertyChangeEvent ignore) {}
131 
132 	/** If true, pass through by default, otherwise filter out by default.
133 	 */
134 	protected boolean passThrough;
135 
136 	/** Table of categories to filter/pass.
137 	 *
138 	 * This table maps all values to {@link #DUMMY}.  In Java2, we can get rid of
139 	 * <code>DUMMY</code> and use a {@link java.util.HashSet} instead.
140 	 */
141 	protected HashMap categories = new HashMap();
142 
143 	/** The DUMMY value for all mappings in the {@link #categories} table.
144 	 */
145 	protected static final Object DUMMY = new Object();
146 
147 	/** The listener on whose behalf we filter.
148 	 */
149 	protected LogListener listener;
150 }
