40  /**
41   * A wrapper class to act as a facade interface to all the different
42   * {@link Metadata} sources given to a PGE.
43   * 
44   * NOTE: 2 ways to update DYNAMIC metadata: 1) Create a key link to a DYNAMIC metadata
45   * key, then change the value of the key link or 2) add metadata then mark the
46   * key as dynamic and commit it.
47   * 
48   * @author bfoster (Brian Foster)
49   * @author mattmann (Chris Mattmann)
50   */
51  public class PgeMetadata {

95     /**
96      * Replaces or creates this {@link PgeMetadata}'s metadata with given
97      * {@link PgeMetadata}'s metadata. Also adds in the list of given
98      * {@link PgeMetadata}'s LOCAL metadata marked for promotion DYNAMIC
99      * metadata and list of key links.
100     * 
101     * @param pgeMetadata
102     *           A {@link PgeMetadata} whose metadata and key links will be added
103     *           to this {@link PgeMetadata}'s metadata and key links.
104     */
105    public void replaceMetadata(PgeMetadata pgeMetadata) {
106       Validate.notNull(pgeMetadata, "pgeMetadata cannot be null");
107 
108       staticMetadata.replaceMetadata(pgeMetadata.staticMetadata);
109       dynamicMetadata.replaceMetadata(pgeMetadata.dynamicMetadata);
110       localMetadata.replaceMetadata(pgeMetadata.localMetadata);
111 
112       keyLinkMap.putAll(pgeMetadata.keyLinkMap);
113       markedAsDynamicMetKeys.addAll(pgeMetadata.markedAsDynamicMetKeys);
114    }

116    /**
117     * Replaces or creates this {@link PgeMetadata}'s metadata with given
118     * {@link PgeMetadata}'s metadata. The provided "group" will be used to
119     * namespace the given {@link PgeMetadata}'s LOCAL metadata when add to this
120     * {@link PgeMetadata}'s LOCAL metadata. It will also namespace given
121     * {@link PgeMetadata}'s key links before adding then to this
122     * {@link PgeMetadata}'s key links. Also add in the list of given
123     * {@link PgeMetadata}'s LOCAL metadata marked for promotion DYNAMIC
124     * metadata.
125     * 
126     * @param pgeMetadata
127     *           A {@link PgeMetadata} whose metadata and key links will be added
128     *           to this {@link PgeMetadata}'s metadata and key links.
129     * @param group
130     *           The namespace which will be used to namespace given
131     *           {@link PgeMetadata}'s LOCAL metadata and key links before being
132     *           added to this {@link PgeMetadata}'s LOCAL metadata and key
133     *           links.
134     */
135    public void replaceMetadata(PgeMetadata pgeMetadata, String group) {
136       Validate.notNull(pgeMetadata, "pgeMetadata cannot be null");
137       Validate.notNull(group, "group cannot be null");
138 
139       staticMetadata.replaceMetadata(pgeMetadata.staticMetadata);
140       dynamicMetadata.replaceMetadata(pgeMetadata.dynamicMetadata);
141       localMetadata.replaceMetadata(group, pgeMetadata.localMetadata);
142 
143       // Namespace link keys that point to either importing
144       // metadata's local key or link key.
145       for (String keyLink : pgeMetadata.keyLinkMap.keySet()) {
146          String key = pgeMetadata.keyLinkMap.get(keyLink);
147          // Check if key is was local key or a link key
148          if (pgeMetadata.localMetadata.containsKey(key)
149                || pgeMetadata.keyLinkMap.containsKey(key)) {
150             key = group + "/" + key;
151          }
152          linkKey(group + "/" + keyLink, key);
153       }
154 
155       // Namespace workflow keys that point to either importing
156       // metadata's local key or link key.
157       for (String key : pgeMetadata.markedAsDynamicMetKeys) {
158          if (pgeMetadata.localMetadata.containsKey(key)
159                || pgeMetadata.keyLinkMap.containsKey(key)) {
160             key = group + "/" + key;
161          }
162          markAsDynamicMetadataKey(key);
163       }
164    }

166    /**
167     * Use to mark LOCAL keys which should be moved into DYNAMIC metadata when
168     * {@link #commitMarkedDynamicMetadataKeys(String...)} is invoked. If no 
169     * args are specified then all LOCAL metadata is marked for move to
170     * DYNAMIC metadata.
171     * 
172     * @param keys
173     *           Keys to mark as to be made DYNAMIC, otherwise if no keys then
174     *           all LOCAL metadata keys are mark for move to DYNAMIC.
175     */
176    public void markAsDynamicMetadataKey(String... keys) {
177       List<String> markedKeys = Lists.newArrayList(keys);
178       if (markedKeys.isEmpty()) {
179          markedKeys.addAll(localMetadata.getAllKeys());
180       }
181       markedAsDynamicMetKeys.addAll(markedKeys);
182    }
183 
184    /**
185     * Use to commit marked LOCAL keys to DYNAMIC keys. Specify a list of keys
186     * only if you want to limit the keys which get committed, otherwise all
187     * marked keys will be moved into DYNAMIC metadata.
188     * 
189     * @param keys
190     *           The list of marked LOCAL metadata keys which should be moved
191     *           into DYNAMIC metadata. If no keys are specified then all marked
192     *           keys are moved.
193     */
194    public void commitMarkedDynamicMetadataKeys(String... keys) {
195       Set<String> commitKeys = Sets.newHashSet(keys);
196       if (commitKeys.isEmpty()) {
197          commitKeys.addAll(markedAsDynamicMetKeys);
198       } else {
199          commitKeys.retainAll(markedAsDynamicMetKeys);
200       }
201       for (String key : commitKeys) {
202          dynamicMetadata.replaceMetadata(key,
203                localMetadata.getAllMetadata(resolveKey(key)));
204          localMetadata.removeMetadata(key);
205          markedAsDynamicMetKeys.remove(key);
206       }
207    }

214    /**
215     * Create a key which is a link to another key, such that if you get the
216     * metadata values for the created link it will return the current metadata
217     * values of the key it was linked to. NOTE: if the key's metadata values
218     * change, then the metadata values for the link key will also be the changed
219     * values. If you want to create a key which holds the current value of a
220     * key, then create a new metadata key.
221     * 
222     * @param keyLink
223     *           The name of the link key you wish to create.
224     * @param key
225     *           The key you which to link to (may also be a key link)
226     */
227    public void linkKey(String keyLink, String key) {
228       Validate.notNull(keyLink, "keyLink cannot be null");
229       Validate.notNull(key, "key cannot be null");
230 
231       localMetadata.removeMetadata(keyLink);
232       keyLinkMap.put(keyLink, key);
233    }

262    /**
263     * Find the actual key whose value will be returned for the given key. If the
264     * given key is a key (not a key link) then the given key will just be
265     * returned, otherwise it will trace through key link mapping to find the key
266     * which the given key link points to.
267     * 
268     * @param key
269     *           The name of a key or key link.
270     * @return The key whose value will be returned for the given key or key
271     *         link.
272     */
273    public String resolveKey(String key) {
274       Validate.notNull(key, "key cannot be null");
275 
276       while (keyLinkMap.containsKey(key)) {
277          key = keyLinkMap.get(key);
278       }
279       return key;
280    }

282    /**
283     * Determines the path by which the given key (if it is a key link) links to
284     * the key whose value it will return. If the given key is a key link and
285     * points to a key then the returning {@link List} will be of size 1 and will
286     * contain just that key. However, if the given key is a key link which
287     * points to another key link then the returning {@link List} will be greater
288     * than 1 (will depend on how many key links are connected before they actual
289     * point to a key. If the given key is a key, then the returning {@link List}
290     * will be empty.
291     * 
292     * @param key
293     *           The path to the key whose value will be returned for the give
294     *           key.
295     * @return A key path {@link List}.
296     */
297    public List<String> getReferenceKeyPath(String key) {
298       Validate.notNull(key, "key cannot be null");
299 
300       List<String> keyPath = Lists.newArrayList();
301       while (keyLinkMap.containsKey(key)) {
302          keyPath.add(key = keyLinkMap.get(key));
303       }
304       return keyPath;
305    }

313    /**
314     * Replace the given key's value with the given value. If the given key is a
315     * key link, then it will update the value of the key it is linked to if that
316     * key is DYNAMIC or LOCAL. If given key is a key link and it links to a
317     * STATIC key, then a new LOCAL key will be create.
318     * 
319     * @param key
320     *           The key or key link for whose value should be replaced.
321     * @param value
322     *           The value to give the given key. Will replace any existing value
323     *           or will be the value of a newly created LOCAL key.
324     */
325    public void replaceMetadata(String key, String value) {
326       Validate.notNull(key, "key cannot be null");
327       Validate.notNull(value, "value cannot be null");
328 
329       String resolveKey = resolveKey(key);
330       // If key is a key link which points to a DYNAMIC key then update the
331       // DYNAMIC key's value.
332       if (keyLinkMap.containsKey(key)
333             && dynamicMetadata.containsKey(resolveKey)) {
334          dynamicMetadata.replaceMetadata(resolveKey, value);
335       } else {
336          localMetadata.replaceMetadata(resolveKey, value);
337       }
338    }

385    /**
386     * Combines STATIC, DYNAMIC, and LOCAL metadata into one metadata object. You
387     * can restrict which metadata you want combined and change the order in
388     * which combining takes place by specifying Type arguments in the order you
389     * which precedence to be observed. For example, if you perform the
390     * following: pgeMetadata.asMetadata(LOCAL, STATIC) then only LOCAL and
391     * STATIC metadata will be combined and LOCAL metadata will trump STATIC
392     * metadata if they both contain the same key. If no arguments are specified
393     * then DEFAULT_COMBINE_ORDER is used.
394     * 
395     * @param types
396     *           The Type hierarchy you which to use when metadata is combined,
397     *           if no args then DEFAULT_COMBINE_ORDER is used.
398     * @return Combined metadata.
399     */
400    public Metadata asMetadata(Type... types) {
401       List<Type> combineOrder = Lists.newArrayList(types);
402       if (combineOrder.isEmpty()) {
403          combineOrder.addAll(DEFAULT_COMBINE_ORDER);
404       }
405 
406       Metadata combinedMetadata = new Metadata();
407       for (Type type : combineOrder) {
408          switch (type) {
409             case DYNAMIC:
410                combinedMetadata.replaceMetadata(dynamicMetadata);
411                break;
412             case STATIC:
413                combinedMetadata.replaceMetadata(staticMetadata);
414                break;
415             case LOCAL:
416                combinedMetadata.replaceMetadata(localMetadata);
417                for (String key : keyLinkMap.keySet()) {
418                   List<String> values = getAllMetadata(key);
419                   if (values != null) {
420                      combinedMetadata.replaceMetadata(key, values);
421                   }
422                }
423                break;
424          }
425       }
426       return combinedMetadata;
427    }

433    /**
434     * Get metadata values for given key. If Types are specified then it provides
435     * the precedence order in which to search for the key. If no Type args are
436     * specified then DEFAULT_QUERY_ORDER will be used. For example if
437     * you pass in Type args: STATIC, LOCAL then STATIC metadata will first be
438     * checked for the key and if it contains it, then it will return the found
439     * value, otherwise it will then check LOCAL metadata for the key and if it
440     * finds the value it will return it, otherwise null.
441     * 
442     * @param key
443     *           The key for whose metadata values should be returned.
444     * @param types
445     *           The type hierarchy which should be used, if no Types specified
446     *           DEFAULT_QUERY_ORDER will be used.
447     * @return Metadata values for given key.
448     */
449    public List<String> getAllMetadata(String key, Type... types) {
450       List<Type> queryOrder = Lists.newArrayList(types);
451       if (queryOrder.isEmpty()) {
452          queryOrder.addAll(DEFAULT_QUERY_ORDER);
453       }
454 
455       String useKey = resolveKey(key);
456       for (Type type : queryOrder) {
457          switch (type) {
458             case DYNAMIC:
459                if (dynamicMetadata.containsKey(useKey)) {
460                   return dynamicMetadata.getAllMetadata(useKey);
461                }
462                break;
463             case STATIC:
464                if (staticMetadata.containsKey(useKey)) {
465                   return staticMetadata.getAllMetadata(useKey);
466                }
467                break;
468             case LOCAL:
469                if (localMetadata.containsKey(useKey)) {
470                   return localMetadata.getAllMetadata(useKey);
471                }
472                break;
473          }
474       }
475       return new Vector<String>();
476    }
