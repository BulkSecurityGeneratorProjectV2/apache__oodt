608    private static ProductCrawler createDummyCrawler(
609          final boolean passesPreconditions, final Metadata productMetadata,
610          final File renamedFile) {
611       return new ProductCrawler() {
612          @Override
613          protected boolean passesPreconditions(File product) {
614             return passesPreconditions;
615          }
616          @Override
617          protected Metadata getMetadataForProduct(File product) {
618             return productMetadata;
619          }
620          @Override
621          protected File renameProduct(File product, Metadata productMetadata)
622                throws Exception {
623             return renamedFile == null ? product : renamedFile;
624          }
625       };
