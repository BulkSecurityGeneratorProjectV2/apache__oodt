110     public void testIngest() {
111         Metadata prodMet;
112 
113         try {
114             URL ingestUrl = this.getClass().getResource("/ingest");
115             URL refUrl = this.getClass().getResource("/ingest/test.txt");
116             URL metUrl = this.getClass().getResource("/ingest/test.txt.met");
117 
118             prodMet = new SerializableMetadata(new FileInputStream(
119                 new File(metUrl.getFile())));
120 
121             // now add the right file location
122             prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
123                 ingestUrl.getFile()).getCanonicalPath());
124             ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
125                 refUrl.getFile()), prodMet);
126         } catch (Exception e) {
127             fail(e.getMessage());
128         }
129 
130         // now make sure that the file is ingested
131         try {
132             XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(
133                     new URL("http://localhost:" + FM_PORT));
134             Product p = fmClient.getProductByName("test.txt");
135             assertNotNull(p);
136             assertEquals(Product.STATUS_RECEIVED, p.getTransferStatus());
137             assertTrue(fmClient.hasProduct("test.txt"));
138         } catch (Exception e) {
139             fail(e.getMessage());
140         }
141     }

209     private void deleteAllFiles(String startDir) {
210         File startDirFile = new File(startDir);
211         File[] delFiles = startDirFile.listFiles();
212 
213         if (delFiles != null && delFiles.length > 0) {
214             for (File delFile : delFiles) {
215                 delFile.delete();
216             }
217         }
218 
219         startDirFile.delete();
220 
221     }

223     private void startXmlRpcFileManager() {
224 
225         Properties properties = new Properties(System.getProperties());
226 
227         // first make sure to load properties for the file manager
228         // and make sure to load logging properties as well
229 
230         // set the log levels
231         URL loggingPropertiesUrl = this.getClass().getResource(
232             "/test.logging.properties");
233         properties.setProperty("java.util.logging.config.file", new File(
234             loggingPropertiesUrl.getFile()).getAbsolutePath());
235 
236         // first load the example configuration
237         try {
238             URL filemgrPropertiesUrl = this.getClass().getResource(
239                 "/filemgr.properties");
240             properties.load(
241                 new FileInputStream(new File(filemgrPropertiesUrl.getFile())));
242         } catch (Exception e) {
243             fail(e.getMessage());
244         }
245 
246         // override the catalog to use: we'll use lucene
247         try {
248             URL ingestUrl = this.getClass().getResource("/ingest");
249             luceneCatLoc = new File(ingestUrl.getFile()).getCanonicalPath()
250                 + "/cat";
251         } catch (Exception e) {
252             fail(e.getMessage());
253         }
254 
255         properties.setProperty("filemgr.catalog.factory",
256                 "org.apache.oodt.cas.filemgr.catalog.LuceneCatalogFactory");
257         properties.setProperty(
258                 "org.apache.oodt.cas.filemgr.catalog.lucene.idxPath",
259                 luceneCatLoc);
260 
261         // now override the repo mgr policy
262         try {
263             URL fmpolicyUrl = this.getClass().getResource("/ingest/fmpolicy");
264             properties.setProperty(
265                     "org.apache.oodt.cas.filemgr.repositorymgr.dirs",
266                     "file://"
267                         + new File(fmpolicyUrl.getFile()).getCanonicalPath());
268         } catch (Exception e) {
269             fail(e.getMessage());
270         }
271 
272         // now override the val layer ones
273         URL examplesCoreUrl = this.getClass().getResource("/examples/core");
274         properties.setProperty("org.apache.oodt.cas.filemgr.validation.dirs",
275                 "file://"
276                     + new File(examplesCoreUrl.getFile()).getAbsolutePath());
277 
278         // set up mime repo path
279         URL mimeTypesUrl = this.getClass().getResource("/mime-types.xml");
280         properties.setProperty(
281                 "org.apache.oodt.cas.filemgr.mime.type.repository", new File(
282                      mimeTypesUrl.getFile()).getAbsolutePath());
283 
284         System.setProperties(properties);
285 
286         try {
287             fm = new XmlRpcFileManager(FM_PORT);
288         } catch (Exception e) {
289             LOG.log(Level.SEVERE, e.getMessage());
290             fail(e.getMessage());
291         }
292     }
