21  import java.io.ByteArrayOutputStream;
22  import java.io.File;
23  import java.io.IOException;
24  import java.util.logging.Level;
25  import java.util.logging.Logger;
26  
27  import net.lingala.zip4j.core.ZipFile;
28  import net.lingala.zip4j.exception.ZipException;
29  import net.lingala.zip4j.model.ZipParameters;
30  import net.lingala.zip4j.util.Zip4jConstants;
31  
32  import org.apache.oodt.cas.metadata.Metadata;
33  import org.apache.oodt.cas.metadata.SerializableMetadata;
34  import org.apache.oodt.cas.product.jaxrs.exceptions.InternalServerErrorException;
35  import org.apache.oodt.cas.product.jaxrs.resources.DatasetResource;
36  import org.apache.oodt.cas.product.jaxrs.resources.MetadataResource;
37  import org.apache.oodt.cas.product.jaxrs.resources.ProductResource;

39  /**
40   * Utility class that zips up a {@link DatasetResource}, including the dataset's
41   * products and metadata.
42   * @author rlaidlaw
43   * @version $Revision$
44   */
45  public class DatasetZipper
46  {
47    private static final Logger LOGGER = Logger.getLogger(DatasetZipper.class
48      .getName());
49  
50    private ProductZipper productZipper = new ProductZipper();
51  
52    /**
53     * Creates a zip archive of the supplied {@link DatasetResource dataset
54     * resource} including all of the dataset's products and metadata.
55     * @param resource the dataset resource to archive
56     * @return the file reference for the zip archive of the dataset resource
57     */
58    public File createZipFile(DatasetResource resource)
59    {
60      try
61      {
62        // Create the working directory if it doesn't already exist.
63        File workingDir = resource.getWorkingDir();
64        if (!workingDir.exists() && !workingDir.mkdirs())
65        {
66          String message = "Unable to create the working directory ("
67            + workingDir.getAbsolutePath() + ") to build the zip file.";
68          LOGGER.log(Level.FINE, message);
69          throw new IOException(message);
70        }
71  
72        // Try to remove a previously created zip file with the same name.
73        File file = new File(workingDir.getCanonicalPath() + "/"
74          + resource.getName() + ".zip");
75        if (file.exists() && !file.delete())
76        {
77          String message = "Unable to delete an existing zip file ("
78            + file.getAbsolutePath()
79            + ") before creating a new zip file with the same name.";
80          LOGGER.log(Level.FINE, message);
81          throw new IOException(message);
82        }
83  
84        // Set up the zip file for the dataset.
85        ZipFile zipFile = new ZipFile(file);
86        ZipParameters parameters = new ZipParameters();
87        parameters.setCompressionMethod(Zip4jConstants.COMP_DEFLATE);
88        parameters.setCompressionLevel(Zip4jConstants.DEFLATE_LEVEL_NORMAL);
89  
90        // Create zip archives for each product and add them to the dataset zip.
91        for (ProductResource productResource : resource.getProductResources())
92        {
93          File refFile = productZipper.createZipFile(productResource);
94          zipFile.addFile(refFile, parameters);
95          if (refFile.exists() && !refFile.delete())
96          {
97            String message = "Unable to delete a temporary product zip ("
98              + refFile.getAbsolutePath()
99              + ") after adding it to the dataset zip.";
100           LOGGER.log(Level.FINE, message);
101           throw new IOException(message);
102         }
103       }
104 
105       // Add the dataset's metadata to the zip.
106       MetadataResource metadataResource = resource.getMetadataResource();
107       Metadata metadata = metadataResource.getMetadata();
108 
109       ByteArrayOutputStream os = new ByteArrayOutputStream();
110       SerializableMetadata serMetadata = new SerializableMetadata(metadata);
111       serMetadata.writeMetadataToXmlStream(os);
112       ByteArrayInputStream bis = new ByteArrayInputStream(os.toByteArray());
113 
114       parameters.setFileNameInZip(resource.getName() + ".met");
115       parameters.setSourceExternalStream(true);
116       zipFile.addStream(bis, parameters);
117 
118       return file;
119     }
120 
121     catch (ZipException e)
122     {
123       String message = "Unable to create a zip archive of the dataset.";
124       LOGGER.log(Level.FINE, message, e);
125       throw new InternalServerErrorException(message + " " + e.getMessage());
126     }
127     catch (IOException e)
128     {
129       String message = "Encountered I/O problems while trying to create a zip "
130         + "archive of the dataset.";
131       LOGGER.log(Level.FINE, message, e);
132       throw new InternalServerErrorException(message + " " + e.getMessage());
133     }
134   }
