32  /** Configuration for an EDA exec-server.
33   *
34   * @author Kelly
35   */
36  public class ExecServerConfig extends Executable implements Documentable {
37  	/** Create an exec-server configuration.
38  	 *
39  	 * @param className Name of the class to execute.
40  	 * @param objectKey Object key under which class will register.
41  	 * @param properties Properties for the server.
42  	 */
43  	public ExecServerConfig(String className, String objectKey, InetAddress preferredHost, Properties properties) {
44  		this.className = className;
45  		this.objectKey = objectKey;
46  		this.preferredHost = preferredHost;
47  		this.properties = properties;
48  	}

50  	/** Create an exec-server configuration.
51  	 *
52  	 * @param xml XML DOM description, must be an &lt;execServer&gt; element.
53  	 * @throws SAXException If <var>xml</var> is invalid.
54  	 * @throws UnknownHostException If the <var>xml</var> refers to an unknown host name.
55  	 */
56  	public ExecServerConfig(Node xml) throws SAXException, UnknownHostException {
57  		properties = new Properties();
58  		preferredHost = org.apache.oodt.commons.net.Net.getLoopbackAddress();
59  		NodeList children = xml.getChildNodes();
60  		className = XML.unwrappedText(children.item(0));
61  		objectKey = XML.unwrappedText(children.item(1));
62  		for (int i = 2; i < children.getLength(); ++i) {
63  			Node child = children.item(i);
64  			String name = child.getNodeName();
65  			if ("host".equals(name)) {
66  			  preferredHost = InetAddress.getByName(XML.unwrappedText(children.item(2)));
67  			} else if ("properties".equals(name)) {
68  			  Configuration.loadProperties(child, properties);
69  			} else {
70  			  throw new SAXException("Unknown node " + name + " in exec server XML");
71  			}
72  		}
73  	}		
74  
75  	/** Create an exec-server configuration.
76  	 *
77  	 * @param xml Serialized XML description.
78  	 * @throws SAXException If we can't parse <var>xml</var>.
79  	 * @throws UnknownHostException If the <var>xml</var> refers to an unknown host name.
80  	 */
81  	public ExecServerConfig(String xml) throws SAXException, UnknownHostException {
82  		this(XML.parse(xml).getDocumentElement());
83  	}

85  	protected String[] getCommandLine() {
86  		String[] commandLine = new String[6 + properties.size()];
87  		commandLine[0] = "java";
88  		commandLine[1] = "-Xms" + initialHeap;
89  		commandLine[2] = "-Xmx" + maxHeap;
90  		int index = 3;
91  	  for (Map.Entry<Object, Object> objectObjectEntry : properties.entrySet()) {
92  		Map.Entry entry = (Map.Entry) objectObjectEntry;
93  		commandLine[index++] = "-D" + entry.getKey() + "=" + entry.getValue();
94  	  }
95  		commandLine[index++] = "org.apache.oodt.commons.ExecServer";
96  		commandLine[index++] = className;
97  		commandLine[index++] = objectKey;
98  		return commandLine;
99  	}

152 	public Node toXML(Document doc) throws DOMException {
153 		Element execServerElement = doc.createElement("execServer");
154 		XML.add(execServerElement, "class", getClassName());
155 		XML.add(execServerElement, "objectKey", getObjectKey());
156 		XML.add(execServerElement, "preferredHost", getPreferredHost().toString());
157 		Configuration.dumpProperties(getProperties(), execServerElement);
158 		return execServerElement;
159 	}
