64  public class RemoteCrawler implements RetrievalMethod {
65  
66      private static final Logger LOG = Logger.getLogger(RemoteCrawler.class
67              .getName());
68  
69      /**
70       * Starts the crawler and creates a default DirStruct if null was supplied
71       * in constructor
72       *
73       * @throws java.net.MalformedURLException
74       * @throws org.apache.oodt.cas.pushpull.exceptions.ProtocolException
75       * @throws org.apache.oodt.cas.pushpull.exceptions.ProtocolFileException
76       */
77      @Override
78     public void processPropFile(FileRetrievalSystem frs, Parser propFileParser,
79              File propFile, DataFilesInfo dfi, DataFileToPropFileLinker linker)
80          throws FileNotFoundException, ParserException, ProtocolException, MalformedURLException,
81          RetrievalMethodException {
82          RemoteSite remoteSite;
83  
84          // parse property file
85          Metadata fileMetadata = new Metadata();
86          VirtualFileStructure vfs = propFileParser.parse(new FileInputStream(
87                  propFile), fileMetadata);
88  
89          // determine RemoteSite
90          DownloadInfo di = dfi.getDownloadInfo();
91          if (!di.isAllowAliasOverride()
92                  || (remoteSite = vfs.getRemoteSite()) == null) {
93            remoteSite = di.getRemoteSite();
94          }
95  
96          // modify vfs to be root based if HOME directory based
97          if (!vfs.isRootBased()) {
98              String homeDirPath = frs.getHomeDir(remoteSite).getPath();
99              VirtualFile root = new VirtualFile(homeDirPath, true);
100             root.addChild(vfs.getRootVirtualFile());
101             vfs = new VirtualFileStructure(homeDirPath + "/"
102                     + vfs.getPathToRoot(), root.getRootDir());
103             frs.changeToHOME(remoteSite);
104         }
105 
106         // initialize variables
107         final String initialCdPath = vfs.getPathToRoot();
108         final VirtualFile vf = vfs.getRootVirtualFile();
109 
110         // change to initial directory (takes care of Linux auto-mounting)
111         frs.changeToDir(initialCdPath, remoteSite);
112 
113         // add starting directory to stack
114         Stack<RemoteSiteFile> files = new Stack<RemoteSiteFile>();
115         files.add(new RemoteSiteFile(frs.getCurrentFile(remoteSite), remoteSite));
116 
117         // start crawling
118         while (!files.isEmpty()) {
119             RemoteSiteFile file = files.peek();
120             try {
121                 // if directory, then add its children to the crawl list
122                 if (file.isDir()) {
123 
124                     // get next page worth of children
125                     List<RemoteSiteFile> children = frs.getNextPage(file,
126                             new ProtocolFileFilter() {
127                                 @Override
128                               public boolean accept(ProtocolFile pFile) {
129                                     return FileRestrictions.isAllowed(new
130                                         ProtocolPath(pFile
131                                             .getPath(), pFile.isDir()), vf);
132                                 }
133                             });
134 
135                     // if directory had more children then add them
136                     if (children.size() > 0) {
137                       files.addAll(children);
138                     }// otherwise remove the directory from the crawl list
139                     else {
140                       files.pop();
141                     }
142 
143                     // if file, then download it
144                 } else {
145                     linker.addPropFileToDataFileLink(propFile, file);
146                     if (!frs.addToDownloadQueue(files.pop(), di
147                             .getRenamingConv(), di.getStagingArea(), dfi
148                             .getQueryMetadataElementName(), di
149                             .deleteFromServer(), fileMetadata)) {
150                       linker.eraseLinks(propFile);
151                     }
152                 }
153 
154             } catch (ToManyFailedDownloadsException e) {
155                 throw new RetrievalMethodException(
156                         "Connection appears to be down. . .unusual number of download failures. . .stopping : "
157                                 + e.getMessage());
158             } catch (CatalogException e) {
159                 throw new RetrievalMethodException(
160                         "Failed to communicate with database : "
161                                 + e.getMessage());
162             } catch (AlreadyInDatabaseException e) {
163                 LOG.log(Level.WARNING, "Skipping file : " + e.getMessage());
164             } catch (UndefinedTypeException e) {
165                 LOG.log(Level.WARNING, "Skipping file : " + e.getMessage());
166             } catch (Exception e) {
167                 linker.markAsFailed(propFile, e.getMessage());
168                 throw new RetrievalMethodException("Uknown error accured while downloading "
169                         + file + " from " + remoteSite + " -- bailing out : "
170                         + e.getMessage(), e);
171             }
172         }
173     }
