111   private void readWrite(Product product) {
112     ByteArrayOutputStream os = new ByteArrayOutputStream();
113     try {
114       XMLUtils.writeXmlToStream(product.toXML(), os);
115     } catch (Exception e) {
116       LOG.log(Level.SEVERE, e.getMessage());
117       fail(e.getMessage());
118     }
119 
120     Product readProduct = null;
121 
122     try {
123       readProduct = new Product(new ByteArrayInputStream(os.toByteArray()));
124     } catch (InstantiationException e) {
125       fail(e.getMessage());
126     }
127 
128     assertNotNull(readProduct);
129     assertEquals("testId", readProduct.getProductId());
130     assertEquals("TestProduct", readProduct.getProductName());
131     assertEquals(Product.STRUCTURE_FLAT, readProduct.getProductStructure());
132     assertEquals(Product.STATUS_RECEIVED, readProduct.getTransferStatus());
133     assertNotNull(readProduct.getProductType());
134     assertNotNull(readProduct.getProductType().getName(), "BogusType");
135     assertNotNull(readProduct.getProductReferences());
136     assertEquals(1, readProduct.getProductReferences().size());
137     Reference readRef = (Reference) readProduct.getProductReferences().get(0);
138     assertNotNull(readRef);
139     assertEquals(readRef.getOrigReference(), "file:///data.orig");
140     assertEquals(readRef.getDataStoreReference(), "file:///data.store");
141     assertEquals(readRef.getFileSize(), 99999L);
142   }
