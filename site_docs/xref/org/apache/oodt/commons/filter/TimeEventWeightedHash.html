32  * <p>
33  * Builds a time sorted hash where the top value has the earliest time.  This hash
34  * is a directional graph.  It builds all possible paths from top node to leaf
35  * nodes.  At any time you can ask for the greatest weighted path in the hash -- that
36  * is, the path from root node to a leaf node which fills the most time. Epsilon is 
37  * TimeEvent objects allowed time overlap (it is of the same unit of measurement that 
38  * is used in you TimeEvent objects added to this hash).
39  * </p>.
40  */

65        events = TimeEvent.getTimeOrderedEvents(events);

83    private void addEvent(TimeEvent newEvent) {
84        TimeEventNode newEventNode = new TimeEventNode(newEvent);
85        MyLinkedHashSet<TimeEventNode> parentNodes = this.findParents(newEvent);
86        MyLinkedHashSet<TimeEventNode> childrenNodes = this.findChildren(newEvent, parentNodes);
87        
88        if (childrenNodes.size() == 0) {
89            leafNodes.add(newEventNode);
90        }
91        
92        newEventNode.addParents(parentNodes);
93        newEventNode.addChildren(childrenNodes);
94        
95    }

97    private MyLinkedHashSet<TimeEventNode> findParents(TimeEvent newEvent) {
98        MyLinkedHashSet<TimeEventNode> parentNodes = new MyLinkedHashSet<TimeEventNode>();
99        MyLinkedHashSet<TimeEventNode> possibleParentNodes = new MyLinkedHashSet<TimeEventNode>();
100       MyLinkedHashSet<TimeEventNode> possibleParentNodesAlreadyChecked = new MyLinkedHashSet<TimeEventNode>();
101       
102       //iterate through possible parent nodes add possible parent child as
103       // possible parents if they do not conflict
104       TimeEventNode curPPN = this.root;
105       while (curPPN != null) {
106           boolean ppnListChanged = false;
107           for (TimeEventNode curChild : curPPN.getChildren()) {
108               //if curChild does not conficts with te, then it is a possible parent
109               if (happensBefore(curChild.getTimeEvent(), newEvent) 
110                       && !possibleParentNodesAlreadyChecked.contains(curChild)) {
111                   possibleParentNodes.add(curChild);
112                   ppnListChanged = true;
113               }
114           }
115           //if all children where not possible parents, then curPPN must be parent
116           if (!ppnListChanged) {
117               parentNodes.add(curPPN);
118           }
119           
120           //get next curPPN
121           possibleParentNodes.remove(curPPN);
122           possibleParentNodesAlreadyChecked.add(curPPN);
123           while((curPPN = possibleParentNodes.get(0)) != null 
124                   && possibleParentNodesAlreadyChecked.contains(curPPN)) {
125               possibleParentNodes.remove(curPPN);
126           }
127       }
128       
129       return parentNodes;
130   }

132   private MyLinkedHashSet<TimeEventNode> findChildren(TimeEvent newEvent, MyLinkedHashSet<TimeEventNode> parentNodes) {
133       MyLinkedHashSet<TimeEventNode> childrenNodes = new MyLinkedHashSet<TimeEventNode>();
134       MyLinkedHashSet<TimeEventNode> possibleChildrenNodes = new MyLinkedHashSet<TimeEventNode>();
135       for (TimeEventNode parent : parentNodes) {
136           possibleChildrenNodes.addAll(parent.getChildren());
137           TimeEventNode curPCN = possibleChildrenNodes.get(0);
138           while (curPCN != null && !parentNodes.contains(curPCN)) {
139               if (happensBefore(newEvent, curPCN.getTimeEvent())) {
140                   childrenNodes.add(curPCN);
141               }else {
142                   possibleChildrenNodes.addAll(curPCN.getChildren());
143               }
144               possibleChildrenNodes.remove(curPCN);
145               curPCN = possibleChildrenNodes.get(0);
146           }
147       }
148       return childrenNodes;
149   }

151   private boolean happensBefore(TimeEvent t1, TimeEvent t2) {
152       long boundaryCheck = t2.getStartTime() - t1.getEndTime();
153       return t1.getStartTime() < t2.getStartTime() && (boundaryCheck + epsilon) > 0;
154   }

156   private WeightedNode getGreatestWeightedPath() {
157       TimeEventNode max = this.leafNodes.get(0);
158       for (TimeEventNode ten : this.leafNodes) {
159          if (ten.getPathWeight() > max.getPathWeight() 
160                  || (ten.getPathWeight() == max.getPathWeight() 
161                          && ten.getPathPriorityWeight() > max.getPathPriorityWeight())) {
162              max = ten;
163          }
164       }
165       WeightedNode root = new WeightedNode(max.getTimeEvent());
166       TimeEventNode curTEN = max.greatestWieghtedParent;
167       while (curTEN != null) {
168           WeightedNode temp = new WeightedNode(curTEN.getTimeEvent());
169           temp.setChild(root);
170           root = temp;
171           curTEN = curTEN.greatestWieghtedParent;
172       }
173       return root.getChild();
174   }

241   private class TimeEventNode {
242 
243       private TimeEvent te;
244       
245       private MyLinkedHashSet<TimeEventNode> children;
246 
247       private MyLinkedHashSet<TimeEventNode> parents;
248       
249       private TimeEventNode greatestWieghtedParent;
250               
251       public TimeEventNode(TimeEvent te) {
252           this.te = te;
253           children = new MyLinkedHashSet<TimeEventNode>();
254           parents = new MyLinkedHashSet<TimeEventNode>();
255       }
256       
257       public long getPathWeight() {
258           if (this.greatestWieghtedParent != null) {
259               return te.getDuration()
260                      + this.greatestWieghtedParent.getPathWeight();
261           } else {
262               return te.getDuration();
263           }
264       }
265 
266       public double getPathPriorityWeight() {
267           if (this.greatestWieghtedParent != null) {
268               return te.getPriority()
269                      + this.greatestWieghtedParent.getPathPriorityWeight();
270           } else {
271               return te.getPriority();
272           }
273       }
274 
275       public TimeEvent getTimeEvent() {
276           return this.te;
277       }
278 
279       public void setTimeEvent(TimeEvent te) {
280           this.te = te;
281       }
282       
283       public void addChild(TimeEventNode child) {            
284           //remove parent to grandchild links
285           for (TimeEventNode parent : this.parents) {
286               parent.children.remove(child);
287               child.parents.remove(parent);
288           }
289           
290           //link to child
291           this.children.add(child);
292           child.parents.add(this);
293           leafNodes.remove(this);
294           
295           //determine child's greatest weighted parent
296           child.greatestWieghtedParent = this;
297           for (TimeEventNode parent : child.parents) {
298               long gwpPaW = child.greatestWieghtedParent.getPathWeight();
299               long pPaW = parent.getPathWeight();
300               double gwpPiW = child.greatestWieghtedParent.getPathPriorityWeight();
301               double pPiW = parent.getPathPriorityWeight();
302               if (pPaW > gwpPaW || (pPaW == gwpPaW && pPiW > gwpPiW)) {
303                   child.greatestWieghtedParent = parent;
304               }
305           }
306       }
307       
308       public void addParent(TimeEventNode parent) {
309           parent.addChild(this);
310       }
311 
312       public void addChildren(Collection<TimeEventNode> children) {
313           for (TimeEventNode child : children) {
314               this.addChild(child);
315           }
316       }
317       
318       public void addParents(Collection<TimeEventNode> parents) {
319           for (TimeEventNode parent : parents) {
320               this.addParent(parent);
321           }
322       }
323 
324       public MyLinkedHashSet<TimeEventNode> getChildren() {
325           return this.children;
326       }
327       
328       public MyLinkedHashSet<TimeEventNode> getParents() {
329           return this.parents;

394   private StringBuffer printNodes(List<TimeEventNode> list,
395           String spacer, long curPathWeight) {
396       StringBuffer output = new StringBuffer("");
397       for (TimeEventNode node : list) {
398           output.append(spacer);
399           output.append(node.te).append(" -- ").append(curPathWeight + node.te.getDuration()).append("\n");
400           output.append(printNodes(node.getChildren().getList(), " " + spacer, curPathWeight + node.te.getDuration()));
401       }
402       return output;
403   }

405     @Override
406     public boolean equals(Object o) {
407         if (this == o) {
408             return true;
409         }
410         if (o == null || getClass() != o.getClass()) {
411             return false;
412         }
413 
414         TimeEventWeightedHash that = (TimeEventWeightedHash) o;
415 
416         if (epsilon != that.epsilon) {
417             return false;
418         }
419         if (root != null ? !root.equals(that.root) : that.root != null) {
420             return false;
421         }
422         return !(leafNodes != null ? !leafNodes.equals(that.leafNodes) : that.leafNodes != null);
423 
424     }

426     @Override
427     public int hashCode() {
428         int result = root != null ? root.hashCode() : 0;
429         result = 31 * result + (int) (epsilon ^ (epsilon >>> 32));
430         result = 31 * result + (leafNodes != null ? leafNodes.hashCode() : 0);
431         return result;
432     }
