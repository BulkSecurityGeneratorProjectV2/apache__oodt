16  package org.apache.oodt.commons.io;
17  
18  import java.io.FilterOutputStream;
19  import java.io.IOException;
20  import java.io.OutputStream;
21  import org.apache.oodt.commons.util.Base64;
22  
23  /** An output stream that encodes its data into RFC-1512 base 64 format.
24   *
25   * Wrap this input stream around another output stream, and all the bytes will be
26   * converted into their base-64 format when you write to it.
27   *
28   * @author Kelly
29   */
30  public class Base64EncodingOutputStream extends FilterOutputStream {
31  	/** Construct a base-64 encoding output stream.
32  	 *
33  	 * @param outputStream The output stream to which to write.
34  	 */
35  	public Base64EncodingOutputStream(OutputStream outputStream) {
36  		super(outputStream);
37  	}
38  
39  	/** Write a byte of data.
40  	 *
41  	 * The byte will be encoded into base-64 format on the output.
42  	 *
43  	 * @param b The byte.
44  	 * @throws IOException If an I/O error occurs.
45  	 */
46  	public void write(int b) throws IOException {
47  		if (buffer == null) {
48  		  throw new IOException("Can't write onto a closed stream");
49  		}
50  		buffer[index++] = (byte) b;
51  		if (index == buffer.length) {
52  		  shipout();
53  		}
54  	}
55  
56  	/** Write a bunch of bytes.
57  	 *
58  	 * The given array of bytes will be encoded into base-64 on the output.
59  	 *
60  	 * @param b The array to write.
61  	 * @param offset Where in the data to start writing.
62  	 * @param length How many bytes to write.
63  	 * @throws IOException If an I/O error occurs.
64  	 */
65  	public void write(byte[] b, int offset, int length) throws IOException {
66  		if (b == null) {
67  		  throw new IllegalArgumentException("Can't write a null array");
68  		}
69  		if (offset < 0 || offset >= b.length) {
70  		  throw new IndexOutOfBoundsException("Can't get bytes at " + offset + " in array with indexes 0.."
71  											  + (b.length - 1));
72  		}
73  		if (length < 0) {
74  		  throw new IllegalArgumentException("Can't write a negative amount of bytes");
75  		}
76  		if (offset + length > b.length) {
77  		  throw new IndexOutOfBoundsException("Can't get bytes beyond edge of array");
78  		}
79  		if (buffer == null) {
80  		  throw new IOException("Can't write onto a closed stream");
81  		}
82  		while (length > 0) {
83  			int avail = buffer.length - index;
84  			int amount = avail < length? avail : length;
85  			System.arraycopy(b, offset, buffer, index, amount);
86  			index += amount;
87  			offset += amount;
88  			length -= amount;
89  			if (index == buffer.length) {
90  			  shipout();
91  			}
92  		}
93  	}
94  
95  	/** Flush the stream.
96  	 *
97  	 * This causes any buffered bytes to be encoded and shipped out to the underlying
98  	 * stream, which is also flushed.
99  	 *
100 	 * @throws IOException If an I/O error occurs.
101 	 */
102 	public void flush() throws IOException {
103 		if (buffer == null) {
104 		  throw new IOException("Can't flush a closed stream");
105 		}
106 		shipout();
107 		out.flush();
108 	}
109 
110 	/** Close the stream.
111 	 *
112 	 * This writes out any unflushed data in base-64 format and closes the underlying
113 	 * stream.
114 	 *
115 	 * @throws IOException If an I/O error occurs.
116 	 */
117 	public void close() throws IOException {
118 		if (buffer == null) {
119 		  throw new IOException("Can't close an already closed stream");
120 		}
121 		flush();
122 		out.close();
123 		out = null;
124 		buffer = null;
125 	}
126 
127 	/** Ship out a bunch of buffered data in base-64 format.
128 	 *
129 	 * This resets the index of the next byte to insert back to zero.
130 	 *
131 	 * @throws IOException If an I/O error occurs.
132 	 */
133 	private void shipout() throws IOException {
134 		byte[] encoded = Base64.encode(buffer, 0, index);
135 		out.write(encoded);
136 		index = 0;
137 	}
138 
139 	/** Size of the output data buffer.  Must be a multiple of 3.
140 	 */
141 	private static final int BUFFER_SIZE = 300;
142 
143 	/** Buffer for output data.
144 	 */
145 	private byte[] buffer = new byte[BUFFER_SIZE];
146 
147 	/** Where we are in the buffer.
148 	 */
149 	private int index = 0;
150 }
