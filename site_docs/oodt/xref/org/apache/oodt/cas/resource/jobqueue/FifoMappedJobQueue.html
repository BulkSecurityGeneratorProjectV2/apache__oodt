57    /**
58     * @see org.apache.oodt.cas.resource.jobqueue.JobQueue.addJob(JobSpec)
59     */
60    public synchronized String addJob(JobSpec spec) throws JobQueueException {
61  
62      // Check if the job is null and if its queue exists
63      if (spec == null) {
64        throw new JobQueueException("A null job was given.");
65      }
66      String queueName = spec.getJob().getQueueName();
67      validateQueueName(queueName);
68  
69      // Check if the jobs queue is full
70      List<String> queue = queues.get(queueName);
71      if (queue.size() == maxQueueSize) {
72        throw new JobQueueException(
73            "The queue " + spec.getJob().getQueueName() + " is full.  The job "
74                + spec.getJob().getId() + " could not be requeued.");
75      }
76  
77      // Add the job to the repository
78      try {
79        this.repo.addJob(spec);
80      } catch (JobRepositoryException e) {
81        throw new JobQueueException(
82            "An error occurred while adding job " + spec.getJob().getId()
83                + " to the job repository: " + e.getMessage());
84      }
85  
86      // Add the job to the queue
87      queue.add(spec.getJob().getId());
88  
89      // Update the jobs status
90      spec.getJob().setStatus(JobStatus.QUEUED);
91      try {
92        this.repo.updateJob(spec);
93      } catch (JobRepositoryException e) {
94        throw new JobQueueException("An error occurred while updating "
95            + "the status of job " + spec.getJob().getId()
96            + " in the job repository: " + e.getMessage());
97      }
98  
99      LOG.log(Level.INFO,
100         "Job [id=" + spec.getJob().getId() + ",name=" + spec.getJob().getName()
101             + "] was added to the job queue in queue " + queueName);
102     return spec.getJob().getId();
103 
104   }

116     validateQueueName(queueName);

121     queue.add(0, spec.getJob().getId());
122 
123     // Set the jobs status
124     spec.getJob().setStatus(JobStatus.QUEUED);
125     try {
126       this.repo.updateJob(spec);

129           + "the status of job " + spec.getJob().getId()
130           + " in the job repository: " + e.getMessage());
131     }

141

143     for (Iterator<Vector<String>> i = queues.values().iterator(); i
144         .hasNext();) {
145       List<String> queue = i.next();
146       for (String jobId : queue) {
147         try {
148           allJobs.add(this.repo.getJobById(jobId));
149         } catch (JobRepositoryException e) {
150           LOG.log(Level.WARNING, "Failed to fetch JobSpec from repo: " + jobId);
151         }
152       }
153     }
154 
155     return allJobs;

205   /**
206    * @see org.apache.oodt.cas.resource.jobqueue.JobQueue.getNextJob()
207    */
208   public synchronized JobSpec getNextJob() {
209 
210     // Check if any queues exist
211     if (this.queues.keySet().size() == 0) {
212       throw new RuntimeException("No queues are defined.");
213     }
214 
215     // Check if all queues are empty
216     if (this.isEmpty()) {
217       throw new RuntimeException("The queue contains no jobs.");
218     }
219 
220     // Look in each queue for a job
221     for (Iterator<String> i = this.queues.keySet().iterator(); i.hasNext();) {
222 
223       // Check if the queue is empty
224       List<String> queue = this.queues.get(i.next());
225       if (!queue.isEmpty()) {
226 
227         // Check jobs from the front of the queue until we find one that
228         // is flagged as ready to schedule
229         for (int index = 0; index < queue.size(); index++) {
230 
231           // Check how the job is flagged
232           String jobId = queue.get(index);
233           JobSpec spec = null;
234           try {
235             spec = this.repo.getJobById(jobId);
236           } catch (JobRepositoryException e) {
237             LOG.log(Level.WARNING,
238                 "Failed to fetch JobSpec from repo: " + jobId);
239           }
240           if (spec.getJob().getReady()) {
241 
242             // Remove the job from the queue
243             queue.remove(index);
244 
245             // Set the status of the fetched job
246             spec.getJob().setStatus(JobStatus.SCHEDULED);
247             try {
248               this.repo.updateJob(spec);
249             } catch (JobRepositoryException e) {
250               LOG.log(Level.WARNING,
251                   "The status of job " + spec.getJob().getId()
252                       + "was not properly set "
253                       + "after being dequeued. Message: " + e.getMessage());
254             }
255 
256             return spec;
257 
258           }
259 
260         }
261 
262       }
263 
264     }
265 
266     return null;
267 
268   }

275     validateQueueName(queueName);
276 
277     // If the queue contains no jobs, return null

283     // Check jobs from the front of the queue until we find one that

286 
287       // Check how the job is flagged
288       String jobId = queue.get(index);
289       JobSpec spec = this.repo.getJobById(jobId);
290       if (spec.getJob().getReady()) {
291 
292         // Remove the job from the queue

296         spec.getJob().setStatus(JobStatus.SCHEDULED);

363   public synchronized void removeJob(JobSpec spec) throws JobQueueException {
364 
365     // Check if the job is null and if its queue exists
366     if (spec == null) {
367       throw new JobQueueException("A null job was given.");
368     }
369     String queueName = spec.getJob().getQueueName();
370     validateQueueName(queueName);
371 
372     // Get the ID of the job
373     String id = spec.getJob().getId();
374 
375     // Find the job in the queue and remove it
376     List<String> queue = this.queues.get(queueName);
377     int index = getIndexInQueue(id, queue);
378     if (index == -1) {
379       LOG.log(Level.WARNING, "No job with ID " + id + "could be removed "
380           + "since it was not found in the queue.");
381     } else {
382       queue.remove(index);
383     }

422   public synchronized void promoteJob(JobSpec spec) throws JobQueueException {
423 
424     // Check if the job is null and if its queue exists
425     if (spec == null) {
426       throw new JobQueueException("A null job was given.");
427     }
428     String queueName = spec.getJob().getQueueName();
429     validateQueueName(queueName);
430 
431     // Get the ID of the job
432     String id = spec.getJob().getId();
433 
434     // Find the job in the queue and move it to the front
435     List<String> queue = this.queues.get(queueName);
436     int index = getIndexInQueue(id, queue);
437     if (index == -1) {
438       LOG.log(Level.WARNING, "No job with ID " + id + "could be promoted "
439           + "since it was not found in the queue.");
440     } else {
441       queue.add(0, queue.remove(index));
442     }

454   public synchronized void promoteKeyValPair(String key, String val)
455       throws JobQueueException, JobRepositoryException {
456 
457     List<JobSpec> specsToPromote = new Vector<JobSpec>();
458 
459     for (Iterator<Vector<String>> i = queues.values().iterator(); i
460         .hasNext();) {
461       List<String> queue = i.next();
462       for (String jobId : queue) {
463         JobSpec spec = null;
464         try {
465           spec = this.repo.getJobById(jobId);
466         } catch (JobRepositoryException e) {
467           LOG.log(Level.WARNING, "Failed to fetch JobSpec from repo: " + jobId);
468         }
469         if (spec.getIn().getMetadata().get(key).equals(val)) {
470           specsToPromote.add(spec);
471         }
472       }
473     }
474 
475     for (JobSpec spec : specsToPromote) {
476       promoteJob(spec);
477     }
478 
479   }

501   private int getIndexInQueue(String id, List<String> queue) {
502     for (int i = 0; i < queue.size(); i++) {
503       if (queue.get(i).equals(id)) {
504         return i;
505       }
506     }
507     return -1;
