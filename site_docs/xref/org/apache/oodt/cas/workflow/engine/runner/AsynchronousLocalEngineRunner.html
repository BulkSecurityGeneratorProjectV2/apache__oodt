39  /**
40   * Runs a local version of a {@link TaskProcessor} asynchronously.
41   * 
42   * @author mattmann (Chris Mattmann)
43   * @author bfoster (Brian Foster)
44   */
45  public class AsynchronousLocalEngineRunner extends AbstractEngineRunnerBase {
46  
47    private static final Logger LOG = Logger
48        .getLogger(AsynchronousLocalEngineRunner.class.getName());
49  
50    public static final int DEFAULT_NUM_THREADS = 25;
51  
52    private final ExecutorService executor;
53    private final Map<String, Thread> workerMap;
54  
55    public AsynchronousLocalEngineRunner() {
56      this(DEFAULT_NUM_THREADS);
57    }
58  
59    public AsynchronousLocalEngineRunner(int numThreads) {
60      super();
61      this.executor = Executors.newFixedThreadPool(numThreads);
62      this.workerMap = new ConcurrentHashMap<String, Thread>();
63    }
64  
65    /*
66     * (non-Javadoc)
67     * 
68     * @see
69     * org.apache.oodt.cas.workflow.engine.runner.EngineRunner#execute(org.apache
70     * .oodt.cas.workflow.engine.processor.TaskProcessor)
71     */
72    @Override
73    public void execute(final TaskProcessor taskProcessor) {
74      Thread worker = new Thread() {
75  
76        @Override
77        public void run() {
78          WorkflowLifecycle lifecycle = getLifecycle(taskProcessor);
79          WorkflowTask workflowTask = getTaskFromProcessor(taskProcessor);
80          WorkflowTaskInstance inst = GenericWorkflowObjectFactory
81              .getTaskObjectFromClassName(workflowTask.getTaskInstanceClassName());
82          try {
83            inst.run(taskProcessor.getWorkflowInstance().getSharedContext(),
84                workflowTask.getTaskConfig());
85            String msg = "Task: [" + workflowTask.getTaskName()
86                + "] for instance id: ["
87                + taskProcessor.getWorkflowInstance().getId()
88                + "] completed successfully";
89            LOG.log(Level.INFO, msg);
90            WorkflowState state = lifecycle.createState("ExecutionComplete", "transition", msg);
91            taskProcessor.getWorkflowInstance().setState(state);
92            persist(taskProcessor.getWorkflowInstance());
93          } catch (Exception e) {
94            LOG.log(Level.SEVERE, e.getMessage());
95            String msg = "Exception executing task: ["
96                + workflowTask.getTaskName() + "]: Message: " + e.getMessage();
97            LOG.log(Level.WARNING, msg);
98            WorkflowState state = lifecycle.createState("Failure", "done", msg);
99            taskProcessor.getWorkflowInstance().setState(state);
100           persist(taskProcessor.getWorkflowInstance());
101         }
102 
103       }
104 
105       /*
106        * (non-Javadoc)
107        * 
108        * @see java.lang.Thread#interrupt()
109        */
110       @SuppressWarnings("deprecation")
111       @Override
112       public void interrupt() {
113         super.interrupt();
114         this.destroy();
115       }
116 
117     };
118 
119     String id = "";
120     synchronized (id) {
121       id = UUID.randomUUID().toString();
122       this.workerMap.put(id, worker);
123       this.executor.execute(worker);
124     }
125   }
126 
127   /*
128    * (non-Javadoc)
129    * 
130    * @see org.apache.oodt.cas.workflow.engine.EngineRunner#shutdown()
131    */
132   @Override
133   public void shutdown() {
134     for (Thread worker : this.workerMap.values()) {
135       if (worker != null) {
136         worker.interrupt();
137       }
138     }
139 
140   }
