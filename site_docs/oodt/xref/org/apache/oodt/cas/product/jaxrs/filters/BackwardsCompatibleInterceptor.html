18  package org.apache.oodt.cas.product.jaxrs.filters;
19  
20  import java.nio.charset.Charset;
21  import java.util.concurrent.ConcurrentHashMap;
22  import java.util.List;
23  import java.util.Map;
24  
25  import org.apache.cxf.interceptor.Fault;
26  import org.apache.cxf.message.Message;
27  import org.apache.cxf.phase.AbstractPhaseInterceptor;
28  import org.apache.cxf.phase.Phase;
29  import org.apache.http.NameValuePair;
30  import org.apache.http.client.utils.URLEncodedUtils;
31  
32  /**
33   * An Apache CXF interceptor for the incoming chain that monitors request URIs
34   * and maps any URIs from the original scheme to the new scheme.
35   * @author rlaidlaw
36   * @version $Revision$
37   */
38  public class BackwardsCompatibleInterceptor extends
39    AbstractPhaseInterceptor<Message>
40  {
41    /**
42     * Constructor that inserts the interceptor at the USER_STREAM phase of the
43     * incoming interceptor chain.
44     */
45    public BackwardsCompatibleInterceptor()
46    {
47      super(Phase.USER_STREAM);
48    }
49  
50    @Override
51    public void handleMessage(Message message) throws Fault
52    {
53      String base = (String) message.get(Message.BASE_PATH);
54      String uri = (String) message.get(Message.REQUEST_URI);
55      String query = (String) message.get(Message.QUERY_STRING);
56  
57      base += base.endsWith("/") ? "" : "/";
58      String request = uri.replaceAll("^" + base, "");
59  
60      // Parse the query string into a map of parameters.
61      // [Note: this will overwrite multiple parameters that have the same name.]
62      List<NameValuePair> params = URLEncodedUtils.parse(query, Charset.forName("UTF-8"));
63      Map<String, String> map = new ConcurrentHashMap<String, String>();
64      for (NameValuePair pair : params)
65      {
66        map.put(pair.getName(), pair.getValue());
67      }
68  
69      // Maps "data?productID=<product ID>" URI to either
70      // "reference.file?productId=<product ID>" or
71      // "product.zip?productId=<product ID>"
72      if(request.equals("data"))
73      {
74        String format = map.get("format");
75        request =
76          "application/x-zip".equals(format) || "application/zip".equals(format)
77          ? "product.zip" : "reference.file";
78  
79        query = "productId=" + map.get("productID");
80        query += map.containsKey("refIndex")
81          ? "&refIndex=" + map.get("refIndex") : "";
82      }
83  
84      // Maps "dataset?typeID=<product type ID>" to
85      // "dataset.zip?productTypeId=<product type ID>"
86      else if(request.equals("dataset") && map.containsKey("typeID"))
87      {
88        request = "dataset.zip";
89        query = "productTypeId=" + map.get("typeID");
90      }
91  
92      // Maps "rdf?type=ALL" or "rdf?id=<product type ID>" to
93      // "dataset.rdf?productTypeId=<ALL or product type ID>"
94      else if (request.equals("rdf"))
95      {
96        request = "dataset.rdf";
97        String type = map.get("type");
98        query = "productTypeId=";
99        query += "ALL".equals(type) ? type : map.get("id");
100     }
101 
102     // Maps "rdf/dataset?type=ALL" or "rdf/dataset?typeID=<product type ID>" to
103     // "dataset.rdf?productTypeId=<ALL or product type ID>"
104     else if (request.equals("rdf/dataset"))
105     {
106       request = "dataset.rdf";
107       String type = map.get("type");
108       query = "productTypeId=";
109       query += "ALL".equals(type) ? type : map.get("typeID");
110       query += map.containsKey("filter") ? "&filter=" + map.get("filter") : "";
111     }
112 
113     // Maps "viewRecent?channel=ALL" or "viewRecent?id=<product type ID>" to
114     // "dataset.rss?productTypeId=<ALL or product type ID>"
115     else if (request.equals("viewRecent"))
116     {
117       request = "dataset.rss";
118       String channel = map.get("channel");
119       query = "productTypeId=";
120       query += "ALL".equals(channel) ? channel : map.get("id");
121       query += map.containsKey("topn") ? "&limit=" + map.get("topn") : "";
122     }
123 
124     // Maps "viewTransfers" to "transfers.rss?productId=ALL"
125     else if (request.equals("viewTransfers"))
126     {
127       request = "transfers.rss";
128       query = "productId=ALL";
129     }
130 
131     // Store the new URI and query in the message map.
132     uri = base + request;
133     message.put(Message.REQUEST_URI, uri);
134     message.put(Message.QUERY_STRING, query);
135   }
