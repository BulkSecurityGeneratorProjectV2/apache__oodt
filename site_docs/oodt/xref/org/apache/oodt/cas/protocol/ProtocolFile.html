17  package org.apache.oodt.cas.protocol;
18  
19  //APACHE imports
20  import org.apache.commons.lang.StringUtils;
21  import org.apache.commons.lang.Validate;
22  
23  /**
24   * A path representing a file/directory over some {@link Protocol}
25   * 
26   * @author bfoster
27   */
28  public class ProtocolFile {
29  
30  	public static final String SEPARATOR = "/";
31  	
32  	private String path;
33  	private boolean isDir;
34  	private ProtocolFile parent;
35  	
36  	public ProtocolFile(String path, boolean isDir) {
37  		this(null, path, isDir);
38  	}
39  	
40  	public ProtocolFile(ProtocolFile parent, String path, boolean isDir) {
41  		this.parent = parent;
42  		this.isDir = isDir;
43  		Validate.notNull(path, "ProtocolFile's path cannot be NULL");
44  		this.path = path.length() > 0 && !path.equals(SEPARATOR) ? StringUtils
45  				.chomp(path, SEPARATOR) : path;
46  	}
47  
48  	/**
49  	 * True is this path is a directory.
50  	 * 
51  	 * @return True if directory, false otherwise
52  	 */
53  	public boolean isDir() {
54  		return isDir;
55  	}
56  
57  	/**
58  	 * Gets the {@link String} representation of this path.
59  	 * 
60  	 * @return The {@link String} representation of this path
61  	 */
62  	public String getPath() {
63  		return path;
64  	}
65  
66  	/**
67  	 * Gets the name of this file this path represents (i.e. '/path/to/file' will
68  	 * return 'file')
69  	 * 
70  	 * @return The name of the file this path represents
71  	 */
72  	public String getName() {
73  		return path.substring(path.lastIndexOf(SEPARATOR) + 1);
74  	}
75  
76  	/**
77  	 * True if this path is a relative path (i.e. does not start with
78  	 * {@link SEPARATOR}).
79  	 * 
80  	 * @return True is this a relative path, false otherwise
81  	 */
82  	public boolean isRelative() {
83  		return !path.startsWith(SEPARATOR);
84  	}
85  
86  	/**
87  	 * Gets the parent {@link ProtocolFile} for this path.
88  	 * 
89  	 * @return The parent {@link ProtocolFile}
90  	 */
91  	public ProtocolFile getParent() {
92  		if (parent != null) {
93  			return parent;
94  		} else {
95  			int index = StringUtils.lastIndexOf(path, SEPARATOR);
96  			return (index > 0) ? new ProtocolFile(StringUtils.substring(path, 0, index), true) : null;
97  		}
98  	}
99  
100 	/**
101 	 * Get Absolute pathed {@link ProtocolFile} version of this {@link ProtocolFile}.
102 	 * 
103 	 * @return the absolute pathed version of this {@link ProtocolFile}
104 	 */
105 	public ProtocolFile getAbsoluteFile() {
106 		if (this.isRelative()) {
107 		  ProtocolFile parent = this.getParent();
108 		  if (parent != null) {
109 				return new ProtocolFile(StringUtils.chomp(parent.getAbsoluteFile().getPath(), SEPARATOR)
110 						+ SEPARATOR + this.getPath(), this.isDir());
111 			}
112 		}
113 		return this;
114 	}
115 
116 	/**
117 	 * {@inheritDoc}
118 	 */
119 	public int hashCode() {
120 		return this.getPath().hashCode();
121 	}
122 
123 	/**
124 	 * {@inheritDoc}
125 	 */
126 	public boolean equals(Object path) {
127 		if (path instanceof ProtocolFile) {
128 			ProtocolFile p = (ProtocolFile) path;
129 			return (p.getAbsoluteFile().getPath()
130 					.equals(this.getAbsoluteFile().getPath()) && p.isDir() == this
131 					.isDir());
132 		}
133 		return false;
134 	}
135 
136 	/**
137 	 * {@inheritDoc}
138 	 */
139 	public String toString() {
140 		return "{parent = '" + this.parent + "', path = '" + path + "', isDir = '" + isDir + "'}";
141 	}
142 }
