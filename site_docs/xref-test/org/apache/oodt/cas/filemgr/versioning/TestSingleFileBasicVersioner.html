43  public class TestSingleFileBasicVersioner extends TestCase {
44  
45  	private String origFileRef = "file:/foo/bar/testfile.txt";
46  	
47  	private String expectedDestFileRef = "file:/foo/bar2/testfile.txt";
48  	
49  	private String productRepoPath = "file:/foo/bar2/";
50  	
51    private Properties initialProperties = new Properties(System.getProperties());
52  
53    public void setUp() throws Exception {
54      Properties properties = new Properties(System.getProperties());
55      URL url = this.getClass().getResource("/mime-types.xml");
56      properties.setProperty("org.apache.oodt.cas.filemgr.mime.type.repository",
57          new File(url.getFile()).getAbsolutePath());
58      System.setProperties(properties);
59    }
60  
61    public void tearDown() throws Exception {
62      System.setProperties(initialProperties);
63    }
64  	
65  	public void testVersioning(){
66  		Product product = new Product();
67  		ProductType type = new ProductType();
68  		type.setProductRepositoryPath(productRepoPath);
69  		product.setProductStructure(Product.STRUCTURE_FLAT);
70  		product.setProductType(type);
71  		
72  		Metadata metadata = new Metadata();
73  		metadata.addMetadata(SingleFileBasicVersioner.FILENAME_FIELD, "testfile.txt");
74  		
75  		Reference r = new Reference();
76  		r.setOrigReference(origFileRef);
77  		
78  		product.getProductReferences().add(r);
79  		
80  		SingleFileBasicVersioner versioner = new SingleFileBasicVersioner();
81  		try{
82  			versioner.createDataStoreReferences(product, metadata);
83  		}
84  		catch(VersioningException e){
85  			fail(e.getMessage());
86  		}
87  		
88  		assertEquals("The generated ref: ["+r.getDataStoreReference()+"] is not equal to the expected ref: ["+expectedDestFileRef+"]", expectedDestFileRef, r.getDataStoreReference());
89  		
90  		
91  	}
