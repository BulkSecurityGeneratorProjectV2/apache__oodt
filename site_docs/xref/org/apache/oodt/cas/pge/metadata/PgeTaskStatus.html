22  /**
23   * Workflow Statuses for {@link PGETaskInstance}.
24   *
25   * @author bfoster (Brian Foster)
26   * @author mattmann (Chris Mattmann)
27   */
28  public enum PgeTaskStatus {
29  	
30  	/**
31  	 * The PGE is staging its input to the job working directory.
32  	 */
33     STAGING_INPUT("PGETask_Staging_Input", "STAGING INPUT"),
34     
35     /**
36      * The PGE is building its internal configuration file.
37      */
38     CONF_FILE_BUILD("PGETask_Building_Config_File", "BUILDING CONFIG FILE"),
39     
40     /**
41      * The PGE is executing.
42      */
43     RUNNING_PGE("PGETask_Running", "PGE EXEC"),
44     
45     /**
46      * The ingest crawler has been created, either a StdProductCrawler
47      * or if {@link PgeTaskMetKeys#MIME_EXTRACTOR_REPO} has been specified
48      * an AutoDetectProductCrawler.
49      */
50     CRAWLING("PGETask_Crawling", "CRAWLING");
51  
52     public static final String USE_LEGACY_STATUS_PROPERTY = "org.apache.oodt.cas.pge.task.status.legacyMode";
53     
54     @VisibleForTesting private String workflowStatusName;
55     @VisibleForTesting private String legacyName;
56  
57     PgeTaskStatus(String workflowStatusName, String legacyName) {
58        this.workflowStatusName = workflowStatusName;
59        this.legacyName = legacyName;
60     }
61  
62     public String getWorkflowStatusName() {
63        return Boolean.getBoolean(USE_LEGACY_STATUS_PROPERTY) ? this.legacyName:this.workflowStatusName;
64     }
65     
66     @Override
67     public String toString() {
68        return getWorkflowStatusName();
69     }
70  }
