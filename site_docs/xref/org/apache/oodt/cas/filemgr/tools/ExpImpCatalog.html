97                      + dUrl + "]");
98              throw new RuntimeException(e);
99          }

101         this.ensureUnique = unique;

106         this.ensureUnique = unique;
107 
108         LOG.log(Level.INFO, "Constructing tool using catalog interfaces");
109         // first load the source prop file
110         try {
111             System.getProperties().load(

142     public void doExpImport(List sourceProductTypes)
143         throws FileManagerException, RepositoryManagerException, CatalogException {
144 
145         if (this.sourceClient != null && this.destClient != null) {
146             // do validation of source/dest types
147             // otherwise, user is on their own discretion
148             List destProductTypes = destClient.getProductTypes();
149 
150             if (!typesExist(sourceProductTypes, destProductTypes)) {
151                 throw new FileManagerException(
152                         "The source product types must be present in the dest file manager!");
153             } else {
154                 LOG
155                         .log(Level.INFO,
156                                 "Source types and Dest types match: beginning processing");
157             }
158         } else {
159           LOG.log(Level.INFO,
160               "Skipping type validation: catalog i/f impls being used.");
161         }
162 
163         // we'll use the get product page method for each product type
164         // paginate through products using source product type
165 
166       for (Object sourceProductType : sourceProductTypes) {
167         ProductType type = (ProductType) sourceProductType;
168         try {
169           exportTypeToDest(type);
170         } catch (CatalogException e) {
171           LOG.log(Level.WARNING, "Error exporting product type: ["
172                                  + type.getName() + "] from source to dest: Message: "
173                                  + e.getMessage(), e);
174           throw e;
175         }
176       }
177 
178     }

180     public void doExpImport() throws RepositoryManagerException, FileManagerException, CatalogException {
181         if (sourceClient == null) {
182           throw new RuntimeException(
183               "Cannot request exp/imp of all product types if no filemgr url specified!");
184         }
185         List sourceProductTypes = sourceClient.getProductTypes();
186         doExpImport(sourceProductTypes);
187     }

189     private void exportTypeToDest(ProductType type) throws CatalogException, RepositoryManagerException {
190         ProductPage page;
191 
192         if (this.srcCatalog != null) {
193             page = srcCatalog.getFirstPage(type);
194         } else {
195             page = sourceClient.getFirstPage(type);
196         }
197 
198         if (page == null) {
199           return;
200         }
201 
202         exportProductsToDest(page.getPageProducts(), type);
203         while (!page.isLastPage()) {
204             if (this.srcCatalog != null) {
205                 page = srcCatalog.getNextPage(type, page);
206             } else {
207               page = sourceClient.getNextPage(type, page);
208             }
209             if (page == null) {
210               break;
211             }
212             exportProductsToDest(page.getPageProducts(), type);
213         }
214     }

216     private void exportProductsToDest(List products, ProductType type)
217         throws CatalogException, RepositoryManagerException {
218         if (products != null && products.size() > 0) {
219           for (Object product : products) {
220             Product p = (Product) product;
221 
222             if (ensureUnique) {
223               boolean hasProduct = safeHasProductTypeByName(p
224                   .getProductName());
225               if (hasProduct) {
226                 LOG.log(Level.INFO, "Skipping product: ["
227                                     + p.getProductName()
228                                     + "]: ensure unique enabled: "
229                                     + "product exists in dest catalog");
230                 continue;
231               }
232             }
233 
234             p.setProductType(type);
235             if (sourceClient != null) {
236               p
237                   .setProductReferences(sourceClient
238                       .getProductReferences(p));
239             } else {
240               p.setProductReferences(srcCatalog.getProductReferences(p));
241             }
242 
243             Metadata met;
244 
245             if (sourceClient != null) {
246               met = sourceClient.getMetadata(p);
247             } else {
248               met = srcCatalog.getMetadata(p);
249             }
250 
251             LOG
252                 .log(
253                     Level.INFO,
254                     "Source Product: ["
255                     + p.getProductName()
256                     + "]: Met Extraction and "
257                     + "Reference Extraction successful: writing to dest file manager");
258 
259             // OODT-543
260             if (sourceClient != null) {
261               // remove the default CAS fields for metadata
262               met.removeMetadata("CAS.ProductId");
263               met.removeMetadata("CAS.ProductReceivedTime");
264               met.removeMetadata("CAS.ProductName");
265             }
266 
267             Product destProduct = new Product();
268             // copy through
269             destProduct.setProductName(p.getProductName());
270             destProduct.setProductStructure(p.getProductStructure());
271             destProduct.setProductType((destClient != null) ? destClient
272                 .getProductTypeById(type.getProductTypeId()) : type);
273             destProduct.setTransferStatus(p.getTransferStatus());
274 
275             LOG.log(Level.INFO, "Cataloging Product: ["
276                                 + p.getProductName() + "]");
277             String destProductId;
278             if (destCatalog != null) {
279               destCatalog.addProduct(destProduct);
280               destProductId = destProduct.getProductId();
281             } else {
282               destProductId = destClient.catalogProduct(destProduct);
283             }
284             LOG.log(Level.INFO, "Catalog successful: dest product id: ["
285                                 + destProductId + "]");
286             destProduct.setProductId(destProductId);
287 
288             LOG.log(Level.INFO, "Adding references for dest product: ["
289                                 + destProductId + "]");
290             destProduct.setProductReferences(p.getProductReferences());
291             if (destCatalog != null) {
292               destCatalog.addProductReferences(destProduct);
293             } else {
294               destClient.addProductReferences(destProduct);
295             }
296             LOG.log(Level.INFO,
297                 "Reference addition successful for dest product: ["
298                 + destProductId + "]");
299 
300             LOG.log(Level.INFO, "Adding metadata for dest product: ["
301                                 + destProductId + "]");
302             if (destCatalog != null) {
303               destCatalog.addMetadata(met, destProduct);
304             } else {
305               destClient.addMetadata(destProduct, met);
306             }
307             LOG.log(Level.INFO,
308                 "Met addition successful for dest product: ["
309                 + destProductId + "]");
310 
311             LOG.log(Level.INFO, "Successful import of product: ["
312                                 + p.getProductName() + "] into dest file manager");
313           }
314         }

335     public static void main(String[] args)
336         throws RepositoryManagerException, FileManagerException, MalformedURLException, InstantiationException,
337         CatalogException {
338         String sourceUrl = null, destUrl = null, srcCatPropFile = null, destCatPropFile = null;
339         boolean unique = false;
340         List types = null;
341 
342         String usage = "ExpImpCatalog [options] \n" + "--source <url>\n"
343                 + "--dest <url>\n " + "--unique\n"
344                 + "[--types <comma separate list of product type names>]\n"
345                 + "[--sourceCatProps <file> --destCatProps <file>]\n";
346 
347         for (int i = 0; i < args.length; i++) {
348             if (args[i].equals("--source")) {
349                 sourceUrl = args[++i];
350             } else if (args[i].equals("--dest")) {
351                 destUrl = args[++i];
352             } else if (args[i].equals("--unique")) {
353                 unique = true;
354             } else if (args[i].equals("--types")) {
355                 String[] typesAndIdsEnc = args[++i].split(",");
356 
357                 types = new Vector(typesAndIdsEnc.length);
358               for (String aTypesAndIdsEnc : typesAndIdsEnc) {
359                 String[] typeIdToks = aTypesAndIdsEnc.split("\\|");
360                 ProductType type = new ProductType();
361                 type.setName(typeIdToks[0]);
362                 type.setProductTypeId(typeIdToks[1]);
363                 types.add(type);
364               }
365             } else if (args[i].equals("--sourceCatProps")) {
366                 srcCatPropFile = args[++i];
367             } else if (args[i].equals("--destCatProps")) {
368                 destCatPropFile = args[++i];
369             }
370         }
371 
372         if (((sourceUrl == null || destUrl == null) && (srcCatPropFile == null || destCatPropFile == null)) || (
373             sourceUrl != null && destUrl != null && (srcCatPropFile != null || destCatPropFile != null))) {
374             System.err.println(usage);
375             System.exit(1);
376         }
377 
378         ExpImpCatalog tool;
379 
380         if (srcCatPropFile != null) {
381             tool = new ExpImpCatalog(srcCatPropFile, destCatPropFile, unique);
382         } else {
383           tool = new ExpImpCatalog(new URL(sourceUrl), new URL(destUrl),
384               unique);
385         }
386 
387         if (types != null && types.size() > 0) {
388             tool.doExpImport(types);
389         } else {
390           tool.doExpImport();
391         }
392     }

394     private boolean typesExist(List sourceList, List destList) {
395         if (sourceList == null || (sourceList.size() == 0)) {
396             return false;
397         }
398 
399         if (destList == null || (destList.size() == 0)) {
400             return false;
401         }
402 
403         // iterate through the source types and try and find the type in the
404         // destList
405       for (Object aSourceList : sourceList) {
406         ProductType type = (ProductType) aSourceList;
407         if (!typeInList(type, destList)) {
408           LOG.log(Level.WARNING, "Source type: [" + type.getName()
409                                  + "] not present in dest file manager");
410           return false;
411         }
412       }
413 
414         return true;
415     }
416 
417     private boolean typeInList(ProductType type, List typeList) {
418         if (typeList == null || (typeList.size() == 0)) {
419             return false;
420         }
421 
422       for (Object aTypeList : typeList) {
423         ProductType destType = (ProductType) aTypeList;
424         if (destType.getProductTypeId().equals(type.getProductTypeId())
425             && destType.getName().equals(type.getName())) {
426           return true;
427         }
428       }
429 
430         return false;
431     }

433     private boolean safeHasProductTypeByName(String productName) {
434 
435         if (destCatalog != null) {
436             try {
437                 return (destCatalog.getProductByName(productName) != null);
438             } catch (CatalogException e) {
439                 LOG.log(Level.SEVERE, e.getMessage());
440                 LOG
441                         .log(Level.WARNING,
442                                 "Exceptiong checking for product type by name: ["
443                                         + productName + "]: Message: "
444                                         + e.getMessage());
445                 return false;
446             }
447         } else {
448             try {
449                 return destClient.hasProduct(productName);
450             } catch (CatalogException e) {
451                 LOG.log(Level.SEVERE, e.getMessage());
452                 LOG
453                         .log(Level.WARNING,
454                                 "Exceptiong checking for product type by name: ["
455                                         + productName + "]: Message: "
456                                         + e.getMessage());
457                 return false;
458             }
459 
460         }
