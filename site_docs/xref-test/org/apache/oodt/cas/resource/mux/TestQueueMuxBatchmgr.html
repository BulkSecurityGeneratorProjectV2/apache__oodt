39  public class TestQueueMuxBatchmgr extends TestCase {
40  
41      private QueueMuxBatchManager queue;
42      private MockBatchManager mock1;
43      private MockBatchManager mock2;
44  
45      protected void setUp() {
46          BackendManager back = new StandardBackendManager();
47          back.addSet("queue-1", null,(mock1 = new MockBatchManager()), null);
48          back.addSet("queue-2", null,(mock2 = new MockBatchManager()), null);
49          queue = new QueueMuxBatchManager(back);       
50      }
51  
52      public void testExecuteRemotely() {
53          try {
54              
55              //Test that the jobs are put in seperate mock-backends based on queues
56              ResourceNode node1 = new ResourceNode();
57              ResourceNode node2 = new ResourceNode();
58      
59              JobSpec spec1 = this.getSpecFromQueue("queue-1");
60              queue.executeRemotely(spec1, node1);
61      
62              JobSpec spec2 = this.getSpecFromQueue("queue-2");
63              queue.executeRemotely(spec2, node2);
64              //Yes...use reference equality, as these must be the exact same object
65              TestCase.assertEquals(spec1,mock1.getCurrentJobSpec());
66              TestCase.assertEquals(spec2,mock2.getCurrentJobSpec());
67              TestCase.assertEquals(node1,mock1.getCurrentResourceNode());
68              TestCase.assertEquals(node2,mock2.getCurrentResourceNode());
69              //Throws exception on bad queue
70              try {
71                  queue.executeRemotely(this.getSpecFromQueue("queue-3"),node1);
72                  TestCase.fail("Failed to throw JobExecutionException on unknown queue.");
73              } catch(JobExecutionException ignored) {}
74          } catch (JobExecutionException e) {
75             TestCase.fail("Unexpected Exception: "+e.getMessage());
76          }
77      }
78  
79      public void testKillJob() {
80          try {
81              ResourceNode node1 = new ResourceNode();
82              ResourceNode node2 = new ResourceNode();
83      
84              JobSpec spec1 = this.getSpecFromQueue("queue-1");
85              queue.executeRemotely(spec1, node1);
86      
87              JobSpec spec2 = this.getSpecFromQueue("queue-2");
88              queue.executeRemotely(spec2, node2);
89              //Make sure that one can kill a job, and the other job is running
90              TestCase.assertTrue(queue.killJob(spec1.getJob().getId(), node1));
91              TestCase.assertEquals(mock1.getCurrentJobSpec(),null);
92              TestCase.assertEquals(mock2.getCurrentJobSpec(),spec2);
93              //Make sure kill fails with bad queue
94              TestCase.assertFalse(queue.killJob(this.getSpecFromQueue("queue-3").getJob().getId(), node1));
95          } catch (JobExecutionException e) {
96              TestCase.fail("Unexpected Exception: "+e.getMessage());
97          }
98      }
99  
100     public void testGetExecNode() {
101         try {
102             ResourceNode node1 = new ResourceNode();
103             ResourceNode node2 = new ResourceNode();
104             node1.setId("Node1-ID");
105             node2.setId("Node2-ID");           
106             JobSpec spec1 = this.getSpecFromQueue("queue-1");
107             queue.executeRemotely(spec1, node1);
108     
109             JobSpec spec2 = this.getSpecFromQueue("queue-2");
110             queue.executeRemotely(spec2, node2);
111             //Make that the execution node is same
112             TestCase.assertEquals(node1.getNodeId(),queue.getExecutionNode(spec1.getJob().getId()));
113             TestCase.assertEquals(node2.getNodeId(),queue.getExecutionNode(spec2.getJob().getId()));
114             //Returns null, if bad-queue
115             TestCase.assertNull(queue.getExecutionNode(this.getSpecFromQueue("queue-3").getJob().getId()));
116         } catch (JobExecutionException e) {
117             TestCase.fail("Unexpected Exception: "+e.getMessage());
118         }
119     }

121     private JobSpec getSpecFromQueue(String queue) {
122         JobSpec spec1 = new JobSpec();
123         Job job1 = new Job();
124         job1.setId("000000100000011-"+queue);
125         job1.setQueueName(queue);
126         spec1.setJob(job1);       
127         return spec1;
128     }
