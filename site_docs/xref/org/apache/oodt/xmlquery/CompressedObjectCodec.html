42  /** A result encoder/decoder for compressed, serialized objects.
43   *
44   * This codec uses a GZIP compressed serialized object format for objects.
45   *
46   * @author Kelly
47   */
48  class CompressedObjectCodec implements Codec {
49  	public Node encode(Object object, Document doc) throws DOMException {
50  		ByteArrayOutputStream byteArray = new ByteArrayOutputStream();
51  		try {
52  			Base64EncodingOutputStream base64 = new Base64EncodingOutputStream(byteArray);
53  			GZIPOutputStream gzip = new GZIPOutputStream(base64);
54  			ObjectOutputStream objStream = new ObjectOutputStream(gzip);
55  			objStream.writeObject(object);
56  			objStream.close();
57  		} catch (IOException ignored) {}
58  		Element value = doc.createElement("resultValue");
59  		value.appendChild(doc.createCDATASection(byteArray.toString()));
60  		return value;
61  	}
62  
63  	public Object decode(Node node) throws ClassNotFoundException, InvalidClassException, StreamCorruptedException,
64  		OptionalDataException {
65  		String encodedValue;
66  		if (node.getFirstChild().getNodeType() == Node.CDATA_SECTION_NODE) {
67  		  encodedValue = node.getFirstChild().getNodeValue();
68  		} else {
69  		  encodedValue = XML.text(node);
70  		}
71  		Object rc = null;
72  		try {
73  			ByteArrayInputStream byteArray = new ByteArrayInputStream(encodedValue.getBytes());
74  			Base64DecodingInputStream base64 = new Base64DecodingInputStream(byteArray);
75  			GZIPInputStream gzip = new GZIPInputStream(base64);
76  			ObjectInputStream objStream = new ObjectInputStream(gzip);
77  			rc = objStream.readObject();
78  			objStream.close();
79  		} catch (InvalidClassException ex) {
80  			throw ex;
81  		} catch (StreamCorruptedException ex) {
82  			throw ex;
83  		} catch (OptionalDataException ex) {
84  			throw ex;
85  		} catch (IOException ignored) {}
86  		return rc;
87  	}
88  
89  	public InputStream getInputStream(Object value) throws IOException {
90  		ByteArrayOutputStream baos = new ByteArrayOutputStream();
91  		ObjectOutputStream oos = new ObjectOutputStream(baos);
92  		oos.writeObject(value);
93  		oos.close();
94  		baos.close();
95  		return new ByteArrayInputStream(baos.toByteArray());
96  	}
97  
98  	public long sizeOf(Object obj) {
99  		try {
100 			CountingOutputStream c = new CountingOutputStream(new NullOutputStream());
101 			ObjectOutputStream stream = new ObjectOutputStream(c);
102 			stream.writeObject(obj);
103 			stream.close();
104 			return c.getBytesWritten();
105 		} catch (IOException ex) {
106 			throw new IllegalStateException("I/O exception " + ex.getClass().getName() + " can't happen, yet did: "
107 				+ ex.getMessage());
108 		}
109 	}
