61    private static final Logger LOG = Logger.getLogger(OFSNFileHandler.class
62        .getName());

64    private static final String CMD_SEPARATOR = ";";
65  
66    // by default return dir size on listing commands
67    private boolean computeDirSize = true;
68    
69    // by default return file size on listing commands
70    private boolean computeFileSize = true;
71  
72    private OFSNFileHandlerConfiguration conf;
73  
74    private Map<String, Object> HANDLER_CACHE;

76    public OFSNFileHandler() throws InstantiationException {
77      // init conf here
78      String xmlConfigFilePath = System.getProperty(OFSN_XML_CONF_FILE_KEY);
79      this.computeDirSize = Boolean.getBoolean(OFSN_COMPUTE_DIR_SIZE);
80      this.computeFileSize = Boolean.getBoolean(OFSN_COMPUTE_FILE_SIZE);
81  
82  
83      if (xmlConfigFilePath == null) {
84        throw new InstantiationException(
85            "Must define xml configuration file path via property : ["
86                + OFSN_XML_CONF_FILE_KEY + "]");
87      }
88  
89      try {
90        this.conf = OFSNFileHandlerConfigurationReader
91            .getConfig(xmlConfigFilePath);
92      } catch (FileNotFoundException e) {
93        throw new InstantiationException("xml configuration file: ["
94            + xmlConfigFilePath + "] not found!");
95      }
96  
97      if (this.conf.getProductRoot() == null) {
98        throw new InstantiationException(
99            "Must define: [productRoot] attribute in XML configuration!");
100     }
101 
102     // used to cache handlers -- map of RT type to Get/List handler instance
103     HANDLER_CACHE = new ConcurrentHashMap<String, Object>();
104   }

106   /*
107    * (non-Javadoc)
108    * 
109    * @see org.apache.oodt.product.QueryHandler#query(org.apache.oodt.xmlquery.XMLQuery)
110    */
111   public XMLQuery query(XMLQuery xmlQuery) throws ProductException {
112     String ofsn = OFSNUtils.extractFieldFromQuery(xmlQuery, OFSN);
113     String cmd = OFSNUtils.extractFieldFromQuery(xmlQuery, RETURN_TYPE);
114     validate(ofsn, cmd);
115     String cmdId = ofsn + CMD_SEPARATOR + cmd;
116     OFSNHandlerConfig cfg = this.conf.getHandlerConfig(cmd);
117     validateHandlerConfig(cfg, cmd);
118 
119     String realPath = this.conf.getProductRoot() + ofsn;
120 
121     if (isListingCmd(cmd)) {
122       ByteArrayOutputStream outStream = new ByteArrayOutputStream();
123       OFSNListHandler handler = getListHandler(cmd, cfg.getClassName());
124       File[] fileList = handler.getListing(realPath);
125       generateOFSNXml(fileList, cfg, outStream);
126       xmlQuery.getResults().add(
127           new Result(cmdId, XML_MIME_TYPE, null, cmdId, Collections.EMPTY_LIST,
128               outStream.toString()));
129     } else if (isGetCmd(cmd)) {
130       OFSNGetHandler handler = getGetHandler(cmd, cfg.getClassName());
131       String rtAndPath = cmd + CMD_SEPARATOR + realPath;
132       String mimeType;
133       
134       // check for and use mimetype conf property if available
135       if (cfg.getHandlerConf().containsKey(PROPERTY_MIMETYPE_ATTR)) {
136     	  MediaType mediaType = MediaType.parse(cfg.getHandlerConf()
137     			  .getProperty(PROPERTY_MIMETYPE_ATTR));
138     	  if (mediaType == null) {
139     		  LOG.log(Level.WARNING, "MIME type ["
140     				  +cfg.getHandlerConf().getProperty(PROPERTY_MIMETYPE_ATTR)+"] specified "
141     				  +"for handler ["+cfg.getClassName()+"] invalid. Defaulting to MIME type ["
142     				  +MediaType.OCTET_STREAM.toString()+"]");
143     		  mediaType = MediaType.OCTET_STREAM;
144     	  }
145     	  mimeType = mediaType.toString();
146       } else { // use default mimetype of product on disk
147           try {
148               mimeType = MimeTypesFactory.create().getMimeType(new File(realPath)).getName();
149           } catch (Exception e) {
150               mimeType = null;
151           }
152       }
153       
154       xmlQuery.getResults().add(
155           new LargeResult(/* id */rtAndPath,/* mimeType */ mimeType, /* profileID */null, 
156         		  /* resourceID */new File(realPath).getName(), Collections.EMPTY_LIST, 
157         		  handler.sizeOf(realPath)));
158     } else {
159       throw new ProductException("return type: [" + cmd + "] is unsupported!");
160     }
161 
162     return xmlQuery;
163 
164   }

178    * @see
179    * org.apache.oodt.product.LargeProductQueryHandler#retrieveChunk(java.lang.String,
180    * long, int)

182   public byte[] retrieveChunk(String id, long offset, int length)
183       throws ProductException {
184     // unmarshall the return type and path
185     String[] rtTypeAndPathArr = id.split(CMD_SEPARATOR);
186     String rtType = rtTypeAndPathArr[0];
187     String filepath = rtTypeAndPathArr[1];
188 
189     OFSNGetHandler handler = getGetHandler(rtType, this.conf
190         .getHandlerClass(rtType));
191     
192     return handler.retrieveChunk(filepath, offset, length);
193   }

195   private void generateOFSNXml(File[] mlsFileList, OFSNHandlerConfig cfg,
196       OutputStream outStream) {
197     XMLUtils.writeXmlToStream(OFSNUtils.getOFSNDoc(Arrays.asList(mlsFileList),
198         cfg, this.conf.getProductRoot(), this.computeDirSize, this.computeFileSize),
199         outStream);
200   }

202   private void validate(String ofsn, String cmd) throws ProductException {
203     if (ofsn == null || cmd == null || (ofsn.equals("")) || (cmd.equals(""))) {
204       throw new ProductException("must specify OFSN and RT parameters!");
205     } else if (!OFSNUtils.validateOFSN(ofsn)) {
206       throw new ProductException("OFSN is invalid");
207     }
208   }
209 
210   private void validateHandlerConfig(OFSNHandlerConfig cfg, String cmd)
211       throws ProductException {
212     if (cfg == null) {
213       throw new ProductException("Unrecognized command: [" + cmd + "]!");
214     }
215   }

217   private OFSNListHandler getListHandler(String rtType, String className) {
218     if (HANDLER_CACHE.containsKey(rtType)) {
219       return (OFSNListHandler) HANDLER_CACHE.get(rtType);
220     } else {
221       OFSNListHandler handler = OFSNObjectFactory.getListHandler(className);
222       LOG.log(Level.INFO, "Getting handler config for RT: ["+rtType+"]");
223       handler.configure(this.conf.getHandlerConfig(rtType).getHandlerConf());
224       HANDLER_CACHE.put(rtType, handler);
225       return handler;
226     }
227   }
228 
229   private OFSNGetHandler getGetHandler(String rtType, String className) {
230     if (HANDLER_CACHE.containsKey(rtType)) {
231       return (OFSNGetHandler) HANDLER_CACHE.get(rtType);
232     } else {
233       OFSNGetHandler handler = OFSNObjectFactory.getGetHandler(className);
234       handler.configure(this.conf.getHandlerConfig(rtType).getHandlerConf());
235       HANDLER_CACHE.put(rtType, handler);
236       return handler;
237     }
238   }
239 
240   private boolean isListingCmd(String cmd) throws ProductException {
241     OFSNHandlerConfig cfg = this.conf.getHandlerConfig(cmd);
242     if (cfg == null) {
243       throw new ProductException("Unrecognized command: [" + cmd + "]!");
244     }
245 
246     return cfg.getType().equals(LISTING_CMD);
247   }
248 
249   private boolean isGetCmd(String cmd) {
250     OFSNHandlerConfig cfg = this.conf.getHandlerConfig(cmd);
251 
252     return cfg.getType().equals(GET_CMD);
253   }
