35  /**
36   * 
37   * @author woollard
38   * @author bfoster
39   * @version $Revision$
40   * 
41   * <p>
42   * An implementation of a {@link Scheduler} that uses a <a
43   * href="http://en.wikipedia.org/wiki/Cache_algorithms">least-recently-used</a>
44   * algorithm for scheduling {@link Job}s.
45   * </p>
46   */
47  public class LRUScheduler implements Scheduler {
48  
49      /* our log stream */
50      private static final Logger LOG = Logger.getLogger(LRUScheduler.class
51              .getName());
52      public static final double DOUBLE = 1000.0;
53  
54      private LRUQueueManager queueManager;
55      
56      /* the monitor we'll use to check the status of the resources */
57      private Monitor myMonitor;
58  
59      /* the batch mgr we'll use to execute jobs */
60      private Batchmgr myBatchmgr;
61  
62      /* our job queue */
63      private JobQueue myJobQueue;
64  
65      /* our wait time between checking the queue */
66      private double waitTime = -1;
67  
68      public LRUScheduler(Monitor m, Batchmgr b, JobQueue q, LRUQueueManager qm) {
69  
70      	queueManager = qm;
71          myMonitor = m;
72          myBatchmgr = b;
73          myJobQueue = q;
74  
75          String waitStr = System.getProperty(
76                  "org.apache.oodt.cas.resource.scheduler.wait.seconds", "20");
77          waitTime = Double.parseDouble(waitStr);
78      }
79  
80      /*
81       * (non-Javadoc)
82       * 
83       * @see java.lang.Runnable#run()
84       */
85      public void run() {
86          for (;;) {
87  
88              try {
89              	long sleepTime = (long)(waitTime * DOUBLE);
90                  Thread.currentThread().sleep(sleepTime);
91              } catch (Exception ignore) {}
92  
93              if (!myJobQueue.isEmpty()) {
94                  JobSpec exec;
95  
96                  try {
97                      exec = myJobQueue.getNextJob();
98                      LOG.log(Level.INFO, "Obtained Job: ["
99                              + exec.getJob().getId()
100                             + "] from Queue: Scheduling for execution");
101                 } catch (Exception e) {
102                     LOG.log(Level.WARNING,
103                             "Error getting next job from JobQueue: Message: "
104                                     + e.getMessage());
105                     continue;
106                 }
107 
108                 try {
109                     schedule(exec);
110                 } catch (Exception e) {
111                     LOG.log(Level.WARNING, "Error scheduling job: ["
112                             + exec.getJob().getId() + "]: Message: "
113                             + e.getMessage());
114                     // place the job spec back on the queue
115                     try {
116                         myJobQueue.requeueJob(exec);
117                     } catch (Exception ignore) {
118                     }
119                 }
120             }
121         }
122 
123     }
124 
125     /*
126      * (non-Javadoc)
127      * 
128      * @see gov.nasa.jpl.oodt.cas.resource.scheduler.Scheduler#schedule(gov.nasa.jpl.oodt.cas.resource.structs.JobSpec)
129      */
130     public synchronized boolean schedule(JobSpec spec)
131             throws SchedulerException {
132         String queueName = spec.getJob().getQueueName();
133         int load = spec.getJob().getLoadValue();
134 
135         ResourceNode node = nodeAvailable(spec);
136 
137         if (node != null) {
138             try {
139                 myMonitor.assignLoad(node, load);
140                 queueManager.usedNode(queueName, node.getNodeId());
141                 
142                 // assign via batch system
143                 LOG.log(Level.INFO, "Assigning job: ["
144                         + spec.getJob().getName() + "] to node: ["
145                         + node.getNodeId() + "]");
146                 try {
147                     myBatchmgr.executeRemotely(spec, node);
148                 } catch (JobExecutionException e) {
149                     LOG.log(Level.WARNING, "Exception executing job: ["
150                             + spec.getJob().getId() + "] to node: ["
151                             + node.getIpAddr() + "]: Message: "
152                             + e.getMessage());
153                     try {
154                         // queue the job back up
155                         LOG.log(Level.INFO, "Requeueing job: ["
156                                 + spec.getJob().getId() + "]");
157                         myJobQueue.requeueJob(spec);
158 
159                         // make sure to decrement the load
160                         myMonitor.reduceLoad(node, load);
161                     } catch (Exception ignore) {
162                     }
163                 }
164             } catch (MonitorException e) {
165                 LOG.log(Level.WARNING, "Exception assigning load to resource "
166                         + "node: [" + node.getNodeId() + "]: load: [" + load
167                         + "]: Message: " + e.getMessage());
168                 throw new SchedulerException(e.getMessage());
169             }
170         } else {
171             // could not find resource, push onto JobQueue
172             try {
173                 myJobQueue.requeueJob(spec);
174             } catch (Exception ignore) {
175             }
176         }
177         return true;
178     }

216     /*
217      * (non-Javadoc)
218      * 
219      * @see gov.nasa.jpl.oodt.cas.resource.scheduler.Scheduler#nodeAvailable(gov.nasa.jpl.oodt.cas.resource.structs.JobSpec)
220      */
221     public synchronized ResourceNode nodeAvailable(JobSpec spec)
222             throws SchedulerException {
223         try {
224 	    	String queueName = spec.getJob().getQueueName();
225 	        int load = spec.getJob().getLoadValue();
226 	
227 	        for (String nodeId : queueManager.getNodes(queueName)) {
228 	            int nodeLoad;
229 	            ResourceNode resNode = null;
230 	
231 	            try {
232 	                resNode = myMonitor.getNodeById(nodeId);
233 	                nodeLoad = myMonitor.getLoad(resNode);
234 	            } catch (MonitorException e) {
235 	                LOG
236 	                        .log(Level.WARNING, "Exception getting load on "
237 	                                + "node: [" + (resNode != null ? resNode.getNodeId() : null)
238 	                                + "]: Message: " + e.getMessage());
239 	                throw new SchedulerException(e.getMessage());
240 	            }
241 	
242 	            if (load <= nodeLoad) {
243 	                return resNode;
244 	            }
245 	        }
246 	
247 	        return null;
248         }catch (Exception e) {
249         	throw new SchedulerException("Failed to find available node for job spec : " + e.getMessage(), e);
250         }
251     }
