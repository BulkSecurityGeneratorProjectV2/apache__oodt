40   * Utility functions to help out in configuring the {@link RSSProductServlet}.

46  public class RSSUtils {
47  
48    public static String getChannelLink(String channelLink, Metadata channelMet) {
49      if (channelLink != null && !channelLink.equals("")) {
50        return PathUtils.replaceEnvVariables(channelLink, channelMet);
51      } else {
52        return channelMet.getMetadata("BaseUrl") + "/rdf/dataset?type="
53        + channelMet.getMetadata("ProductType") + "&typeID="
54        + channelMet.getMetadata("ProductTypeId");
55      }
56    }
57  
58    public static Element emitRSSTag(RSSTag tag, Metadata prodMet, Document doc,
59        Element item) {
60      String outputTag = tag.getName();
61      if (outputTag.contains(" ")) {
62        outputTag = StringUtils.join(WordUtils.capitalizeFully(outputTag).split(
63            " "));
64      }
65      Element rssMetElem = XMLUtils.addNode(doc, item, outputTag);
66  
67      // first check if there is a source defined, if so, use that as the value
68      if (tag.getSource() != null) {
69        rssMetElem.appendChild(doc.createTextNode(StringEscapeUtils.escapeXml(PathUtils.replaceEnvVariables(
70            tag.getSource(), prodMet))));
71      }
72  
73      // check if there are attributes defined, and if so, add to the attributes
74      for (RSSTagAttribute attr : tag.getAttrs()) {
75        rssMetElem.setAttribute(attr.getName(), PathUtils.replaceEnvVariables(
76            attr.getValue(), prodMet));
77      }
78  
79      return rssMetElem;
80    }
81  
82    public static RSSConfig initRSS(ServletConfig conf)
83        throws FileNotFoundException {
84      return RSSConfigReader.readConfig(new File(PathUtils
85          .replaceEnvVariables(conf.getServletContext().getInitParameter(
86              (RSS_CONTEXT_CONF_KEY)))));
87    }
