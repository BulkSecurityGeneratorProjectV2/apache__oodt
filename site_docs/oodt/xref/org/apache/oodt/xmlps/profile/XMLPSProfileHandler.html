42  /**
43   * 
44   * <p>
45   * An implementation of a {@link ProfileHandler} that extends the capabilities
46   * of the {@link XMLPSProductHandler}, and uses the XML Specification defined
47   * by the mapping file to represent its field mapping information.
48   * </p>.

50  public class XMLPSProfileHandler extends XMLPSProductHandler implements
51          ProfileHandler {

53      private DBMSExecutor executor;
54  
55      /* our log stream */
56      private static final Logger LOG = Logger.getLogger(XMLPSProfileHandler.class
57              .getName());
58  
59      private String resLocationSpec;
60  
61      public XMLPSProfileHandler() throws InstantiationException {
62          super(null);
63          String mappingFilePath = System
64                  .getProperty("org.apache.oodt.xmlps.profile.xml.mapFilePath");
65  
66          if (mappingFilePath == null) {
67              throw new InstantiationException(
68                      "Need to specify path to xml mapping file!");
69          }
70  
71          try {
72              mapping = MappingReader.getMapping(mappingFilePath);
73          } catch (Exception e) {
74              throw new InstantiationException(
75                      "Unable to parse profile mapping xml file: ["
76                              + mappingFilePath + "]: reason: "
77                              + e.getMessage());
78          }
79  
80          // load the db properties file
81                  // if one exists: otherwise, don't bother and just print out the SQL to
82          // the console.
83          // 
84          String dbPropFilePath = System
85                  .getProperty("org.apache.oodt.xmlps.profile.xml.dbPropFilePath");
86          if (dbPropFilePath != null) {
87              try {
88                  System.getProperties()
89                          .load(new FileInputStream(dbPropFilePath));
90              } catch (FileNotFoundException e) {
91                  // TODO Auto-generated catch block
92                  LOG.log(Level.SEVERE, e.getMessage());
93              } catch (IOException e) {
94                  LOG.log(Level.SEVERE, e.getMessage());
95                  throw new InstantiationException(e.getMessage());
96              }
97  
98              executor = new DBMSExecutor();
99          }
100 
101         this.resLocationSpec = System
102                 .getProperty("org.apache.oodt.xmlps.profile.xml.resLocationSpec");
103     }

105     /*
106      * (non-Javadoc)
107      * 
108      * @see org.apache.oodt.profile.handlers.ProfileHandler#findProfiles(org.apache.oodt.xmlquery.XMLQuery)
109      */
110     public List<Profile> findProfiles(XMLQuery query) throws ProfileException {
111         List<QueryElement> whereSet = query.getWhereElementSet();
112         List<QueryElement> selectSet = query.getSelectElementSet();
113         try {
114             translateToDomain(selectSet, true);
115             translateToDomain(whereSet, false);
116         } catch (Exception e) {
117             LOG.log(Level.SEVERE, e.getMessage());
118             throw new ProfileException(e.getMessage());
119         }
120         return queryAndPackageProfiles(query);
121     }

141     protected List<Profile> queryAndPackageProfiles(XMLQuery query) {
142         Stack<QueryElement> queryStack = HandlerQueryParser
143                 .createQueryStack(query.getWhereElementSet());
144         Expression parsedQuery = HandlerQueryParser.parse(queryStack,
145                 this.mapping);
146         List<Profile> profs = null;
147 
148         StringBuilder sqlBuf = new StringBuilder("SELECT *");
149         sqlBuf.append(" FROM ");
150         sqlBuf.append(mapping.getDefaultTable());
151         sqlBuf.append(" ");
152 
153         if (mapping.getNumTables() > 0) {
154             for (String tableName : mapping.getTableNames()) {
155                 if (tableName.equals(mapping.getDefaultTable())) {
156                     continue;
157                 }
158                 DatabaseTable tbl = mapping.getTableByName(tableName);
159                 sqlBuf.append("INNER JOIN ");
160                 sqlBuf.append(tbl.getName());
161                 sqlBuf.append(" ON ");
162                 sqlBuf.append(tbl.getName());
163                 sqlBuf.append(".");
164                 sqlBuf.append(tbl.getJoinFieldName());
165                 sqlBuf.append(" = ");
166                 sqlBuf.append(tbl.getDefaultTableJoin());
167                 sqlBuf.append(".");
168                 sqlBuf.append(tbl.getDefaultTableJoinFieldName());
169                 sqlBuf.append(" ");
170             }
171         }
172 
173         if (parsedQuery != null) {
174             sqlBuf.append(" WHERE ");
175             sqlBuf.append(parsedQuery.evaluate());
176         }
177 
178         LOG.log(Level.INFO, sqlBuf.toString());
179 
180         if (executor != null) {
181             try {
182                 profs = executor.executeLocalQuery(this.mapping, sqlBuf
183                         .toString(), this.resLocationSpec);
184 
185             } catch (SQLException e) {
186                 LOG.log(Level.SEVERE, e.getMessage());
187                 LOG.log(Level.WARNING, "Error executing sql: ["
188                         + sqlBuf.toString() + "]: Message: " + e.getMessage());
189             }
190         }
191 
192         return profs;
193     }

198     }
