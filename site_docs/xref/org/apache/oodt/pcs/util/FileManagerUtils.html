159   public List queryAllTypes(Query query, List excludeTypeList) {
160     List productTypes = safeGetProductTypes();
161     List products = new Vector();
162 
163     if (productTypes != null && productTypes.size() > 0) {
164       for (Object productType : productTypes) {
165         ProductType type = (ProductType) productType;
166         if (excludeTypeList != null && excludeTypeList.contains(type.getName())) {
167           continue;
168         }
169 
170         List prods = safeIssueQuery(query, type);
171         if (prods != null && prods.size() > 0) {
172           products.addAll(prods);
173         }
174       }
175     }
176 
177     return products;
178 
179   }

295   public Product getLatestProduct(Query query, ProductType type) {
296     List products;
297 
298     try {
299       products = fmgrClient.query(query, type);
300       if (products != null && products.size() > 0) {
301         Product p = (Product) products.get(0);
302         p.setProductReferences(safeGetProductReferences(p));
303         return p;
304       }
305     } catch (Exception e) {
306       LOG.log(Level.SEVERE, "Unable to obtain products: query: [" + query
307           + "]: Message: " + e.getMessage(), e);

363   public String getFilePath(Product prod) {
364     if (prod.getProductReferences() == null) {
365       prod.setProductReferences(safeGetProductReferences(prod));
366     }
367 
368     if (prod.getProductReferences() == null
369         || (prod.getProductReferences() != null && prod.getProductReferences()
370             .size() == 0)) {
371       return "N/A";
372     }
373 
374     // get the first ref
375     Reference r = (Reference) prod.getProductReferences().get(0);
376     return safeGetFileFromUri(r.getDataStoreReference()).getAbsolutePath();
377   }

463   public static Reference getRootReference(String productName, List refs) {

476   public static boolean isRootDir(Reference r, String prodName) {
477     if (!r.getOrigReference().endsWith("/")) {
478       return false;
479     } else {
480       String referenceURI = r.getOrigReference();
481       int lastFolderInPathStartIdx = referenceURI.substring(0,
482           referenceURI.length() - 1).lastIndexOf("/") + 1;
483       String lastFolderInPath = referenceURI.substring(
484           lastFolderInPathStartIdx, referenceURI.length() - 1);
485       return lastFolderInPath.startsWith(prodName);
486     }
