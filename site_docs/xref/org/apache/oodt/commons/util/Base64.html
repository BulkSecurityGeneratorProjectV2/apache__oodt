27  /** Base 64 encoding and decoding.
28   *
29   * This class provides methods for RFC-1521 specified "base 64" encoding and decoding of
30   * arbitrary data.  Pass a byte array into the {@link #encode} method and you'll get a
31   * byte array result where all of the bytes are printable ASCII values.  Pass that result
32   * into {@link #decode} and you'll get your original byte array.
33   *
34   * <p>Sincere thanks to Tom Daley for providing a sample encoder algorithm and a great
35   * explanation of how RFC-1521 is supposed to work.
36   *
37   * @author Kelly
38   */
39  public class Base64 {
40  	/** Encode into base 64.
41  	 *
42  	 * Encode the given data into RFC-1521 base 64.  Encoding a null array gives a
43  	 * null result.
44  	 *
45  	 * @param data The data to encode.
46  	 * @return Base-64 encoded <var>data</var>.
47  	 */
48  	public static byte[] encode(final byte[] data) {
49  		return encode(data, 0, data.length);
50  	}

52  	/** Encode into base 64.
53  	 *
54  	 * Encode the given data into RFC-1521 base 64.  Encoding a null array gives a
55  	 * null result.  Start encoding at the given offset and go for the given amount of
56  	 * bytes.
57  	 *
58  	 * @param data The data to encode.
59  	 * @param offset Where to start looking for data to encode.
60  	 * @param length How much data to encode.
61  	 * @return Base-64 encoded <var>data</var>
62  	 */
63  	public static byte[] encode(final byte[] data, int offset, int length) {
64  		if (data == null) {
65  		  return null;
66  		}
67  		if (offset < 0 || offset > data.length) {
68  		  throw new IndexOutOfBoundsException("Can't encode at index " + offset + " which is beyond array bounds 0.."
69  											  + data.length);
70  		}
71  		if (length < 0) {
72  		  throw new IllegalArgumentException("Can't encode a negative amount of data");
73  		}
74  		if (offset + length > data.length) {
75  		  throw new IndexOutOfBoundsException("Can't encode beyond right edge of array");
76  		}
77  		
78  		int i, j;
79  		byte dest[] = new byte[((length+2)/3)*4];
80  
81  		// Convert groups of 3 bytes into 4.
82  		for (i = offset, j = 0; i < offset + length - 2; i += 3) {
83  			dest[j++] = (byte) ((data[i] >>> 2) & 077);
84  			dest[j++] = (byte) ((data[i+1] >>> 4) & 017 | (data[i] << 4) & 077);
85  			dest[j++] = (byte) ((data[i+2] >>> 6) & 003 | (data[i+1] << 2) & 077);
86  			dest[j++] = (byte) (data[i+2] & 077);
87  		}
88  
89  		// Convert any leftover bytes.
90  		if (i < offset + length) {
91  			dest[j++] = (byte) ((data[i] >>> 2) & 077);
92  			if (i < offset + length - 1) {
93  				dest[j++] = (byte) ((data[i+1] >>> 4) & 017 | (data[i] << 4) & 077);
94  				dest[j++] = (byte) ((data[i+1] << 2) & 077);
95  			} else {
96  			  dest[j++] = (byte) ((data[i] << 4) & 077);
97  			}
98  		}
99  
100 		// Now, map those onto base 64 printable ASCII.
101 		for (i = 0; i <j; i++) {
102 			if      (dest[i] < 26) {
103 			  dest[i] = (byte) (dest[i] + 'A');
104 			} else if (dest[i] < 52) {
105 			  dest[i] = (byte) (dest[i] + 'a' - 26);
106 			} else if (dest[i] < 62) {
107 			  dest[i] = (byte) (dest[i] + '0' - 52);
108 			} else if (dest[i] < 63) {
109 			  dest[i] = (byte) '+';
110 			} else {
111 			  dest[i] = (byte) '/';
112 			}
113 		}
114 
115 		// Pad the result with and we're done.
116 		for (; i < dest.length; i++) {
117 		  dest[i] = (byte) '=';
118 		}
119 		return dest;
120 	}

122 	/** Decode from base 64.
123 	 *
124 	 * Decode the given RFC-1521 base 64 encoded bytes into the original data they
125 	 * represent.  Decoding null data gives a null result.
126 	 *
127 	 * @param data Base-64 encoded data to decode.
128 	 * @return Decoded <var>data</var>.
129 	 */
130 	public static byte[] decode(final byte[] data) {
131 		return decode(data, 0, data.length);
132 	}

134 	/** Decode from base 64.
135 	 *
136 	 * Decode the given RFC-1521 base 64 encoded bytes into the original data they
137 	 * represent.  Decoding null data gives a null result.
138 	 *
139 	 * @param data Base-64 encoded data to decode.
140 	 * @param offset Where to start looking for data to decode.
141 	 * @param length How much data to decode.
142 	 * @return Decoded <var>data</var>.
143 	 */
144 	public static byte[] decode(final byte[] data, int offset, int length) {
145 		if (data == null) {
146 		  return null;
147 		}
148 		if (offset < 0 || offset >= data.length) {
149 		  throw new IndexOutOfBoundsException("Can't decode at index " + offset + " which is beyond array bounds 0.."
150 											  + (data.length - 1));
151 		}
152 		if (length < 0) {
153 		  throw new IllegalArgumentException("Can't decode a negative amount of data");
154 		}
155 		if (offset + length > data.length) {
156 		  throw new IndexOutOfBoundsException("Can't decode beyond right edge of array");
157 		}
158 
159 		// Ignore any padding at the end.
160 		int tail = offset + length - 1;
161 		while (tail >= offset && data[tail] == '=') {
162 		  --tail;
163 		}
164 		byte dest[] = new byte[tail + offset + 1 - length/4];
165 
166 		// First, convert from base-64 ascii to 6 bit bytes.
167 		for (int i = offset; i < offset+length; i++) {
168 			if      (data[i] == '=') {
169 			  data[i] = 0;
170 			} else if (data[i] == '/') {
171 			  data[i] = 63;
172 			} else if (data[i] == '+') {
173 			  data[i] = 62;
174 			} else if (data[i] >= '0' && data[i] <= '9') {
175 			  data[i] = (byte) (data[i] - ('0' - 52));
176 			} else if (data[i] >= 'a'  &&  data[i] <= 'z') {
177 			  data[i] = (byte) (data[i] - ('a' - 26));
178 			} else if (data[i] >= 'A'  &&  data[i] <= 'Z') {
179 			  data[i] = (byte) (data[i] - 'A');
180 			}
181 		}
182 
183 		// Map those from 4 6-bit byte groups onto 3 8-bit byte groups.
184 		int i, j;
185 		for (i = 0 + offset, j = 0; j < dest.length - 2; i += 4, j += 3) {
186 			dest[j]   = (byte) (((data[i] << 2) & 255) | ((data[i+1] >>> 4) & 003));
187 			dest[j+1] = (byte) (((data[i+1] << 4) & 255) | ((data[i+2] >>> 2) & 017));
188 			dest[j+2] = (byte) (((data[i+2] << 6) & 255) | (data[i+3] & 077));
189 		}
190 
191 		// And get the leftover ...
192 		if (j < dest.length) {
193 		  dest[j] = (byte) (((data[i] << 2) & 255) | ((data[i + 1] >>> 4) & 003));
194 		}
195 		if (++j < dest.length) {
196 		  dest[j] = (byte) (((data[i + 1] << 4) & 255) | ((data[i + 2] >>> 2) & 017));
197 		}
198 
199 		// That's it.
200 		return dest;
201 	}

209 	/** Command-line runner that encodes or decodes.
210 	 *
211 	 * @param argv Command-line arguments.
212 	 */
213 	public static void main(String[] argv) throws IOException {
214 		if (argv.length < 1 || argv.length > 2) {
215 			System.err.println("Usage: encode|decode [file]");
216 			System.exit(1);
217 		}
218 		boolean encode = true;
219 		if ("encode".equals(argv[0])) {
220 		  encode = true;
221 		} else if ("decode".equals(argv[0])) {
222 		  encode = false;
223 		} else {
224 			System.err.println("Specify either \"encode\" or \"decode\"");
225 			System.exit(1);
226 		}
227 		InputStream source = argv.length == 2? new BufferedInputStream(new FileInputStream(argv[1])) : System.in;
228 		InputStream in;
229 		OutputStream out;
230 		if (encode) {
231 			in = source;
232 			out = new Base64EncodingOutputStream(System.out);
233 		} else {
234 			in = new Base64DecodingInputStream(source);
235 			out = System.out;
236 		}
237 		byte[] buf = new byte[512];
238 		int numRead;
239 		while ((numRead = in.read(buf)) != -1) {
240 		  out.write(buf, 0, numRead);
241 		}
242 		in.close();
243 		out.close();
244 		System.exit(0);
245 	}
