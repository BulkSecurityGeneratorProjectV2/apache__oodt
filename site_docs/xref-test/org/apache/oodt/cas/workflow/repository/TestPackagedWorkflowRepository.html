95    @Test
96    public void testDetectInnerWorkflows() {
97      assertNotNull(this.repo);
98      List<String> events = null;
99  
100     try {
101       events = this.repo.getRegisteredEvents();
102     } catch (Exception e) {
103       fail(e.getMessage());
104     }
105 
106     boolean foundFour = false, foundThree = false;
107     for (String event : events) {
108       List<Workflow> workflows = null;
109 
110       try {
111         workflows = this.repo.getWorkflowsForEvent(event);
112       } catch (Exception e) {
113         fail(e.getMessage());
114       }
115 
116       assertNotNull(workflows);
117       assertTrue(workflows.size() > 0);
118       if (workflows.size() == 3) {
119         foundThree = true;
120       } else if (workflows.size() == 4) {
121         foundFour = true;
122       }
123 
124     }
125 
126     assertTrue(foundThree);
127     assertTrue(foundFour);
128   }
