17  package org.apache.oodt.cas.cli.option.handler;
18  
19  //OODT imports
20  import org.apache.oodt.cas.cli.action.CmdLineAction;
21  import org.apache.oodt.cas.cli.option.CmdLineOption;
22  import org.apache.oodt.cas.cli.option.CmdLineOptionInstance;
23  
24  /**
25   * Handles a {@link CmdLineOption}'s values in relation to given
26   * {@link CmdLineAction}s.
27   * 
28   * @author bfoster (Brian Foster)
29   */
30  public interface CmdLineOptionHandler {
31  
32     /**
33      * Called after handler construction to allow handler to setup
34      * state before it is required to handle the option later. This
35      * is also called when help is run so allows registration to take
36      * place if necessary for help analysis.
37      */
38     void initialize(CmdLineOption option);
39  
40     void handleOption(CmdLineAction selectedAction,
41                       CmdLineOptionInstance optionInstance);
42  
43     /**
44      * Gets the {@link CmdLineOptionHandler}s help message when associated with
45      * given {@link CmdLineOption}.
46      * 
47      * @param option
48      *           The {@link CmdLineOption} to which this
49      *           {@link CmdLineOptionHandler} was associated with
50      * @return The help message for this {@link CmdLineOptionHandler}
51      */
52     String getHelp(CmdLineOption option);
53  
54     /**
55      * If this handler causes the argument descriptor to be different for
56      * certain {@link CmdLineAction}s, then should return the arg
57      * description here.
58      */
59     String getArgDescription(CmdLineAction action,
60                              CmdLineOption option);
61  }
