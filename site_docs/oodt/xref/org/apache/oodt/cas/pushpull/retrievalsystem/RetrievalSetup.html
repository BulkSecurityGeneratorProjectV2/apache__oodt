65      public RetrievalSetup(Config config, SiteInfo siteInfo) {
66          this.downloadingProps = false;
67          this.config = config;
68          this.siteInfo = siteInfo;
69          alreadyProcessedPropFiles = new HashSet<File>();
70          classToRmMap = new ConcurrentHashMap<Class<RetrievalMethod>, RetrievalMethod>();
71          linker = new DataFileToPropFileLinker();
72      }

74      public void retrieveFiles(PropFilesInfo pfi, final DataFilesInfo dfi) {
75  
76          FileRetrievalSystem dataFilesFRS = null;
77          try {
78              this.startPropFileDownload(pfi);
79  
80              (dataFilesFRS = new FileRetrievalSystem(config, siteInfo))
81                      .initialize();
82              dataFilesFRS.registerDownloadListener(linker);
83  
84              File[] propFiles;
85              while ((propFiles = getCurrentlyDownloadedPropFiles(pfi)).length > 0
86                      || downloadingProps) {
87                  for (File propFile : propFiles) {
88                      try {
89                          Parser parser = pfi.getParserForFile(propFile);
90                          Class<RetrievalMethod> rmClass = config.getParserInfo()
91                                  .getRetrievalMethod(parser);
92                          RetrievalMethod rm;
93                          if ((rm = this.classToRmMap.get(rmClass)) == null) {
94                              LOG.log(Level.INFO, "Creating '"
95                                      + rmClass.getCanonicalName()
96                                      + "' to download data files");
97                              rm = PushPullObjectFactory
98                                      .createNewInstance(rmClass);
99                              this.classToRmMap.put(rmClass, rm);
100                         }
101                         rm.processPropFile(dataFilesFRS, parser, propFile, dfi,
102                                 linker);
103                     } catch (ParserException e) {
104                         LOG.log(Level.SEVERE, "Failed to parse property file "
105                                 + propFile + " : " + e.getMessage(), e);
106                         linker.markAsFailed(propFile,
107                                 "Failed to parse property file " + propFile
108                                         + " : " + e.getMessage());
109                     } catch (Exception e) {
110                         LOG.log(Level.SEVERE,
111                                 "Failed to finish downloading per property files "
112                                         + propFile.getAbsolutePath() + " : "
113                                         + e.getMessage(), e);
114                         linker.markAsFailed(propFile,
115                                 "Error while downloading per property file "
116                                         + propFile.getAbsolutePath() + " : "
117                                         + e.getMessage());
118                     }
119                 }
120 
121                 dataFilesFRS.waitUntilAllCurrentDownloadsAreComplete();
122 
123                 for (File propFile : propFiles) {
124                     try {
125                         if (pfi.getLocalDir().equals(pfi.getOnSuccessDir())
126                                 || pfi.getLocalDir().equals(pfi.getOnFailDir())) {
127                           alreadyProcessedPropFiles.add(propFile);
128                         }
129                         this.movePropsFileToFinalDestination(pfi, propFile,
130                                 linker.getErrorsAndEraseLinks(propFile));
131                     } catch (Exception e) {
132                         LOG.log(Level.SEVERE, e.getMessage());
133                         LOG.log(Level.SEVERE,
134                                 "Error occurred while writing errors to error dir for file '"
135                                         + propFile + "' : " + e.getMessage());
136                     }
137                 }
138             }
139 
140         } catch (Exception e) {
141             LOG.log(Level.SEVERE, e.getMessage());
142         } finally {
143             if (dataFilesFRS != null) {
144               dataFilesFRS.shutdown();
145             }
146             alreadyProcessedPropFiles.clear();
147             linker.clear();
148         }
149     }

151     private void startPropFileDownload(final PropFilesInfo pfi) {
152         if (pfi.needsToBeDownloaded()) {
153             this.downloadingProps = true;
154             new Thread(new Runnable() {
155 
156                 @Override
157                public void run() {
158                     FileRetrievalSystem frs = null;
159                     try {
160                         (frs = new FileRetrievalSystem(
161                                 RetrievalSetup.this
162                                         .createPropFilesConfig(config
163                                                 .getProtocolInfo()), siteInfo))
164                                 .initialize();
165 
166                         LinkedList<File> propDirStructFiles = pfi
167                                 .getDownloadInfoPropFiles();
168                         for (File dirStructFile : propDirStructFiles) {
169                             Parser parser = pfi.getParserForFile(dirStructFile);
170                             Class<RetrievalMethod> rmClass = config
171                                     .getParserInfo().getRetrievalMethod(parser);
172                             RetrievalMethod rm;
173                             if ((rm = RetrievalSetup.this.classToRmMap
174                                     .get(rmClass)) == null) {
175                                 LOG.log(Level.INFO, "Creating '"
176                                         + rmClass.getCanonicalName()
177                                         + "' to download property files");
178                                 rm = PushPullObjectFactory
179                                         .createNewInstance(rmClass);
180                                 RetrievalSetup.this.classToRmMap.put(rmClass,
181                                         rm);
182                             }
183                             rm.processPropFile(frs, parser, dirStructFile,
184                                     new DataFilesInfo(null, pfi
185                                             .getDownloadInfo()), linker);
186                         }
187                     } catch (Exception e) {
188                         LOG.log(Level.SEVERE, e.getMessage());
189                     } finally {
190                         if (frs != null) {
191                           frs.shutdown();
192                         }
193                         RetrievalSetup.this.downloadingProps = false;
194                     }
195                 }
196 
197             }).start();
198 
199             // give property file download a 5 sec head start
200             LOG
201                     .log(
202                             Level.INFO,
203                             "Waiting data download thread for 5 secs to give the property files download thread a head start");
204             synchronized (this) {
205                 try {
206                     this.wait(TIMEOUT);
207                 } catch (Exception ignored) {
208                 }
209             }
210         }
211     }

213     private File[] getCurrentlyDownloadedPropFiles(final PropFilesInfo pfi) {
214         File[] files = pfi.getLocalDir().listFiles(new FileFilter() {
215             @Override
216             public boolean accept(File pathname) {
217                 return pfi.getParserForFile(pathname) != null
218                         && !(pathname.getName().startsWith("Downloading_")
219                                 || pathname.getName().endsWith(
220                                         RetrievalSetup.this.config
221                                                 .getMetFileExtension()) || alreadyProcessedPropFiles
222                                 .contains(pathname));
223             }
224         });
225         return files == null ? new File[0] : files;
226     }

228     private Config createPropFilesConfig(ProtocolInfo pi) {
229         Config propConfig = this.config.clone();
230         ProtocolInfo propPI = pi.clone();
231         propPI.setPageSize(-1);
232         propConfig.setProtocolInfo(propPI);
233         propConfig.setUseTracker(false);
234         propConfig.setIngester(null);
235         propConfig.setOnlyDownloadDefinedTypes(false);
236         return propConfig;
237     }

239     private void movePropsFileToFinalDestination(PropFilesInfo pfi,
240             File dirstructFile, String errorMsgs) throws IOException {
241         File metFile = new File(
242             String.format("%s.%s", dirstructFile.getAbsolutePath(), config.getMetFileExtension()));
243         if (pfi.getDeleteOnSuccess() && errorMsgs == null) {
244             dirstructFile.delete();
245             metFile.delete();
246             return;
247         }
248         File moveToDir = pfi.getFinalDestination(errorMsgs == null);
249         moveToDir.mkdirs();
250         File newLoc = new File(moveToDir, dirstructFile.getName());
251         dirstructFile.renameTo(newLoc);
252         metFile.renameTo(new File(
253             String.format("%s.%s", newLoc.getAbsolutePath(), config.getMetFileExtension())));
254         if (errorMsgs != null) {
255             File errorFile = new File(newLoc.getParentFile(), dirstructFile
256                     .getName()
257                     + ".errors");
258             errorFile.createNewFile();
259             
260             PrintStream ps = new PrintStream(new FileOutputStream(errorFile));
261             ps.print(errorMsgs);
262             ps.println();
263             ps.close();
264         }
265     }
