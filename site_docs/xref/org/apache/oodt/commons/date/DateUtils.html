101     public static int getLeapSecsForDate(Calendar utcCal) throws CommonsException {
102         long timeInMillis = utcCal.getTimeInMillis();
103         for (int i = dateAndLeapSecs.length - 1; i >= 0; i--) {
104             if (dateAndLeapSecs[i][IndexType.DATE.index] < timeInMillis) {
105                 return (int) dateAndLeapSecs[i][IndexType.LEAP_SECS.index];
106             }
107         }
108         throw new CommonsException("No Leap Second found for given date!");
109     }

111     public static synchronized Calendar toTai(Calendar cal) throws CommonsException {
112         Calendar taiCal = Calendar.getInstance(createTaiTimeZone(getLeapSecsForDate(cal)));
113         taiCal.setTimeInMillis(cal.getTimeInMillis() + getLeapSecsForDate(cal) * 1000);
114         return taiCal;
115     }

117     private static synchronized Calendar taiToUtc(Calendar taiCal) {
118         Calendar calUtc = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
119         calUtc.setTimeInMillis(taiCal.getTimeInMillis() - taiCal.getTimeZone().getRawOffset());
120         return calUtc;
121     }

127     public static synchronized Calendar toLocal(Calendar cal) {
128         if (cal.getTimeZone().getID().equals("TAI")) {
129             return taiToLocal(cal);
130         } else {
131             Calendar calLocal = Calendar.getInstance();
132             calLocal.setTimeInMillis(cal.getTimeInMillis());
133             return calLocal;
134         }

137     public static synchronized Calendar toUtc(Calendar cal) {
138         if (cal.getTimeZone().getID().equals("TAI")) {
139             return taiToUtc(cal);

141             Calendar calUtc = Calendar.getInstance(TimeZone.getTimeZone("UTC"));

163     public static String toString(Calendar cal) {
164         String timeZoneId = cal.getTimeZone().getID();
165         if (timeZoneId.equals("UTC")) {
166             return utcFormat.format(cal.getTime());
167         }else if (timeZoneId.equals("TAI")) {
168             return taiFormat.format(cal.getTime()) + "-0000" 
169                 + (cal.getTimeZone().getRawOffset() / 1000);
170         }else {
171             return localFormat.format(cal.getTime());
172         }
173     }

181     public static synchronized Calendar toCalendar(String calString, FormatType formatType)
182             throws ParseException {
183         Calendar cal = Calendar.getInstance();
184         switch (formatType) {
185             case LOCAL_FORMAT:
186                 cal.setTimeInMillis(localFormat.parse(calString).getTime());
187                 break;
188             case TAI_FORMAT:
189                 cal.setTimeZone(createTaiTimeZone(Integer.parseInt(calString
190                     .substring(calString.length() - 2))));
191                 calString = calString.substring(0, calString.length() - 5);
192                 cal.setTimeInMillis(taiFormat.parse(calString).getTime());
193                 break;
194             case UTC_FORMAT:
195                 cal.setTimeZone(TimeZone.getTimeZone("UTC"));
196                 cal.setTimeInMillis(utcFormat.parse(calString).getTime());
197                 break;
198             default:
199                 cal.setTimeInMillis(localFormat.parse(calString).getTime());
200         }
201         return cal;
202     }

212     public static long getTimeInMillis(Calendar cal, Calendar epoch) throws CommonsException {
213         long epochDiffInMilli;
214         /**
215          * Fixes date conversion issues preventing tests passing in the UK but working elsewhere in the world.
216          */
217         if(julianEpoch.getTimeZone().getID().equals("Europe/London")){
218             epochDiffInMilli = epoch.getTimeInMillis() - (julianEpoch.getTimeInMillis()+julianEpoch.getTimeZone().getOffset(julianEpoch.getTimeInMillis())) ;
219         }else {
220             epochDiffInMilli = epoch.getTimeInMillis() - julianEpoch.getTimeInMillis() ;
221         }
222         if (cal.getTimeZone().getID().equals("TAI")) {
223             epochDiffInMilli += getLeapSecsForDate(epoch) * 1000;
224         }
225         long milliseconds = cal.getTimeInMillis();
226         return milliseconds - epochDiffInMilli;
227     }

230         return new SimpleTimeZone(leapSecs * 1000, "TAI");
