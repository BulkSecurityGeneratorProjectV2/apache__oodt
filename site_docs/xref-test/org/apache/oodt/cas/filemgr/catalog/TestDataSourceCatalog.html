267     public void testFirstProductOnlyOnFirstPage() {
268         // add catPageSize of the test Product
269         // then add a product called "ShouldBeFirstForPage.txt"
270         // make sure it's the first one on the 1st page
271 
272         Product testProd = getTestProduct();
273         Metadata met = getTestMetadata("test");
274 
275         for (int i = 0; i < catPageSize; i++) {
276             try {
277                 myCat.addProduct(testProd);
278                 myCat.addMetadata(met, testProd);
279             } catch (Exception e) {
280                 fail(e.getMessage());
281             }
282         }
283 
284         testProd.setProductName("ShouldBeFirstForPage.txt");
285         met.replaceMetadata("CAS.ProdutName", "ShouldBeFirstForPage.txt");
286 
287         try {
288             myCat.addProduct(testProd);
289             myCat.addMetadata(met, testProd);
290 
291         } catch (Exception e) {
292             fail(e.getMessage());
293         }
294 
295         try {
296             assertNotNull(myCat.getProducts());
297             assertEquals(21, myCat.getProducts().size());
298         } catch (Exception e) {
299             fail(e.getMessage());
300         }
301 
302         ProductType type = new ProductType();
303         type.setProductTypeId("urn:oodt:GenericFile");
304         type.setName("GenericFile");
305         assertNotNull(myCat.getFirstPage(type));
306         assertNotNull(myCat.getFirstPage(type).getPageProducts());
307         assertEquals(catPageSize, myCat.getFirstPage(type).getPageProducts()
308                 .size());
309         assertNotNull(myCat.getFirstPage(type).getPageProducts().get(0));
310         assertEquals("ShouldBeFirstForPage.txt", ((Product)myCat.getFirstPage(type).getPageProducts().get(0)).getProductName());
311         ProductPage page = myCat.getNextPage(type, myCat.getFirstPage(type));
312         assertNotNull(page);
313         assertNotNull(page.getPageProducts());
314         assertEquals(1, page.getPageProducts().size());
315         assertEquals(2, page.getTotalPages());
316         assertNotNull(page.getPageProducts().get(0));
317         Product retProd = ((Product) page.getPageProducts().get(0));
318         assertEquals("test", retProd.getProductName());
319 
320     }

437     public void testQuery() throws Exception {
438         // ingest first file
439         Product testProduct = getTestProduct();
440         testProduct.setProductId("23");
441         Metadata prodMet = new Metadata();
442         URL ingestUrl = this.getClass().getResource(
443             "/ingest");
444         prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
445             ingestUrl.getFile()).getCanonicalPath());
446         prodMet.addMetadata(CoreMetKeys.FILENAME, "test-file-1.txt");
447         prodMet.addMetadata("CAS.ProductName", "TestFile1");
448         prodMet.addMetadata(CoreMetKeys.PRODUCT_TYPE, "GenericFile");
449         prodMet.addMetadata("NominalDate", "2008-01-20");
450         prodMet.addMetadata("DataVersion", "3.6");
451         myCat.addMetadata(prodMet, testProduct);
452 
453         // ingest second file
454         testProduct.setProductId("24");
455         prodMet.replaceMetadata(CoreMetKeys.FILENAME, "test-file-2.txt");
456         prodMet.replaceMetadata("CAS.ProductName", "TestFile2");
457         myCat.addMetadata(prodMet, testProduct);
458 
459         // ingest thrid file
460         testProduct.setProductId("25");
461         prodMet.replaceMetadata(CoreMetKeys.FILENAME, "test-file-2.txt");
462         prodMet.replaceMetadata("CAS.ProductName", "TestFile3");
463         prodMet.replaceMetadata("DataVersion", "4.6");
464         myCat.addMetadata(prodMet, testProduct);
465 
466         // perform first query
467         Query query = new Query();
468         query
469                 .addCriterion(SqlParser
470                         .parseSqlWhereClause("CAS.ProductName != 'TestFile3' AND (Filename == 'test-file-1.txt' OR Filename == 'test-file-2.txt') AND CAS.ProductName != 'TestFile2'"));
471         System.out.println(query);
472         List<String> productIds = myCat.query(query, testProduct
473                 .getProductType());
474         System.out.println(productIds);
475         assertEquals("[23]", productIds.toString());
476 
477         // perform second query
478         query = new Query();
479         query
480                 .addCriterion(SqlParser
481                         .parseSqlWhereClause("Filename == 'test-file-1.txt' OR (Filename == 'test-file-2.txt' AND CAS.ProductName != 'TestFile2')"));
482         System.out.println(query);
483         productIds = myCat.query(query, testProduct.getProductType());
484         System.out.println(productIds);
485         assertEquals("[25, 23]", productIds.toString());
486 
487         // perform second query
488         query = new Query();
489         query
490                 .addCriterion(SqlParser
491                         .parseSqlWhereClause("NominalDate == '2008-01-20' AND DataVersion >= '3.6' AND DataVersion <= '4.0'"));
492         System.out.println(query);
493         productIds = myCat.query(query, testProduct.getProductType());
494         System.out.println(productIds);
495         assertEquals("[24, 23]", productIds.toString());
496     }
