20  import java.util.ArrayList;

22  import java.util.HashSet;
23  import java.util.List;

26  /**
27   * @author woollard
28   * @author bfoster

31   * <p>
32   * A Free Text Criteria element for a Query to the Catalog.
33   * </p>
34   * 
35   */
36  public class FreeTextQueryCriteria extends QueryCriteria {
37  
38      private static final long serialVersionUID = 1L;
39  
40      private String elementName;
41  
42      private List<String> values;
43  
44      private static final String[] noiseWords = { "a", "all", "am", "an", "and",
45              "any", "are", "as", "at", "be", "but", "can", "did", "do", "does",
46              "for", "from", "had", "has", "have", "here", "how", "i", "if",
47              "in", "is", "it", "no", "not", "of", "on", "or", "so", "that",
48              "the", "then", "there", "this", "to", "too", "up", "use", "what",
49              "when", "where", "who", "why", "you" };
50  
51      private static HashSet<String> noiseWordHash;
52  
53      /**
54       * Default constructor.
55       */
56      public FreeTextQueryCriteria() {
57          elementName = "";
58          values = new ArrayList<String>();
59  
60          noiseWordHash = new HashSet<String>();
61          Collections.addAll(noiseWordHash, noiseWords);
62      }
63  
64      /**
65       * Constructor for the FreeTextQueryECriteria Class.
66       * 
67       * @param v
68       *            A List of terms to search for.
69       */
70      public FreeTextQueryCriteria(String elementName, List<String> v) {
71          this.elementName = elementName;
72          values = v;
73  
74          noiseWordHash = new HashSet<String>();
75          Collections.addAll(noiseWordHash, noiseWords);
76      }
77  
78      /**
79       * Accessor method for the values of the element to search on.
80       * 
81       * @return The values of the element to search on as a List of Strings.
82       */
83      public List<String> getValues() {
84          return values;
85      }
86  
87      /**
88       * Mutator method for the values of the element to search on. This method
89       * should be used when keywords have been parsed out of user-entered free
90       * text. The query will JOIN on all of these values. In order to add
91       * unparsed free text to a Query, see the addFreeText method of this class.
92       *
93       */
94      public void setValue(List<String> v) {
95          this.values = v;
96      }
97  
98      /**
99       * A method for adding a value to search on to the list of values.
100      * 
101      * @param v
102      *            The value to be added to the search as a String.
103      */
104     public void addValue(String v) {
105         values.add(v);
106     }
107 
108     /**
109      * A method for adding unparsed free text to the FreeTextCriteria. Free text
110      * entered as a string is tokenized and punctuation and common words are
111      * dropped before the values are added to the query. In order to query for
112      * pre-parsed keywords, see the setValues method of this class.
113      * 
114      * @param text
115      *            The free text to be parsed and searched on.
116      */
117     public void addFreeText(String text) {
118         // remove punctuation from the text
119         text = text.replaceAll("\\p{Punct}+", "");
120 
121         // tokenize string using default delimiters
122         StringTokenizer tok = new StringTokenizer(text);
123         String token;
124 
125         // filter noise words and add to values vector
126         while (tok.hasMoreElements()) {
127             token = tok.nextToken();
128             if (!noiseWordHash.contains(token)) {
129                 values.add(token);
130             }
131         }
132     }
133 
134     /**
135      * Implementation of the abstract method inherited from QueryCriteria for
136      * accessing the element ID to search on.
137      * 
138      * @return The element ID to search on as a String.
139      */
140     public String getElementName() {
141         return elementName;
142     }
143 
144     /**
145      * Implementation of the abstract method inherited from QueryCriteria for
146      * mutating the element ID to search on.
147      *
148      */
149     public void setElementName(String elementName) {
150         this.elementName = elementName;
151     }
152 
153     /**
154      * Implementation of the abstract method inherited from QueryCriteria for
155      * generating a human-parsable string version of the query criteria. Note
156      * that the returned String follows the Lucene query language.
157      * 
158      * @return The query as a String.
159      */
160     public String toString() {
161         StringBuilder serial = new StringBuilder(elementName + ":(");
162         for (String value : values) {
163             serial.append("+").append(value);
164         }
165         serial.append(")");
166         return serial.toString();
167     }
