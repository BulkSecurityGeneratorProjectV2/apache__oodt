38  /** XML-RPC utilities.
39   *
40   * This class provides several XML-RPC utilities.
41   *
42   * @author Kelly

44  public class XMLRPC {
45  	/** Build an XML-RPC method call.
46  	 *
47  	 * This may throw {@link IllegalArgumentException} if the <var>params</var>
48  	 * contains an object that's incompatible with XML-RPC.
49  	 *
50  	 * @param name Name of the method to call.
51  	 * @param params Parameters to pass to the call.
52  	 * @return An XML string encapsulationg the call.
53  	 * @throws DOMException If we can't construct the call.
54  	 */
55  	public static byte[] createMethodCall(String name, Collection params) throws DOMException {
56  		Document doc = XML.createDocument();
57  		Element methodCallElement = doc.createElement("methodCall");
58  		doc.appendChild(methodCallElement);
59  		XML.add(methodCallElement, "methodName", name);
60  		if (params != null && !params.isEmpty()) {
61  			Element paramsElement = doc.createElement("params");
62  			methodCallElement.appendChild(paramsElement);
63  		  for (Object param : params) {
64  			paramsElement.appendChild(createValueElement(doc, param));
65  		  }
66  		}
67  		return XML.serialize(doc).getBytes();
68  	}

70  	/** Create a &lt;value&gt; element for an XML-RPC method.
71  	 *
72  	 * This may throw {@link IllegalArgumentException} if the <var>value</var> is
73  	 * incompatible with XML-RPC.
74  	 *
75  	 * @param doc Owning document.
76  	 * @param value The value.
77  	 * @throws DOMException If we can't construct the &lt;value&gt;.
78  	 */
79  	private static Element createValueElement(Document doc, Object value) throws DOMException {
80  		if (value == null) {
81  		  throw new IllegalArgumentException("Nulls not supported in XML-RPC");
82  		}
83  		Element valueElement = doc.createElement("value");
84  		if (value instanceof Integer || value instanceof Short) {
85  			XML.add(valueElement, "int", value.toString());
86  		} else if (value instanceof Boolean) {
87  			XML.add(valueElement, "boolean", (Boolean) value ? "1" : "0");
88  		} else if (value instanceof String) {
89  			Element stringElement = doc.createElement("string");
90  			valueElement.appendChild(stringElement);
91  			stringElement.appendChild(doc.createCDATASection(value.toString()));
92  		} else if (value instanceof Float || value instanceof Double) {
93  			XML.add(valueElement, "double", value.toString());
94  		} else if (value instanceof Date) {
95  			XML.add(valueElement, "dateTime.iso8601", ISO8601_FORMAT.format((Date) value));
96  		} else if (value instanceof byte[]) {
97  			Element base64Element = doc.createElement("base64");
98  			valueElement.appendChild(base64Element);
99  			base64Element.appendChild(doc.createCDATASection(new String(Base64.encode((byte[])value))));
100 		} else if (value instanceof Map) {
101 			Element structElement = doc.createElement("struct");
102 			valueElement.appendChild(structElement);
103 			Map map = (Map) value;
104 		  for (Object o : map.entrySet()) {
105 			Element memberElement = doc.createElement("member");
106 			valueElement.appendChild(memberElement);
107 			Map.Entry entry = (Map.Entry) o;
108 			if (!(entry.getKey() instanceof String)) {
109 			  throw new IllegalArgumentException("Keys in maps for XML-RPC structs must be Strings");
110 			}
111 			XML.add(memberElement, "name", entry.getKey().toString());
112 			memberElement.appendChild(createValueElement(doc, entry.getValue()));
113 		  }
114 		} else if (value instanceof Collection) {
115 			Element arrayElement = doc.createElement("array");
116 			valueElement.appendChild(arrayElement);
117 			Element dataElement = doc.createElement("data");
118 			arrayElement.appendChild(dataElement);
119 			Collection collection = (Collection) value;
120 		  for (Object aCollection : collection) {
121 			dataElement.appendChild(createValueElement(doc, aCollection));
122 		  }
123 		} else {
124 		  throw new IllegalArgumentException(value.getClass().getName() + " not supported in XML-RPC");
125 		}
126 		return valueElement;
127 	}

129 	/** Parse an XML-RPC method response.
130 	 *
131 	 * @param response The response data.
132 	 * @return The value contained in the <var>response</var>.
133 	 * @throws XMLRPCFault If the <var>response</var> contained a fault.
134 	 */
135 	public static Object parseResponse(byte[] response) throws XMLRPCFault {
136 		try {
137 			DOMParser parser = XML.createDOMParser();
138 			parser.setFeature("http://xml.org/sax/features/validation", false);
139 			parser.parse(new InputSource(new ByteArrayInputStream(response)));
140 			Document doc = parser.getDocument();
141 			doc.normalize();
142 			XML.removeComments(doc);
143 			Element methodResponseElement = doc.getDocumentElement();
144 			if (!"methodResponse".equals(methodResponseElement.getNodeName())) {
145 			  throw new SAXException("Not a <methodResponse> document");
146 			}
147 			Node child = methodResponseElement.getFirstChild();
148 			if ("params".equals(child.getNodeName())) {
149 				return parseValue(child.getFirstChild().getFirstChild());
150 			} else if ("fault".equals(child.getNodeName())) {
151 				try {
152 					Map map = (Map) parseValue(child.getFirstChild());
153 					throw new XMLRPCFault(((Integer) map.get("faultCode")).intValue(),
154 						(String) map.get("faultString"));
155 				} catch (ClassCastException ex) {
156 					throw new SAXException("XML-RPC <fault> invalid");
157 				}
158 			} else {
159 			  throw new SAXException("XML-RPC response does not contain <params> or <fault>");
160 			}
161 		} catch (SAXException ex) {
162 			throw new IllegalArgumentException(ex.getMessage());
163 		} catch (IOException ex) {
164 			throw new RuntimeException("Unexpected I/O exception that shouldn't happen, but did: " + ex.getMessage());
165 		}
166 	}

168 	/** Parse an XML-RPC &lt;value&gt;.
169 	 *
170 	 * @param node The &lt;value&gt; node.
171 	 * @return The Java value of <var>node</var>.
172 	 */
173 	private static Object parseValue(Node node) {
174 		String n = node.getNodeName();
175 		if (!"value".equals(n)) {
176 		  throw new IllegalArgumentException("Expecting a <value>, not a <" + n + ">");
177 		}
178 		Node t = node.getFirstChild();
179 		n = t.getNodeName();
180 
181 		// Default if there's no nested element is a String.
182 		if (t.getNodeType() == Node.TEXT_NODE || t.getNodeType() == Node.CDATA_SECTION_NODE) {
183 			return t.getNodeValue();
184 		}
185 
186 		// Figure out what the type is from the nested element.
187 		String txt = XML.unwrappedText(t);
188 		if ("i4".equals(n) || "int".equals(n)) {
189 			return Integer.valueOf(txt);
190 		} else if ("boolean".equals(n)) {
191 			if ("1".equals(txt)) {
192 			  return true;
193 			} else if ("0".equals(txt)) {
194 			  return false;
195 			} else {
196 			  throw new IllegalArgumentException(n + " does not contain a 0 or 1");
197 			}
198 		} else if ("string".equals(n)) {
199 			return txt;
200 		} else if ("double".equals(n)) {
201 			return new Double(txt);
202 		} else if ("dateTime.iso8601".equals(n)) {
203 			try {
204 				return ISO8601_FORMAT.parse(txt);
205 			} catch (ParseException ex) {
206 				throw new IllegalArgumentException(n + " does not contain an ISO8601 format date/time");
207 			}
208 		} else if ("base64".equals(n)) {
209 			return Base64.decode(txt.getBytes());
210 		} else if ("struct".equals(n)) {
211 			Map m = new ConcurrentHashMap();
212 			NodeList memberNodes = t.getChildNodes();
213 			for (int i = 0; i < memberNodes.getLength(); ++i) {
214 				Node memberNode = memberNodes.item(i);
215 				if (!"member".equals(memberNode.getNodeName())) {
216 				  throw new IllegalArgumentException(n + " contains <" + memberNode.getNodeName()
217 													 + ">, not <member>");
218 				}
219 				Node nameNode = memberNode.getFirstChild();
220 				if (nameNode == null || !"name".equals(nameNode.getNodeName())) {
221 				  throw new IllegalArgumentException("<member> missing <name> element");
222 				}
223 				Node valueNode = nameNode.getNextSibling();
224 				if (valueNode == null || !"value".equals(valueNode.getNodeName())) {
225 				  throw new IllegalArgumentException("<member> missing <value> element");
226 				}
227 				m.put(XML.unwrappedText(nameNode), parseValue(valueNode));
228 			}
229 			return m;
230 		} else if ("array".equals(n)) {
231 			Node dataNode = t.getFirstChild();
232 			if (dataNode == null || !"data".equals(dataNode.getNodeName())) {
233 			  throw new IllegalArgumentException("<array> missing <data> element");
234 			}
235 			NodeList children = dataNode.getChildNodes();
236 			List x = new ArrayList(children.getLength());
237 			for (int i = 0; i < children.getLength(); ++i) {
238 			  x.add(parseValue(children.item(i)));
239 			}
240 			return x;
241 		} else {
242 		  throw new IllegalArgumentException("Illegal type " + n + " in <value>");
243 		}
244 	}
