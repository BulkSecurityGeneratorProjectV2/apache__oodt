69     * Should return the list of available, Queued, {@link WorkflowProcessor}s.
70     * 
71     * @return the list of available, Queued, {@link WorkflowProcessor}s.
72     */
73    public synchronized List<WorkflowProcessor> getProcessors() {
74      WorkflowInstancePage page;
75      try {
76        page = repo.getPagedWorkflows(1);
77      } catch (Exception e) {
78        LOG.log(Level.SEVERE, e.getMessage());
79        LOG.log(Level.WARNING, "Unable to load workflow processors: Message: "
80            + e.getMessage());
81        return null;
82      }
83  
84      List<WorkflowProcessor> processors = new Vector<WorkflowProcessor>(
85          page.getPageWorkflows() != null ? page.getPageWorkflows().size() : 0);
86      for (WorkflowInstance inst : (List<WorkflowInstance>) (List<?>) page
87          .getPageWorkflows()) {
88        if (!inst.getState().getCategory().getName().equals("done")) {
89          WorkflowProcessor processor;
90          try {
91            processor = fromWorkflowInstance(inst);
92          } catch (Exception e) {
93            LOG.log(Level.SEVERE, e.getMessage());
94            LOG.log(Level.WARNING,
95                "Unable to convert workflow instance: [" + inst.getId()
96                    + "] into WorkflowProcessor: Message: " + e.getMessage());
97            continue;
98          }
99          if (processor != null) {
100           processors.add(processor);
101         }
102       }
103     }
104 
105     return processors;
106   }

109   public synchronized void persist(WorkflowInstance inst) {
110     try {
111       if (inst.getId() == null || (inst.getId().equals(""))) {
112         // we have to persist it by adding it
113         // rather than updating it
114         repo.addWorkflowInstance(inst);
115       } else {
116         // persist by update
117         repo.updateWorkflowInstance(inst);
118       }
119     } catch (InstanceRepositoryException e) {
120       LOG.log(Level.SEVERE, e.getMessage());
121       LOG.log(Level.WARNING,
122           "Unable to update workflow instance: [" + inst.getId()
123               + "] with status: [" + inst.getState().getName() + "]: Message: "
124               + e.getMessage());
125     }
126   }

128   private WorkflowProcessor fromWorkflowInstance(WorkflowInstance inst) {
129     WorkflowProcessor processor = null;
130     if (processorCache.containsKey(inst.getId())) {
131       return processorCache.get(inst.getId());
132     } else {
133       if (inst.getParentChildWorkflow().getGraph() == null) {
134         LOG.log(Level.SEVERE,
135             "Unable to process Graph for workflow instance: [" + inst.getId()
136                 + "]");
137         return null;
138       }
139 
140       if (isCompositeProcessor(inst)) {
141         processor = getProcessorFromInstanceGraph(inst, lifecycle);
142         WorkflowState processorState = getLifecycle(
143             inst.getParentChildWorkflow()).createState(
144             "Loaded",
145             "initial",
146             "Sequential Workflow instance with id: [" + inst.getId()
147                 + "] loaded by processor queue.");
148         inst.setState(processorState);
149         persist(inst);
150 
151         // handle its pre-conditions
152         for (WorkflowCondition cond : inst.getParentChildWorkflow()
153             .getPreConditions()) {
154           WorkflowInstance instance = new WorkflowInstance();
155           WorkflowState condWorkflowState = lifecycle
156               .getDefaultLifecycle()
157               .createState(
158                   "Null",
159                   "initial",
160                   "Sub Pre Condition Workflow created by Workflow Processor Queue for workflow instance: "
161                       + "[" + inst.getId() + "]");
162           instance.setState(condWorkflowState);
163           instance.setPriority(inst.getPriority());
164           WorkflowTask conditionTask = toConditionTask(cond);
165           instance.setCurrentTaskId(conditionTask.getTaskId());
166           Graph condGraph = new Graph();
167           condGraph.setExecutionType("condition");
168           condGraph.setCond(cond);
169           condGraph.setTask(conditionTask);
170           ParentChildWorkflow workflow = new ParentChildWorkflow(condGraph);
171           workflow.setId("pre-cond-workflow-"
172               + inst.getParentChildWorkflow().getId());
173           workflow.setName("Pre Condition Workflow-" + cond.getConditionName());
174           workflow.getTasks().add(conditionTask);
175           instance.setParentChildWorkflow(workflow);
176           this.addToModelRepo(workflow);
177           persist(instance);
178           WorkflowProcessor subProcessor = fromWorkflowInstance(instance);
179           processor.getSubProcessors().add(subProcessor);
180           synchronized (processorCache) {
181             processorCache.put(instance.getId(), subProcessor);
182           }
183         }
184 
185         // handle its tasks
186         for (WorkflowTask task : inst.getParentChildWorkflow().getTasks()) {
187           WorkflowInstance instance = new WorkflowInstance();
188           WorkflowState taskWorkflowState = lifecycle.getDefaultLifecycle()
189               .createState(
190                   "Null",
191                   "initial",
192                   "Sub Task Workflow created by Workflow Processor Queue for workflow instance: "
193                       + "[" + inst.getId() + "]");
194           instance.setState(taskWorkflowState);
195           instance.setPriority(inst.getPriority());
196           instance.setCurrentTaskId(task.getTaskId());
197           Graph taskGraph = new Graph();
198           taskGraph.setExecutionType("task");
199           taskGraph.setTask(task);
200           ParentChildWorkflow workflow = new ParentChildWorkflow(taskGraph);
201           workflow.setId("task-workflow-"
202               + inst.getParentChildWorkflow().getId());
203           workflow.setName("Task Workflow-" + task.getTaskName());
204           workflow.getTasks().add(task);
205           workflow.getGraph().setTask(task);
206           instance.setParentChildWorkflow(workflow);
207           this.addToModelRepo(workflow);
208           persist(instance);
209           WorkflowProcessor subProcessor = fromWorkflowInstance(instance);
210           processor.getSubProcessors().add(subProcessor);
211           synchronized (processorCache) {
212             processorCache.put(instance.getId(), subProcessor);
213           }
214         }
215 
216         // handle its post conditions
217         for (WorkflowCondition cond : inst.getParentChildWorkflow()
218             .getPostConditions()) {
219           WorkflowInstance instance = new WorkflowInstance();
220           WorkflowState condWorkflowState = lifecycle
221               .getDefaultLifecycle()
222               .createState(
223                   "Null",
224                   "initial",
225                   "Sub Post Condition Workflow created by Workflow Processor Queue for workflow instance: "
226                       + "[" + inst.getId() + "]");
227           instance.setState(condWorkflowState);
228           instance.setPriority(inst.getPriority());
229           WorkflowTask conditionTask = toConditionTask(cond);
230           instance.setCurrentTaskId(conditionTask.getTaskId());
231           Graph condGraph = new Graph();
232           condGraph.setExecutionType("condition");
233           condGraph.setCond(cond);
234           condGraph.setTask(conditionTask);
235           ParentChildWorkflow workflow = new ParentChildWorkflow(condGraph);
236           workflow.setId("post-cond-workflow-"
237               + inst.getParentChildWorkflow().getId());
238           workflow
239               .setName("Post Condition Workflow-" + cond.getConditionName());
240           workflow.getTasks().add(conditionTask);
241           instance.setParentChildWorkflow(workflow);
242           this.addToModelRepo(workflow);
243           persist(instance);
244           WorkflowProcessor subProcessor = fromWorkflowInstance(instance);
245           processor.getSubProcessors().add(subProcessor);
246           synchronized (processorCache) {
247             processorCache.put(instance.getId(), subProcessor);
248           }
249         }
250 
251       } else {
252         // it's not a composite workflow, and it's either just a task processor
253         // or a condition processor
254         if (inst.getParentChildWorkflow().getGraph().getExecutionType()
255             .equals("task")) {
256           processor = new TaskProcessor(lifecycle, inst);
257           WorkflowState taskProcessorState = getLifecycle(
258               inst.getParentChildWorkflow()).createState(
259               "Loaded",
260               "initial",
261               "Task Workflow instance with id: [" + inst.getId()
262                   + "] loaded by processor queue.");
263           inst.setState(taskProcessorState);
264 
265           // handle its pre-conditions
266           for (WorkflowCondition cond : inst.getParentChildWorkflow()
267               .getGraph().getTask().getPreConditions()) {
268             WorkflowInstance instance = new WorkflowInstance();
269             WorkflowState condWorkflowState = lifecycle
270                 .getDefaultLifecycle()
271                 .createState(
272                     "Null",
273                     "initial",
274                     "Sub Pre Condition Workflow for Task created by Workflow Processor Queue for workflow instance: "
275                         + "[" + inst.getId() + "]");
276             instance.setState(condWorkflowState);
277             instance.setPriority(inst.getPriority());
278             WorkflowTask conditionTask = toConditionTask(cond);
279             instance.setCurrentTaskId(conditionTask.getTaskId());
280             Graph condGraph = new Graph();
281             condGraph.setExecutionType("condition");
282             condGraph.setCond(cond);
283             condGraph.setTask(conditionTask);
284             ParentChildWorkflow workflow = new ParentChildWorkflow(condGraph);
285             workflow.setId("pre-cond-workflow-"
286                 + inst.getParentChildWorkflow().getGraph().getTask()
287                     .getTaskId());
288             workflow.setName("Task Pre Condition Workflow-"
289                 + cond.getConditionName());
290             workflow.getTasks().add(conditionTask);
291             instance.setParentChildWorkflow(workflow);
292             this.addToModelRepo(workflow);
293             persist(instance);
294             WorkflowProcessor subProcessor = fromWorkflowInstance(instance);
295             processor.getSubProcessors().add(subProcessor);
296             synchronized (processorCache) {
297               processorCache.put(instance.getId(), subProcessor);
298             }
299           }
300 
301           // handle its post-conditions
302           for (WorkflowCondition cond : inst.getParentChildWorkflow()
303               .getGraph().getTask().getPostConditions()) {
304             WorkflowInstance instance = new WorkflowInstance();
305             WorkflowState condWorkflowState = lifecycle
306                 .getDefaultLifecycle()
307                 .createState(
308                     "Null",
309                     "initial",
310                     "Sub Post Condition Workflow for Task created by Workflow Processor Queue for workflow instance: "
311                         + "[" + inst.getId() + "]");
312             instance.setState(condWorkflowState);
313             instance.setPriority(inst.getPriority());
314             WorkflowTask conditionTask = toConditionTask(cond);
315             instance.setCurrentTaskId(conditionTask.getTaskId());
316             Graph condGraph = new Graph();
317             condGraph.setExecutionType("condition");
318             condGraph.setCond(cond);
319             condGraph.setTask(conditionTask);
320             ParentChildWorkflow workflow = new ParentChildWorkflow(condGraph);
321             workflow.setId("post-cond-workflow-"
322                 + inst.getParentChildWorkflow().getGraph().getTask()
323                     .getTaskId());
324             workflow.setName("Task Post Condition Workflow-"
325                 + cond.getConditionName());
326             workflow.getTasks().add(conditionTask);
327             instance.setParentChildWorkflow(workflow);
328             this.addToModelRepo(workflow);
329             persist(instance);
330             WorkflowProcessor subProcessor = fromWorkflowInstance(instance);
331             processor.getSubProcessors().add(subProcessor);
332             synchronized (processorCache) {
333               processorCache.put(instance.getId(), subProcessor);
334             }
335           }
336 
337         } else if (inst.getParentChildWorkflow().getGraph().getExecutionType()
338             .equals("condition")) {
339           processor = new ConditionProcessor(lifecycle, inst);
340           WorkflowState condProcessorState = getLifecycle(
341               inst.getParentChildWorkflow()).createState(
342               "Loaded",
343               "initial",
344               "Condition Workflow instance with id: [" + inst.getId()
345                   + "] loaded by processor queue.");
346           inst.setState(condProcessorState);
347         }
348         persist(inst);
349       }
350 
351       synchronized (processorCache) {
352         processorCache.put(inst.getId(), processor);
353       }
354       return processor;
355     }

385   private boolean isCompositeProcessor(WorkflowInstance instance) {
386     if (instance.getParentChildWorkflow().getGraph() != null
387         && instance.getParentChildWorkflow().getGraph().getExecutionType() != null
388         && !instance.getParentChildWorkflow().getGraph().getExecutionType()
389             .equals("")) {
390       return instance.getParentChildWorkflow().getGraph().getExecutionType()
391           .equals("parallel")
392           || instance.getParentChildWorkflow().getGraph().getExecutionType()
393               .equals("sequential");
394     } else {
395       // we don't have a Graph to work with, so we'll default to whether or not
396       // so we'll assume this is a workflow instance delivered to us by the
397       // instRep
398       // which doesn't understand Graphs yet (TODO: make instRep understand
399       // graphs
400       // and persist them)
401       // so the simple solution is to check whether or not the ID starts with
402       // task-workflow or pre-cond or post-cond
403       return !(instance.getParentChildWorkflow().getId()
404           .startsWith("task-workflow")
405           || instance.getParentChildWorkflow().getId().startsWith("pre-cond") || instance
406           .getParentChildWorkflow().getId().startsWith("post-cond"));
407     }
408   }

410   private WorkflowProcessor getProcessorFromInstanceGraph(
411       WorkflowInstance instance, WorkflowLifecycleManager lifecycle) {
412     Graph graph = instance.getParentChildWorkflow().getGraph();
413     if (graph != null && graph.getExecutionType() != null
414         && graph.getExecutionType().equals("sequential")) {
415       return new SequentialProcessor(lifecycle, instance);
416     } else {
417       return new ParallelProcessor(lifecycle, instance);
418     }
419   }

421   private synchronized WorkflowTask toConditionTask(WorkflowCondition cond){    
422     String taskId = cond.getConditionId()+"-task"; // TODO: this is incompat with DataSourceWorkflowRepository
423     WorkflowTask condTask = safeGetTaskById(taskId);
424     if(condTask != null) {
425       return condTask;
426     }
427     condTask = new WorkflowTask();
428     condTask.setTaskId(taskId);
429     condTask.setTaskInstanceClassName(ConditionTaskInstance.class.getCanonicalName());
430     condTask.setTaskName(cond.getConditionName()+" Task");
431     WorkflowTaskConfiguration config = new WorkflowTaskConfiguration();
432     config.getProperties().putAll(cond.getCondConfig().getProperties());
433     // this one is a special one that will be removed by the ConditionTaskInstance class
434     config.addConfigProperty("ConditionClassName", cond.getConditionInstanceClassName()); 
435     condTask.setTaskConfig(config);
436     this.addTaskToModelRepo(condTask);
437     return condTask;
438   }

440   private WorkflowTask safeGetTaskById(String taskId){
441     WorkflowTask task = null;
442       try{
443         if((task = this.modelRepo.getTaskById(taskId)) != null){
444           return task;
445         }
446       }
447       catch(RepositoryException e){
448         LOG.log(Level.SEVERE, e.getMessage());
449       }
450     
451     return null;
452   }
