16  package org.apache.oodt.commons.io;
17  
18  import java.io.Writer;
19  import java.io.IOException;
20  import org.apache.oodt.commons.util.*;
21  
22  /** Log messages to a <code>Writer</code>.
23   *
24   * This class defines a {@link LogListener} that logs its messages to a character stream
25   * {@link Writer}.  This formats and writes to the output stream all log events passed to
26   * {@link LogListener#messageLogged}, one per line.  Each line is separated with the
27   * system's line separator characters, specified by the <code>line.separator</code> system
28   * property.
29   *
30   * <p>It ignores all events passed to {@link LogListener#streamStarted} and
31   * {@link LogListener#streamStopped}.
32   *
33   * <p>It quietly hides all IO errors that may occur during writing.
34   *
35   * @see Log
36   * @author Kelly
37   */
38  public class WriterLogger implements LogListener {
39  	/** Constructor given an output stream.
40  	 *
41  	 * Construct a writer logger that writes message events to the given output stream
42  	 * and flushes the stream after every logged message.
43  	 *
44  	 * @param outputStream The output stream to which to write events.
45  	 */
46  	public WriterLogger(java.io.OutputStream outputStream) {
47  		this(new java.io.BufferedWriter(new java.io.OutputStreamWriter(outputStream)), /*autoFlush*/ true);
48  	}
49  
50  	/** Constructor given a writer.
51  	 *
52  	 * Construct a writer logger that writes message events to the given writer and
53  	 * flushes the stream after every logged message.
54  	 *
55  	 * @param writer The writer to which to write events.
56  	 */
57  	public WriterLogger(Writer writer) {
58  		this(writer, /*autoFlush*/ true);
59  	}
60  
61  	/** General constructor.
62  	 *
63  	 * Construct a writer logger that writes message events to the given writer and
64  	 * optionally flushes the stream after every logged message.
65  	 *
66  	 * @param writer The writer to which to write events.
67  	 * @param autoFlush If true, call flush on the writer after every message
68  	 * logged. If false, don't call flush.
69  	 */
70  	public WriterLogger(Writer writer, boolean autoFlush) {
71  		if (writer == null) {
72  		  throw new IllegalArgumentException("Can't write to a null writer");
73  		}
74  		this.writer = writer;
75  		this.autoFlush = autoFlush;
76  		this.lineSep = System.getProperty("line.separator", "\n");
77  	}
78  
79  	/** Close the writer.
80  	 *
81  	 * This closes the writer to which this logger was logging.  Future message events
82  	 * are ignored and not written.
83  	 */
84  	public final void close() {
85  		if (writer == null) {
86  		  return;
87  		}
88  		try {
89  			writer.close();
90  		} catch (IOException ignore) {}
91  		writer = null;
92  	}
93  
94  	/** Log a message to the writer.
95  	 *
96  	 * This method first calls {@link #formatMessage} to format the message, and then
97  	 * writes the message to the output stream.  
98  	 * 
99  	 * @param event The event describing the message that was logged.
100 	 */
101 	public final void messageLogged(LogEvent event) {
102 		if (writer == null) {
103 		  return;
104 		}
105 		try {
106 			writer.write(formatMessage(event.getTimestamp(), (String) event.getSource(), event.getCategory(),
107 				event.getMessage()) + lineSep);
108 			if (autoFlush) {
109 			  writer.flush();
110 			}
111 		} catch (IOException ignore) {}
112 	}
113 
114 	/** Format a message for logging.
115 	 *
116 	 * This method formats a message as follows: 
117 	 *
118 	 * <p><code><var>epochTime</var> (<var>time</var>) <var>source</var> <var>category</var>:
119 	 * <var>message</var> 
120 	 *
121 	 * <p>where <var>epochTime</var> is the time in milliseconds since midnight, 1st
122 	 * January 1970 GMT, <var>time</var> is human-readable time, <var>source</var> is
123 	 * the source of the message, <var>category</var> is the category under which the
124 	 * message was logged, and <var>message</var> is the message.
125 	 *
126 	 * <p>You can override this method and provide your own formatting.
127 	 *
128 	 * @param timestamp Timestamp for the message.
129 	 * @param source Source of the message.
130 	 * @param category Category of the message.
131 	 * @param message The message text.
132 	 */
133 	protected String formatMessage(java.util.Date timestamp, String source, Object category, String message) {
134 		return ("\n" + DateConvert.isoFormat(timestamp) + " " + source + " " + category + ": " + message);
135 	}
136 
137 	/** Ignore the stream started event.
138 	 *
139 	 * @param event The event to ignore.
140 	 */
141 	public void streamStarted(LogEvent event) {
142 	}
143 
144 	/** Ignore the stream stopped event.
145 	 *
146 	 * @param event The event to ignore.
147 	 */
148 	public void streamStopped(LogEvent event) {
149 	}
150 
151 	public void propertyChange(java.beans.PropertyChangeEvent ignore) {}
152 
153 	/** The writer to which we write log messages.
154 	 *
155 	 * If null, then we were closed.  This is protected so you can extend this class
156 	 * and log other events normally ignored.
157 	 */
158 	protected Writer writer;
159 	
160 	/** If true, flush after every message logged.
161 	 *
162 	 * This is protected so you can extend this class and log other events normally ignored.
163 	 */
164 	protected boolean autoFlush;
165 
166 	/** What the line separator is.
167 	 *
168 	 * This is protected so you can extend this class and log other events normally ignored.
169 	 */
170 	protected String lineSep;
171 }
