55    @Override
56    protected void setUp() throws Exception {
57      ctrl = EasyMock.createNiceControl();
58      rs = ctrl.createMock(ResultSet.class);
59      con = ctrl.createMock(Connection.class);
60      rsmet = ctrl.createMock(ResultSetMetaData.class);
61      expect(rs.next()).andReturn(true);
62      expect(rsmet.getColumnCount()).andReturn(4).anyTimes();
63      expect(rs.getMetaData()).andReturn(rsmet).anyTimes();
64      expect(rs.getString(1)).andReturn(ID);
65      expect(rs.getString(2)).andReturn(LAST);
66      expect(rs.getString(3)).andReturn(FIRST);
67      expect(rs.getString(4)).andReturn(DOB);
68      expect(rs.next()).andReturn(true);
69      expect(rs.getString(1)).andReturn(ID1);
70      expect(rs.getString(2)).andReturn(LAST1);
71      expect(rs.getString(3)).andReturn(FIRST1);
72      expect(rs.getString(4)).andReturn(DOB1);
73      expect(rs.next()).andReturn(false);
74      ctrl.replay();
75      result = new CDEResult(rs,con);
76    }
