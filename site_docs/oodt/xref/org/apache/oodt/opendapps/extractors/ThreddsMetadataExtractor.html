50  /**
51   * Implementation of {@link MetadataExtractor} that extracts metadata from a
52   * Thredds dataset.
53   * 
54   * @author Luca Cinquini
55   * 
56   */
57  public class ThreddsMetadataExtractor implements MetadataExtractor {
58  
59    private static Logger LOG = Logger.getLogger(ThreddsMetadataExtractor.class
60        .getName());
61  
62    // constant missing for 4.2 version of NetCDF library
63    public final static String SERVICE_TYPE_NCML = "NCML";
64    
65    /**
66     * The source of metadata to be extracted.
67     */
68    private final InvDataset dataset;
69  
70    public ThreddsMetadataExtractor(final InvDataset dataset) {
71      this.dataset = dataset;
72    }
73  
74    public void extract(Metadata met, OpendapConfig config) {
75  
76      LOG.log(Level.INFO, "Crawling catalog URL=" + dataset.getCatalogUrl()
77          + " dataset ID=" + dataset.getID());
78  
79      ProfileUtils.addIfNotNull(met, "Authority", dataset.getAuthority());
80      ProfileUtils.addIfNotNull(met, "CatalogUrl", dataset.getCatalogUrl());
81      try {
82      	ProfileUtils.addIfNotNull(met, "Host", (new URL(dataset.getCatalogUrl())).getHost() );
83      } catch(MalformedURLException e) {
84      	LOG.log(Level.WARNING, e.getMessage());
85      }
86      ProfileUtils.addIfNotNull(met, "DatasetFullName", dataset.getFullName());
87      if (dataset.getContributors() != null) {
88        for (Contributor contributor : dataset.getContributors()) {
89          ProfileUtils.addIfNotNull(met, "Contributor", contributor.getName());
90        }
91      }
92  
93      if (dataset.getCreators() != null) {
94        for (Source source : dataset.getCreators()) {
95          ProfileUtils.addIfNotNull(met, "Creator", source.getName());
96        }
97      }
98  
99      if (dataset.getDataFormatType() != null) {
100       ProfileUtils.addIfNotNull(met, "DataFormatType", dataset
101           .getDataFormatType().toString());
102     }
103 
104     if (dataset.getDataType() != null) {
105       ProfileUtils.addIfNotNull(met, "DataType", dataset.getDataType()
106           .toString());
107     }
108 
109     if (dataset.getDates() != null) {
110       for (DateType dateType : dataset.getDates()) {
111         String dateString = null;
112         try {
113           dateString = ProfileUtils.toISO8601(dateType.getDate());
114         } catch (Exception e) {
115           LOG.log(Level.WARNING,
116               "Error converting date: [" + dateType.getDate() + "]: Message: "
117                   + e.getMessage());
118         }
119         ProfileUtils.addIfNotNull(met, "Dates", dateString);
120       }
121     }
122 
123     if (dataset.getDocumentation() != null) {
124       for (InvDocumentation doc : dataset.getDocumentation()) {
125       	// textual documentation
126       	if (StringUtils.hasText(doc.getInlineContent())) {
127       		if (StringUtils.hasText(doc.getType())) {
128       			// use specific documentation type, when available
129       			ProfileUtils.addIfNotNull(met, doc.getType(), doc.getInlineContent());
130       		} else {
131       			// otherwise use generic "Documentation" tag
132       			ProfileUtils.addIfNotNull(met, "Documentation", doc.getInlineContent());
133       		}
134       	}
135       	// hyperlinked documentation
136       	if (StringUtils.hasText(doc.getXlinkHref())) {
137       		String tuple = this.encodeXlinkTuple(doc.getXlinkHref(), doc.getXlinkTitle(), doc.getType());
138       		ProfileUtils.addIfNotNull(met, "Xlink", tuple);
139       	}
140       	
141       }
142     }
143 
144     ProfileUtils.addIfNotNull(met, "FullName", dataset.getFullName());
145     GeospatialCoverage geoCoverage = dataset.getGeospatialCoverage();
146     
147     if (geoCoverage != null) {
148     	
149       LatLonRect bbox = geoCoverage.getBoundingBox();
150       if (bbox != null) {
151         ProfileUtils.addIfNotNull(met, "SouthwestBC", bbox.getLowerLeftPoint()
152             .toString());
153         ProfileUtils.addIfNotNull(met, "NorthwestBC", bbox.getUpperLeftPoint()
154             .toString());
155         ProfileUtils.addIfNotNull(met, "NortheastBC", bbox.getUpperRightPoint()
156             .toString());
157         ProfileUtils.addIfNotNull(met, "SoutheastBC", bbox.getLowerRightPoint()
158             .toString());
159       }
160       
161       // try north south, east west
162       if (geoCoverage.getNorthSouthRange() != null) {        	
163         Range nsRange = geoCoverage.getNorthSouthRange();
164         ProfileUtils.addIfNotNull(met, "NorthSouthRangeStart",
165             String.valueOf(nsRange.getStart()));
166         ProfileUtils.addIfNotNull(met, "NorthSouthRangeResolution",
167             String.valueOf(nsRange.getResolution()));
168         ProfileUtils.addIfNotNull(met, "NorthSouthRangeSize",
169             String.valueOf(nsRange.getSize()));
170         ProfileUtils.addIfNotNull(met, "NorthSouthRangeUnits",
171             nsRange.getUnits());
172         ProfileUtils.addIfNotNull(met, "NorthSouthRangeStop",
173             String.valueOf(nsRange.getStart()+nsRange.getSize()));
174       }
175 
176       if (geoCoverage.getEastWestRange() != null) {
177         Range nsRange = geoCoverage.getEastWestRange();
178         ProfileUtils.addIfNotNull(met, "EastWestRangeStart",
179             String.valueOf(nsRange.getStart()));
180         ProfileUtils.addIfNotNull(met, "EastWestRangeResolution",
181             String.valueOf(nsRange.getResolution()));
182         ProfileUtils.addIfNotNull(met, "EastWestRangeSize",
183             String.valueOf(nsRange.getSize()));
184         ProfileUtils.addIfNotNull(met, "EastWestRangeUnits",
185             nsRange.getUnits());
186         ProfileUtils.addIfNotNull(met, "EastWestRangeStop",
187             String.valueOf(nsRange.getStart()+nsRange.getSize()));
188       }      
189 
190       ProfileUtils.addIfNotNull(met, "GeospatialCoverageLatitudeResolution",
191           String.valueOf(dataset.getGeospatialCoverage().getLatResolution()));
192       ProfileUtils.addIfNotNull(met, "GeospatialCoverageLongitudeResolution",
193           String.valueOf(dataset.getGeospatialCoverage().getLonResolution()));
194       
195       // add geo-spatial coverage alternative form
196       ProfileUtils.addIfNotNull(met, "GeospatialCoverageLatSouth", String.valueOf(dataset.getGeospatialCoverage().getLatSouth()));
197       ProfileUtils.addIfNotNull(met, "GeospatialCoverageLatNorth", String.valueOf(dataset.getGeospatialCoverage().getLatNorth()));   
198       ProfileUtils.addIfNotNull(met, "GeospatialCoverageLonWest", String.valueOf(dataset.getGeospatialCoverage().getLonWest()));
199       ProfileUtils.addIfNotNull(met, "GeospatialCoverageLonEast", String.valueOf(dataset.getGeospatialCoverage().getLonEast()));
200       
201       if (dataset.getGeospatialCoverage().getNames() != null) {
202         for (Vocab gName : dataset.getGeospatialCoverage().getNames()) {
203           ProfileUtils.addIfNotNull(met, "GeospatialCoverage", gName.getText());
204         }
205       }
206 
207     }
208 
209     ProfileUtils.addIfNotNull(met, "History", dataset.getHistory());
210     if (dataset.getKeywords() != null) {
211       for (Vocab vocab : dataset.getKeywords()) {
212         ProfileUtils.addIfNotNull(met, "Keywords", vocab.getText());
213       }
214     }
215     ProfileUtils.addIfNotNull(met, "Name", dataset.getName());
216     ProfileUtils.addIfNotNull(met, "Processing", dataset.getProcessing());
217     if (dataset.getProjects() != null) {
218       for (Vocab vocab : dataset.getProjects()) {
219         ProfileUtils.addIfNotNull(met, "Projects", vocab.getText());
220       }
221     }
222 
223     if (dataset.getProperties() != null) {
224       for (InvProperty prop : dataset.getProperties()) {
225         ProfileUtils.addIfNotNull(met, prop.getName(), prop.getValue());
226       }
227     }
228 
229     if (dataset.getPublishers() != null) {
230       for (Source source : dataset.getPublishers()) {
231       	// Note: use "Publisher" (singular) as from the OODT profile specification
232         ProfileUtils.addIfNotNull(met, "Publisher", source.getName());
233       }
234     }
235 
236     ProfileUtils.addIfNotNull(met, "RestrictAccess",
237         dataset.getRestrictAccess());
238     if (dataset.getTimeCoverage() != null) {
239       String startDateTimeStr = null, endDateTimeStr = null;
240       try {
241         startDateTimeStr = ProfileUtils.toISO8601(dataset.getTimeCoverage()
242             .getStart().getDate());
243         endDateTimeStr = ProfileUtils.toISO8601(dataset.getTimeCoverage()
244             .getEnd().getDate());
245       } catch (Exception e) {
246         LOG.log(
247             Level.WARNING,
248             "Error converting start/end date time strings: Message: "
249                 + e.getMessage());
250       }
251 
252       ProfileUtils.addIfNotNull(met, "StartDateTime", startDateTimeStr);
253       ProfileUtils.addIfNotNull(met, "EndDateTime", endDateTimeStr);
254     }
255 
256     if (dataset.getTimeCoverage() != null
257         && dataset.getTimeCoverage().getResolution() != null) {
258       ProfileUtils.addIfNotNull(met, "TimeCoverageResolution", dataset
259           .getTimeCoverage().getResolution().getText());
260     }
261     // dataset unique ID
262     ProfileUtils.addIfNotNull(met, "UniqueID", dataset.getUniqueID());
263 
264     // dataset ID is typically not null
265     ProfileUtils.addIfNotNull(met, "ID", dataset.getID());
266 
267     // generate a UUID for each dataset, to be used as profile ID
268     ProfileUtils.addIfNotNull(met, "UUID", UUID.randomUUID().toString());
269 
270     // store access services - only the OpenDAP endpoint for now
271     for (InvAccess access : dataset.getAccess()) {    	
272       String url = access.getStandardUri().toString();
273     	String type = access.getService().getServiceType().toString();
274     	String name = access.getService().getName();
275     	
276     	// add opendap access URL
277     	if (type.equalsIgnoreCase(ServiceType.OPENDAP.toString())) {
278     		// store opendap URL
279     		ProfileUtils.addIfNotNull(met,"OpendapUrl",url);
280     	  // note: special processing of opendap endpoints since URL in thredds catalog is unusable without a suffix
281     		ProfileUtils.addIfNotNull(met,"Access", this.encodeAccessTuple(url+".html", ProfileUtils.MIME_TYPE_OPENDAP_HTML, type));
282     	} 
283     }
284     // add TREDDS XML catalog URL
285     String url = dataset.getCatalogUrl(); // catalog_url.xml#dataset_id
286     ProfileUtils.addIfNotNull(met,"Access", this.encodeAccessTuple(url, ProfileUtils.MIME_TYPE_THREDDS, "Catalog/XML"));
287     ProfileUtils.addIfNotNull(met,"Access", this.encodeAccessTuple(url.replaceAll("\\.xml", ".html"), ProfileUtils.MIME_TYPE_HTML, "Catalog/HTML"));
288 
289   }

291 	/**
292 	 * Utility method that joins the parts of an xlink tuple (href, title, type) with a delimiting character.
293 	 * @param href : the xlink URL, must be not null
294 	 * @param title : the xlink title, may be null
295 	 * @param type : the xlink type, may be null
296 	 * @return
297 	 */
298 	private String encodeXlinkTuple(final String href, final String title, final String type) {
299       return href + ProfileUtils.CHAR + (StringUtils.hasText(title) ? title : "Reference") + ProfileUtils.CHAR + (
300           StringUtils.hasText(type) ? type : "HTML");
301 	}
302 	
303 	/**
304 	 * Utility method that joins the part of A THREDDS access point (url, service type, service name) with a delimiting character
305 	 * @param url : the access URL
306 	 * @param type : the service type, mapped to a mime type
307 	 * @param name : the service name
308 	 * @return
309 	 */
310 	private String encodeAccessTuple(final String url, final String type, final String name) {
311       return url + ProfileUtils.CHAR + (StringUtils.hasText(type) ? type : "") + ProfileUtils.CHAR + (
312           StringUtils.hasText(name) ? name : "");
313 	}
