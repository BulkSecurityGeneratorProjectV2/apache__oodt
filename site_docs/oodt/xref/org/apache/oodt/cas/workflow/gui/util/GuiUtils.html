47    public static boolean isSubGraph(ModelGraph graph, ModelGraph subGraph) {
48      if (graph.equals(subGraph)) {
49        return true;
50      }
51      for (ModelGraph child : graph.getChildren()) {
52        if (isSubGraph(child, subGraph)) {
53          return true;
54        }
55      }
56      if (graph.getPreConditions() != null) {
57        if (isSubGraph(graph.getPreConditions(), subGraph)) {
58          return true;
59        }
60      }
61      if (graph.getPostConditions() != null) {
62        if (isSubGraph(graph.getPostConditions(), subGraph)) {
63          return true;
64        }
65      }
66      return false;
67    }

69    public static void updateGraphModelId(ViewState state, String id,
70        String newModelId) {
71      ModelGraph graph = find(state.getGraphs(), id);
72      if (graph.getParent() != null
73          && graph.getParent().getModel().getExcusedSubProcessorIds()
74              .contains(graph.getModel().getModelId())) {
75        graph.getParent().getModel().getExcusedSubProcessorIds()
76            .remove(graph.getModel().getModelId());
77        graph.getParent().getModel().getExcusedSubProcessorIds().add(newModelId);
78      }
79      graph.getModel().setModelId(newModelId);
80    }

90    public static List<ModelGraph> findRootGraphs(List<ModelGraph> graphs) {
91      List<ModelGraph> rootGraphs = new Vector<ModelGraph>();
92      for (ModelGraph graph : graphs) {
93        if (rootGraphs.size() == 0) {
94          rootGraphs.add(graph);
95        } else {
96          if (find(rootGraphs, graph.getModel().getModelId()) == null) {
97            rootGraphs.add(graph);
98          }
99        }
100     }
101     for (int i = 0; i < rootGraphs.size(); i++) {
102       ModelGraph rootGraph = rootGraphs.get(i);
103       for (int j = 0; j < rootGraphs.size(); j++) {
104         if (i != j
105             && rootGraphs.get(j).recursiveFind(
106                 rootGraph.getModel().getModelId()) != null) {
107           rootGraphs.remove(i--);
108           break;
109         }
110       }
111 
112     }
113     return rootGraphs;
114   }

116   public static ModelGraph findRoot(List<ModelGraph> rootGraphs,
117       ModelGraph graph) {
118     for (ModelGraph rootGraph : rootGraphs) {
119       if (graph.equals(rootGraph)) {
120         return rootGraph;
121       } else if (graph.getParent() != null) {
122         ModelGraph root = findRoot(rootGraphs, graph.getParent());
123         if (root != null) {
124           return root;
125         }
126       }
127     }
128     return null;
129   }

152   public static ModelGraph removeNode(List<ModelGraph> graphs, ModelNode node) {
153     for (int i = 0; i < graphs.size(); i++) {
154       if (graphs.get(i).getModel().equals(node)) {
155         return graphs.remove(i);
156       } else {
157         ModelGraph graph = removeNode(graphs.get(i), node);
158         if (graph != null) {
159           return graph;
160         }
161       }
162     }
163     return null;
164   }

166   public static ModelGraph removeNode(ModelGraph graph, ModelNode node) {
167     Stack<ModelGraph> stack = new Stack<ModelGraph>();
168     stack.add(graph);
169     while (!stack.empty()) {
170       ModelGraph curGraph = stack.pop();
171       if (curGraph.getModel().equals(node)) {
172         curGraph.setParent(null);
173         return curGraph;
174       } else {
175         stack.addAll(curGraph.getChildren());
176         if (curGraph.getPreConditions() != null) {
177           stack.add(curGraph.getPreConditions());
178         }
179         if (curGraph.getPostConditions() != null) {
180           stack.add(curGraph.getPostConditions());
181         }
182       }
183     }
184     return null;
185   }

195   public static List<Line> findSequentialLines(final ModelGraph graph) {
196     Vector<Line> lines = new Vector<Line>();
197     if (graph.getChildren().size() > 0) {
198       Stack<ModelGraph> stack = new Stack<ModelGraph>();
199       stack.add(graph);
200       while (!stack.empty()) {
201         ModelGraph curGraph = stack.pop();
202         if (curGraph.getModel().getExecutionType().equals("sequential")) {
203           for (int i = 0; i < curGraph.getChildren().size() - 1; i++) {
204             lines.add(new Line(curGraph.getChildren().get(i).getModel(),
205                 curGraph.getChildren().get(i + 1).getModel()));
206           }
207         }
208         stack.addAll(curGraph.getChildren());
209       }
210     }
211     return lines;
212   }

222   public static List<Line> findLines(final ModelGraph graph) {
223     Vector<Line> lines = new Vector<Line>();
224     if (graph.getChildren().size() > 0) {
225       Stack<ModelGraph> graphs = new Stack<ModelGraph>();
226       graphs.add(graph);
227       while (!graphs.empty()) {
228         ModelGraph curGraph = graphs.pop();
229         if (curGraph.getModel().isParentType()) {
230 
231           if (curGraph.getChildren().size() == 0) {
232             curGraph.addChild(new ModelGraph(createDummyNode()));
233           }
234 
235           List<Line> relaventLines = getRelaventLines(lines, curGraph
236               .getModel().getId());
237           for (Line relaventLine : relaventLines) {
238             int index = lines.indexOf(relaventLine);
239             if (curGraph.getModel().getExecutionType().toLowerCase()
240                 .equals("sequential")) {
241               lines.remove(index);
242               if (curGraph.getChildren().size() > 0) {
243                 if (relaventLine.getFromModel().equals(curGraph.getModel())) {
244                   lines.add(new Line(curGraph.getChildren()
245                                              .get(curGraph.getChildren().size() - 1).getModel(),
246                       relaventLine.getToModel()));
247                 } else {
248                   lines.add(new Line(relaventLine.getFromModel(), curGraph
249                       .getChildren().get(0).getModel()));
250                 }
251               }
252             } else if (curGraph.getModel().getExecutionType().toLowerCase()
253                 .equals("parallel")) {
254               lines.remove(index);
255               if (relaventLine.getFromModel().equals(curGraph.getModel())) {
256                 for (ModelGraph child : curGraph.getChildren()) {
257                   lines.add(new Line(child.getModel(), relaventLine
258                       .getToModel()));
259                 }
260               } else {
261                 for (ModelGraph child : curGraph.getChildren()) {
262                   lines.add(new Line(relaventLine.getFromModel(), child
263                       .getModel()));
264                 }
265               }
266             }
267           }
268 
269           if (curGraph.getModel().getExecutionType().toLowerCase()
270               .equals("sequential")) {
271             for (int i = 0; i < curGraph.getChildren().size(); i++) {
272               if (i == curGraph.getChildren().size() - 1) {
273                 lines.add(new Line(curGraph.getChildren().get(i).getModel(),
274                     null));
275               } else {
276                 lines.add(new Line(curGraph.getChildren().get(i).getModel(),
277                     curGraph.getChildren().get(i + 1).getModel()));
278               }
279             }
280           } else if (curGraph.getModel().getExecutionType().toLowerCase()
281               .equals("parallel")) {
282             for (int i = 0; i < curGraph.getChildren().size(); i++) {
283               lines
284                   .add(new Line(curGraph.getChildren().get(i).getModel(), null));
285             }
286           }
287           graphs.addAll(curGraph.getChildren());
288         }
289       }
290     } else {
291       lines.add(new Line(graph.getModel(), null));
292     }
293     return lines;
294   }

335   public static List<Line> getRelaventLines(List<Line> lines, String id) {
336     List<Line> relaventLines = new Vector<Line>();
337     for (Line line : lines) {
338       if ((line.getFromModel() != null && line.getFromModel().getId()
339                                               .equals(id))
340           || (line.getToModel() != null && line.getToModel().getId().equals(id))) {
341         relaventLines.add(line);
342       }
343     }
344     return relaventLines;
345   }
