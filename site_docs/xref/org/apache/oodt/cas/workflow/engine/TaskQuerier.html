39  /**
40   * 
41   * The purpose of this class is to constantly pop off tasks that are run to run
42   * and made available by the {@link WorkflowProcessorQueue}, and then to set
43   * their state to Executing (running Category), so they will be picked up on the
44   * next WorkflowState change, and end up executing.
45   * 
46   * @author mattmann
47   * @author bfoster
48   * @version $Revision$
49   * 
50   */
51  public class TaskQuerier implements Runnable {
52  
53    private boolean running;
54  
55    private WorkflowProcessorQueue processorQueue;
56  
57    private List<WorkflowProcessor> runnableProcessors;
58  
59    private PrioritySorter prioritizer;
60  
61    private WorkflowInstanceRepository repo;
62    
63    private long waitSeconds;
64  
65    private static final Logger LOG = Logger.getLogger(TaskQuerier.class
66        .getName());
67  
68    /**
69     * Constructs a new TaskQuerier with the given {@link WorkflowProcessorQueue},
70     * and with the associated {@link PrioritySorter} which acts as a sorter of
71     * the runnable {@link WorkflowProcessor}s.
72     * 
73     * @param processorQueue
74     *          The associated set of queued Workflow Tasks.
75     * @param prioritizer
76     *          The prioritizer to use to sort the ready-to-run Workflow Tasks.
77     * 
78     * @param repo
79     *          The {@link WorkflowInstanceRepository} to save the state of
80     *          WorkflowInstances.
81     *          
82     * @param waitSeconds The default amount of seconds to wait while dispositioning
83     *        processors
84     */
85    public TaskQuerier(WorkflowProcessorQueue processorQueue,
86        PrioritySorter prioritizer, WorkflowInstanceRepository repo, long waitSeconds) {
87      this.running = true;
88      this.processorQueue = processorQueue;
89      this.runnableProcessors = new Vector<WorkflowProcessor>();
90      this.prioritizer = prioritizer;
91      this.repo = repo;
92      this.waitSeconds = waitSeconds;
93    }
94  
95    /**
96     * Marches through the set of processors that are currently in the Processor
97     * queue, checks to see if they are NOT in the done state, or if they are
98     * currently in the holding state. If either of those are true, the processor
99     * is popped off the queue, and then added to the runnableProcessors list (in
100    * a synchronized fashion), and then their state is set to Executing
101    * (category, running).
102    * 
103    * Finally the runnableProcessors list is sorted according to the given
104    * {@link #prioritizer}.
105    */
106   public void run() {
107     LOG.log(Level.FINE, "TaskQuerier configured with wait seconds: ["+this.waitSeconds+"]");
108     while (running) {
109       List<WorkflowProcessor> processors = processorQueue.getProcessors();
110       List<WorkflowProcessor> processorsToRun = new Vector<WorkflowProcessor>();
111 
112       for (WorkflowProcessor processor : processors) {
113         // OK now get its lifecycle
114         WorkflowProcessorHelper helper = new WorkflowProcessorHelper(
115             processor.getLifecycleManager());
116         WorkflowLifecycle lifecycle = helper
117             .getLifecycleForProcessor(processor);
118 
119         LOG.log(Level.FINE, "TaskQuerier: dispositioning processor with id: ["
120             + processor.getWorkflowInstance().getId() + "]: state: "
121             + processor.getWorkflowInstance().getState());
122 
123         if (!processor.isAnyCategory("done", "holding")
124             && !processor.isAnyState("Executing")
125             && processor.getRunnableWorkflowProcessors().size() > 0) {
126           for (TaskProcessor tp : processor.getRunnableWorkflowProcessors()) {
127             WorkflowState state = lifecycle.createState("WaitingOnResources",
128                 "waiting", "Added to Runnable queue");
129             tp.getWorkflowInstance().setState(state);
130             persist(tp.getWorkflowInstance());
131             LOG.log(Level.INFO, "Added processor with priority: ["
132                 + tp.getWorkflowInstance().getPriority() + "]");
133             processorsToRun.add(tp);
134           }
135 
136           if (processorsToRun.size() > 1) {
137             prioritizer.sort(processorsToRun);
138           }
139 
140           synchronized (runnableProcessors) {
141             if (running) {
142               runnableProcessors = processorsToRun;
143             }
144           }
145 
146         } else {
147           // simply call nextState and persist it
148           LOG.log(Level.FINE, "Processor for workflow instance: ["
149               + processor.getWorkflowInstance().getId()
150               + "] not ready to Execute or already Executing: "
151               + "advancing it to next state.");
152           processor.nextState();
153           persist(processor.getWorkflowInstance());
154         }
155       }
156       
157       try{
158         Thread.currentThread().sleep(waitSeconds*1000);
159       }
160       catch(InterruptedException ignore){}
161       
162     }
163   }

187   /**
188    * Gets the next available {@link TaskProcessor} from the {@link List} of
189    * {@link #runnableProcessors}. Removes that {@link TaskProcessor} from the
190    * actual {@link #runnableProcessors} {@link List}.
191    * 
192    * @return The next available {@link TaskProcessor} from the {@link List} of
193    *         {@link #runnableProcessors}.
194    */
195   public TaskProcessor getNext() {
196     if (getRunnableProcessors().size() == 0) {
197       return null;
198     }
199     return (TaskProcessor) getRunnableProcessors().remove(0);
200   }

202   private synchronized void persist(WorkflowInstance instance) {
203     if (this.repo != null) {
204       try {
205         if (instance.getId() == null || (instance.getId().equals(""))) {
206           // we have to persist it by adding it
207           // rather than updating it
208           repo.addWorkflowInstance(instance);
209         } else {
210           // persist by update
211           repo.updateWorkflowInstance(instance);
212         }
213       } catch (InstanceRepositoryException e) {
214         LOG.log(Level.SEVERE, e.getMessage());
215         LOG.log(Level.WARNING, "Unable to update workflow instance: ["
216             + instance.getId() + "] status to [" + instance.getState().getName()
217             + "]. Message: " + e.getMessage());
218       }
219     }
220   }
