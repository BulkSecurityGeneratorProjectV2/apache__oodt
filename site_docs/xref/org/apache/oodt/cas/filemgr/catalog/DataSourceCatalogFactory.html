18  package org.apache.oodt.cas.filemgr.catalog;
19  
20  //OODT imports
21  import org.apache.oodt.commons.database.DatabaseConnectionBuilder;
22  import org.apache.oodt.cas.filemgr.util.GenericFileManagerObjectFactory;
23  import org.apache.oodt.cas.filemgr.validation.ValidationLayer;
24  import org.apache.oodt.cas.metadata.util.PathUtils;
25  
26  //JDK imports
27  import javax.sql.DataSource;
28  
29  /**
30   * @author mattmann
31   * @version $Revision$
32   * 
33   * <p>
34   * A Factory class for creating {@link DataSourceCatalog}s.
35   * </p>
36   * 
37   */
38  public class DataSourceCatalogFactory implements CatalogFactory {
39    public static final int VAL = 20;
40    public static final long VAL1 = 5L;
41    /* our data source */
42      protected DataSource dataSource = null;
43  
44      /* our validation layer */
45      protected ValidationLayer validationLayer = null;
46  
47      /* field Id Str: should product_type_id and element_id be quoted? */
48      protected boolean fieldIdStr = false;
49  
50      /*
51       * page size: size of the pages used by the catalog to paginate products
52       * back to the user
53       */
54      protected int pageSize = -1;
55  
56      /* the amount of minutes to allow between updating the product cache */
57      protected long cacheUpdateMinutes = -1L;
58      
59      /* Whether or not to enforce strict definition of metadata fields:
60    	 * 'lenient=false' means that all metadata fields need to be explicitly defined in the XML configuration file */
61      protected boolean lenientFields = false;
62    	
63      /* Flag to indicate whether the "product_id" key type should be treated as a string */
64      protected boolean productIdString = false;
65  
66      protected boolean orderedValues = false;
67  
68      /**
69       * <p>
70       * Default Constructor
71       * </p>.
72       */
73      public DataSourceCatalogFactory() {
74      	
75      	  // instantiate data source
76          String jdbcUrl, user, pass, driver;
77  
78          jdbcUrl = PathUtils
79                  .replaceEnvVariables(System
80                          .getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.jdbc.url"));
81          user = PathUtils
82                  .replaceEnvVariables(System
83                          .getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.jdbc.user"));
84          pass = PathUtils
85                  .replaceEnvVariables(System
86                          .getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.jdbc.pass"));
87          driver = PathUtils
88                  .replaceEnvVariables(System
89                          .getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.jdbc.driver"));
90  
91          dataSource = DatabaseConnectionBuilder.buildDataSource(user, pass,
92                  driver, jdbcUrl);
93  
94          this.configure();
95  
96      }
97  
98      /**
99       * <p>
100      * Constructs a new Factory from the specified {@link DataSource}.
101      * </p>
102      * 
103      * @param ds
104      *            The DataSource to construct this factory from.
105      */
106     public DataSourceCatalogFactory(DataSource ds) {
107        
108     		this.dataSource = ds;
109         
110      		this.configure();
111     		
112     }
113     
114     /** Method to configure the factory (and validation layer) from the environment properties,
115      *  before any Catalog instance is created 
116      **/
117     private void configure() {
118     
119   		lenientFields = Boolean.parseBoolean( System.getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.lenientFields", "false") );
120   		if (!lenientFields) {
121 	        String validationLayerFactoryClass = System
122                 .getProperty("filemgr.validationLayer.factory",
123                         "org.apache.oodt.cas.filemgr.validation.DataSourceValidationLayerFactory");
124 	        validationLayer = GenericFileManagerObjectFactory
125               .getValidationLayerFromFactory(validationLayerFactoryClass);
126   		} 
127   		
128       fieldIdStr = Boolean
129       	.getBoolean("org.apache.oodt.cas.filemgr.catalog.datasource.quoteFields");
130 
131 			pageSize = Integer
132                 .getInteger(
133                     "org.apache.oodt.cas.filemgr.catalog.datasource.pageSize",
134                     VAL);
135 			cacheUpdateMinutes = Long
136                 .getLong(
137                     "org.apache.oodt.cas.filemgr.catalog.datasource.cacheUpdateMinutes",
138                     VAL1);
139 			
140 			productIdString = Boolean.parseBoolean( 
141 				System.getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.productId.string", "false") );
142   	
143 			orderedValues = Boolean.parseBoolean(System.getProperty("org.apache.oodt.cas.filemgr.catalog.datasource.orderedValues"));
144     }
145 
146     /*
147      * (non-Javadoc)
148      * 
149      * @see org.apache.oodt.cas.filemgr.catalog.CatalogFactory#createCatalog()
150      */
151     public Catalog createCatalog() {
152     	if (validationLayer==null) {
153     			return new LenientDataSourceCatalog(dataSource, null, fieldIdStr,
154 							    pageSize, cacheUpdateMinutes, productIdString, orderedValues);
155     	} else {
156         return new DataSourceCatalog(dataSource, validationLayer, fieldIdStr,
157 				     pageSize, cacheUpdateMinutes, productIdString, orderedValues);
158     	}
159     }
