51    public WorkflowViewer(String id, String wmUrlStr, final String workflowId,
52        final Class<? extends WebPage> taskPage) {
53      super(id);
54      this.wm = new WorkflowMgrConn(wmUrlStr);
55      final IModel<Workflow> wModel = new LoadableDetachableModel<Workflow>() {
56  
57        @Override
58        protected Workflow load() {
59          return wm.safeGetWorkflowById(workflowId);
60        }
61      };
62  
63      add(new Label("workflow_id", new PropertyModel<Workflow>(wModel, "id")));
64      add(new Label("workflow_name", new PropertyModel<Workflow>(wModel, "name")));
65  
66      int numTasks = wModel.getObject().getTasks().size();
67      String width = String.valueOf((82 * numTasks) + (115 * (numTasks - 1)));
68      String height = "82";
69      WebMarkupContainer wTable = new WebMarkupContainer("workflow_model_tbl");
70      wTable.add(new SimpleAttributeModifier("width", width));
71      wTable.add(new SimpleAttributeModifier("height", height));
72  
73      wTable.add(new ListView<WorkflowTask>("tasks", new ListModel<WorkflowTask>(
74          wModel.getObject().getTasks())) {
75  
76        @Override
77        protected void populateItem(ListItem<WorkflowTask> item) {
78          final PageParameters params = new PageParameters();
79          params.add("id", item.getModelObject().getTaskId());
80          Link taskLink = new Link("task_link") {
81            /*
82             * (non-Javadoc)
83             * 
84             * @see org.apache.wicket.markup.html.link.Link#onClick()
85             */
86            @Override
87            public void onClick() {
88              setResponsePage(taskPage, params);
89            }
90          };
91          String taskName = summarizeWords(item.getModelObject().getTaskName(),
92              16, 16);
93          taskLink.add(new Label("task_name", taskName).setRenderBodyOnly(true));
94          item.add(taskLink);
95  
96          if (item.getIndex() == wModel.getObject().getTasks().size() - 1) {
97            item.add(new WebMarkupContainer("task_arrow").setVisible(false));
98          } else {
99            item.add(new WebMarkupContainer("task_arrow"));
100         }
101 
102       }
103     });
104     add(wTable);
105 
106   }

108   /**
109    * <p>
110    * Summarizes a given String of words (the <code>orig</code> parameter), and
111    * limits the size of the individual words in the string by the given
112    * <code>wordThreshold</code>, and limits the final size of the final
113    * summarized word string by the given <code>maxLengthTotal</code>.
114    * </p>
115    * 
116    * @param orig
117    *          The original String to summarize.
118    * @param wordThreshhold
119    *          The maximum amount of characters for any given word in the string.
120    * @param maxLengthTotal
121    *          The maximum final size of the summarized set of words.
122    * @return A summarized string.
123    */
124   private String summarizeWords(String orig, int wordThreshhold,
125       int maxLengthTotal) {
126     String[] words = orig.split(" ");
127     StringBuilder summarizedString = new StringBuilder();
128 
129     for (String word : words) {
130       summarizedString.append(word.substring(0, Math.min(wordThreshhold, word
131           .length())));
132       summarizedString.append(" ");
133     }
134     
135     // add '...' to end of summarized string if applicable
136     if (summarizedString.length() > maxLengthTotal) {
137     	return summarizedString.substring(0,
138             Math.min(maxLengthTotal, summarizedString.length()) - 3) + "...";
139     } else {
140       return summarizedString.toString();
141     }
142   }
