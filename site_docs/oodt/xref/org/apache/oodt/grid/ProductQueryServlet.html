37  /**
38   * Product query servlet handles product queries.  It always returns the first matching
39   * product, if any.  If no handler can provide a product, it returns 404 Not Found.  If
40   * there are no query handlers, it returns 404 Not Found.
41   *
42   */
43  public class ProductQueryServlet extends QueryServlet {
44  
45    public static final int INT = 512;
46  
47    /** {@inheritDoc} */
48  	protected List getServers(Configuration config) {
49  		return config.getProductServers();
50  	}
51  
52  	/** {@inheritDoc} */
53  	protected void handleQuery(XMLQuery query, List handlers, HttpServletRequest req, HttpServletResponse res)
54  		throws IOException {
55  		if (handlers.isEmpty()) {
56  			res.sendError(HttpServletResponse.SC_NOT_FOUND, "no query handlers available to handle query");
57  			return;
58  		}
59  
60  		try {								       // OK, let's try
61  		  for (Object handler1 : handlers) {           // Try each query handler
62  			QueryHandler handler = (QueryHandler) handler1;           // Get the query handler
63  			query = handler.query(query);                   // Give it the query
64  			if (!query.getResults().isEmpty()) {               // Did it give any result?
65  			  Result result = (Result) query.getResults().get(0); // Yes, get the result
66  			  deliverResult(handler, result, res);           // And deliver it
67  			  return;                           // Done!
68  			}
69  		  }
70  		} catch (ProductException ex) {
71            if (ex instanceof HttpRedirectException) {
72              HttpRedirectException hre = (HttpRedirectException) ex;
73              res.sendRedirect(hre.getLocation());
74            } else {
75              res.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, ex.getMessage());
76            }
77            return;
78  		}
79  
80  		res.sendError(HttpServletResponse.SC_NOT_FOUND, "no matching products from any query handler");
81  	}

83  	/**
84  	 * Deliver a result.  This streams the product data.
85  	 *
86  	 * @param handler Which handler produced the result.
87  	 * @param result The result.
88  	 * @param res The HTTP response to which to send the result data.
89  	 * @throws IOException if an error occurs.
90  	 */
91  	private void deliverResult(QueryHandler handler, Result result, HttpServletResponse res) throws IOException {
92  		characterize(handler, result, res);					       // First, describe it using HTTP headers
93  		if (result instanceof LargeResult) {				       // Is it a large result?
94  			LargeResult lr = (LargeResult) result;			       // Yes, this is gonna take some special work
95  			LargeProductQueryHandler lpqh = (LargeProductQueryHandler) handler; // First treat 'em as large
96  			ProductRetriever retriever = new ProductRetriever(lpqh);       // Large ones need a retriever
97  			lr.setRetriever(retriever);				       // Set the retriever
98  		}
99  		BufferedInputStream in = null;					       // Start with no input stream
100 		try {								       // Then try ...
101 			in = new BufferedInputStream(result.getInputStream());	       // To open the input stream
102 			byte[] buf = new byte[INT];				       // And a byte buffer for data
103 			int num;						       // And a place to count data
104 			while ((num = in.read(buf)) != -1)			       // While we read
105 			{
106 			  res.getOutputStream().write(buf, 0, num);           // We write
107 			}
108 			res.getOutputStream().flush();				       // Flush to commit response
109 		} finally {							       // And finally
110 			if (in != null) {
111 			  try {                           // If we opened it
112 				in.close();                           // Close it
113 			  } catch (IOException ignore) {
114 			  }                       // Ignoring any error during closing
115 			}
116 		}								       // Because come on, it's just closing!
117 	}									       // For fsck's sake!

119 	/**
120 	 * Characterize a result by using HTTP headers.
121 	 *
122 	 * @param result Result to characterize.
123 	 * @param res HTTP response to set headers in.
124 	 */
125 	private void characterize(QueryHandler handler, Result result, HttpServletResponse res) {
126 		String contentType = result.getMimeType();			       // Grab the content type
127 		res.setContentType(contentType);				       // Set it
128 		long size = result.getSize();					       // Grab the size
129 		if (size >= 0) {
130 		  res.addHeader("Content-Length", String.valueOf(size));           // Don't use setContentLength(int)
131 		}
132 		if (!displayable(contentType))					       // Finally, if a browser can't show it
133 		{
134 		  this.suggestFilename(handler, result, res);               // Then suggest a save-as filename
135 		}
136 	}

138 	/**
139 	 * Tell if a result is displayable.  This compares its MIME type to a list of MIME
140 	 * types commonly displayble by browsers.
141 	 *
142 	 * @param contentType MIME type.
143 	 * @return a <code>boolean</code> value.
144 	 */
145 	protected static boolean displayable(String contentType) {
146 	  for (String DISPLAYABLE_TYPE : DISPLAYABLE_TYPES) {
147 		if (DISPLAYABLE_TYPE.equals(contentType)) {
148 		  return true;     // Does it match?
149 		}
150 	  }
151 		return false;							       // None of 'em do, it's not displayable
152 	}

154 	/**
155 	 * We can suggest a filename (if the client happens to be a browser) using a
156 	 * content-disposition header.
157 	 *
158 	 * @param res a <code>HttpServletResponse</code> value.
159 	 */
160 	protected void suggestFilename(QueryHandler handler, Result result, HttpServletResponse res) {
161 		
162 		String resource = result.getResourceID();
163 		if (resource == null || resource.length() == 0) {
164 		  resource = "product.dat";
165 		}
166 		
167 		// suggest some names based on resource mime type
168 		String contentType = res.getContentType();
169 		
170 		// "zip" mime types
171 		if (contentType!=null) {
172 			if (contentType.equals("application/x-compressed") || 
173 				contentType.equals("application/x-zip-compressed") || 
174 				contentType.equals("application/zip") || 
175 				contentType.equals("multipart/x-zip") ) {
176 				
177 				// resource = resource.replaceAll("\\..+",".zip"); // replace extension with .zip 
178 				resource = "products_" + resource + ".zip"; 				
179 			}
180 		}
181 		
182 		// set "Content-disposition" header
183 		res.addHeader("Content-disposition", "attachment; filename=\"" + resource + "\"");
184 		
185 	}

187 	/**
188 	 * MIME types commonly displayable by browsers.
189 	 */
190 	private static final String[] DISPLAYABLE_TYPES = {
191 		"text/plain", "text/richtext", "text/enriched", "text/tab-separated-values", "text/html", "text/xml", "text/rtf",
192 		"message/rfc822", "message/partial", "message/external-body", "message/news", "message/http",
193 		"message/delivery-status", "message/disposition-notification", "message/s-http", "application/rtf",
194 		"application/pdf", "image/jpeg", "image/gif", "image/tiff", "image/png", "audio/basic", "audio/32kadpcm",
195 		"audio/mpeg", "video/mpeg", "video/quicktime"
196 	};

198 	/**
199 	 * Retriever that retrieves product data over a method call boundary to a large
200 	 * product query handler.
201 	 */
202 	private static class ProductRetriever implements Retriever {
203 		/**
204 		 * Creates a new <code>ProductRetriever</code> instance.
205 		 *
206 		 * @param handler a <code>LargeProductQueryHandler</code> value.
207 		 */
208 		public ProductRetriever(LargeProductQueryHandler handler) {
209 			this.handler = handler;
210 		}
211 
212 		/** {@inheritDoc} */
213 		public byte[] retrieveChunk(String id, long offset, int len) throws ProductException {
214 			return handler.retrieveChunk(id, offset, len);
215 		}
216 
217 		/** {@inheritDoc} */
218 		public void close(String id) {
219 			handler.close(id);
220 		}
221 
222 		/** Handler to use. */
223 		private LargeProductQueryHandler handler;
224 	}
