188     private void doIngest() {
189         Metadata prodMet;
190 
191         try {
192             URL ingestUrl = this.getClass().getResource("/ingest");
193             URL refUrl = this.getClass().getResource("/ingest/test.txt");
194             URL metUrl = this.getClass().getResource("/ingest/test.txt.met");
195 
196             prodMet = new SerializableMetadata(new FileInputStream(
197                 new File(metUrl.getFile())));
198 
199             // now add the right file location
200             prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
201                 ingestUrl.getFile()).getCanonicalPath());
202             ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
203                 refUrl.getFile()), prodMet);
204         } catch (Exception e) {
205             LOG.log(Level.SEVERE, e.getMessage());
206             fail(e.getMessage());
207         }
208 
209         // now make sure that the file is ingested
210         try {
211             XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(
212                     new URL("http://localhost:" + FM_PORT));
213             Product p = fmClient.getProductByName("test.txt");
214             assertNotNull(p);
215             assertEquals(Product.STATUS_RECEIVED, p.getTransferStatus());
216             assertTrue(fmClient.hasProduct("test.txt"));
217         } catch (Exception e) {
218             fail(e.getMessage());
219         }
220     }

222     private void deleteAllFiles(String startDir) {
223         File startDirFile = new File(startDir);
224         File[] delFiles = startDirFile.listFiles();
225 
226         if (delFiles != null && delFiles.length > 0) {
227             for (File delFile : delFiles) {
228                 delFile.delete();
229             }
230         }
231 
232         startDirFile.delete();
233 
234     }

236     private void startXmlRpcFileManager() {
237 
238         Properties properties = new Properties(System.getProperties());
239 
240         // first make sure to load properties for the file manager
241         // and make sure to load logging properties as well
242 
243         // set the log levels
244         URL loggingPropertiesUrl = this.getClass().getResource(
245             "/test.logging.properties");
246         properties.setProperty("java.util.logging.config.file", new File(
247             loggingPropertiesUrl.getFile()).getAbsolutePath());
248 
249         // first load the example configuration
250         try {
251           URL filemgrPropertiesUrl = this.getClass().getResource(
252               "/filemgr.properties");
253           properties.load(
254               new FileInputStream(new File(filemgrPropertiesUrl.getFile())));
255         } catch (Exception e) {
256             fail(e.getMessage());
257         }
258 
259         // override the catalog to use: we'll use lucene
260         try {
261             URL ingestUrl = this.getClass().getResource("/ingest");
262             luceneCatLoc = new File(ingestUrl.getFile()).getCanonicalPath()
263                 + "/cat";
264         } catch (Exception e) {
265             fail(e.getMessage());
266         }
267 
268         properties.setProperty("filemgr.catalog.factory",
269                 "org.apache.oodt.cas.filemgr.catalog.LuceneCatalogFactory");
270         properties.setProperty(
271                 "org.apache.oodt.cas.filemgr.catalog.lucene.idxPath",
272                 luceneCatLoc);
273 
274         // now override the repo mgr policy
275         try {
276             URL fmpolicyUrl = this.getClass().getResource("/ingest/fmpolicy");
277             properties.setProperty(
278                     "org.apache.oodt.cas.filemgr.repositorymgr.dirs",
279                     "file://"
280                         + new File(fmpolicyUrl.getFile()).getCanonicalPath());
281         } catch (Exception e) {
282             fail(e.getMessage());
283         }
284 
285         // now override the val layer ones
286         URL examplesCoreUrl = this.getClass().getResource("/examples/core");
287         properties.setProperty("org.apache.oodt.cas.filemgr.validation.dirs",
288                 "file://"
289                     + new File(examplesCoreUrl.getFile()).getAbsolutePath());
290 
291         // set up mime repo path
292         URL mimeTypesUrl = this.getClass().getResource("/mime-types.xml");
293         properties.setProperty(
294                 "org.apache.oodt.cas.filemgr.mime.type.repository", new File(
295                      mimeTypesUrl.getFile()).getAbsolutePath());
296 
297         System.setProperties(properties);
298 
299         try {
300             fm = new XmlRpcFileManager(FM_PORT);
301         } catch (Exception e) {
302             fail(e.getMessage());
303         }
304     }
