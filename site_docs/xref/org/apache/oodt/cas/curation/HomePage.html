18  package org.apache.oodt.cas.curation;
19  
20  import org.apache.oodt.cas.curation.login.LoginPage;
21  import org.apache.oodt.cas.curation.workbench.WorkbenchPage;
22  import org.apache.wicket.Application;
23  import org.apache.wicket.PageParameters;
24  import org.apache.wicket.markup.html.WebPage;
25  import org.apache.wicket.markup.html.basic.Label;
26  import org.apache.wicket.markup.html.link.Link;
27  
28  public class HomePage extends WebPage {
29    
30    public HomePage() {
31      super();
32      CurationApp app = ((CurationApp) Application.get());
33      String loggedInLabel;
34      String logoutLabel = "Log Out";
35      CurationSession session = (CurationSession)getSession();
36  
37      if (session.isLoggedIn()) {
38        loggedInLabel = "Logged in as " + session.getLoginUsername()+ ".";
39        add(new Label("loggedin_label", loggedInLabel));
40        add(new Label("logout_label", logoutLabel).setVisible(false));
41        add(new Link<String>("login_link") {
42          @Override
43          public void onClick() {
44            PageParameters params = new PageParameters();
45            params.add("action", "login");
46            setResponsePage(LoginPage.class, params);
47          }
48        }.setVisible(false));
49        add(new Link<String>("logout_link") {
50          @Override
51          public void onClick() {
52            PageParameters params = new PageParameters();
53            params.add("action", "logout");
54            setResponsePage(LoginPage.class, params);
55          }
56        });
57      } else {
58        loggedInLabel = "Not logged in.";
59        add(new Label("loggedin_label", loggedInLabel));
60        add(new Label("logout_label", logoutLabel).setVisible(false));
61        add(new Link<String>("login_link") {
62          @Override
63          public void onClick() {
64            PageParameters params = new PageParameters();
65            params.add("action", "login");
66            setResponsePage(LoginPage.class, params);
67          }
68        });
69        add(new Link<String>("logout_link") {
70          @Override
71          public void onClick() {
72            PageParameters params = new PageParameters();
73            params.add("action", "logout");
74            setResponsePage(LoginPage.class, params);
75          }
76        }.setVisible(false));
77      }
78  
79      String projectName = app.getProjectName() + " CAS Curation";
80      add(new Label("project_name", projectName));
81      add(new Label("crumb_name", "Main"));
82      add(new Link<String>("home_link"){
83        @Override
84        public void onClick() {
85          setResponsePage(WorkbenchPage.class);
86        }
87      });
88    }
