38  /**
39   * @author mattmann
40   * @author bfoster
41   * @version $Revision$
42   * 
43   * <p>
44   * An abstract base class for data transfers that uses an in-memory ConcurrentHashMap to
45   * keep track of data transfer status information.
46   * </p>
47   * 
48   */
49  public class TransferStatusTracker {
50      /* ConcurrentHashMap containing a list of current product transfers */
51      protected ConcurrentHashMap<String, Product> currentProductTransfers = new ConcurrentHashMap<String, Product>();
52  
53      /* our catalog object */
54      private Catalog catalog = null;
55  
56      /* our log stream */
57      private static final Logger LOG = Logger.getLogger(TransferStatusTracker.class
58              .getName());
59  
60      /**
61       * <p>
62       * Default Constructor
63       * </p>.
64       * 
65       * @param catalog
66       *            The Catalog interface handed to it by the File Manager. It
67       *            needs this object to look up reference information.
68       */
69      public TransferStatusTracker(Catalog catalog) {
70          this.catalog = catalog;
71      }
72  
73      public FileTransferStatus getCurrentFileTransfer() {
74          List<FileTransferStatus> transfers = getCurrentFileTransfers();
75  
76          if (transfers != null && transfers.size() > 0) {
77              return (FileTransferStatus) transfers.get(0);
78          } else {
79              return null;
80          }
81      }
82  
83      public void transferringProduct(Product product) {
84          currentProductTransfers.put(product.getProductId(), product);
85      }
86  
87      public List<FileTransferStatus> getCurrentFileTransfers() {
88          List<FileTransferStatus> currTransfers = new Vector<FileTransferStatus>();
89  
90          for (Map.Entry<String, Product> productId : currentProductTransfers.entrySet()) {
91              Product p = productId.getValue();
92  
93              // get its references
94              List<Reference> refs = quietGetReferences(p);
95  
96              if (refs != null && refs.size() > 0) {
97                  for (Reference r : refs) {
98                      long bytesTransferred = getBytesTransferred(r);
99  
100                     if (bytesTransferred > 0
101                         && bytesTransferred < r.getFileSize() && !isDir(r)) {
102                         FileTransferStatus status = new FileTransferStatus();
103                         status.setBytesTransferred(bytesTransferred);
104                         status.setFileRef(r);
105                         status.setParentProduct(p);
106                         currTransfers.add(status);
107                     }
108                 }
109             }
110         }
111 
112         return currTransfers;
113     }
114 
115     public double getPctTransferred(Product product) {
116         // get its references
117         List<Reference> refs = quietGetReferences(product);
118         long totalBytesTransferred = 0L;
119         long totalProductSize = 0L;
120 
121         if (refs.size() > 0) {
122             for (Reference r : refs) {
123                 long bytesTransferred = getBytesTransferred(r);
124 
125                 if (!isDir(r)) {
126                     // only add this if > 0
127                     if (bytesTransferred > 0) {
128                         totalBytesTransferred += bytesTransferred;
129                     }
130 
131                     // add this no matter what
132                     totalProductSize += r.getFileSize();
133                 }
134 
135             }
136         }
137 
138         return ((double) ((1.0 * totalBytesTransferred) / (1.0 * totalProductSize)));
139 
140     }
141 
142     public double getPctTransferred(Reference ref) {
143         long bytesTransferred = getBytesTransferred(ref);
144         return ((double) ((1.0 * bytesTransferred) / (1.0 * ref.getFileSize())));
145     }
146 
147     public void removeProductTransferStatus(Product product) {
148         if (currentProductTransfers.get(product.getProductId()) != null) {
149             currentProductTransfers.remove(product.getProductId());
150         }
151     }
152 
153     public boolean isTransferComplete(Product product) {
154         return getPctTransferred(product) == 1.0;
155     }
156 
157     private long getBytesTransferred(Reference r) {
158         File destFile;
159 
160         try {
161             destFile = new File(new URI(r.getDataStoreReference()));
162             return destFile.length();
163         } catch (URISyntaxException e) {
164             LOG.log(Level.WARNING,
165                     "URISyntaxException when checking size of destFile: ["
166                             + r.getDataStoreReference() + "]: Message: "
167                             + e.getMessage());
168             return -1L;
169         }
170     }
171 
172     private List<Reference> quietGetReferences(Product p) {
173         List<Reference> refs;
174 
175         try {
176             refs = catalog.getProductReferences(p);
177         } catch (CatalogException e) {
178             LOG.log(Level.SEVERE, e.getMessage());
179             LOG.log(Level.WARNING, "Error retreiving references for product: ["
180                     + p.getProductId()
181                     + "] from catalog in transfer status tracker: Message: "
182                     + e.getMessage());
183             refs = new Vector<Reference>();
184         }
185 
186         return refs;
187     }
188 
189     private boolean isDir(Reference r) {
190         File fileRef;
191 
192         try {
193             fileRef = new File(new URI(r.getDataStoreReference()));
194             return fileRef.isDirectory();
195         } catch (URISyntaxException e) {
196             return false;
197         }
198     }
