228     private void ingestTestFiles() {
229         Metadata prodMet;
230         StdIngester ingester = new StdIngester(transferServiceFacClass);

232         try {
233             URL ingestUrl = this.getClass().getResource("/ingest");
234             URL refUrl = this.getClass().getResource("/ingest/test.txt");
235             URL metUrl = this.getClass().getResource("/ingest/test.txt.met");

237             prodMet = new SerializableMetadata(new FileInputStream(
238                 new File(metUrl.getFile())));
239 
240             // now add the right file location
241             prodMet.addMetadata(CoreMetKeys.FILE_LOCATION, new File(
242                 ingestUrl.getFile()).getCanonicalPath());
243             ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
244                 refUrl.getFile()), prodMet);
245             
246             prodMet.replaceMetadata(CoreMetKeys.PRODUCT_NAME, "TestTypeFile");
247             prodMet.replaceMetadata(CoreMetKeys.PRODUCT_TYPE, "TestType");
248             ingester.ingest(new URL("http://localhost:" + FM_PORT), new File(
249                 refUrl.getFile()), prodMet);

251             fail(e.getMessage());

255     private void deleteAllFiles(String startDir) {
256         File startDirFile = new File(startDir);
257         File[] delFiles = startDirFile.listFiles();
258 
259         if (delFiles != null && delFiles.length > 0) {
260             for (File delFile : delFiles) {
261                 delFile.delete();
262             }
263         }
264 
265         startDirFile.delete();
266 
267     }

269     private XmlRpcFileManager startXmlRpcFileManager(int port, String catPath) {
270 
271         Properties properties = new Properties(System.getProperties());
272 
273         XmlRpcFileManager fileMgr = null;
274         
275         // first make sure to load properties for the file manager
276         // and make sure to load logging properties as well
277 
278         // set the log levels
279         URL loggingPropertiesUrl = this.getClass().getResource(
280             "/test.logging.properties");
281         properties.setProperty("java.util.logging.config.file", new File(
282             loggingPropertiesUrl.getFile()).getAbsolutePath());
283         
284         
285         // first load the example configuration
286         try {
287             URL filemgrPropertiesUrl = this.getClass().getResource(
288                 "/filemgr.properties");
289             properties.load(
290                 new FileInputStream(new File(filemgrPropertiesUrl.getFile())));
291         } catch (Exception e) {
292             fail(e.getMessage());
293         }
294 
295         // override the catalog to use: we'll use lucene
296         properties.setProperty("filemgr.catalog.factory",
297                 "org.apache.oodt.cas.filemgr.catalog.LuceneCatalogFactory");
298         properties.setProperty(
299                 "org.apache.oodt.cas.filemgr.catalog.lucene.idxPath",
300                 catPath);
301 
302         // now override the repo mgr policy
303         URL fmpolicyUrl = this.getClass().getResource("/ingest/fmpolicy");
304         try {
305           properties.setProperty(
306               "org.apache.oodt.cas.filemgr.repositorymgr.dirs",
307               "file://" + new File(fmpolicyUrl.getFile()).getCanonicalPath());
308         } catch (Exception e) {
309             fail(e.getMessage());
310         }
311 
312         // now override the val layer ones
313         properties.setProperty("org.apache.oodt.cas.filemgr.validation.dirs",
314             "file://" + new File(fmpolicyUrl.getFile()).getAbsolutePath());
315 
316         // set up mime repo path
317         URL mimeTypesUrl = this.getClass().getResource("/mime-types.xml");
318         properties.setProperty(
319             "org.apache.oodt.cas.filemgr.mime.type.repository",
320             new File(mimeTypesUrl.getFile()).getAbsolutePath());
321 
322         System.setProperties(properties);
323 
324         try {
325             fileMgr = new XmlRpcFileManager(port);
326         } catch (Exception e) {
327             fail(e.getMessage());
328         }
329 
330         return fileMgr;
331     }
