38  /**

40   * Test harness for the {@link DirectoryProductVersioner}.

42   * @author mattmann
43   * @version $Revision$

45   */
46  public class TestDirectoryBasedProductVersioner extends TestCase {
47  
48    private static Logger LOG = Logger.getLogger(TestDirectoryBasedProductVersioner.class.getName());
49    private Properties initialProperties = new Properties(System.getProperties());
50  
51    public void setUp() throws Exception {
52      Properties properties = new Properties(System.getProperties());
53      URL url = this.getClass().getResource("/mime-types.xml");
54      properties.setProperty("org.apache.oodt.cas.filemgr.mime.type.repository",
55          new File(url.getFile()).getAbsolutePath());
56      System.setProperties(properties);
57    }
58  
59    public void tearDown() throws Exception {
60      System.setProperties(initialProperties);
61    }
62  
63    public void testVersioner() {
64      DirectoryProductVersioner versioner = new DirectoryProductVersioner();
65      Product p = Product.getDefaultFlatProduct("test", "urn:oodt:GenericFile");
66      p.setProductStructure(Product.STRUCTURE_HIERARCHICAL);
67      p.getProductType().setProductRepositoryPath("file:///home/files");
68      Reference r = new Reference("file:///tmp/dir1", null, 4L);
69      Reference r2 = new Reference("file:///tmp/dir1/file1.txt", null, 20L);
70      p.getProductReferences().add(r);
71      p.getProductReferences().add(r2);
72      Metadata met = new Metadata();
73      met.addMetadata(CoreMetKeys.FILENAME, "dir1");
74      try {
75        versioner.createDataStoreReferences(p, met);
76      } catch (Exception e) {
77        LOG.log(Level.SEVERE, e.getMessage());
78        fail(e.getMessage());
79      }
80  
81      assertNotNull(p.getProductReferences());
82      assertEquals(2, p.getProductReferences().size());
83      assertEquals("file:/home/files/dir1/", p.getProductReferences().get(0)
84          .getDataStoreReference());
85      assertEquals("file:/home/files/dir1/file1.txt", p.getProductReferences()
86          .get(1).getDataStoreReference());
87  
88    }
