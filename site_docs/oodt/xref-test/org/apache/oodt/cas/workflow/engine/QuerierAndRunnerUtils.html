48  public class QuerierAndRunnerUtils {
49  
50    private int dateGen;
51    
52    public QuerierAndRunnerUtils() {
53      this.dateGen = 0;
54    }
55    
56    public WorkflowTask getTask(File testDir){
57      WorkflowTask task = new WorkflowTask();
58      task.setConditions(Collections.emptyList());
59      task.setRequiredMetFields(Collections.emptyList());
60      task.setTaskId("urn:cas:workflow:tester");
61      task.setTaskInstanceClassName(SimpleTester.class.getName());
62      task.setTaskName("Tester");
63      WorkflowTaskConfiguration config = new WorkflowTaskConfiguration();
64      config.addConfigProperty("TestDirPath",
65          testDir.getAbsolutePath().endsWith("/") ? testDir.getAbsolutePath()
66              : testDir.getAbsolutePath() + "/");
67      task.setTaskConfig(config);
68      return task;
69    }
70  
71    public WorkflowProcessor getProcessor(double priority, String stateName,
72        String categoryName) throws InstantiationException,
73        IllegalAccessException, IOException, IllegalArgumentException, SecurityException, InvocationTargetException, NoSuchMethodException {
74      WorkflowLifecycleManager lifecycleManager = new WorkflowLifecycleManager(
75          "./src/main/resources/examples/wengine/wengine-lifecycle.xml");
76      WorkflowInstance inst = new WorkflowInstance();
77      Date sd = new Date();
78      sd.setTime(sd.getTime() + (this.dateGen * 5000));
79      this.dateGen++;
80      inst.setStartDate(sd);
81      inst.setId("task-winst-" + priority);
82      ParentChildWorkflow workflow = new ParentChildWorkflow(new Graph());
83      workflow.getTasks().add(getTask(getTmpPath()));
84      inst.setParentChildWorkflow(workflow);
85      inst.setPriority(Priority.getPriority(priority));
86      inst.setCurrentTaskId(workflow.getTasks().get(0).getTaskId());
87      inst.setParentChildWorkflow(workflow);
88      WorkflowProcessorBuilder builder = WorkflowProcessorBuilder
89          .aWorkflowProcessor().withLifecycleManager(lifecycleManager)
90          .withPriority(priority).withInstance(inst);
91      TaskProcessor taskProcessor = (TaskProcessor) builder
92          .build(TaskProcessor.class);
93      taskProcessor.getWorkflowInstance().setState(lifecycleManager.getDefaultLifecycle().createState(
94          stateName, categoryName, ""));    
95      return taskProcessor;
96    }
97    
98    private File getTmpPath() throws IOException{
99      File testDir;
100     String parentPath = File.createTempFile("test", "txt").getParentFile().getAbsolutePath();
101     parentPath = parentPath.endsWith("/") ? parentPath:parentPath + "/";
102     String testJobDirPath = parentPath + "jobs";
103     testDir = new File(testJobDirPath);
104     testDir.mkdirs();
105     return testDir;
106   }
