17  package org.apache.oodt.cas.crawl.cli.action;
18  
19  //OODT imports
20  import org.apache.oodt.cas.cli.action.CmdLineAction;
21  import org.apache.oodt.cas.cli.exception.CmdLineActionException;
22  import org.apache.oodt.cas.crawl.ProductCrawler;
23  import org.apache.oodt.cas.crawl.daemon.CrawlDaemon;
24  
25  //Spring imports
26  import org.springframework.context.support.FileSystemXmlApplicationContext;
27  
28  /**
29   * A {@link CmdLineAction} which is responsible for launching crawlers.
30   * 
31   * @author bfoster (Brian Foster)
32   */
33  public class CrawlerLauncherCliAction extends CmdLineAction {
34  
35     private String beanRepo;
36     private String crawlerId;
37  
38     public CrawlerLauncherCliAction() {
39        this.beanRepo = System.getProperty("org.apache.oodt.cas.crawl.bean.repo",
40              "classpath:/org/apache/oodt/cas/crawl/crawler-config.xml");
41     }
42  
43     @Override
44     public void execute(ActionMessagePrinter printer)
45           throws CmdLineActionException {
46  
47        FileSystemXmlApplicationContext appContext = new FileSystemXmlApplicationContext(
48              this.beanRepo);
49  
50        try {
51           ProductCrawler pc = (ProductCrawler) appContext
52                 .getBean(crawlerId != null ? crawlerId : getName());
53           pc.setApplicationContext(appContext);
54           if (pc.getDaemonPort() != -1 && pc.getDaemonWait() != -1) {
55              new CrawlDaemon(pc.getDaemonWait(), pc, pc.getDaemonPort())
56                    .startCrawling();
57           } else {
58              pc.crawl();
59           }
60        } catch (Exception e) {
61           throw new CmdLineActionException("Failed to launch crawler : "
62                 + e.getMessage(), e);
63        }
64     }
65  
66     public void setCrawlerId(String crawlerId) {
67        this.crawlerId = crawlerId;
68     }
69  }
