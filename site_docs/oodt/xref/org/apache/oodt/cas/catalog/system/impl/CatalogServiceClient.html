71  	protected List<PluginURL> transferToServerSide(List<PluginURL> clientSideUrls) throws CatalogServiceException {
72  		try {
73  			URL customUrlStorageDir = this.communicationChannelClient.getPluginStorageDir();
74  			System.out.println("Got Tmp dir : " + customUrlStorageDir);
75  			Vector<PluginURL> serverSideUrls = new Vector<PluginURL>();
76  			for (PluginURL pluginURL : clientSideUrls) {
77  				PluginURL serverSideURL = new PluginURL(pluginURL.getCatalogId(), new URL(customUrlStorageDir, new File(pluginURL.getURL().getFile()).getName()));
78  				System.out.println("generated server side url : " + customUrlStorageDir);
79  				this.communicationChannelClient.transferUrl(pluginURL.getURL(), serverSideURL.getURL());
80  				serverSideUrls.add(serverSideURL);
81  			}
82  			return serverSideUrls;
83  		} catch (Exception e) {
84  			throw new CatalogServiceException(e.getMessage(), e);
85  		}
86  	}

215 	public List<TransactionalMetadata> getAllPages(QueryPager queryPager)
216 			throws CatalogServiceException {
217 		try {
218 			List<TransactionalMetadata> metadata = new Vector<TransactionalMetadata>();
219 			if (queryPager.getTotalPages() > 0) {
220 				queryPager.setPageInfo(new PageInfo(this.autoPagerSize, PageInfo.FIRST_PAGE));
221 				while (!queryPager.isLastPage()) {
222 					metadata.addAll(this.communicationChannelClient.getNextPage(queryPager));
223 					queryPager.incrementPageNumber();
224 				}
225 				metadata.addAll(this.communicationChannelClient.getNextPage(queryPager));
226 			}
227 			return metadata;
228 		} catch (Exception e) {
229 			throw new CatalogServiceException(e.getMessage(), e);
230 		}
231 	}
