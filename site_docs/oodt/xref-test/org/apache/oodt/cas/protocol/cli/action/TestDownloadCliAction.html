33  /**
34   * Test class for {@link DownloadCliAction}.
35   * 
36   * @author bfoster (Brian Foster)
37   */
38  public class TestDownloadCliAction extends TestCase {
39     
40     private ProtocolManager pm;
41     
42     @Override
43     public void setUp() {
44        pm = new ProtocolManager(new MockSpringProtocolConfig());
45     }
46     
47     public void testCreateProtocol() throws URISyntaxException {
48        DownloadCliAction da = new DownloadCliAction();
49        da.setUrl("http://localhost/some/file");
50        Protocol protocol = da.createProtocol(pm);
51        assertNotNull(protocol);
52        assertTrue(protocol instanceof MockProtocol);
53        assertEquals("http1", ((MockProtocol) protocol).getFactoryId());
54     }
55     
56     public void testCreateProtocolFile() throws URISyntaxException {
57        DownloadCliAction da = new DownloadCliAction();
58        da.setUrl("http://localhost/some/file");
59        ProtocolFile file = da.createProtocolFile();
60        assertNotNull(file);
61        assertEquals("/some/file", file.getPath());
62        assertFalse(file.isRelative());
63        assertFalse(file.isDir());
64     }
65     
66     public void testFileDownload() throws Exception {
67        DownloadCliAction da = new DownloadCliAction();
68        da.setUrl("http://localhost/some/file");
69        da.setProtocolManager(pm);
70        da.execute(new ActionMessagePrinter());
71        assertNotNull(da.getUsedProtocol());
72        assertTrue(da.getUsedProtocol() instanceof MockProtocol);
73        assertEquals(1, ((MockProtocol) da.getUsedProtocol()).getGetFiles().size());
74        assertEquals("/some/file", ((MockProtocol) da.getUsedProtocol()).getGetFiles().get(0).getPath());
75     }
