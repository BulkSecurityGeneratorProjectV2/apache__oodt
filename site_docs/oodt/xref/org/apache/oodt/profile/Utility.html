52  	static void addProperty(Model model, Resource resource, Property property, Object value, ProfileAttributes profAttr,
53  		URI uri) {
54  
55  		if (value == null || value.toString().length() == 0) {
56  		  return;
57  		}
58  
59  		Object obj;
60  		if (value instanceof Collection) {
61  			Collection<?> collection = (Collection<?>) value;
62  			if (collection.isEmpty()) {
63  			  return;
64  			}
65  			Bag bag = model.createBag(uri + "_" + property.getLocalName() + "_bag");
66  		  for (Object aCollection : collection) {
67  			bag.add(aCollection);
68  		  }
69  			resource.addProperty(property, bag);
70  			obj = bag;
71  		} else {
72  			resource.addProperty(property, value.toString());
73  			obj = value;
74  		}
75  
76  		Resource reification = model.createResource(uri + "_" + property.getLocalName() + "_reification");
77  
78  		reification.addProperty(rdfSubject, resource);
79  		reification.addProperty(rdfPredicate, property);
80  		reification.addProperty(rdfObject, obj.toString());
81  		reification.addProperty(rdfType, rdfStatement);
82  
83  		addPotentiallyNullReifiedStatement(reification, edmID, profAttr.getVersion());
84  		addPotentiallyNullReifiedStatement(reification, edmVersion, profAttr.getVersion());
85  		addPotentiallyNullReifiedStatement(reification, edmType, profAttr.getType());
86  		addPotentiallyNullReifiedStatement(reification, edmStatus, profAttr.getStatusID());
87  		addPotentiallyNullReifiedStatement(reification, edmSecurity, profAttr.getSecurityType());
88  		addPotentiallyNullReifiedStatement(reification, edmParent, profAttr.getParent());
89  		addPotentiallyNullReifiedStatement(reification, edmRegAuth, profAttr.getRegAuthority());
90  
91  		List<?> children = profAttr.getChildren();
92  		if (!children.isEmpty()) {
93  			Bag bag = model.createBag(uri + "_" + property.getLocalName() + "_childrenBag");
94  		  for (Object aChildren : children) {
95  			bag.add(aChildren);
96  		  }
97  			reification.addProperty(edmChild, bag);
98  		}
99  
100 		List<?> revNotes = profAttr.getRevisionNotes();
101 		if (!revNotes.isEmpty()) {
102 			Seq seq = model.createSeq(uri + "_" + property.getLocalName() + "_revNotesSeq");
103 		  for (Object revNote : revNotes) {
104 			seq.add(revNote);
105 		  }
106 			reification.addProperty(edmRevNote, seq);
107 		}
108 	}
109 
110 	private static void addPotentiallyNullReifiedStatement(Resource reification, Property property, Object value) {
111 		if (value == null || value.toString().length() == 0) {
112 		  return;
113 		}
114 		reification.addProperty(property, value.toString());
115 	}
