43  /**
44   * 
45   * @author bfoster
46   * @version $Revision$
47   *
48   * <p>Use to evaluate whether your regular expressions are 
49   * accepting the correct strings
50   * </p>.
51   */
52  public class ExpressionValidator extends JPanel {
53    private static Logger LOG = Logger.getLogger(ExpressionValidator.class.getName());
54  	private static final long serialVersionUID = -2840589940304298547L;
55  
56  	private JLabel exprLabel;
57  	private JLabel validateLabel;
58  	private JLabel resultLabel;
59  	
60  	private JFormattedTextField exprField;
61  	private JFormattedTextField validateField;
62  	private JFormattedTextField resultField;
63  
64  	public ExpressionValidator() {
65  		super(new BorderLayout());
66  
67  		exprLabel = new JLabel("Regular Expression: ");
68  		validateLabel = new JLabel("Validate String: ");
69  		resultLabel = new JLabel("Result: ");
70  
71  		exprField = new JFormattedTextField();
72  		exprField.setColumns(60);
73  
74  		validateField = new JFormattedTextField();
75  		validateField.setColumns(60);
76  
77  		resultField = new JFormattedTextField();
78  		resultField.setColumns(60);
79  		resultField.setEditable(false);
80  		resultField.setForeground(Color.red);
81  
82  		exprLabel.setLabelFor(exprField);
83  		validateLabel.setLabelFor(validateField);
84  		resultLabel.setLabelFor(resultField);
85  
86  		JPanel labelPane = new JPanel(new GridLayout(0, 1));
87  		labelPane.add(exprLabel);
88  		labelPane.add(validateLabel);
89  		labelPane.add(resultLabel);
90  
91  		JPanel fieldPane = new JPanel(new GridLayout(0, 1));
92  		fieldPane.add(exprField);
93  		fieldPane.add(validateField);
94  		fieldPane.add(resultField);
95  
96  		JPanel buttonPane = new JPanel(new BorderLayout());
97  		JButton validate = new JButton("Validate");
98  		validate.addActionListener(new ActionListener() {
99  			public void actionPerformed(ActionEvent event) {
100 				try {
101 					boolean value = ((String) validateField.getText()).matches((String) exprField.getText());
102 					resultField.setText(value + "");
103 				}catch (Exception e) {
104 					LOG.log(Level.SEVERE, e.getMessage());
105 				}
106 			}
107 		});
108 		buttonPane.add(validate, BorderLayout.EAST);
109 		
110 		setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
111 		add(labelPane, BorderLayout.WEST);
112 		add(fieldPane, BorderLayout.CENTER);
113 		add(buttonPane, BorderLayout.EAST);
114 	}
115 
116 	public static void main(String[] args) {
117 		SwingUtilities.invokeLater(new Runnable() {
118 			public void run() {
119 			    Dimension dim = Toolkit.getDefaultToolkit().getScreenSize();
120 				JFrame frame = new JFrame("Regular Expression Evaluator");
121 				frame.setSize(dim.width / 2, dim.height / 2);
122 			    frame.setLocation(dim.width / 3, dim.height / 3);
123 				frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
124 				frame.add(new ExpressionValidator());
125 				frame.pack();
126 				frame.setVisible(true);
127 			}
128 		});
129 	}
