44  /**
45   * A {@link Provider} that writes {@link ProductResource product resources} to
46   * output streams for HTTP responses with "application/rss+xml" content-type.
47   * @author rlaidlaw
48   * @version $Revision$
49   */
50  @Provider
51  @Produces("application/rss+xml")
52  public class ProductRssWriter extends RssWriter
53    implements MessageBodyWriter<ProductResource>
54  {
55    private static final Logger LOGGER = Logger.getLogger(ProductRssWriter.class
56      .getName());
57  
58  
59  
60    @Override
61    public long getSize(ProductResource resource, Class<?> type, Type genericType,
62      Annotation[] annotations, MediaType mediaType)
63    {
64      return -1;
65    }
66  
67  
68  
69    @Override
70    public boolean isWriteable(Class<?> type, Type genericType,
71      Annotation[] annotations, MediaType mediaType)
72    {
73      return true;
74    }
75  
76  
77  
78    @Override
79    public void writeTo(ProductResource resource, Class<?> type, Type genericType,
80      Annotation[] annotations, MediaType mediaType,
81      MultivaluedMap<String, Object> httpHeaders, OutputStream entityStream)
82      throws IOException, WebApplicationException
83    {
84      try
85      {
86        String productName = resource.getProductName();
87        String productId = resource.getProductId();
88        String productTypeName = resource.getProductTypeName();
89  
90        String base = getBaseUri() + "product";
91        String query = "?productId=" + productId;
92        String idLink = base + query;
93        String selfLink = base + ".rss" + query;
94        String currentDate = dateFormatter.format(new Date());
95  
96        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
97        factory.setNamespaceAware(true);
98        Document doc = factory.newDocumentBuilder().newDocument();
99        Element rss = XMLUtils.addNode(doc, doc, "rss");
100       XMLUtils.addAttribute(doc, rss, "version", "2.0");
101       XMLUtils.addAttribute(doc, rss, "xmlns:atom",
102         "http://www.w3.org/2005/Atom");
103 
104       Element channel = XMLUtils.addNode(doc, rss, "channel");
105       XMLUtils.addNode(doc, channel, "title", productName);
106 
107       Element atomLink = XMLUtils.addNode(doc, channel, "atom:link");
108       XMLUtils.addAttribute(doc, atomLink, "href", selfLink);
109       XMLUtils.addAttribute(doc, atomLink, "rel", "self");
110       XMLUtils.addAttribute(doc, atomLink, "type", "application/rss+xml");
111 
112       XMLUtils.addNode(doc, channel, "link", idLink);
113       XMLUtils.addNode(doc, channel, "description", productTypeName);
114 
115       XMLUtils.addNode(doc, channel, "language", LANGUAGE);
116       XMLUtils.addNode(doc, channel, "copyright", COPYRIGHT);
117       XMLUtils.addNode(doc, channel, "pubDate", currentDate);
118       XMLUtils.addNode(doc, channel, "category", productTypeName);
119       XMLUtils.addNode(doc, channel, "generator", GENERATOR);
120       XMLUtils.addNode(doc, channel, "lastBuildDate", currentDate);
121 
122       for (ReferenceResource referenceResource : resource
123         .getReferenceResources())
124       {
125         int refIndex = referenceResource.getRefIndex();
126         String referenceBase = getBaseUri() + "reference";
127         String referenceQuery = "?productId=" + productId + "&refIndex="
128           + refIndex;
129         String referenceLink = referenceBase + ".file" + referenceQuery;
130         String referenceIdLink = referenceBase + referenceQuery;
131 
132         Element item = XMLUtils.addNode(doc, channel, "item");
133         XMLUtils.addNode(doc, item, "title", "reference (" + refIndex + ")");
134         XMLUtils.addNode(doc, item, "link", referenceLink);
135         XMLUtils.addNode(doc, item, "description",
136           "A file manager reference entity");
137         XMLUtils.addNode(doc, item, "guid", referenceIdLink);
138       }
139 
140       XMLUtils.writeXmlToStream(doc, entityStream);
141     }
142     catch (ParserConfigurationException e)
143     {
144       String message = "Unable to create RSS XML document for RSS response.";
145       LOGGER.log(Level.WARNING, message, e);
146       throw new InternalServerErrorException(message);
147     }
148   }
149 }
