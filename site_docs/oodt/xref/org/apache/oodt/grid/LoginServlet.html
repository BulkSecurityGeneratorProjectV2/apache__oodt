18  package org.apache.oodt.grid;
19  
20  import java.io.IOException;
21  import java.util.Arrays;
22  import javax.servlet.ServletException;
23  import javax.servlet.http.HttpServletRequest;
24  import javax.servlet.http.HttpServletResponse;
25  
26  /**
27   * Controller servlet that authenticates the administrator password.
28   * 
29   */
30  public class LoginServlet extends GridServlet {
31  	/**
32  	 * Handle authentication from an administrator.
33  	 *
34  	 * @param req a <code>HttpServletRequest</code> value.
35  	 * @param res a <code>HttpServletResponse</code> value.
36  	 * @throws ServletException if an error occurs.
37  	 * @throws IOException if an error occurs.
38  	 */
39  	public void doPost(HttpServletRequest req, HttpServletResponse res) throws ServletException, IOException {
40  		Configuration config = getConfiguration();			       // Get configuration
41  		if (!approveAccess(config, req, res)) {
42  		  return;                   // Do https, localhost checking first
43  		}
44  
45  		ConfigBean cb = getConfigBean(req);				       // Get bean
46  		if (cb.isAuthentic()) {						       // Already authentic?
47  			req.getRequestDispatcher("config.jsp").forward(req, res);      // Back to the config page with you!
48  			return;
49  		}
50  
51  		String password = req.getParameter("password");			       // Get submitted password
52  		if (password == null) {
53  		  password = "";                       // If none, use an empty string
54  		}
55  		byte[] bytes = password.getBytes();				       // Get the bytes
56  		if (!Arrays.equals(config.getPassword(), bytes)) {		       // Compare to stored password bytes 
57  			cb.setMessage("Password incorrect");			       // Not equal!  Set message.
58  			throw new ServletException(new AuthenticationRequiredException());
59  		} else {
60  			cb.setMessage("");					       // Equal, clear message
61  			cb.setAuthentic(true);					       // You are now authenticated
62  			req.getRequestDispatcher("config.jsp").forward(req, res);      // To the config page with you!
63  		}
64  	}
65  }
