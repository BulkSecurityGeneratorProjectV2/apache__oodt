153     public void testAddAndGetMetadata() throws SQLException, MalformedURLException, ConnectionException {
154         Metadata met = new Metadata();
155         met.addMetadata("DataVersion", "4.0");
156         met.addMetadata("ProductName", "test");
157         Product testProduct = getTestProduct();
158 
159         XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(new URL("http://localhost:" + FILEMGR_PORT));
160         try {
161             testProduct.setProductType(fmClient.getProductTypeByName("GenericFile"));
162             testProduct.setProductId(fmClient.ingestProduct(testProduct, met, false));
163         } catch (Exception e) {
164             LOG.log(Level.SEVERE, e.getMessage());
165             fail(e.getMessage());
166         }
167 
168         try {
169             Metadata retMet = fmClient.getMetadata(testProduct);
170             assertNotNull(retMet);
171             assertTrue(retMet.containsKey("DataVersion"));
172             assertEquals("4.0", retMet
173                     .getMetadata("DataVersion"));
174         } catch (CatalogException e) {
175             fail(e.getMessage());
176         }
177         
178         Statement statement = publicDataSource.getConnection().createStatement();
179         ResultSet rs = statement.executeQuery("SELECT metadata_value FROM GenericFile_metadata WHERE element_id = 'urn:test:DataVersion';");
180         rs.next();
181         assertEquals(rs.getString("metadata_value"), "04.00");
182     }

210     public void testGetCatalogAndOrigValuesAndGetCatalogQuery() throws ConnectionException, RepositoryManagerException, XmlRpcException, IOException {
211         Metadata met = new Metadata();
212         met.addMetadata("DataVersion", "4.0");
213         met.addMetadata("ProductName", "test");
214         Product testProduct = getTestProduct();
215 
216         XmlRpcFileManagerClient fmClient = new XmlRpcFileManagerClient(new URL("http://localhost:" + FILEMGR_PORT));
217         ProductType genericFile = fmClient.getProductTypeByName("GenericFile");
218         assertEquals("04.00", (met = fmClient.getCatalogValues(met, genericFile)).getMetadata("DataVersion"));
219         assertEquals("4.0", fmClient.getOrigValues(met, genericFile).getMetadata("DataVersion"));
220         
221         Query query = new Query();
222         TermQueryCriteria termQuery = new TermQueryCriteria();
223         termQuery.setElementName("DataVersion");
224         termQuery.setValue("4.0");
225         query.addCriterion(termQuery);
226         query = fmClient.getCatalogQuery(query, genericFile);
227         assertEquals("04.00", ((TermQueryCriteria) query.getCriteria().get(0)).getValue());
228     }

341         /*
342          * (non-Javadoc)
343          * 
344          * @see org.apache.oodt.cas.filemgr.catalog.DataSourceCatalog#pagedQuery(org.apache.oodt.cas.filemgr.structs.Query,
345          *      org.apache.oodt.cas.filemgr.structs.ProductType, int)
346          */
347         public ProductPage pagedQuery(Query query, ProductType type, int pageNum)
348                 throws CatalogException {
349             int totalPages = PaginationUtils.getTotalPage(getResultListSize(
350                     query, type), this.pageSize);
351 
352             /*
353              * if there are 0 total pages in the result list size then don't
354              * bother returning a valid product page instead, return blank
355              * ProductPage
356              */
357             if (totalPages == 0) {
358                 return ProductPage.blankPage();
359             }
360 
361             ProductPage retPage = new ProductPage();
362             retPage.setPageNum(pageNum);
363             retPage.setPageSize(this.pageSize);
364             retPage.setTotalPages(totalPages);
365 
366             List productIds = paginateQuery(query, type, pageNum);
367 
368             if (productIds != null && productIds.size() > 0) {
369                 List products = new Vector(productIds.size());
370 
371                 for (Object productId1 : productIds) {
372                     String productId = (String) productId1;
373                     Product p = getProductById(productId);
374                     products.add(p);
375                 }
376 
377                 retPage.setPageProducts(products);
378             }
379 
380             return retPage;
381         }

409             if (currentPage == null) {
410                 return getFirstPage(type);

412 
413             if (currentPage.isLastPage()) {
414                 return currentPage;
415             }
416 
417             ProductPage nextPage = null;
418             Query query = new Query();
419

421                 nextPage = pagedQuery(query, type, currentPage.getPageNum() + 1);
422             } catch (CatalogException e) {
423                 LOG.log(Level.WARNING, "Exception getting next page: Message: "
424                         + e.getMessage());
425             }
426 
427             return nextPage;
428         }

430         private List paginateQuery(Query query, ProductType type, int pageNum)
431                 throws CatalogException {
432             Connection conn = null;
433             Statement statement = null;
434             ResultSet rs = null;
435 
436             List productIds = null;
437             boolean doSkip = true;
438             int numResults = -1;
439 
440             if (pageNum == -1) {
441                 doSkip = false;
442             } else {
443                 numResults = getResultListSize(query, type);
444             }
445 
446             try {
447                 conn = dataSource.getConnection();
448                 statement = conn.createStatement(
449                         ResultSet.TYPE_SCROLL_INSENSITIVE,
450                         ResultSet.CONCUR_READ_ONLY);
451 
452                 String getProductSql;
453                 String tableName = type.getName() + "_metadata";
454                 String subSelectQueryBase = "SELECT product_id FROM "
455                         + tableName + " ";
456                 StringBuilder fromClause = new StringBuilder("FROM " + tableName
457                         + " p ");
458                 StringBuilder whereClause = new StringBuilder("WHERE ");
459 
460                 boolean gotFirstClause = false;
461                 int clauseNum = 0;
462 
463                 if (query.getCriteria() != null
464                         && query.getCriteria().size() > 0) {
465                     for (QueryCriteria criteria : query.getCriteria()) {
466                         clauseNum++;
467 
468                         String elementIdStr;
469 
470                         if (fieldIdStringFlag) {
471                             elementIdStr = "'" + this.getValidationLayer().getElementByName(criteria.getElementName())
472                                                      .getElementId() + "'";
473                         } else {
474                             elementIdStr =
475                                 this.getValidationLayer().getElementByName(criteria.getElementName()).getElementId();
476                         }
477 
478                         String clause;
479 
480                         if (!gotFirstClause) {
481                             clause = "(p.element_id = " + elementIdStr
482                                      + " AND ";
483                             if (criteria instanceof TermQueryCriteria) {
484                                 clause += " metadata_value LIKE '%"
485                                           + ((TermQueryCriteria) criteria)
486                                               .getValue() + "%') ";
487                             } else if (criteria instanceof RangeQueryCriteria) {
488                                 String startVal = ((RangeQueryCriteria) criteria)
489                                     .getStartValue();
490                                 String endVal = ((RangeQueryCriteria) criteria)
491                                     .getEndValue();
492                                 boolean inclusive = ((RangeQueryCriteria) criteria)
493                                     .getInclusive();
494 
495                                 if ((startVal != null && !startVal.equals(""))
496                                     || (endVal != null && !endVal
497                                     .equals(""))) {
498                                     clause += " metadata_value ";
499 
500                                     boolean gotStart = false;
501 
502                                     if (startVal != null
503                                         && !startVal.equals("")) {
504                                         if (inclusive) {
505                                             clause += ">= '" + startVal + "'";
506                                         } else {
507                                             clause += "> '" + startVal + "'";
508                                         }
509                                         gotStart = true;
510                                     }
511 
512                                     if (endVal != null && !endVal.equals("")) {
513                                         if (gotStart) {
514                                             if (inclusive) {
515                                                 clause += " AND metadata_value <= '"
516                                                           + endVal + "'";
517                                             } else {
518                                                 clause += " AND metadata_value < '"
519                                                           + endVal + "'";
520                                             }
521                                         } else if (inclusive) {
522                                             clause += "<= '" + endVal + "'";
523                                         } else {
524                                             clause += "< '" + endVal + "'";
525                                         }
526                                     }
527 
528                                     clause += ") ";
529                                 }
530                             }
531 
532                             whereClause.append(clause);
533                             gotFirstClause = true;
534                         } else {
535                             String subSelectTblName = "p" + clauseNum;
536                             String subSelectQuery = subSelectQueryBase
537                                                     + "WHERE (element_id = " + elementIdStr
538                                                     + " AND ";
539                             if (criteria instanceof TermQueryCriteria) {
540                                 subSelectQuery += " metadata_value LIKE '%"
541                                                   + ((TermQueryCriteria) criteria)
542                                                       .getValue() + "%')";
543                             } else if (criteria instanceof RangeQueryCriteria) {
544                                 String startVal = ((RangeQueryCriteria) criteria)
545                                     .getStartValue();
546                                 String endVal = ((RangeQueryCriteria) criteria)
547                                     .getEndValue();
548 
549                                 if (startVal != null || endVal != null) {
550                                     subSelectQuery += " metadata_value ";
551 
552                                     boolean gotStart = false;
553 
554                                     if (startVal != null
555                                         && !startVal.equals("")) {
556                                         subSelectQuery += ">= '" + startVal
557                                                           + "'";
558                                         gotStart = true;
559                                     }
560 
561                                     if (endVal != null && !endVal.equals("")) {
562                                         if (gotStart) {
563                                             subSelectQuery += " AND metadata_value <= '"
564                                                               + endVal + "'";
565                                         } else {
566                                             subSelectQuery += "<= '" + endVal
567                                                               + "'";
568                                         }
569                                     }
570 
571                                     subSelectQuery += ") ";
572 
573                                 }
574                             }
575                             fromClause.append("INNER JOIN (").append(subSelectQuery).append(") ")
576                                       .append(subSelectTblName).append(" ON ").append(subSelectTblName)
577                                       .append(".product_id = p.product_id ");
578 
579                         }
580                     }
581                 }
582                 getProductSql = "SELECT DISTINCT p.product_id " + fromClause.toString();
583                 if (gotFirstClause) {
584                     getProductSql += whereClause.toString();
585                 }
586 
587                 LOG.log(Level.FINE, "catalog query: executing: "
588                         + getProductSql);
589 
590                 rs = statement.executeQuery(getProductSql);
591                 productIds = new Vector();
592 
593                 if (doSkip) {
594                     int startNum = (pageNum - 1) * pageSize;
595 
596                     if (startNum > numResults) {
597                         startNum = 0;
598                     }
599 
600                     // must call next first, or else no relative cursor
601                     if (rs.next()) {
602                         // grab the first one
603                         int numGrabbed;
604 
605                         if (pageNum == 1) {
606                             numGrabbed = 1;
607                             productIds.add(rs.getString("product_id"));
608                         } else {
609                             numGrabbed = 0;
610                         }
611 
612                         // now move the cursor to the correct position
613                         if(pageNum != 1){
614                             rs.relative(startNum-1);
615                         }
616 
617                         // grab the rest
618                         while (rs.next() && numGrabbed < pageSize) {
619                             String productId = rs.getString("product_id");
620                             productIds.add(productId);
621                             numGrabbed++;
622                         }
623                     }
624 
625                 } else {
626                     while (rs.next()) {
627                         String productId = rs.getString("product_id");
628                         productIds.add(productId);
629                     }
630                 }
631 
632                 if (productIds.size() == 0) {
633                     productIds = null;
634                 }
635 
636             } catch (Exception e) {
637                 LOG.log(Level.SEVERE, e.getMessage());
638                 LOG.log(Level.WARNING, "Exception performing query. Message: "
639                         + e.getMessage());
640                 try {
641                     conn.rollback();
642                 } catch (SQLException e2) {
643                     LOG.log(Level.SEVERE,
644                             "Unable to rollback query transaction. Message: "
645                                     + e2.getMessage());
646                 }
647                 throw new CatalogException(e.getMessage());
648             } finally {
649 
650                 if (rs != null) {
651                     try {
652                         rs.close();
653                     } catch (SQLException ignore) {
654                     }
655 
656                 }
657 
658                 if (statement != null) {
659                     try {
660                         statement.close();
661                     } catch (SQLException ignore) {
662                     }
663 
664                 }
665 
666                 if (conn != null) {
667                     try {
668                         conn.close();
669 
670                     } catch (SQLException ignore) {
671                     }
672 
673                 }
674             }
675 
676             return productIds;
677         }

680                 throws CatalogException {
681             Connection conn = null;
682             Statement statement = null;
683             ResultSet rs = null;
684 
685             int resultCount = 0;
686 
687             try {
688                 conn = dataSource.getConnection();
689                 statement = conn.createStatement();
690 
691                 String getProductSql;
692                 String tableName = type.getName() + "_metadata";
693                 String subSelectQueryBase = "SELECT product_id FROM "
694                         + tableName + " ";
695                 StringBuilder fromClause = new StringBuilder("FROM " + tableName
696                         + " p ");
697                 StringBuilder whereClause = new StringBuilder("WHERE ");
698 
699                 boolean gotFirstClause = false;
700                 int clauseNum = 0;
701 
702                 if (query.getCriteria() != null
703                         && query.getCriteria().size() > 0) {
704                     for (QueryCriteria criteria : query.getCriteria()) {
705                         clauseNum++;
706 
707                         String elementIdStr;
708 
709                         if (fieldIdStringFlag) {
710                             elementIdStr = "'" + this.getValidationLayer().getElementByName(criteria.getElementName())
711                                                      .getElementId() + "'";
712                         } else {
713                             elementIdStr =
714                                 this.getValidationLayer().getElementByName(criteria.getElementName()).getElementId();
