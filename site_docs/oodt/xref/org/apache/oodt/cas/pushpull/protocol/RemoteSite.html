98          if (obj instanceof RemoteSite) {
99              RemoteSite rs = (RemoteSite) obj;
100             try {
101                 return (rs.alias.equals(this.alias) && rs.url.toURI().equals(this.url.toURI())

103                         .equals(this.password) && rs.maxConnections == this.maxConnections);
