16  package org.apache.oodt.commons.util;
17  
18  import java.util.*;
19  import java.beans.*;
20  
21  /** Property manager manages updates to properties.
22   *
23   * @author Kelly
24   */
25  public class PropertyMgr {
26  	/** Add a listener for property changes.
27  	 *
28  	 * @param listener Listener to add.
29  	 */
30  	public static void addPropertyChangeListener(PropertyChangeListener listener) {
31  		listeners.add(listener);
32  	}
33  
34  	/** Remove a listener's interest in property changes.
35  	 *
36  	 * @param listener Listener to remove.
37  	 */
38  	public static void removePropertyChangeListener(PropertyChangeListener listener) {
39  		listeners.remove(listener);
40  	}
41  
42  	/** Set a property and notify listeners.
43  	 *
44  	 * @param key Property's name.
45  	 * @param value New property's value.
46  	 */
47  	public static void setProperty(String key, String value) {
48  		String oldValue = System.getProperty(key);
49  		System.setProperty(key, value);
50  		if (!listeners.isEmpty()) {
51  			PropertyChangeEvent event = new PropertyChangeEvent(System.getProperties(), key, oldValue, value);
52  		  for (Object listener1 : listeners) {
53  			PropertyChangeListener listener = (PropertyChangeListener) listener1;
54  			listener.propertyChange(event);
55  		  }
56  		}
57  	}
58  
59  	/** Property change listeners. */
60  	private static List listeners = new ArrayList();
61  }
