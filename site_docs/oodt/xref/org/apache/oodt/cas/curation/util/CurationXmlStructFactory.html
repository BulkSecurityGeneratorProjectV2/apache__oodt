37  /**

39   * Until we solve the serialization problem in CAS 1.8.0 for adding metadata and

47  public class CurationXmlStructFactory {

49    public static void writeProductTypeXmlDocument(
50        List<ProductType> productTypes, String xmlFilePath) throws UnsupportedEncodingException, CurationException {
51      XMLUtils.writeXmlFile(getProductTypeXmlDocument(productTypes), xmlFilePath);
52    }
53  
54    public static Document getProductTypeXmlDocument(
55        List<ProductType> productTypes) throws UnsupportedEncodingException, CurationException {
56      Document doc = XmlStructFactory.getProductTypeXmlDocument(productTypes);
57  
58      // for every product type, i want to add in the versioner info and the
59      // met extractor info
60  
61      Element rootElem = doc.getDocumentElement();
62      NodeList typeNodeList = rootElem.getElementsByTagName("type");
63      if (typeNodeList != null && typeNodeList.getLength() > 0) {
64        for (int i = 0; i < typeNodeList.getLength(); i++) {
65          Element typeElem = (Element) typeNodeList.item(i);
66          augmentElement(productTypes, typeElem, doc);
67        }
68      }
69  
70      return doc;
71    }
72  
73    private static void augmentElement(List<ProductType> productTypes,
74        Element typeElem, Document doc) throws UnsupportedEncodingException, CurationException {
75      String productTypeName = typeElem.getAttribute("name");
76      ProductType type = getType(productTypes, productTypeName);
77  
78      Element metadataRootElem = doc.createElement("metadata");
79      
80      for (Object metKey : type.getTypeMetadata().getMap().keySet()) {
81        String key = (String) metKey;
82        List<String> vals = type.getTypeMetadata().getAllMetadata(key);
83  
84        Element metadataElem = doc.createElement("keyval");
85        Element keyElem = doc.createElement("key");
86        keyElem.appendChild(doc.createTextNode(URLEncoder.encode(key, "UTF-8")));
87  
88        metadataElem.appendChild(keyElem);
89        for (String val : vals) {
90          Element valElem = doc.createElement("val");
91          if (val == null) {
92            throw new CurationException("Attempt to write null value "
93                + "for property: [" + key + "]: val: [null]");
94          }
95  
96          valElem
97              .appendChild(doc.createTextNode(URLEncoder.encode(val, "UTF-8")));
98          metadataElem.appendChild(valElem);
99        }
100 
101       metadataRootElem.appendChild(metadataElem);
102       typeElem.appendChild(metadataRootElem);
103 
104     }
105 
106   }
107 
108   private static ProductType getType(List<ProductType> types, String name) {
109     for (ProductType type : types) {
110       if (type.getName().equals(name)) {
111         return type;
112       }
113     }
114 
115     return null;
116   }
