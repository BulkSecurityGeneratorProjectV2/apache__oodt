50  	/** Test the {@link XML#createDOMParser} and {@link XML#serialize} methods.
51  	 */
52  	public void testDOM() throws Exception {
53  		DOMParser p = XML.createDOMParser();
54  		p.parse(inputSource);
55  		Document doc = p.getDocument();
56  		doc.normalize();
57  		assertEquals("log", doc.getDocumentElement().getTagName());
58  		java.util.zip.CRC32 crc = new java.util.zip.CRC32();
59  		String result = XML.serialize(doc);
60  		crc.update(result.getBytes());
61  		long value = crc.getValue();
62  		assertTrue("Stringified DOM document CRC mismatch, got value = " + value, 3880488030L == value || 2435419114L == value || /* added by Chris Mattmann: pretty print fix */3688328384L == value || /* other newline treatment */ 750262163L == value || 3738296466L == value /* Apache incubator warmed up the file, so it suffered thermal expansion */ || 1102069581L == value /* lewismc and his ALv2 header. */);
63  	}

76  	/** Test the {@link XML#dump(PrintWriter,Node)} method. */
77  	public void testDump() throws Exception {
78  		DOMParser p = XML.createDOMParser();
79  		p.parse(inputSource);
80  		Document doc = p.getDocument();
81  		StringWriter stringWriter = new StringWriter();
82  		PrintWriter printWriter = new PrintWriter(stringWriter);
83  		XML.dump(printWriter, doc);
84  		printWriter.close();
85  		java.util.zip.CRC32 crc = new java.util.zip.CRC32();
86  		crc.update(stringWriter.getBuffer().toString().getBytes());
87  		long value = crc.getValue();
88  		assertTrue("Dumped DOM tree CRC mismatch; got " + value, value == 828793L || value == 2241317601L || value == 3208931170L /* lewismc and his ALv2 header */);
89  	}

106 	public void testAddString() throws Exception {
