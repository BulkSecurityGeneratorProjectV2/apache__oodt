314             .getProductReferences().get(0).getOrigReference().endsWith(ref));

347    public void testLuceneQuery() {
348       String query = "ProductId:TestProductId";
349       cmdLineUtility
350             .run(("--url http://localhost:9000 --operation --luceneQuery"
351                   + " --query " + query).split(" "));
352       MethodCallDetails methodCallDetails = client.getLastMethodCallDetails();
353       assertEquals("complexQuery", methodCallDetails.getMethodName());
354       assertEquals("ProductId", ((TermQueryCriteria) ((Query) methodCallDetails
355             .getArgs().get(0)).getCriteria().get(0)).getElementName());
356       assertEquals("TestProductId",
357             ((TermQueryCriteria) ((Query) methodCallDetails.getArgs().get(0))
358                   .getCriteria().get(0)).getValue());
359 
360       OptionPropertyRegister.clearRegister();
361 
362       String reducedMetadataKeys = "ProductId ProductType";
363       String outputFormat = "$ProductId";
364       String reducedProductTypes = "TestProductType";
365       String sortBy = "ProductId";
366       String delimiter = ",";
367       cmdLineUtility
368             .run(("--url http://localhost:9000 --operation --luceneQuery"
369                   + " --query " + query + " --reducedMetadataKeys " + reducedMetadataKeys
370                   + " --outputFormat " + outputFormat
371                   + " --reducedProductTypes " + reducedProductTypes
372                   + " --sortBy " + sortBy
373                   + " --delimiter " + delimiter).split(" "));
374       methodCallDetails = client.getLastMethodCallDetails();
375       assertEquals("complexQuery", methodCallDetails.getMethodName());
376       ComplexQuery complexQuery = (ComplexQuery) methodCallDetails.getArgs().get(0);
377       assertEquals("ProductId", ((TermQueryCriteria) complexQuery.getCriteria().get(0)).getElementName());
378       assertEquals("TestProductId",
379             ((TermQueryCriteria) complexQuery.getCriteria().get(0)).getValue());
380       assertEquals(Lists.newArrayList(reducedMetadataKeys.split(" ")), complexQuery.getReducedMetadata());
381       assertEquals(outputFormat, complexQuery.getToStringResultFormat());
382       assertEquals(Lists.newArrayList(reducedProductTypes.split(" ")), complexQuery.getReducedProductTypeNames());
383       assertEquals(sortBy, complexQuery.getSortByMetKey());
384    }
