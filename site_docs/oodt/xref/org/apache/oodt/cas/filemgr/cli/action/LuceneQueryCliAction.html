41  /**
42   * A {@link CmdLineAction} which converts a Lucene-like query into a File
43   * Manager {@link Query} and returns the results.
44   * 
45   * @author bfoster (Brian Foster)
46   */
47  public class LuceneQueryCliAction extends AbstractQueryCliAction {
48  
49     private static final String FREE_TEXT_BLOCK = "__FREE__";
50  
51     private String query;
52     private List<String> reducedProductTypes;
53     private List<String> reducedMetadataKeys;
54  
55     public LuceneQueryCliAction() {
56        super();
57     }
58  
59     @Override
60     public ComplexQuery getQuery() throws ParseException, CatalogException, QueryFormulationException {
61        Validate.notNull(query, "Must specify query");
62  
63        ComplexQuery complexQuery = new ComplexQuery();
64        complexQuery.setCriteria(Lists.newArrayList(generateCASQuery(parseQuery(query))));
65        complexQuery.setReducedProductTypeNames(reducedProductTypes);
66        complexQuery.setReducedMetadata(reducedMetadataKeys);
67        return complexQuery;
68     }
69  
70     public void setQuery(String query) {
71        this.query = query;
72     }
73  
74     public void setReducedProductTypes(List<String> reducedProductTypes) {
75        this.reducedProductTypes = reducedProductTypes;
76     }
77  
78     public void setReducedMetadataKeys(List<String> reducedMetadataKeys) {
79        this.reducedMetadataKeys = reducedMetadataKeys;
80     }
81  
82     private Query parseQuery(String query) throws ParseException {
83        // note that "__FREE__" is a control work for free text searching
84        return (Query) new QueryParser(FREE_TEXT_BLOCK, new WhitespaceAnalyzer()).parse(query);
85     }
86  
87     private QueryCriteria generateCASQuery(Query luceneQuery)
88         throws CatalogException, QueryFormulationException {
89        if (luceneQuery instanceof TermQuery) {
90           Term t = ((TermQuery) luceneQuery).getTerm();
91           if (t.field().equals(FREE_TEXT_BLOCK)) {
92              throw new CatalogException("Free text blocks not supported!");
93           } else {
94              return new TermQueryCriteria(t.field(), t.text());
95           }
96        } else if (luceneQuery instanceof PhraseQuery) {
97           Term[] t = ((PhraseQuery) luceneQuery).getTerms();
98           if (t[0].field().equals(FREE_TEXT_BLOCK)) {
99              throw new CatalogException("Free text blocks not supported!");
100          } else {
101             BooleanQueryCriteria bqc = new BooleanQueryCriteria();
102             bqc.setOperator(BooleanQueryCriteria.AND);
103             for (Term aT : t) {
104                bqc.addTerm(new TermQueryCriteria(aT.field(), aT
105                    .text()));
106             }
107             return bqc;
108          }
109       } else if (luceneQuery instanceof TermRangeQuery) {
110          BytesRef startT = ((TermRangeQuery) luceneQuery).getLowerTerm();
111          BytesRef endT = ((TermRangeQuery) luceneQuery).getUpperTerm();
112          //TODO CHECK Inclusive
113          boolean inc = false;
114          if(((TermRangeQuery) luceneQuery).includesLower() && ((TermRangeQuery) luceneQuery).includesUpper()){
115             inc = true;
116          }
117          return new RangeQueryCriteria(((TermRangeQuery) luceneQuery).getField(), startT
118                .utf8ToString(), endT.utf8ToString(), inc);
119       } else if (luceneQuery instanceof BooleanQuery) {
120          List<BooleanClause> clauses = ((BooleanQuery) luceneQuery).clauses();
121          BooleanQueryCriteria bqc = new BooleanQueryCriteria();
122          bqc.setOperator(BooleanQueryCriteria.AND);
123          for (BooleanClause clause : clauses) {
124             if (clause.getOccur().equals(BooleanClause.Occur.SHOULD)) {
125                bqc.setOperator(BooleanQueryCriteria.OR);
126             }
127             bqc.addTerm(generateCASQuery(clause.getQuery()));
128          }
129          return bqc;
130       } else {
131          throw new CatalogException(
132                "Error parsing query! Cannot determine clause type: ["
133                      + luceneQuery.getClass().getName() + "] !");
134       }
135    }
