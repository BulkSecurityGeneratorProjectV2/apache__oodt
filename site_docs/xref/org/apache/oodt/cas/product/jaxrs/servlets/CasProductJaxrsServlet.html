72    /**
73     * Initializes the file manager client and stores it as a "client" context
74     * attribute by retrieving a value for the file manager URL from the servlet
75     * context.
76     * @param context the servlet context
77     * @throws ServletException if the servlet context parameter for the file
78     * manager working directory path cannot be found
79     */
80    private void initializeClient(ServletContext context) throws ServletException
81    {
82      try
83      {
84        URL url;
85        String urlParameter = context.getInitParameter("filemgr.url");
86        if (urlParameter != null)
87        {
88          // Get the file manager URL from the context parameter.
89          url = new URL(PathUtils.replaceEnvVariables(urlParameter));
90        }
91        else
92        {
93          // Try the default URL for the file manager.
94          LOGGER.log(Level.WARNING, "Unable to find a servlet context parameter "
95            + "for the file manager URL.");
96          url = new URL("http://localhost:9000");
97        }
98  
99        // Attempt to connect the client to the file manager and if successful
100       // store the client as a context attribute for other objects to access.
101       XmlRpcFileManagerClient client = new XmlRpcFileManagerClient(url);
102       context.setAttribute("client", client);
103     }
104     catch (MalformedURLException e)
105     {
106       String message = "Encountered a malformed URL for the file manager.";
107       LOGGER.log(Level.SEVERE, message, e);
108       throw new ServletException(message);
109     }
110     catch (ConnectionException e)
111     {
112       String message =
113         "Client could not establish a connection to the file manager.";
114       LOGGER.log(Level.SEVERE, message, e);
115       throw new ServletException(message);
116     }
117   }

121   /**
122    * Initializes the file manager working directory path and stores it as a
123    * "workingDir" context attribute by retrieving a value from the servlet
124    * context.
125    * @param context the servlet context
126    * @throws ServletException if the servlet context parameter for the file
127    * manager working directory path cannot be found
128    */
129   private void initializeWorkingDir(ServletContext context)
130     throws ServletException
131   {
132     String workingDirPath = context.getInitParameter("filemgr.working.dir");
133     if (workingDirPath != null)
134     {
135       // Validate the path.
136       File workingDir = new File(PathUtils.replaceEnvVariables(workingDirPath));
137       if (workingDir.exists() && workingDir.isDirectory())
138       {
139         context.setAttribute("workingDir", workingDir);
140         LOGGER.log(Level.FINE, "The file manager's working directory has been "
141           + "set up as " + workingDir.getAbsolutePath());
142       }
143       else
144       {
145         LOGGER.log(Level.SEVERE, "Unable to locate the working directory for "
146           + "the file manager.");
147       }
148     }
149     else
150     {
151       String message = "Unable to find a servlet context parameter for the file"
152         + " manager working directory path.";
153       LOGGER.log(Level.SEVERE, message);
154       throw new ServletException(message);
155     }
156   }

160   /**
161    * Initializes the output configurations for various different formats and
162    * stores them in "configurations" attributes by retrieving values from the
163    * servlet context.
164    * @param context the servlet context
165    * @throws IOException if the specified file cannot be found or read
166    */
167   public void initializeConfigurations(ServletContext context)
168   {
169     Map<String, RdfConfiguration> rdfConfigurations =
170       new ConcurrentHashMap<String, RdfConfiguration>();
171     Map<String, RssConfiguration> rssConfigurations =
172       new ConcurrentHashMap<String, RssConfiguration>();
173 
174     Enumeration<String> enumeration = context.getInitParameterNames();
175     while (enumeration.hasMoreElements())
176     {
177       String parameterName = enumeration.nextElement();
178       if(parameterName.startsWith("configuration"))
179       {
180         String[] values = parameterName.split("\\.");
181         if (values.length == CONFIG_PARAM_LENGTH)
182         {
183           String format = values[CONFIG_PARAM_FORMAT];
184           String name = values[CONFIG_PARAM_NAME];
185           String value = PathUtils.replaceEnvVariables(
186             context.getInitParameter(parameterName));
187 
188           try
189           {
190             if ("rdf".equals(format))
191             {
192               RdfConfiguration configuration = new RdfConfiguration();
193               configuration.initialize(new File(value));
194               rdfConfigurations.put(name, configuration);
195             }
196             else if ("rss".equals(format))
197             {
198               RssConfiguration configuration = new RssConfiguration();
199               configuration.initialize(new File(value));
200               rssConfigurations.put(name, configuration);
201             }
202           }
203           catch (IOException e)
204           {
205             LOGGER.log(Level.WARNING, "The configuration '" + parameterName
206               + "'could not be initialized (value: " + value + ").", e);
207           }
208         }
209         else
210         {
211           LOGGER.log(Level.FINE,
212             "Configuration context parameter could not be parsed.");
213         }
214       }
215     }
216 
217     context.setAttribute("rdfConfigurations", rdfConfigurations);
218     context.setAttribute("rssConfigurations", rssConfigurations);
219   }
