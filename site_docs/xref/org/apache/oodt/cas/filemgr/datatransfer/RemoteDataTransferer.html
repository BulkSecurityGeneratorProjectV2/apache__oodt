39  /**
40   * @author mattmann
41   * @author bfoster
42   * @version $Revision$
43   * 
44   *          <p>
45   *          An implementation of the {@link DataTransfer} interface that
46   *          transfers files to a remote file manager over XML-RPC, using the
47   *          File Manager Client.
48   *          </p>
49   * 
50   */
51  public class RemoteDataTransferer implements DataTransfer {
52  
53    public static final int NUM_BYTES = 1024;
54    /*
55          * the url pointer to the file manager that we'll remotely transfer the file
56          * to
57          */
58     private URL fileManagerUrl = null;
59  
60     /*
61      * the size of the chunks that files should be transferred over XML-RPC using
62      */
63     private int chunkSize = 1024;
64  
65     /* our file manager client */
66     private XmlRpcFileManagerClient client = null;
67  
68     /* our log stream */
69     private static final Logger LOG = Logger
70           .getLogger(RemoteDataTransferer.class.getName());
71  
72     /**
73       * 
74       */
75     public RemoteDataTransferer(int chunkSz) {
76        this.chunkSize = chunkSz;
77     }
78  
79     /*
80      * (non-Javadoc)
81      * 
82      * @see
83      * org.apache.oodt.cas.filemgr.datatransfer.DataTransfer#setFileManagerUrl
84      * (java.net.URL)
85      */
86     public void setFileManagerUrl(URL url) {
87        try {
88           client = new XmlRpcFileManagerClient(url);
89           this.fileManagerUrl = url;
90           LOG.log(Level.INFO, "Remote Data Transfer to: ["
91                 + client.getFileManagerUrl().toString() + "] enabled");
92        } catch (ConnectionException e) {
93           LOG.log(Level.WARNING, "Connection exception for filemgr: [" + url
94                 + "]");
95        }
96     }
97  
98     /*
99      * (non-Javadoc)
100     * 
101     * @see
102     * org.apache.oodt.cas.filemgr.datatransfer.DataTransfer#transferProduct(
103     * org.apache.oodt.cas.filemgr.structs.Product)
104     */
105    public void transferProduct(Product product) throws DataTransferException,
106          IOException {
107 
108       if (fileManagerUrl == null) {
109          throw new DataTransferException(
110                "No file manager url specified for remote data transfer: cannot transfer product: ["
111                      + product.getProductName() + "]!");
112       }
113 
114       quietNotifyTransferProduct(product);
115 
116       // for each file reference, transfer the file to the remote file manager
117      for (Reference r : product.getProductReferences()) {
118        // test whether or not the reference is a directory or a file
119        File refFile;
120        try {
121          refFile = new File(new URI(r.getOrigReference()));
122        } catch (URISyntaxException e) {
123          LOG.log(Level.WARNING,
124              "Unable to test if reference: [" + r.getOrigReference()
125              + "] is a directory: skipping it");
126          continue;
127        }
128 
129        if (!refFile.isDirectory()) {
130          LOG.log(Level.FINE, "Reference: [" + r.getOrigReference()
131                              + "] is file: transferring it");
132 
133          try {
134            remoteTransfer(r, product);
135          } catch (URISyntaxException e) {
136            LOG.log(Level.WARNING,
137                "Error transferring file: [" + r.getOrigReference()
138                + "]: URISyntaxException: " + e.getMessage());
139          }
140        } else {
141          LOG.log(
142              Level.FINE,
143              "RemoteTransfer: skipping reference: ["
144              + refFile.getAbsolutePath() + "] of product: ["
145              + product.getProductName() + "]: ref is a directory");
146        }
147      }
148 
149       quietNotifyProductTransferComplete(product);
150 
151    }
152 
153    /*
154     * (non-Javadoc)
155     * 
156     * @see
157     * org.apache.oodt.cas.filemgr.datatransfer.DataTransfer#retrieveProduct(org.
158     * apache.oodt.cas.filemgr.structs.Product, java.io.File)
159     */
160    public void retrieveProduct(Product product, File directory)
161          throws DataTransferException, IOException {
162       for (Reference reference : product.getProductReferences()) {
163          FileOutputStream fOut = null;
164          try {
165             File dataStoreFile = new File(new URI(
166                   reference.getDataStoreReference()));
167             File dest = new File(directory, dataStoreFile.getName());
168             fOut = new FileOutputStream(dest, false);
169             LOG.log(
170                   Level.INFO,
171                   "RemoteDataTransfer: Copying File: " + "fmp:"
172                         + dataStoreFile.getAbsolutePath() + " to " + "file:"
173                         + dest.getAbsolutePath());
174             byte[] fileData;
175             int offset = 0;
176             while (true) {
177                fileData = (byte[]) client.retrieveFile(
178                      dataStoreFile.getAbsolutePath(), offset, NUM_BYTES);
179                if (fileData.length <= 0) {
180                  break;
181                }
182                fOut.write(fileData);
183                if (fileData.length < NUM_BYTES) {
184                  break;
185                }
186                offset += NUM_BYTES;
187             }
188          } catch (Exception e) {
189             throw new DataTransferException("", e);
190          } finally {
191             try {
192                fOut.close();
193             } catch (Exception ignored) {
194             }
195          }
196       }
197    }

199    @Override
200    public void deleteProduct(Product product) throws DataTransferException, IOException {
201      for (Reference ref : product.getProductReferences()) {
202        File dataFile = new File(URI.create(ref.getDataStoreReference()).toURL().getPath());
203        if (!dataFile.delete()) {
204         throw new IOException(String.format("Failed to delete file %s - delete returned false",
205             dataFile));
206        }
207      }
208    }

210    private void remoteTransfer(Reference reference, Product product)
211          throws URISyntaxException {
212       // get the file path
213       File origFile = new File(new URI(reference.getOrigReference()));
214       File destFile = new File(new URI(reference.getDataStoreReference()));
215       String origFilePath = origFile.getAbsolutePath();
216       String destFilePath = destFile.getAbsolutePath();
217 
218       // read the file in chunk by chunk
219 
220       byte[] buf = new byte[chunkSize];
221 
222       FileInputStream is = null;
223 
224       try {
225          is = new FileInputStream(origFile);
226          int offset = 0;
227          int numBytes;
228 
229          // remove the file if it already exists: this operation
230          // is an overwrite
231          if (!client.removeFile(destFilePath)) {
232             LOG.log(Level.WARNING,
233                   "RemoteDataTransfer: attempt to perform overwrite of dest file: ["
234                         + destFilePath + "] failed");
235          }
236 
237          while ((numBytes = is.read(buf, offset, chunkSize)) != -1) {
238             client.transferFile(destFilePath, buf, offset, numBytes);
239          }
240       } catch (IOException e) {
241          LOG.log(Level.WARNING,
242                "Error opening input stream to read file to transfer: Message: "
243                      + e.getMessage());
244       } catch (DataTransferException e) {
245          LOG.log(
246                Level.WARNING,
247                "DataTransferException when transfering file: [" + origFilePath
248                      + "] to [" + destFilePath + "]: Message: "
249                      + e.getMessage());
250       } finally {
251          if (is != null) {
252             try {
253                is.close();
254             } catch (Exception ignore) {
255             }
256 
257          }
258       }
259    }

261    private void quietNotifyTransferProduct(Product p) {
262       try {
263          client.transferringProduct(p);
264       } catch (DataTransferException e) {
265          LOG.log(Level.SEVERE, e.getMessage());
266          LOG.log(Level.WARNING,
267                "Error notifying file manager of product transfer initiation for product: ["
268                      + p.getProductId() + "]: Message: " + e.getMessage());
269       }
270    }
271 
272    private void quietNotifyProductTransferComplete(Product p) {
273       try {
274          client.removeProductTransferStatus(p);
275       } catch (DataTransferException e) {
276          LOG.log(Level.SEVERE, e.getMessage());
277          LOG.log(Level.WARNING,
278                "Error notifying file manager of product transfer completion for product: ["
279                      + p.getProductId() + "]: Message: " + e.getMessage());
280       }
281    }
