227   public synchronized List<TaskProcessor> getRunnableWorkflowProcessors() {
228     Vector<TaskProcessor> runnableTasks = new Vector<TaskProcessor>();
229 
230     // evaluate pre-conditions
231     if (!this.passedPreConditions()) {
232       for (WorkflowProcessor subProcessor : this.getPreConditions()
233           .getRunnableSubProcessors()) {
234         for (TaskProcessor tp : subProcessor.getRunnableWorkflowProcessors()) {
235           runnableTasks.add(tp);
236         }
237       }
238 
239     } else if (this.isDone().getName().equals("ResultsBail")) {
240       for (WorkflowProcessor subProcessor : this.getRunnableSubProcessors()) {
241         runnableTasks.addAll(subProcessor.getRunnableWorkflowProcessors());
242       }
243     } else if (!this.passedPostConditions()) {
244       for (WorkflowProcessor subProcessor : this.getPostConditions()
245           .getRunnableSubProcessors()) {
246         for (TaskProcessor tp : subProcessor.getRunnableWorkflowProcessors()) {
247           runnableTasks.add(tp);
248         }
249       }
250 
251     }
252 
253     return runnableTasks;
254   }

256   /**
257    * Advances this WorkflowProcessor to its next {@link WorkflowState}.
258    */
259   public void nextState() {
260     if (this.workflowInstance != null
261         && this.workflowInstance.getState() != null) {
262       WorkflowState currState = this.workflowInstance.getState();
263       WorkflowState nextState = null;
264       if (currState.getName().equals("Null")) {
265         nextState = this.helper.getLifecycleForProcessor(this).createState(
266             "Loaded",
267             "initial",
268             "Workflow Processor: nextState: " + "loading workflow instance: ["
269                 + this.workflowInstance.getId() + "]");
270       } else if (currState.getName().equals("Loaded")) {
271         nextState = this.helper.getLifecycleForProcessor(this).createState(
272             "Queued",
273             "initial",
274             "Workflow Processor: nextState: " + "queueing instance: ["
275                 + this.workflowInstance.getId() + "]");
276       } else if (currState.getName().equals("Queued")) {
277         if (!this.passedPreConditions()) {
278           nextState = this.helper.getLifecycleForProcessor(this).createState(
279               "PreConditionEval",
280               "running",
281               "Workflow Processor: nextState: "
282                   + "running preconditiosn for workflow instance: ["
283                   + this.workflowInstance.getId() + "]");
284         } else {
285           if (this.isDone().getName().equals("ResultsSuccess")) {
286             nextState = this.helper.getLifecycleForProcessor(this).createState(
287                 "Success",
288                 "done",
289                 "Workflow Processor: nextState: " + "workflow instance: ["
290                     + this.workflowInstance.getId()
291                     + "] completed successfully");
292           }
293         }
294       } else if (currState.getName().equals("Executing")) {
295         if(this.isDone().getName().equals("ResultsSuccess")){
296         nextState = this.helper.getLifecycleForProcessor(this).createState(
297             "Success",
298             "done",
299             "Workflow Processor: nextState: " + "workflow instance: ["
300                 + this.workflowInstance.getId() + "] completed successfully");
301         }
302       }
303       else if(currState.getName().equals("ExecutionComplete")){
304         nextState = this.helper.getLifecycleForProcessor(this).createState(
305             "Success",
306             "done",
307             "Workflow Processor: nextState: " + "workflow instance: ["
308                 + this.workflowInstance.getId() + "] completed successfully");        
309       }
310 
311       if (nextState != null) {
312         this.workflowInstance.setState(nextState);
313       }
314 
315     } else {
316       this.workflowInstance.setState(helper.getLifecycleForProcessor(this)
317           .createState(
318               "Unknown",
319               "holding",
320               "The Workflow Processor for instance : ["
321                   + this.getWorkflowInstance().getId() + "] "
322                   + "had a null state"));
323     }
324   }

385   /**
386    * First checks to see if any of this Processor's {@link #subProcessors} have
387    * arrived in a state within the done category. If so the method determines if
388    * any of the done {@link #subProcessors} are in Failure state. If so, the
389    * method compares the number of Failed sub-processors against
390    * {@link #minReqSuccessfulSubProcessors}, and if it is greater than it,
391    * returns a ResultsFailure {@link WorkflowState}. Otherwise, the method scans
392    * the failed sub-processors, and checks to see if all of them have been
393    * excused. If they haven't, then a ResultFailure state is returned. Finally,
394    * the method checks to ensure that all sub processors are in the done
395    * category. If they are, a ResultsSuccess {@link WorkflowState} is returned,
396    * otherwise, a ResultsBail state is returned.
397    * 
398    * @return A {@link WorkflowState}, according to the method description.
399    */
400   protected WorkflowState isDone() {
401     if (this.helper.containsCategory(this.getSubProcessors(), "done")) {
402       List<WorkflowProcessor> failedSubProcessors = this.helper
403           .getWorkflowProcessorsByState(this.getSubProcessors(), "Failure");
404       if (this.minReqSuccessfulSubProcessors != -1
405           && failedSubProcessors.size() > (this.getSubProcessors().size() - this.minReqSuccessfulSubProcessors)) {
406         return lifecycleManager.getDefaultLifecycle().createState(
407             "ResultsFailure", "results",
408             "More than the allowed number of sub-processors failed");
409       }
410       for (WorkflowProcessor subProcessor : failedSubProcessors) {
411         if (!this.getExcusedSubProcessorIds().contains(
412             subProcessor.getWorkflowInstance().getId())) {
413           return lifecycleManager.getDefaultLifecycle().createState(
414               "ResultsFailure",
415               "results",
416               "Sub processor: [" + subProcessor.getWorkflowInstance().getId()
417                   + "] failed.");
418         }
419       }
420       if (this.helper
421           .allProcessorsSameCategory(this.getSubProcessors(), "done")) {
422         return lifecycleManager.getDefaultLifecycle().createState(
423             "ResultsSuccess",
424             "results",
425             "Workflow Processor: processing instance id: ["
426             + workflowInstance.getId() + "] is Done.");
427       }
428     }
429     return lifecycleManager.getDefaultLifecycle().createState(
430         "ResultsBail",
431         "results",
432         "All sub-processors for Workflow Processor handling workflow id: ["
433             + workflowInstance.getId() + "] are " + "not complete");
434   }
