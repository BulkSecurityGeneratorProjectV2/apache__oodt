35  	/** Test the fixed buffer output stream's methods with various illegal arguments.
36  	 */
37  	public void testIllegalArgs() throws IOException {
38  		try {
39  			new FixedBufferOutputStream(-1);
40  			fail("Illegal size failed to throw exception");
41  		} catch (IllegalArgumentException ignored) {}
42  		FixedBufferOutputStream out = new FixedBufferOutputStream(100);
43  		try {
44  			out.write(null);
45  			fail("Writing a null byte array failed to throw exception");
46  		} catch (NullPointerException ignored) {}
47  		try {
48  			out.write(null, 0, 10);
49  			fail("Writing a null byte array failed to throw exception");
50  		} catch (NullPointerException ignored) {}
51  		try {
52  			out.write(TEST_DATA, -1, 10);
53  			fail("Writing with negative offset failed to throw exception");
54  		} catch (IndexOutOfBoundsException ignored) {}
55  		try {
56  			out.write(TEST_DATA, TEST_DATA.length + 1, 10);
57  			fail("Writing with offset past end of array failed to throw exception");
58  		} catch (IndexOutOfBoundsException ignored) {}
59  		try {
60  			out.write(TEST_DATA, 0, -1);
61  			fail("Writing with negative length array failed to throw exception");
62  		} catch (IndexOutOfBoundsException ignored) {}
63  		try {
64  			out.write(TEST_DATA, 2, 5);
65  			fail("Writing with offset and length exceeding end of array failed to throw exception");
66  		} catch (IndexOutOfBoundsException ignored) {}
67  	}
