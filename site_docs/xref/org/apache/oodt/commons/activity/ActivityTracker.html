18  package org.apache.oodt.commons.activity;
19  
20  import java.util.ArrayList;
21  import java.util.Collection;
22  import java.util.List;
23  import java.util.StringTokenizer;
24  
25  /**
26   * Tracker of activities.  This is the main class applications use to keep track
27   * activities.  To track an activity, call the {@link #createActivity} method to generate
28   * a new {@link Activity}, then log {@link Incident}s against it with the {@link
29   * Activity#log} method.  Finally, top the activity with {@link Activity#stop}.
30   *
31   * <p>The <code>ActivityTracker</code> generates activities using an {@link
32   * ActivityFactory}.  It configures the factory by examining the system properties.  The
33   * property <code>org.apache.oodt.commons.activity.factories</code> (or the
34   * <code>activity.factories</code> property for those that prefer an abbreviated name and
35   * the prior one isn't defined) is a comma- (or vertical bar-) seperated list of class
36   * names.  Each class is expected to implement the {@link ActivityFactory} interface.  If
37   * there are none (or neither property is defined), then a special null factory is used
38   * that throws away all incidents.  If more than one's defined, then a special factory is
39   * used that multiplexes all incidents to each one.
40   *
41   * @author Kelly
42   * @version $Revision: 1.1 $
43   */
44  public class ActivityTracker {
45  	/**
46  	 * Generate an activity.
47  	 *
48  	 * @return an new {@link Activity} instance.
49  	 */
50  	public static Activity createActivity() {
51  		return factory.createActivity();
52  	}
53  
54  	/** What actually makes Activities. */
55  	static ActivityFactory factory;
56  
57  	/**
58  	 * Examine the prope
59  	 *
60  	 * @throws Exception if an error occurs.
61  	 */
62  	static void initializeFactories() throws IllegalAccessException, InstantiationException, ClassNotFoundException {
63  		String facNames = System.getProperty("org.apache.oodt.commons.activity.factories", System.getProperty("activity.factories", ""));
64  		List factories = new ArrayList();
65  		for (StringTokenizer tokens = new StringTokenizer(facNames, ",|"); tokens.hasMoreTokens();) {
66  			String factoryName = tokens.nextToken();
67  			Class factoryClass = Class.forName(factoryName);
68  			factories.add(factoryClass.newInstance());
69  		}
70  		if (factories.isEmpty()) {
71  		  factory = new NullActivityFactory();
72  		} else factory =
73  			factories.size() == 1 ? (ActivityFactory) factories.get(0) : new CompositeActivityFactory(factories);
74  	}
75  
76  	/**
77  	 * Initialize the <code>factory</code>.
78  	 */
79  	static {
80  		try {
81  			initializeFactories();
82  		} catch (RuntimeException ex) {
83  			throw ex;
84  		} catch (Exception ex) {
85  			throw new IllegalStateException("Cannot initialize activity factories: " + ex.getMessage());
86  		}
87  	}
88  
89  	/**
90  	 * Special factory that creates only null activities.  Null activities toss out every incident.
91  	 */
92  	private static class NullActivityFactory implements ActivityFactory {
93  		/**
94  		 * Create null activities.
95  		 *
96  		 * @return Null activity.
97  		 */
98  		public Activity createActivity() {
99  			return new NullActivity();
100 		}
101 	}
102 
103 	/**
104 	 * Special factory that multiplexes activites.  This factory takes a collection of
105 	 * other factories.  It generates activities that then fantail incidents to other
106 	 * factories' activities.
107 	 */
108 	private static class CompositeActivityFactory implements ActivityFactory {
109 		/**
110 		 * Creates a new {@link CompositeActivityFactory} instance.
111 		 *
112 		 * @param factories a {@link Collection} of ActivityFactories.
113 		 */
114 		private CompositeActivityFactory(Collection factories) {
115 			this.factories = factories;
116 		}
117 
118 		/**
119 		 * Create an activity that fantails to other factories' activities.
120 		 *
121 		 * @return an {@link Activity} value.
122 		 */
123 		public Activity createActivity() {
124 			List activities = new ArrayList();
125 		  for (Object factory1 : factories) {
126 			ActivityFactory factory = (ActivityFactory) factory1;
127 			activities.add(factory.createActivity());
128 		  }
129 			return new CompositeActivity(activities);
130 		}
131 
132 		/** Collection of {@link ActivityFactory} instances. */
133 		private Collection factories;
134 	}
135 }
