82       * (non-Javadoc)
83       * 
84       * @see org.apache.oodt.cas.filemgr.catalog.Catalog#addMetadata(org.apache.oodt.cas.metadata.Metadata,
85       *      org.apache.oodt.cas.filemgr.structs.Product)
86       */
87      public synchronized void addMetadata(Metadata m, Product product)
88              throws CatalogException {
89      	
90      		// replace "CAS.ProductId"
91      		m.removeMetadata("CAS.ProductId");
92      		m.addMetadata("CAS.ProductId", product.getProductId());
93      	
94      		// map containing metadata type (id, name) pairs
95          Map<String, String> metadataTypes = getMetadataTypes(m, product);
96  
97          // loop over metadata types
98          for (Map.Entry<String, String> metadataId : metadataTypes.entrySet()) {
99          	String metadataName = metadataId.getValue();
100         	
101             List<String> values = m.getAllMetadata(metadataName);
102 
103             if (values == null) {
104                 LOG.log(Level.WARNING, "No Metadata specified for product ["
105                         + product.getProductName() + "] for required field ["
106                         + metadataName
107                         + "]: Attempting to continue processing metadata");
108                 continue;
109             }
110 
111           for (String value : values) {
112             try {
113               addMetadataValue(metadataId, product, value);
114             } catch (Exception e) {
115               LOG.log(Level.SEVERE, e.getMessage());
116               LOG
117                   .log(
118                       Level.WARNING,
119                       "Exception ingesting metadata. Error inserting field: ["
120                       + metadataId
121                       + "=>"
122                       + value
123                       + "]: for product: ["
124                       + product.getProductName()
125                       + "]: Message: "
126                       + e.getMessage()
127                       + ": Attempting to continue processing metadata");
128             }
129           }
130         }
131 
132     }

134     // Utility method to return a map of metadata (field id, field name)
135     private Map<String, String> getMetadataTypes(Metadata m, Product product) throws CatalogException {
136     	
137   		// map containing metadata type (id, name) pairs
138       Map<String, String> metadataTypes = new LinkedHashMap<String,String>();
139 
140       if (getValidationLayer()!=null) {
141       	// validation layer: add valid metadata elements 
142         try {
143             for (Element element : getValidationLayer().getElements(product.getProductType())) {
144             	metadataTypes.put(element.getElementId(), element.getElementName());
145             }
146             
147         } catch (ValidationLayerException e) {
148             LOG.log(Level.SEVERE, e.getMessage());
149             throw new CatalogException(
150                     "ValidationLayerException when trying to obtain element list for product type: "
151                             + product.getProductType().getName()
152                             + ": Message: " + e.getMessage(), e);
153         }
154         
155       } else {
156       	
157       	// no validation layer: add ALL metadata elements
158       	// use (key, key) pairs (i.e. metadata id == metadata name)
159       	for (String key : m.getAllKeys()) {
160       		metadataTypes.put(key, key);
161       	}
162       	
163       }
164       
165       return metadataTypes;
166       
167     }

175     public synchronized void removeMetadata(Metadata m, Product product)
176             throws CatalogException {
177     	
178   			// map containing metadata type (id, name) pairs
179       	Map<String, String> metadataTypes = getMetadataTypes(m, product);
180             
181         // loop over metadata types
182         for (Map.Entry<String, String> metadataId : metadataTypes.entrySet()) {
183            	String metadataName = metadataId.getValue();
184             	
185             List<String> values = m.getAllMetadata(metadataName);
186 
187             if (values != null) {
188               for (String value : values) {
189                 try {
190                   removeMetadataValue(metadataId.getKey(), product, value);
191                 } catch (Exception e) {
192                   LOG.log(Level.SEVERE, e.getMessage());
193                   LOG
194                       .log(
195                           Level.WARNING,
196                           "Exception removing metadata. Error deleting field: ["
197                           + metadataId
198                           + "=>"
199                           + value
200                           + "]: for product: ["
201                           + product.getProductName()
202                           + "]: Message: "
203                           + e.getMessage()
204                           + ": Attempting to continue processing metadata");
205                 }
206               }
207             }
208         }

211     public Metadata getMetadata(Product product) throws CatalogException {
212         Connection conn = null;
213         Statement statement = null;
214         ResultSet rs = null;
215         Metadata m = null;
216 
217         try {
218             conn = dataSource.getConnection();
219             statement = conn.createStatement();
220 
221             String metadataSql = "SELECT * FROM "
222                     + product.getProductType().getName() + "_metadata "
223 		+ "WHERE product_id = '" + product.getProductId()+"'";
224             if(this.orderedValues) {
225               metadataSql += " ORDER BY pkey";
226             }
227 
228             LOG.log(Level.FINE, "getMetadata: Executing: " + metadataSql);
229             rs = statement.executeQuery(metadataSql);
230             
231             // parse SQL results
232             m = populateProductMetadata(rs, product);
233 
234         } catch (Exception e) {
235             LOG.log(Level.SEVERE, e.getMessage());
236             LOG.log(Level.WARNING, "Exception getting metadata. Message: "
237                     + e.getMessage());
238             throw new CatalogException(e.getMessage(), e);
239         } finally {
240 
241             if (rs != null) {
242                 try {
243                     rs.close();
244                 } catch (SQLException ignore) {
245                 }
246 
247             }
248 
249             if (statement != null) {
250                 try {
251                     statement.close();
252                 } catch (SQLException ignore) {
253                 }
254 
255             }
256 
257             if (conn != null) {
258                 try {
259                     conn.close();
260 
261                 } catch (SQLException ignore) {
262                 }
263 
264             }
265         }
266 
267         return m;
268     }

270     /** Method to populate the product metadata from a SQL ResultSet. **/
271     private Metadata populateProductMetadata(ResultSet rs, Product product) throws CatalogException, SQLException {
272      
273     	Metadata m = new Metadata();
274 
275       if (getValidationLayer()!=null) {
276         
277       	// validation layer: retrieve valid metadata elements
278         List<Element> elements;
279 
280       	try {
281             elements = getValidationLayer().getElements(product.getProductType());
282         } catch (ValidationLayerException e) {
283             LOG.log(Level.SEVERE, e.getMessage());
284             throw new CatalogException(
285                     "ValidationLayerException when trying to obtain element list for product type: "
286                             + product.getProductType().getName()
287                             + ": Message: " + e.getMessage(), e);
288         }
289 
290         while (rs.next()) {
291           for (Element e : elements) {
292             // right now, we just support STRING
293             String elemValue = rs.getString("metadata_value");
294             String elemId = rs.getString("element_id");
295 
296             if (elemId.equals(e.getElementId())) {
297               elemValue = (elemValue != null ? elemValue : "");
298               m.addMetadata(e.getElementName(), elemValue);
299             }
300           }
301         }
302       
303       } else {
304       	
305       	// no validation layer - add all (name, value) pairs for this product_id query
306         while (rs.next()) {
307             // right now, we just support STRING
308             String elemValue = rs.getString("metadata_value");
309             String elemId = rs.getString("element_id");
310 
311             elemValue = (elemValue != null ? elemValue : "");
312             m.addMetadata(elemId, elemValue);    
313         }
314         
315       }
316       
317       return m;
318       
319     }

321     public Metadata getReducedMetadata(Product product, List<String> elems) throws CatalogException {
322         Connection conn = null;
323         Statement statement = null;
324         ResultSet rs = null;
325         Metadata m = null;
326 
327         try {
328             conn = dataSource.getConnection();
329             statement = conn.createStatement();
330 
331             StringBuilder elementIds = new StringBuilder("");
332             if (elems.size() > 0) {
333             	
334             	  if (getValidationLayer()!=null) {
335             	  	// validation layer: column "element_id" contains the element identifier (e.g. "urn:oodt:ProductReceivedTime")
336 	                elementIds.append(" AND (element_id = '" + this.getValidationLayer().getElementByName(elems.get(0))
337                                                             .getElementId() + "'");
338 	                for (int i = 1; i < elems.size(); i++) {
339                       elementIds.append(
340                           " OR element_id = '" + this.getValidationLayer().getElementByName(elems.get(i)).getElementId()
341                           + "'");
342                     }
343 	                elementIds.append(")");
344             	 
345             	  } else {
346             	  	// no validation layer: column "element_id" contains the element name (e.g. "CAS.ProductReceivedTime")
347 	                elementIds.append(" AND (element_id = '" + elems.get(0) + "'");
348 	                for (int i = 1; i < elems.size(); i++) {
349                       elementIds.append(" OR element_id = '" + elems.get(i) + "'");
350                     }
351 	                elementIds.append(")");
352 	                
353             	  }
354             	  
355             }
356             String metadataSql = "SELECT element_id,metadata_value FROM "
357                     + product.getProductType().getName() + "_metadata"
358 		+ " WHERE product_id = " + quoteIt(product.getProductId()) + elementIds.toString();
359             if(this.orderedValues) {
360               metadataSql += " ORDER BY pkey";
361             }
362 
363             LOG.log(Level.FINE, "getMetadata: Executing: " + metadataSql);
364             rs = statement.executeQuery(metadataSql);
365 
366             // parse SQL results
367             m = populateProductMetadata(rs, product);
368 
369 
370         } catch (Exception e) {
371             LOG.log(Level.SEVERE, e.getMessage());
372             LOG.log(Level.WARNING, "Exception getting metadata. Message: "
373                     + e.getMessage());
374             throw new CatalogException(e.getMessage(), e);
375         } finally {
376 
377             if (rs != null) {
378                 try {
379                     rs.close();
380                 } catch (SQLException ignore) {
381                 }
382 
383             }
384 
385             if (statement != null) {
386                 try {
387                     statement.close();
388                 } catch (SQLException ignore) {
389                 }
390 
391             }
392 
393             if (conn != null) {
394                 try {
395                     conn.close();
396 
397                 } catch (SQLException ignore) {
398                 }
399 
400             }
401         }
402 
403         return m;
404     }

406     private synchronized void addMetadataValue(Map.Entry<String, String> key,
407             Product product, String value) throws CatalogException {
408 
409         Connection conn = null;
410         Statement statement = null;
411 
412         String metadataTable = product.getProductType().getName() + "_metadata";
413                 
414 
415         try {
416             conn = dataSource.getConnection();
417             conn.setAutoCommit(false);
418             statement = conn.createStatement();
419 
420             // build up the sql statement
421           StringBuilder valueClauseSql = new StringBuilder();
422 
423           valueClauseSql.append("VALUES ");
424 
425             // now do the value clause
426             if (fieldIdStringFlag) {
427                 valueClauseSql.append("(").append(quoteIt(product.getProductId())).append(", '").append(key)
428                               .append("', '").append(value).append("')");
429             } else {
430                 valueClauseSql.append("(").append(product.getProductId()).append(", ").append(key).append(", '")
431                               .append(value).append("')");
432             }
433 
434             String metaIngestSql = ("INSERT INTO " + metadataTable
435                                     + " (product_id, element_id, metadata_value) ")
436                     + valueClauseSql.toString();
437             LOG
438                     .log(Level.FINE, "addMetadataValue: Executing: "
439                             + metaIngestSql);
440             statement.execute(metaIngestSql);
441             
442             // synchronize CAS.ProductReceivedTime with products.product_datetime
443             if (key.equals("CAS.ProductReceivedTime") && this.productIdString) {
444             		// convert from "2012-12-18T09:00:14.068-08:00" --> "2012-12-18T09:00:14.068-0800" --> "2012-12-18T10:00:14"
445             		String datetime = dbFormat.format( isoFormat.parse( value.replaceAll(":00$", "00") ));
446             		String updateDateTimeSql = "UPDATE products SET product_datetime='"+datetime+"' WHERE product_id="+quoteIt(product.getProductId());
447             		LOG.log(Level.FINE, "addMetadataValue: Executing: "+updateDateTimeSql);
448             		statement.execute(updateDateTimeSql);
449             }
450             
451             conn.commit();
452         } catch (Exception e) {
453             LOG.log(Level.SEVERE, e.getMessage());
454             LOG.log(Level.WARNING, "Exception adding metadata value. Message: "
455                     + e.getMessage());
456             try {
457               if (conn != null) {
458                 conn.rollback();
459               }
460             } catch (SQLException e2) {
461                 LOG.log(Level.SEVERE,
462                         "Unable to rollback add metadata value. Message: "
463                                 + e2.getMessage());
464             }
465             throw new CatalogException(e.getMessage(), e);
466         } finally {
467             if (statement != null) {
468                 try {
469                     statement.close();
470                 } catch (SQLException ignore) {
471                 }
472 
473             }
474 
475             if (conn != null) {
476                 try {
477                     conn.close();
478 
479                 } catch (SQLException ignore) {
480                 }
481 
482             }
483         }
484     }

554     /**
555      * Overridden method from superclass to allow for null validation layer.
556      */
557     protected int getResultListSize(Query query, ProductType type)
558             throws CatalogException {
559         Connection conn = null;
560         Statement statement = null;
561         ResultSet rs = null;
562 
563         int resultCount = 0;
564 
565         try {
566             conn = dataSource.getConnection();
567             statement = conn.createStatement();
568 
569             String getProductSql;
570             String tableName = type.getName() + "_metadata";
571             String subSelectQueryBase = "SELECT product_id FROM " + tableName
572                     + " ";
573           StringBuilder fromClause = new StringBuilder("FROM " + tableName
574                     + " p ");
575             StringBuilder whereClause = new StringBuilder("WHERE ");
576 
577             boolean gotFirstClause = false;
578             int clauseNum = 0;
579 
580             if (query.getCriteria() != null && query.getCriteria().size() > 0) {
581               for (QueryCriteria criteria : query.getCriteria()) {
582                 clauseNum++;
583 
584                 String elementIdStr;
585 
586                 if (fieldIdStringFlag) {
587                   if (getValidationLayer() != null) {
588                     elementIdStr =
589                         "'" + this.getValidationLayer().getElementByName(criteria.getElementName()).getElementId()
590                         + "'";
591                   } else {
592                     elementIdStr = "'" + criteria.getElementName() + "'";
593                   }
594                 } else {
595                   if (getValidationLayer() != null) {
596                     elementIdStr = this.getValidationLayer().getElementByName(criteria.getElementName()).getElementId();
597                   } else {
598                     elementIdStr = criteria.getElementName();
599                   }
600                 }
601 
602                 String clause;
603 
604                 if (!gotFirstClause) {
605                   clause = "(p.element_id = " + elementIdStr + " AND ";
606                   if (criteria instanceof TermQueryCriteria) {
607                     clause += " metadata_value LIKE '%"
608                               + ((TermQueryCriteria) criteria).getValue()
609                               + "%') ";
610                   } else if (criteria instanceof RangeQueryCriteria) {
611                     String startVal = ((RangeQueryCriteria) criteria)
612                         .getStartValue();
613                     String endVal = ((RangeQueryCriteria) criteria)
614                         .getEndValue();
615                     boolean inclusive = ((RangeQueryCriteria) criteria)
616                         .getInclusive();
617 
618                     if ((startVal != null && !startVal.equals(""))
619                         || (endVal != null && !endVal.equals(""))) {
620                       clause += " metadata_value ";
621 
622                       boolean gotStart = false;
623 
624                       if (startVal != null && !startVal.equals("")) {
625                         if (inclusive) {
626                           clause += ">= '" + startVal + "'";
627                         } else {
628                           clause += "> '" + startVal + "'";
629                         }
630                         gotStart = true;
631                       }
632 
633                       if (endVal != null && !endVal.equals("")) {
634                         if (gotStart) {
635                           if (inclusive) {
636                             clause += " AND metadata_value <= '"
637                                       + endVal + "'";
638                           } else {
639                             clause += " AND metadata_value < '"
640                                       + endVal + "'";
641                           }
642                         } else if (inclusive) {
643                           clause += "<= '" + endVal + "'";
644                         } else {
645                           clause += "< '" + endVal + "'";
646                         }
647                       }
648 
649                       clause += ") ";
650                     }
651                   }
652                   whereClause.append(clause);
653                   gotFirstClause = true;
654                 } else {
655                   String subSelectTblName = "p" + clauseNum;
656                   String subSelectQuery = subSelectQueryBase
657                                           + "WHERE (element_id = " + elementIdStr
658                                           + " AND ";
659                   if (criteria instanceof TermQueryCriteria) {
660                     subSelectQuery += " metadata_value LIKE '%"
661                                       + ((TermQueryCriteria) criteria).getValue()
662                                       + "%')";
663                   } else if (criteria instanceof RangeQueryCriteria) {
664                     String startVal = ((RangeQueryCriteria) criteria)
665                         .getStartValue();
666                     String endVal = ((RangeQueryCriteria) criteria)
667                         .getEndValue();
668 
669                     if (startVal != null || endVal != null) {
670                       subSelectQuery += " metadata_value ";
671 
672                       boolean gotStart = false;
673 
674                       if (startVal != null && !startVal.equals("")) {
675                         subSelectQuery += ">= '" + startVal + "'";
676                         gotStart = true;
677                       }
678 
679                       if (endVal != null && !endVal.equals("")) {
680                         if (gotStart) {
681                           subSelectQuery += " AND metadata_value <= '"
682                                             + endVal + "'";
683                         } else {
684                           subSelectQuery += "<= '" + endVal + "'";
685                         }
686                       }
687 
688                       subSelectQuery += ") ";
689 
690                     }
691                   }
692 
693                   fromClause.append("INNER JOIN (").append(subSelectQuery).append(") ").append(subSelectTblName)
694                             .append(" ON ").append(subSelectTblName).append(".product_id = p.product_id ");
695 
696                 }
697               }
698             }
699 
700             getProductSql = "SELECT COUNT(DISTINCT p.product_id) AS numResults " + fromClause.toString();
701             if (gotFirstClause) {
702                 getProductSql += whereClause.toString();
703             }
704 
705 
706             LOG.log(Level.FINE, "catalog get num results: executing: "
707                     + getProductSql);
708 
709             rs = statement.executeQuery(getProductSql);
710 
711             while (rs.next()) {
712                 resultCount = rs.getInt("numResults");
713             }
714 
715         } catch (Exception e) {
716             LOG.log(Level.SEVERE, e.getMessage());
717             LOG.log(Level.WARNING,
718                     "Exception performing get num results. Message: "
719                             + e.getMessage());
720             try {
721               if (conn != null) {
722                 conn.rollback();
723               }
724             } catch (SQLException e2) {
725                 LOG.log(Level.SEVERE,
726                         "Unable to rollback get num results transaction. Message: "
727                                 + e2.getMessage());
728             }
729             throw new CatalogException(e.getMessage(), e);
730         } finally {
731 
732             if (rs != null) {
733                 try {
734                     rs.close();
735                 } catch (SQLException ignore) {
736                 }
737 
738             }
739 
740             if (statement != null) {
741                 try {
742                     statement.close();
743                 } catch (SQLException ignore) {
744                 }
745 
746             }
747 
748             if (conn != null) {
749                 try {
750                     conn.close();
751 
752                 } catch (SQLException ignore) {
753                 }
