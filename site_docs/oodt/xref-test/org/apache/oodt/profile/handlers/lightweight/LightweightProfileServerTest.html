54  	/**
55  	 * Test the searching methods.
56  	 */
57  	public void testSearching() {
58  		try {
59  			List profiles;
60  
61  			// Try a positive search.
62  			profiles = doSearch("TEST > 2 AND TEST < 30");
63  			assertEquals(1, profiles.size());
64  			SearchableProfile profile = (SearchableProfile) profiles.get(0);
65  			assertEquals("PROFILE1", profile.getProfileAttributes().getID());
66  			Map elements = profile.getProfileElements();
67  			assertEquals(3, elements.size());
68  			ProfileElement element = (ProfileElement) elements.get("TEST");
69  			assertNotNull(element);
70  			assertEquals("3.14159", element.getID());
71  
72  			// Now a negative one.
73  			profiles = doSearch("NONEXISTENT = 452712917824812123125100884");
74  			assertEquals(0, profiles.size());
75  
76  			// Now one that has multiple elements from one profile
77  			profiles = doSearch("TEST <= 14 AND TEST2 >= 10");
78  			assertEquals(1, profiles.size());
79  			profile = (SearchableProfile) profiles.get(0);
80  			assertEquals("PROFILE1", profile.getProfileAttributes().getID());
81  			elements = profile.getProfileElements();
82  			assertEquals(3, elements.size());
83  			assertTrue(elements.containsKey("TEST"));
84  			assertTrue(elements.containsKey("TEST2"));
85  
86  			// And again, but with OR instead of AND
87  			profiles = doSearch("TEST <= 14 OR TEST2 >= 10");
88  			assertEquals(1, profiles.size());
89  			profile = (SearchableProfile) profiles.get(0);
90  			assertEquals("PROFILE1", profile.getProfileAttributes().getID());
91  			elements = profile.getProfileElements();
92  			assertEquals(3, elements.size());
93  			assertTrue(elements.containsKey("TEST"));
94  			assertTrue(elements.containsKey("TEST2"));
95  
96  			// And again, but with one of the elements being not found
97  			profiles = doSearch("NONEXISTENT = 123456789 OR TEST2 <= 1000000");
98  			assertEquals(1, profiles.size());
99  			profile = (SearchableProfile) profiles.get(0);
100 			assertEquals("PROFILE1", profile.getProfileAttributes().getID());
101 			elements = profile.getProfileElements();
102 			assertEquals(3, elements.size());
103 			assertTrue(elements.containsKey("TEST2"));
104 
105 			// And again, but spanning profiles
106 			profiles = doSearch("TEST2 = 48 OR TEST4 = 192");
107 			assertEquals(2, profiles.size());
108 		  for (Object profile1 : profiles) {
109 			profile = (SearchableProfile) profile1;
110 			elements = profile.getProfileElements();
111 			assertEquals(3, elements.size());
112 			if (profile.getProfileAttributes().getID().equals("PROFILE1")) {
113 			  assertTrue(elements.containsKey("TEST2"));
114 			} else if (profile.getProfileAttributes().getID().equals("PROFILE2")) {
115 			  assertTrue(elements.containsKey("TEST4"));
116 			} else {
117 			  fail("Profile \"" + profile.getProfileAttributes().getID() + "\" matched, but shouldn't");
118 			}
119 		  }
120 
121 			// And again, but with a query on the "from" part.
122 			profiles = doSearch("( TEST2 = 48 OR TEST4 = 192 ) AND Creator = Alice");
123 			assertEquals(1, profiles.size());
124 			profile = (SearchableProfile) profiles.get(0);
125 			assertEquals("PROFILE1", profile.getProfileAttributes().getID());
126 
127 			// And again, but on a nonenumerated element with no min/max values.
128 			profiles = doSearch("TEST5 = GEEBA");
129 			assertEquals(1, profiles.size());
130 			profile = (SearchableProfile) profiles.get(0);
131 			assertEquals("PROFILE2", profile.getProfileAttributes().getID());
132 
133 			// And again, but with "RETURN =" parts.
134 			//fail("Not yet implemented");
135 
136 		} catch (ProfileException ex) {
137 			fail("Profile server failed with excepton: " + ex.getMessage());
138 		}
139 	}

141 	/**
142 	 * Execute the given search.
143 	 *
144 	 * @param expr The search experssion.
145 	 * @return List of matching profiles.
146 	 * @throws ProfileException If the profile server fails.
147 	 */
148 	private List doSearch(String expr) throws ProfileException {
149 		XMLQuery query = new XMLQuery(expr, "test1", "LightweightProfileServerTest",
150 			"This query is to test the LightweightProfileServer", /*ddId*/null, /*resultModeId*/null,
151 			/*propType*/null, /*propLevels*/null, XMLQuery.DEFAULT_MAX_RESULTS);
152 		return server.findProfiles(query);
153 	}
