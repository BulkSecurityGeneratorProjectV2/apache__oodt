36  /**
37   * {@link ProtocolAction} for transferring a file from one host to another
38   * 
39   * @author bfoster (Brian Foster)
40   */
41  public class CrossProtocolTransferCliAction extends ProtocolCliAction {
42  
43     private URI fromUri;
44     private URI toUri;
45     private ProtocolVerifierFactory verifierFactory;
46  
47     @Override
48     public void execute(ActionMessagePrinter printer)
49           throws CmdLineActionException {
50        try {
51           ProtocolVerifier verifier = null;
52           if (verifierFactory != null) {
53              verifier = verifierFactory.newInstance();
54           }
55  
56           File localFile = createTempDownloadFile();
57           if (localFile == null) {
58              throw new Exception("Failed to create tempory local file");
59           }
60  
61           ProtocolManager protocolManager = getProtocolManager();
62           Protocol fromProtocol = protocolManager.getProtocolBySite(fromUri,
63                 getAuthentication(fromUri), verifier);
64           if (fromProtocol == null) {
65              throw new Exception("Failed to get protocol for 'from' URI '"
66                    + fromUri + "'");
67           }
68  
69           Protocol toProtocol = protocolManager.getProtocolBySite(toUri,
70                 getAuthentication(toUri), verifier);
71           if (toProtocol == null) {
72              throw new Exception("Failed to get protocol for 'to' URI '" + toUri
73                    + "'");
74           }
75  
76           fromProtocol
77                 .get(new ProtocolFile(fromUri.getPath(), false), localFile);
78           toProtocol.put(localFile, new ProtocolFile(toUri.getPath(), false));
79        } catch (Exception e) {
80           throw new CmdLineActionException(
81                 "Failed to transfer between 2 protocols : " + e.getMessage(), e);
82        }
83     }
84  
85     public void setFromUri(String fromUri) throws URISyntaxException {
86        this.fromUri = new URI(fromUri);
87     }
88  
89     public void setToUri(String toUri) throws URISyntaxException {
90        this.toUri = new URI(toUri);
91     }
92  
93     public void setVerifierFactory(ProtocolVerifierFactory verifierFactory) {
94        this.verifierFactory = verifierFactory;
95     }
96  
97     private File createTempDownloadFile() {
98        File bogusFile = null;
99        try {
100          bogusFile = File.createTempFile("bogus", "bogus");
101          File tmpDir = new File(bogusFile.getParentFile(), "ProtocolTransfer/"
102                + UUID.randomUUID().toString());
103          tmpDir.mkdirs();
104          return new File(tmpDir, "temp_file");
105       } catch (Exception e) {
106          return null;
107       } finally {
108          try {
109             bogusFile.delete();
110          } catch (Exception ignored) {
111          }
112       }
113    }
114 
115    private Authentication getAuthentication(URI uri) {
116       if (uri.getUserInfo() != null) {
117          String[] userInfo = uri.getUserInfo().split("\\:");
118          return new BasicAuthentication(userInfo[0], userInfo[1]);
119       } else {
120          return new NoAuthentication();
121       }
122    }
