17  package org.apache.oodt.cas.filemgr.structs.query;
18  
19  import org.apache.oodt.cas.filemgr.metadata.CoreMetKeys;
20  import org.apache.oodt.cas.filemgr.structs.Product;
21  import org.apache.oodt.cas.metadata.Metadata;
22  
23  import java.util.Arrays;
24  import java.util.Collections;
25  import java.util.List;
26  
27  import junit.framework.TestCase;
28  
29  public class TestQueryResultComparator extends TestCase {
30  
31    private QueryResult qr1;
32    private QueryResult qr2;
33    private QueryResult qrfilename;
34    private QueryResult qrowners;
35    private QueryResult qrnull;
36    private QueryResult qrnull2;
37  
38    @Override
39    protected void setUp() throws Exception {
40      Product p = new Product();
41      p.setProductId("TestProductID");
42      Metadata m = new Metadata();
43      m.addMetadata(CoreMetKeys.FILENAME, "datafile.dat");
44      m.addMetadata("Owners", Arrays.asList("Chad", "Cam"));
45      qr1 = new QueryResult(p, m);
46  
47      Metadata m2 = new Metadata();
48      m2.addMetadata(CoreMetKeys.FILENAME, "textfile.dat");
49      m2.addMetadata("Owners", Arrays.asList("Bob", "Billy"));
50      qr2 = new QueryResult(p, m2);
51  
52      Metadata filename = new Metadata();
53      filename.addMetadata(CoreMetKeys.FILENAME, "foo.txt");
54      qrfilename = new QueryResult(p, filename);
55  
56      Metadata owners = new Metadata();
57      owners.addMetadata("Owners", Arrays.asList("Dave", "Dan"));
58      qrowners = new QueryResult(p, owners);
59  
60      qrnull = new QueryResult(p, new Metadata());
61      qrnull2 = new QueryResult(p, new Metadata());
62    }
63  
64    public void testCompare() {
65      List<QueryResult> list = Arrays.asList(qr1, qrowners, qrnull2, qr2, qrnull, qrfilename);
66      QueryResultComparator c = new QueryResultComparator();
67      c.setSortByMetKey(CoreMetKeys.FILENAME);
68  
69      Collections.sort(list, c);
70      assertEquals(qr1, list.get(0));
71      assertEquals(qrfilename, list.get(1));
72      assertEquals(qr2, list.get(2));
73      List<QueryResult> sub = list.subList(3, list.size());
74      assertEquals(3, sub.size());
75      assertTrue(sub.contains(qrnull));
76      assertTrue(sub.contains(qrnull2));
77      assertTrue(sub.contains(qrowners));
78  
79      c.setSortByMetKey("Owners");
80      Collections.sort(list, c);
81      assertEquals(qr2, list.get(0));
82      assertEquals(qr1, list.get(1));
83      assertEquals(qrowners, list.get(2));
84      sub = list.subList(3, list.size());
85      assertEquals(3, sub.size());
86      assertTrue(sub.contains(qrfilename));
87      assertTrue(sub.contains(qrnull));
88      assertTrue(sub.contains(qrnull2));
89    }
