18  package org.apache.oodt.commons.activity;
19  
20  import java.util.ArrayList;
21  import java.util.Collections;
22  import java.util.List;
23  import java.util.Timer;
24  import java.util.TimerTask;
25  
26  /**
27   * A history of incidents to be stored in long-term storage.
28   *
29   * The <code>History</code> tracks Incidents that occurred during the course of an
30   * activity.  It's a temporary container for incidents before the activity is deemed
31   * complete and stored into whatever the long-term {@link Storage} is.  A
32   * <code>History</code> starts out with at least one {@link Incident} received from
33   * (usually remote) processing.  Other incidents arrive through {@link #addIncident} and
34   * must have the same ID.  Eventually, one of those incidents will be an
35   * <code>ActivityStopped</code> incident, which then schedules the History to commit
36   * itself.
37   *
38   * <p>It doesn't commit itself right away though; other incidents may arrive that are part
39   * of the activity out of order, so there's a delay of {@link #closeTime} milliseconds
40   * before it's sent to {@link Storage}.  Other incidents will be recorded during this delay.
41   *
42   * <p>It's also possible that the <code>ActivityStopped</code> incident never arrives from
43   * the process that deems the activity stopped.  In that case, the idle timer (which is
44   * reset whenever an incident arrives) determines when to put the <code>History</code>
45   * into its commital state.  The stop can arrive after this point (and will be recorded),
46   * but it otherwise doesn't affect the commit-to-close time.
47   *
48   * @author Kelly
49   * @version $Revision: 1.1 $
50   */
51  final class History {
52  	/**
53  	 * Creates a new {@link History} instance.
54  	 *
55  	 * @param initialIncident Incident which gives the history its ID.
56  	 * @param storage Where to store the incidents when the history's done.
57  	 */
58  	History(Incident initialIncident, Storage storage) {
59  		id = initialIncident.getActivityID();
60  		this.storage = storage;
61  		incidents.add(initialIncident);
62  		scheduleExpiration();
63  	}
64  
65  	/**
66  	 * Add an incident to this history.
67  	 *
68  	 * @param incident an {@link Incident} value.
69  	 */
70  	synchronized void addIncident(Incident incident) {
71  		if (!incident.getActivityID().equals(id)) {
72  		  throw new IllegalArgumentException("Incident's activity ID " + incident.getActivityID()
73  											 + " doesn't match History's ID " + id);
74  		}
75  		if (incidents == null) {
76  		  return;
77  		}
78  		incidents.add(incident);
79  		if (expireHistoryTask != null) {
80  		  expireHistoryTask.cancel();
81  		}
82  		if (incident instanceof ActivityStopped) {
83  		  commit();
84  		} else if (closeHistoryTask == null) {
85  		  scheduleExpiration();
86  		}
87  	}
88  
89  	/**
90  	 * Commit this history by starting the commit-to-close timer.
91  	 */
92  	private void commit() {
93  		if (closeHistoryTask != null) {
94  		  return;
95  		}
96  		TIMER.schedule(closeHistoryTask = new CloseHistoryTask(), closeTime);
97  	}
98  
99  	/**
100 	 * Schedule expiration of this history due to idleness.
101 	 */
102 	private void scheduleExpiration() {
103 		TIMER.schedule(expireHistoryTask = new ExpireHistoryTask(), idleTime);
104 	}
105 
106 	/**
107 	 * Close this history out by storing the incidents to long-term storage.
108 	 */
109 	private synchronized void close() {
110  		Collections.sort(incidents);
111 		storage.store(id, incidents);
112 		incidents = null;
113 	}
114 
115 	/** History's ID. */
116 	private String id;
117 
118 	/** In what long-term storage to store incidents. */
119 	private Storage storage;
120 
121 	/** Timer task that expires an idle history.  If null, this history is in commit-to-close state. */
122 	private ExpireHistoryTask expireHistoryTask;
123 
124 	/** Timer task to close a history.  If nonnull, this history is in commit-to-close state. */ 
125 	private CloseHistoryTask closeHistoryTask;
126 
127 	/** Incidents to store. */
128 	private List incidents = new ArrayList();
129 
130 	/**
131 	 * Timer task that expires (commits and closes) an idle history.
132 	 */
133 	private class ExpireHistoryTask extends TimerTask {
134 		public void run() {
135 			synchronized (History.this) {
136 				// Check to see if a new expireHistoryTask got generated.
137 				// If it did, it means a new incident got logged while we
138 				// were about to expire the history.  But if the history's
139 				// current expiration task is this task, then we're clear
140 				// to commit the history, and any incidents that arrive
141 				// won't reset it.
142 				if (expireHistoryTask == this) {
143 				  commit();
144 				}
145 			}
146 		}
147 	}
148 
149 	/**
150 	 * Timer task that gives time for more incidents to arrive before closing forever a history.
151 	 */
152 	private class CloseHistoryTask extends TimerTask {
153 		public void run() {
154 			close();
155 		}
156 	}
157 
158 	/** Only timer we'll ever need. */
159  	private static final Timer TIMER = new Timer(/*isDaemon*/true);
160 
161 	/** How many milliseconds to wait before giving up on an idle history. */
162 	static long idleTime = Long.getLong("org.apache.oodt.commons.activity.History.idle", 5 * 60 * 1000);
163 
164 	/** How many milliseconds to wait to give a history extra time to receive incidents before saving it to storage. */
165 	static long closeTime = Long.getLong("org.apache.oodt.commons.activity.History.close", 5 * 60 * 1000);
166 }
