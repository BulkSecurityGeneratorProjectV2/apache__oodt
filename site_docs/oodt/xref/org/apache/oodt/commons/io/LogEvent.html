16  package org.apache.oodt.commons.io;
17  
18  import java.util.Date;
19  import java.util.EventObject;
20  
21  /** Logging event.
22   *
23   * A logging event is generated and delivered to all registered log listeners when a
24   * message was logged, a logging stream was started, or when a logging stream was stopped.
25   *
26   * <p>Use the various query methods to determine the details of the event.  The event
27   * source (from {@link EventObject#getSource}) is always a {@link String}.
28   *
29   * @see Log
30   * @see LogListener
31   * @author Kelly
32   */
33  public class LogEvent extends EventObject {
34  	/** Create a "message logged" event.
35  	 *
36  	 * @param timestamp The message's timestamp.
37  	 * @param source The source label of the message.
38  	 * @param category The message's category.
39  	 * @param message The message.
40  	 */
41  	public LogEvent(Date timestamp, String source, Object category, String message) {
42  		super(source);
43  		type = MSG_LOGGED;
44  		this.timestamp = timestamp;
45  		this.category = category;
46  		this.message = message;
47  	}
48  
49  	/** Create a "stream started" event.
50  	 *
51  	 * @param stream The name of the stream.
52  	 * @param timestamp The time the stream started.
53  	 * @param source A string identifying who or what started the stream.
54  	 */
55  	public LogEvent(String stream, Date timestamp, String source) {
56  		super(source);
57  		type = STREAM_STARTED;
58  		this.stream = stream;
59  		this.timestamp = timestamp;
60  	}
61  
62  	/** Create a "stream stopped" event.
63  	 *
64  	 * @param stream The name of the stream.
65  	 */
66  	public LogEvent(String stream) {
67  		super(stream);
68  		type = STREAM_STOPPED;
69  		this.stream = stream;
70  	}
71  
72  	/** Get the category.
73  	 *
74  	 * For message logged events, this is the category for the message. For all other
75  	 * events, this returns null.
76  	 *
77  	 * @return The category, or null.
78  	 */
79  	public Object getCategory() {
80  		return category;
81  	}
82  
83  	/** Get the message.
84  	 *  
85  	 * For message logged events, this is the actual message text. For all other
86  	 * events, this returns null.
87  	 *
88  	 *@return The message, or null.
89  	 */
90  	public String getMessage() {
91  		return message;
92  	}
93  
94  	/** Get the stream.
95  	 *
96  	 * For stream started and stopped events, this is the name of the stream that was
97  	 * started or stopped.  For message logged events, this is null.
98  	 *
99  	 * @return The stream, or null.
100 	 */
101 	public String getStream() {
102 		return stream;
103 	}
104 
105 	/** Get the timestamp.
106 	 *
107 	 * For message logged and stream started events, this is the timestamp of the
108 	 * event.  For stream stopped events, this is null.
109 	 *
110 	 * @return The timestamp, or null.
111 	 */
112 	public Date getTimestamp() {
113 		return timestamp;
114 	}
115 
116 	/** Return a string representation of this event.
117 	 *
118 	 * @return A string identifying the type of the event.
119 	 */
120 	public String toString() {
121 		switch (type) {
122 			case MSG_LOGGED:
123 				return "Log Event (message logged)";
124 			case STREAM_STARTED:
125 				return "Log Event (stream started)";
126 			case STREAM_STOPPED:
127 				return "Log Event (stream stopped)";
128 			default:
129 				return "Unknown Log Event Type";
130 		}
131 	}
132 
133 	/** The timestamp: null if this is a "stream stopped" event.
134 	 */
135 	private Date timestamp;
136 
137 	/** The stream: null for logging.
138 	 */
139 	private String stream;
140 
141 	/** The category: null for stream started/stopped events.
142 	 */
143 	private Object category;
144 
145 	/** The message: nonnull only for "message logged" events.
146 	 */
147 	private String message;
148 
149 	/** The type of event this is.
150 	 *
151 	 * This gets one of the values <code>MSG_LOGGED</code>, <code>STREAM_STARTED</code>, or <code>STREAM_STOPPED</code>.
152 	 */
153 	private int type;
154 
155 	/** The "message logged" event type.
156 	 *
157 	 * @see #type
158 	 */
159 	private static final int MSG_LOGGED = 1;
160 
161 	/** The "stream started" event type.
162 	 *
163 	 * @see #type
164 	 */
165 	private static final int STREAM_STARTED = 2;
166 
167 	/** The "stream stopped" event type.
168 	 *
169 	 * @see #type
170 	 */
171 	private static final int STREAM_STOPPED = 3;
172 }
