19  package org.apache.oodt.profile.handlers.lightweight;

21  import java.util.List;
22  import org.apache.oodt.profile.EnumeratedProfileElement;
23  import org.apache.oodt.profile.ObjectFactory;
24  import org.apache.oodt.profile.Profile;
25  import org.apache.oodt.profile.ProfileAttributes;
26  import org.apache.oodt.profile.RangedProfileElement;
27  import org.apache.oodt.profile.ResourceAttributes;
28  import org.apache.oodt.profile.UnspecifiedProfileElement;
29  import org.w3c.dom.Element;

31  /**
32   * Factory that yields searchable profile objects.
33   *
34   * @author Kelly
35   * @version $Revision: 1.1.1.1 $
36   */
37  class SearchableObjectFactory implements ObjectFactory {
38  	public RangedProfileElement createRangedProfileElement(Profile profile, String name, String id, String desc, String type,
39  		String unit, List synonyms, boolean obligation, int maxOccurrence, String comments, String min, String max) {
40  		return new SearchableRangedProfileElement((SearchableProfile) profile, name, id, desc, type, unit, synonyms,
41  			obligation, maxOccurrence, comments, min, max);
42  	}
43  	public UnspecifiedProfileElement createUnspecifiedProfileElement(Profile profile, String name, String id, String desc,
44  		String type, String unit, List synonyms, boolean obligation, int maxOccurrence, String comments) {
45  		return new SearchableUnspecifiedProfileElement((SearchableProfile) profile, name, id, desc, type, unit, synonyms,
46  			obligation, maxOccurrence, comments);
47  	}
48  	public EnumeratedProfileElement createEnumeratedProfileElement(Profile profile, String name, String id, String desc,
49  		String type, String unit, List synonyms, boolean obligation, int maxOccurrence, String comments, List values) {
50  		return new SearchableEnumeratedProfileElement((SearchableProfile) profile, name, id, desc, type, unit, synonyms,
51  			obligation, maxOccurrence, comments, values);
52  	}
53  	public Profile createProfile(Element node) {
54  		return new SearchableProfile(node, this);
55  	}
56  	public ProfileAttributes createProfileAttributes(Element node) {
57  		return new ProfileAttributes(node);
58  	}
59  	public ResourceAttributes createResourceAttributes(Profile profile, Element node) {
60  		return new SearchableResourceAttributes((SearchableProfile) profile, node);
61  	}
62  }
