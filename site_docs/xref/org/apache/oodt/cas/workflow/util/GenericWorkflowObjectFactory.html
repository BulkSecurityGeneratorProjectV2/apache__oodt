380 		  for (Object workflow : workflows) {

393 	/**
394 	 * <p>Creates an exact copy of the specified {@link Workflow} <code>w</code>,
395 	 * allocating new memory for the new object, and then returning it. The Workflow's
396 	 * {@link WorkflowTask}s and {@link WorkflowCondition}s on those tasks are also constructed
397 	 * anew, and copied from their original instances.</p>
398 	 *
399 	 * @param w The Workflow object to create a copy of.
400 	 * @return A copy of the specified Workflow.
401 	 */
402 	public static Workflow copyWorkflow(Workflow w){
403 		Workflow newWorkflow;
404 
405 
406 		newWorkflow = getWorkflowObjectFromClassName(w.getClass().getName());
407 
408 
409 		//copy through
410 		newWorkflow.setName(w.getName());
411 		newWorkflow.setId(w.getId());
412 		newWorkflow.setTasks(copyTasks(w.getTasks()));
413 
414 		return newWorkflow;
415 	}
