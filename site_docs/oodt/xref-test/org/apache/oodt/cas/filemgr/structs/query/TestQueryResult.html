17  package org.apache.oodt.cas.filemgr.structs.query;
18  
19  import org.apache.oodt.cas.filemgr.metadata.CoreMetKeys;
20  import org.apache.oodt.cas.filemgr.structs.Product;
21  import org.apache.oodt.cas.metadata.Metadata;
22  
23  import java.util.Arrays;
24  import java.util.List;
25  
26  import junit.framework.TestCase;
27  
28  public class TestQueryResult extends TestCase {
29  
30    private QueryResult qr;
31  
32    @Override
33    protected void setUp() throws Exception {
34      Product p = new Product();
35      p.setProductId("TestProductId");
36      Metadata m = new Metadata();
37      m.addMetadata(CoreMetKeys.FILENAME, "datafile.dat");
38      m.addMetadata("Owners", Arrays.asList("Bob", "Billy"));
39      qr = new QueryResult(p, m);
40    }
41  
42    public void testToString() {
43      String str = qr.toString();
44      assertTrue(str.contains("datafile.dat"));
45      assertTrue(str.contains("Bob,Billy"));
46      assertEquals(",", str.replace("datafile.dat", "").replace("Bob,Billy",""));
47    }
48  
49    public void testToStringFormat() {
50      qr.setToStringFormat("$Owners,$Filename");
51      assertEquals("Bob,Billy,datafile.dat", qr.toString());
52  
53      qr.setToStringFormat("$Filename\t$Owners");
54      List<String> list = Arrays.asList(qr.toString().split("\t"));
55      assertEquals(2, list.size());
56      assertEquals("datafile.dat", list.get(0));
57      assertEquals("Bob,Billy", list.get(1));
58    }
59  
60    public void testToStringEmpty() {
61      qr.setMetadata(new Metadata());
62      assertEquals("", qr.toString());
63    }
64  
65    public void testEquals() {
66      assertEquals(qr, qr);
67      assertFalse(qr.equals(null));
68      assertFalse(qr.equals(0));
69  
70      QueryResult metnull = new QueryResult(qr.getProduct(), null);
71      assertFalse(metnull.equals(qr));
72      assertFalse(qr.equals(metnull));
73      assertEquals(metnull, new QueryResult(qr.getProduct(), null));
74  
75      QueryResult prodnull = new QueryResult(null, new Metadata(qr.getMetadata()));
76      assertFalse(prodnull.equals(qr));
77      assertFalse(qr.equals(prodnull));
78      assertEquals(prodnull, new QueryResult(null, new Metadata(qr.getMetadata())));
79  
80      QueryResult equal = new QueryResult(qr.getProduct(), new Metadata(qr.getMetadata()));
81      assertEquals(qr, equal);
82      assertEquals(equal, qr);
83  
84      QueryResult fmt = new QueryResult(qr.getProduct(), new Metadata(qr.getMetadata()));
85      fmt.setToStringFormat("blah");
86      assertFalse(qr.equals(fmt));
87      assertFalse(fmt.equals(qr));
88  
89      QueryResult fmt2 = new QueryResult(qr.getProduct(), new Metadata(qr.getMetadata()));
90      fmt2.setToStringFormat("blah");
91      assertEquals(fmt, fmt2);
92      assertEquals(fmt2, fmt);
93    }
