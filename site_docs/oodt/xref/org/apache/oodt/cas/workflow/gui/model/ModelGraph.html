39  public class ModelGraph {

41    private ModelGraph parent;
42    private ModelNode model;
43    private ModelGraph preConditions;
44    private Vector<ModelGraph> children;
45    private ModelGraph postConditions;
46    private boolean isPreCondition;
47    private boolean isPostCondition;

49    public ModelGraph(ModelNode model) {
50      this.isPreCondition = isPostCondition = false;
51      this.model = model;
52      this.children = new Vector<ModelGraph>();
53      if (this.model.isParentType()) {
54        this.addChild(new ModelGraph(GuiUtils.createDummyNode()));
55      }
56    }
57  
58    public void setIsRef(boolean isRef) {
59      Stack<ModelGraph> stack = new Stack<ModelGraph>();
60      stack.add(this);
61      while (!stack.empty()) {
62        ModelGraph curGraph = stack.pop();
63        curGraph.getModel().setIsRef(isRef);
64        stack.addAll(curGraph.getChildren());
65        if (curGraph.getPreConditions() != null) {
66          stack.add(curGraph.getPreConditions());
67        }
68        if (curGraph.getPostConditions() != null) {
69          stack.add(curGraph.getPostConditions());
70        }
71      }
72    }
73  
74    public boolean isCondition() {
75      return this.isPreCondition || this.isPostCondition;
76    }
77  
78    public boolean isExcused() {
79      return this.parent != null
80          && this.parent.getModel().getExcusedSubProcessorIds()
81              .contains(this.getModel().getModelId());
82    }

84    public String getId() {
85      return model.getId();
86    }

88    public void setParent(ModelGraph parent) {
89      if (this.parent != null) {
90        if (this.isCondition() && !this.parent.isCondition()) {
91          if (this.isPreCondition) {
92            this.parent.preConditions = null;
93          } else {
94            this.parent.postConditions = null;
95          }
96        } else {
97          this.parent.removeChild(this);
98        }
99      }
100     this.parent = parent;
101     if (!this.getModel().isRef() && parent != null && parent.getModel().isRef()) {
102       this.getModel().setIsRef(true);
103     }
104   }

118   public List<ModelGraph> getPathFromRootParent() {
119     Vector<ModelGraph> path = new Vector<ModelGraph>();
120     ModelGraph curGraph = this;
121     while (curGraph != null) {
122       path.add(0, curGraph);
123       curGraph = this.parent;
124     }
125     return path;
126   }

128   public void addChild(ModelGraph graph) {
129     if (this.children.size() == 1
130         && GuiUtils.isDummyNode(this.children.get(0).getModel())) {
131       this.children.clear();
132     }
133     this.children.add(graph);
134     graph.setParent(this);
135   }
136 
137   public void removeChild(ModelGraph graph) {
138     this.children.remove(graph);
139     this.getModel().getExcusedSubProcessorIds()
140         .remove(graph.getModel().getModelId());
141     graph.parent = null;
142     if (this.children.size() == 0) {
143       this.addChild(new ModelGraph(GuiUtils.createDummyNode()));
144     }
145   }

147   public List<ModelGraph> getChildren() {
148     if (this.getModel().isParentType() && children.size() == 0) {
149       this.addChild(new ModelGraph(GuiUtils.createDummyNode()));
150     }
151     return children;
152   }

154   public boolean hasChildren() {
155     if (this.children.size() == 1
156         && GuiUtils.isDummyNode(this.children.get(0).getModel())) {
157       return false;
158     } else {
159       return this.children.size() > 0;
160     }

171   public Metadata getInheritedStaticMetadata(ViewState state) {
172     Metadata m = new Metadata();
173     if (this.parent != null) {
174       m.replaceMetadata(this.parent.getInheritedStaticMetadata(state));
175       if (this.parent.getModel().getStaticMetadata() != null) {
176         m.replaceMetadata(this.parent.getModel().getStaticMetadata());
177       }
178       if (this.parent.getModel().getExtendsConfig() != null) {
179         for (String configGroup : this.parent.getModel().getExtendsConfig()) {
180           m.replaceMetadata(state.getGlobalConfigGroups().get(configGroup)
181                                  .getMetadata());
182         }
183       }
184     }
185     return m;
186   }

192   public void setPreConditions(ModelGraph preConditions) {
193     if (this.preConditions != null) {
194       this.preConditions.setParent(null);
195     }
196     Stack<ModelGraph> stack = new Stack<ModelGraph>();
197     stack.add(preConditions);
198     while (!stack.empty()) {
199       ModelGraph graph = stack.pop();
200       graph.isPreCondition = true;
201       stack.addAll(graph.getChildren());
202     }
203     (this.preConditions = preConditions).setParent(this);
204   }

215     stack.add(postConditions);

218       graph.isPostCondition = true;
219       stack.addAll(graph.getChildren());
220     }

222   }

224   public ModelGraph recursiveFind(String id) {
225     Stack<ModelGraph> stack = new Stack<ModelGraph>();
226     stack.add(this);
227     while (!stack.empty()) {
228       ModelGraph curGraph = stack.pop();
229       if (curGraph.getId().equals(id)) {
230         return curGraph;
231       }
232       stack.addAll(curGraph.getChildren());
233       if (curGraph.getPreConditions() != null) {
234         stack.add(curGraph.getPreConditions());
235       }
236       if (curGraph.getPostConditions() != null) {
237         stack.add(curGraph.getPostConditions());
238       }
239     }
240     return null;
241   }

243   public ModelGraph recursiveFindByModelId(String modelId) {
244     Stack<ModelGraph> stack = new Stack<ModelGraph>();
245     stack.add(this);
246     while (!stack.empty()) {
247       ModelGraph curGraph = stack.pop();
248       if (curGraph.getModel().getModelId().equals(modelId)) {

255       if (curGraph.getPostConditions() != null) {
256         stack.add(curGraph.getPostConditions());

258     }

260   }

262   public List<ModelGraph> getLeafNodes() {
263     Vector<ModelGraph> leafNodes = new Vector<ModelGraph>();
264     Stack<ModelGraph> stack = new Stack<ModelGraph>();
265     stack.add(this);
266     while (!stack.empty()) {
267       ModelGraph curGraph = stack.pop();
268       if (curGraph.getChildren().size() == 0) {
269         leafNodes.add(curGraph);
270       } else {
271         stack.addAll(curGraph.getChildren());
272       }
273     }
274     return leafNodes;
275   }

293   public ModelGraph clone() {
294     ModelNode cloneNode = this.model.clone();
295     ModelGraph clone = new ModelGraph(cloneNode);
296     for (ModelGraph child : this.children) {
297       clone.addChild(child.clone());
298     }
299     if (this.preConditions != null) {
300       clone.setPreConditions(this.preConditions.clone());
301     }
302     if (this.postConditions != null) {
303       clone.setPostConditions(this.postConditions.clone());
304     }
305     return clone;
