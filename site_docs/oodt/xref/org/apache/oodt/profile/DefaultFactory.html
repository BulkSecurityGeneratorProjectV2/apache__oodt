16  package org.apache.oodt.profile;
17  
18  import java.util.List;
19  import org.w3c.dom.Element;
20  
21  /**
22   * Default factory for profile objects.
23   *
24   * This factory creates the typical profile objects {@link RangedProfileElement}, {@link
25   * UnspecifiedProfileElement}, {@link EnumeratedProfileElement}, {@link Profile}, {@link
26   * ProfileAttributes}, and {@link ResourceAttributes}.
27   *
28   * @author Kelly
29   * @version $Revision: 1.1.1.1 $
30   */
31  class DefaultFactory implements ObjectFactory {
32  	public RangedProfileElement createRangedProfileElement(Profile profile, String name, String id, String desc, String type,
33  		String unit, List synonyms, boolean obligation, int maxOccurrence, String comments, String min, String max) {
34  		return new RangedProfileElement(profile, name, id, desc, type, unit, synonyms, obligation, maxOccurrence,
35  			comments, min, max);
36  	}
37  	public UnspecifiedProfileElement createUnspecifiedProfileElement(Profile profile, String name, String id, String desc,
38  		String type, String unit, List synonyms, boolean obligation, int maxOccurrence, String comments) {
39  		return new UnspecifiedProfileElement(profile, name, id, desc, type, unit, synonyms, obligation, maxOccurrence,
40  			comments);
41  	}
42  	public EnumeratedProfileElement createEnumeratedProfileElement(Profile profile, String name, String id, String desc,
43  		String type, String unit, List synonyms, boolean obligation, int maxOccurrence, String comments, List values) {
44  		return new EnumeratedProfileElement(profile, name, id, desc, type, unit, synonyms, obligation, maxOccurrence,
45  			comments, values);
46  	}
47  	public Profile createProfile(Element node) {
48  		return new Profile(node, this);
49  	}
50  	public ProfileAttributes createProfileAttributes(Element node) {
51  		return new ProfileAttributes(node);
52  	}
53  	public ResourceAttributes createResourceAttributes(Profile profile, Element node) {
54  		return new ResourceAttributes(profile, node);
55  	}
56  }
