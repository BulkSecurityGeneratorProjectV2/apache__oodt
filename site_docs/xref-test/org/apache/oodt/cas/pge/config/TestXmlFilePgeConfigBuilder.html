36  /**
37   * Test class for {@link XmlFilePgeConfigBuilder}.
38   *
39   * @author bfoster (Brian Foster)
40   */
41  public class TestXmlFilePgeConfigBuilder extends TestCase {
42  
43     public static final String CONFIG_FILE = "src/test/resources/pge-config.xml";
44  
45     public void testBuild() throws IOException {
46        XmlFilePgeConfigBuilder builder = new XmlFilePgeConfigBuilder();
47        PgeMetadata pgeMetadata = new PgeMetadata();
48        pgeMetadata.replaceMetadata(CONFIG_FILE_PATH, CONFIG_FILE);
49        pgeMetadata.replaceMetadata("INPUT_FILE_1", "src/test/resources/data-file-1.txt");
50        pgeMetadata.replaceMetadata("INPUT_FILE_2", "src/test/resources/data-file-2.txt");
51        pgeMetadata.replaceMetadata("WORKING_DIR", "/tmp");
52        pgeMetadata.markAsDynamicMetadataKey();
53        pgeMetadata.commitMarkedDynamicMetadataKeys();
54        PgeConfig pgeConfig = builder.build(pgeMetadata);
55  
56        // Verify metadata checks out.
57        assertEquals(40, pgeMetadata.asMetadata().getAllKeys().size());
58        assertEquals(CONFIG_FILE, pgeMetadata.getMetadata(CONFIG_FILE_PATH));
59        assertEquals("src/test/resources/data-file-1.txt", pgeMetadata.getMetadata("INPUT_FILE_1"));
60        assertEquals("src/test/resources/data-file-2.txt", pgeMetadata.getMetadata("INPUT_FILE_2"));
61        assertEquals("/tmp", pgeMetadata.getMetadata("WORKING_DIR"));
62        assertEquals(">", pgeMetadata.getMetadata("commons/GreaterThan"));
63        assertEquals("<", pgeMetadata.getMetadata("commons/LessThan"));
64        assertEquals("&", pgeMetadata.getMetadata("commons/Ampersand"));
65        assertEquals("'", pgeMetadata.getMetadata("commons/Apostrophe"));
66        assertEquals("\"", pgeMetadata.getMetadata("commons/QuotationMark"));
67        assertEquals("org.apache.oodt.cas.pge.writers.TextConfigFileWriter", pgeMetadata.getMetadata("writers/TextWriter"));
68        assertEquals("org.apache.oodt.cas.pge.writers.CsvConfigFileWriter", pgeMetadata.getMetadata("writers/CsvWriter"));
69        assertEquals("Simple", pgeMetadata.getMetadata("GREETING_ENUM"));
70        assertEquals("true", pgeMetadata.getMetadata("FORCE_STAGING"));
71        assertEquals("Custom", pgeMetadata.getMetadata("CUSTOM_GREETING_ENUM"));
72        assertEquals("<Custom Greeting Here>", pgeMetadata.getMetadata("CUSTOM_GREETING_ENUM_VALUE"));
73        assertEquals("/tmp", pgeMetadata.getMetadata("WorkingDir"));
74        assertEquals("Simple", pgeMetadata.getMetadata("GreetingEnum"));
75        assertEquals("Custom", pgeMetadata.getMetadata("CustomGreetingEnum"));
76        assertEquals("<Custom Greeting Here>", pgeMetadata.getMetadata("CustomGreetingEnumValue"));
77        assertEquals("src/test/resources/data-file-1.txt", pgeMetadata.getMetadata("InputFile1"));
78        assertEquals("src/test/resources/data-file-2.txt", pgeMetadata.getMetadata("InputFile2"));
79        assertEquals("true", pgeMetadata.getMetadata("ForceStaging"));
80        assertEquals(Lists.newArrayList("/tmp/staging/data-file-1.txt", "/tmp/staging/data-file-2.txt"), pgeMetadata.getAllMetadata("InputFiles"));
81        assertEquals("/tmp/config", pgeMetadata.getMetadata("ConfigDir"));
82        assertEquals("/tmp/output", pgeMetadata.getMetadata("OutputDir"));
83        assertEquals("/tmp/staging", pgeMetadata.getMetadata("StagingDir"));
84        assertEquals("dyn-input.txt", pgeMetadata.getMetadata("DynInput/Text/Name"));
85        assertEquals("dyn-input.csv", pgeMetadata.getMetadata("DynInput/CSV/Name"));
86        assertEquals("/tmp/config/dyn-input.txt", pgeMetadata.getMetadata("TextInputFile"));
87        assertEquals("/tmp/config/dyn-input.csv", pgeMetadata.getMetadata("CsvInputFile"));
88        assertEquals("PgeOutput.txt", pgeMetadata.getMetadata("OutputFileName"));
89        assertEquals("\n      [Greeting],\n      This is a template for text file [TextInputFile].\n      This template was written at: [DATE.UTC].\n      [Signature]\n    ", pgeMetadata.getMetadata("TextFileTemplate"));
90        assertEquals("Hello", pgeMetadata.getMetadata("SimpleGreeting"));
91        assertEquals("Hi", pgeMetadata.getMetadata("CasualGreeting"));
92        assertEquals("<Custom Greeting Here>", pgeMetadata.getMetadata("CustomGreeting"));
93        assertEquals("Hello", pgeMetadata.getMetadata("Greating"));
94        assertEquals("-bfost", pgeMetadata.getMetadata("Signature"));
95        assertEquals(Lists.newArrayList("File1.txt","File2.dat","File3.xml"), pgeMetadata.getAllMetadata("AuxInputFiles"));
96        assertEquals(Lists.newArrayList("true","false","true"), pgeMetadata.getAllMetadata("IsText"));
97        assertEquals(Lists.newArrayList("AuxInputFiles","IsText"), pgeMetadata.getAllMetadata("CsvHeader"));
98  
99        // Verify pgeconfig checks out.
100       List<DynamicConfigFile> dynConfigFiles = pgeConfig.getDynamicConfigFiles();
101       assertEquals(2, dynConfigFiles.size());
102       assertEquals("org.apache.oodt.cas.pge.writers.TextConfigFileWriter", dynConfigFiles.get(0).getWriterClass());
103       assertEquals("/tmp/config/dyn-input.txt", dynConfigFiles.get(0).getFilePath());
104       assertEquals(Lists.newArrayList("\n      [Greeting],\n      This is a template for text file [TextInputFile].\n      This template was written at: [DATE.UTC].\n      [Signature]\n    "), Arrays.asList(dynConfigFiles.get(0).getArgs()));
105       assertEquals("org.apache.oodt.cas.pge.writers.CsvConfigFileWriter", dynConfigFiles.get(1).getWriterClass());
106       assertEquals("/tmp/config/dyn-input.csv", dynConfigFiles.get(1).getFilePath());
107       assertEquals(Lists.newArrayList("AuxInputFiles,IsText"), Arrays.asList(dynConfigFiles.get(1).getArgs()));
108 
109       assertEquals(5, pgeConfig.getExeCmds().size());
110       assertEquals("echo /tmp/config/dyn-input.txt > PgeOutput.txt", pgeConfig.getExeCmds().get(0));
111       assertEquals("echo /tmp/config/dyn-input.csv >> PgeOutput.txt", pgeConfig.getExeCmds().get(1));
112       assertEquals("if ( ! -e src/test/resources/data-file-1.txt || ! -e src/test/resources/data-file-2.txt ) then", pgeConfig.getExeCmds().get(2));
113       assertEquals("  exit 1", pgeConfig.getExeCmds().get(3));
114       assertEquals("endif", pgeConfig.getExeCmds().get(4));
115       assertEquals("/tmp", pgeConfig.getExeDir());
116       assertEquals("csh", pgeConfig.getShellType());
117 
118       FileStagingInfo fileStagingInfo = pgeConfig.getFileStagingInfo();
119       assertEquals(Lists.newArrayList("src/test/resources/data-file-2.txt", "src/test/resources/data-file-1.txt"), fileStagingInfo.getFilePaths());
120       assertEquals("/tmp/staging", fileStagingInfo.getStagingDir());
121       assertEquals(true, fileStagingInfo.isForceStaging());
122 
123       List<OutputDir> outputDirs = pgeConfig.getOuputDirs();
124       assertEquals(1, outputDirs.size());
125       assertEquals(true, outputDirs.get(0).isCreateBeforeExe());
126       assertEquals("/tmp/output", outputDirs.get(0).getPath());
127    }
