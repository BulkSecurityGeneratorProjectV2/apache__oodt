18  import java.util.Date;

20  /** A writer for log messages.
21   *
22   * Objects of this class let you write messages into the logging facility. You get these
23   * objects by calling one of the <code>get</code> methods of class {@link Log} and use it
24   * as you would use a {@link java.io.PrintWriter}. This class automatically flushes a message to
25   * the logging facility whenever you call one of the <code>println</code> methods, or
26   * {@link #flush}. It prints all values and objects using the platform's default character
27   * encoding.
28   *
29   * <p>Note that the <code>println</code> methods of this class don't actually write
30   * <em>any</em> line separation characters into the log.  Log listeners will want the
31   * messages without such characters anyway, so this is the correct behavior.
32   *
33   * @see Log
34   * @author Kelly
35   */
36  public class LogWriter extends java.io.Writer {
37  
38    public static final int CAPACITY = 80;
39  
40    /** Constructor.
41  	 *
42  	 * @param timestamp The time for messages logged with the returned writer.
43  	 * @param source The source of the log message.
44  	 * @param category The messages' category.
45  	 */
46  	LogWriter(Date timestamp, String source, Object category) {
47  		buf = new StringBuffer(/*length*/  CAPACITY);
48  		lock = buf;
49  		this.timestamp = timestamp;
50  		this.source = source;
51  		this.category = category;
52  		flushed = false;
53  	}

55  	/** Write a single character.
56  	 *
57  	 * This writes a single character (the low order 16 bits of c) to the log
58  	 * writer. The character isn't flushed to the logging facility until you call one
59  	 * of the <code>println</code> methods or {@link #flush}.
60  	 *
61  	 * @param c The character to write.
62  	 */
63  	public void write(int c) {
64  		if (buf == null) {
65  		  return;
66  		}
67  		buf.append((char) c);
68  	}
69  
70  	/** Write a portion of an array of characters.
71  	 *
72  	 * This writes the given character array starting at offset and going for length bytes.
73  	 *
74  	 * @param array The character array to write.
75  	 * @param offset Where in the array to get characters to write.
76  	 * @param length How many characters to write.
77  	 */
78  	public void write(char[] array, int offset, int length) {
79  		if (buf == null || length == 0) {
80  		  return;
81  		}
82  		if (offset < 0 || offset > array.length || length < 0 || (offset+length) > array.length || (offset+length) < 0) {
83  		  throw new IndexOutOfBoundsException("Can't write " + length + " characters at " + offset
84  											  + " from array whose length is " + array.length);
85  		}
86  		buf.append(array, offset, length);
87  	}
88  
89  	/** Write an array of characters.
90  	 *
91  	 * This writes the entire given array.
92  	 *
93  	 * @param array Array of characters to write.
94  	 */
95  	public void write(char[] array) {
96  		write(array, 0, array.length);
97  	}

99  	/** Write a portion of a string.
100 	 *
101 	 * This writes length characters from the given string, starting offset characters
102 	 * into it.
103 	 *
104 	 * @param string The string to write.
105 	 * @param offset Where in the string to get characters to write.
106 	 * @param length How many characters to write.
107 	 */
108 	public void write(String string, int offset, int length) {
109 		if (buf == null || length == 0) {
110 		  return;
111 		}
112 		buf.append(string.substring(offset, offset + length));
113 	}
114 
115 	/** Write a string.
116 	 *
117 	 * This writes the entire given string.
118 	 *
119 	 * @param string String to write.
120 	 */
121 	public void write(String string) {
122 		write(string, 0, string.length());
123 	}

125 	/** Flush the log writer.
126 	 *
127 	 * This sends any text sent to the writer on its way to the logging facility, and beyond.
128 	 */
129 	public void flush() {
130 		if (buf == null) {
131 		  return;
132 		}
133 		Log.logMessage(timestamp, source, category, buf.toString());
134 		buf.setLength(0);
135 		flushed = true;
136 	}
137 
138 	/** Close the log writer.
139 	 *
140 	 * <p>This flushes any remaining text to the logging facility and then shuts down
141 	 * the log writer. You can't use it again after that (but closing a previously
142 	 * closed log writer is OK).
143 	 */
144 	public void close() {
145 		flush();
146 		buf = null;
147 	}

328 	public void println(Throwable t) {
329 		if (t == null) {
330 		  println("Null throwable");
331 		} else {
332 			StackTraceElement[] frames = t.getStackTrace();
333 				println(t.getClass().getName() + ":");
334 			  for (StackTraceElement frame : frames) {
335 				println(frame);
336 			  }
337 
338 		}
339 	}
