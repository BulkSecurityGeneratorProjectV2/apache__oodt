89     public void testClientTransTrueAndFlatProduct() throws CmdLineActionException, IOException {

92        cliAction.setQuery(QUERY);

110       assertEquals(START_DATE_TIME_MET_KEY, clientSetComplexQuery.getQueryFilter().getStartDateTimeMetKey());

116    public void testClientTransTrueAndFlatProductAndNoOutputFormat() throws CmdLineActionException, IOException {
117       ActionMessagePrinter printer = new ActionMessagePrinter();
118       MockSqlQueryCliAction cliAction = new MockSqlQueryCliAction();
119       cliAction.setQuery(QUERY);
120       cliAction.setSortBy(SORT_BY);
121       cliAction.setDelimiter(DELIMITER);
122       cliAction.setFilterAlgor(FILTER_ALGOR);
123       cliAction.setStartDateTimeMetKey(START_DATE_TIME_MET_KEY);
124       cliAction.setEndDateTimeMetKey(END_DATE_TIME_MET_KEY);
125       cliAction.setPriorityMetKey(PRIORITY_DATE_TIME_MET_KEY);
126       cliAction.setVersionConverter(VERSION_CONV);
127       cliAction.execute(printer);
128       assertEquals(2, printer.getPrintedMessages().size());
129       String msg = printer.getPrintedMessages().get(0);
130       assertTrue(msg.contains("data.dat"));
131       assertTrue(msg.contains("Bob,Billy"));
132       assertEquals(",", msg.replace("data.dat","").replace("Bob,Billy",""));
133       assertEquals("\n", printer.getPrintedMessages().get(1));
134       assertEquals(SORT_BY, clientSetComplexQuery.getSortByMetKey());
135       assertNull(clientSetComplexQuery.getToStringResultFormat());
136       assertNull(clientSetComplexQuery.getReducedProductTypeNames());
137       assertNull(clientSetComplexQuery.getReducedMetadata());
138       assertEquals(FILTER_ALGOR, clientSetComplexQuery.getQueryFilter().getFilterAlgor());
139       assertEquals(START_DATE_TIME_MET_KEY, clientSetComplexQuery.getQueryFilter().getStartDateTimeMetKey());
140       assertEquals(END_DATE_TIME_MET_KEY, clientSetComplexQuery.getQueryFilter().getEndDateTimeMetKey());
141       assertEquals(PRIORITY_DATE_TIME_MET_KEY, clientSetComplexQuery.getQueryFilter().getPriorityMetKey());
142       assertEquals(VERSION_CONV, clientSetComplexQuery.getQueryFilter().getConverter());
143    }

145    public class MockSqlQueryCliAction extends SqlQueryCliAction {
146       @Override
147       public XmlRpcFileManagerClient getClient() throws MalformedURLException,
148             ConnectionException {
149          return new XmlRpcFileManagerClient(new URL("http://localhost:9000"),
150                false) {
151             @Override
152             public List<QueryResult> complexQuery(ComplexQuery complexQuery) {
153                clientSetComplexQuery = complexQuery;
154                Product p = new Product();
155                p.setProductId("TestProductId");
156                Metadata m = new Metadata();
157                m.addMetadata("Filename", TEST_FILENAME);
158                m.addMetadata("Owners", Lists.newArrayList("Bob", "Billy"));
159                QueryResult qr = new QueryResult(p, m);
160                qr.setToStringFormat(complexQuery.getToStringResultFormat());
161                return Lists.newArrayList(qr);
162             }
163          };
164       }
165    }
