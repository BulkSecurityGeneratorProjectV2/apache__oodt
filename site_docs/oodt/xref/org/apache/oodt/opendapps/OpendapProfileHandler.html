49   * A generic reusable OODT {@link ProfileHandler} for use in extracting metadata
50   * from OPeNDAP and THREDDS-accessible datasets.
51   * 
52   */
53  public class OpendapProfileHandler implements ProfileHandler {
54  
55    private static final String PROFILE_HANDLER_ID = "OODT OPeNDAP Profile Handler";
56  
57    private static Logger LOG = Logger.getLogger(OpendapProfileHandler.class
58        .getName());
59  
60    private OpendapConfig conf;
61  
62    public OpendapProfileHandler(){
63    }
64  
65    /**
66     * Implementation of interface method
67     */
68    public List<Profile> findProfiles(XMLQuery xmlQuery) throws ProfileException {
69      String configFileLoc = null;
70      String q = xmlQuery.getKwdQueryString();
71      if (q.contains("ConfigUrl=")){
72      	Pattern parameterPattern = Pattern.compile("ConfigUrl=(.+?)( .*)?$");
73      	Matcher fileMatch = parameterPattern.matcher(q);
74      	while (fileMatch.find()) {
75      		configFileLoc = fileMatch.group(1);
76      	}
77      } else {
78      	configFileLoc = System.getProperty("org.apache.oodt.opendap.config.filePath");
79      }
80      
81      if (configFileLoc.isEmpty()){
82      	throw new ProfileException(
83      		"Configuration file not found. Please specify in System property opendap.config.filePath or as URL parameter ConfigUrl");
84      } else {
85      	try {
86      		this.conf = OpendapConfigReader.read(configFileLoc);
87      	} catch (FileNotFoundException e) {
88      		throw new ProfileException("FileNotFoundException: File not found!");
89      	} catch (MalformedURLException e) {
90      		throw new ProfileException("MalformedURLException: please fix file URL");
91      	}
92      }
93      
94      List<Profile> profiles = new Vector<Profile>();
95      List<DapRoot> roots = this.conf.getRoots();
96  	  
97      // loop over THREDDS catalogs
98      for (DapRoot root : roots) {
99      	LOG.log(Level.INFO,"Parsing DapRoot="+root.getDatasetUrl());
100 
101       DatasetExtractor d = new DatasetExtractor(xmlQuery, root.getCatalogUrl()
102           .toExternalForm(), root.getDatasetUrl().toExternalForm(), conf);
103       if (d.getDapUrls() != null) {
104         for (String opendapUrl : d.getDapUrls()) {
105         	
106           // wrap the profile generation in try-catch to avoid stopping the whole harvesting process in case an exception is thrown
107           try {
108 
109           	LOG.log(Level.FINE,"Connecting to opendapurl="+opendapUrl);
110   
111             Profile profile = new Profile();
112             DConnect dConn;
113             try {
114               dConn = new DConnect(opendapUrl, true);
115             } catch (FileNotFoundException e) {
116               LOG.log(Level.WARNING, "Opendap URL not found: [" + opendapUrl
117                   + "]: Message: " + e.getMessage());
118               throw new ProfileException("Opendap URL not found: [" + opendapUrl
119                   + "]: Message: " + e.getMessage());
120             }
121 
122             // retrieve already extracted THREDDS metadata
123             Metadata datasetMet = d.getDatasetMet(opendapUrl);
124             
125             // extract DAS metadata
126             MetadataExtractor dasExtractor = new DasMetadataExtractor(dConn);
127             dasExtractor.extract(datasetMet, conf);
128             
129             // extract NcML metadata, if available
130            if (datasetMet.containsKey(ThreddsMetadataExtractor.SERVICE_TYPE_NCML)) {
131             	// retrieve URL of NcML document, previously stored
132             	final String ncmlUrl = datasetMet.getMetadata(ThreddsMetadataExtractor.SERVICE_TYPE_NCML);
133             	MetadataExtractor ncmlExtractor = new NcmlMetadataExtractor(ncmlUrl);
134             	ncmlExtractor.extract(datasetMet, conf);
135             }
136             
137             // debug: write out all metadata entries
138             for (String key : datasetMet.getAllKeys()) {
139           	  LOG.log(Level.FINER, "Metadata key="+key+" value="+datasetMet.getMetadata(key));
140             }
141          
142             // <resAttributes>
143             profile.setResourceAttributes(ProfileUtils.getResourceAttributes(
144                 this.conf, opendapUrl, dConn, datasetMet));
145             
146             // <profAttributes>
147             profile.setProfileAttributes(ProfileUtils
148                 .getProfileAttributes(this.conf, datasetMet));
149             // <profElement>
150             profile.getProfileElements().putAll(
151                 ProfileUtils.getProfileElements(this.conf, dConn, datasetMet, profile));
152             profiles.add(profile);
153             LOG.log(Level.FINE, "Added profile id="+profile.getProfileAttributes().getID());
154             
155             
156           } catch(Exception e) {
157           	// in case of exception, don't harvest this dataset, but keep going
158           	LOG.log(Level.WARNING,"Error while building profile for opendapurl="+opendapUrl); 
159           	LOG.log(Level.WARNING,e.getMessage());
160           }
161 
162         }
163       }
164     }
165     return profiles;
166   }
